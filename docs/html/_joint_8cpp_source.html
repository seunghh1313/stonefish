74 Scalar Joint::getFeedback(unsigned int dof)
75 {
76  if(dof > 5)
77  return Scalar(0);
78 
79  if(constraint != nullptr)
80  {
81  btJointFeedback* fb = constraint->getJointFeedback();
82  if(dof < 3)
83  return fb->m_appliedForceBodyA[dof];
84  else
85  return fb->m_appliedTorqueBodyA[dof-3];
86  }
87  else if(mbConstraint != nullptr)
88  {
89  return mbConstraint->getAppliedImpulse(dof);
90  }
91  else
92  return Scalar(0);
93 }

95 void Joint::AddToSimulation(SimulationManager* sm)
96 {
97  if(constraint != nullptr)
98  {
99  //Force feedback
100  btJointFeedback* fb = new btJointFeedback();
101  constraint->enableFeedback(true);
102  constraint->setJointFeedback(fb);
103 
104  //Breaking
105  constraint->setBreakingImpulseThreshold(BT_LARGE_FLOAT);
106 
107  //Solver setup
108  Scalar erp, stopErp;
109  sm->getJointErp(erp, stopErp);
110 
111  if(constraint->getConstraintType() == D6_SPRING_2_CONSTRAINT_TYPE)
112  {
113  for(int i=0; i<6; ++i) // Go through all axes
114  {
115  constraint->setParam(BT_CONSTRAINT_ERP, erp, i);
116  constraint->setParam(BT_CONSTRAINT_STOP_ERP, stopErp, i);
117  constraint->setParam(BT_CONSTRAINT_CFM, 0.0, i);
118  constraint->setParam(BT_CONSTRAINT_STOP_CFM, 0.0, i);
119  }
120  }
121  else // Use default axis
122  {
123  constraint->setParam(BT_CONSTRAINT_ERP, erp);
124  constraint->setParam(BT_CONSTRAINT_STOP_ERP, stopErp); //Avoid explosion by softening the joint limits
125  constraint->setParam(BT_CONSTRAINT_CFM, 0.0);
126  constraint->setParam(BT_CONSTRAINT_STOP_CFM, 0.0);
127  }
128 
129  //Add joint to dynamics world
130  sm->getDynamicsWorld()->addConstraint(constraint, !collisionEnabled);
131  }
132  else if(mbConstraint != nullptr)
133  {
134  Scalar erp, stopErp;
135  sm->getJointErp(erp, stopErp);
136  mbConstraint->setErp(erp);
137  sm->getDynamicsWorld()->addMultiBodyConstraint(mbConstraint);
138  }
139 }
