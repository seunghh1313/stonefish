35 FeatherstoneEntity::FeatherstoneEntity(std::string uniqueName, unsigned int totalNumOfLinks, SolidEntity* baseSolid, bool fixedBase) : Entity(uniqueName)
36 {
37  Scalar M = baseSolid->getAugmentedMass();
38  Vector3 I = baseSolid->getAugmentedInertia();
39  multiBody = new btMultiBody(totalNumOfLinks - 1, M, I, fixedBase, true);
40  multiBody->setBaseWorldTransform(Transform::getIdentity());
41  multiBody->setAngularDamping(Scalar(0));
42  multiBody->setLinearDamping(Scalar(0));
43  multiBody->setMaxAppliedImpulse(BT_LARGE_FLOAT);
44  multiBody->setMaxCoordinateVelocity(Scalar(1000));
45  multiBody->useRK4Integration(false); //Enabling RK4 causes unreallistic energy accumulation (strange motions in 0 gravity)
46  multiBody->useGlobalVelocities(false); //See previous comment
47  multiBody->setHasSelfCollision(false); //No self collision by default
48  multiBody->setUseGyroTerm(true);
49  multiBody->setCanSleep(true);
50 
51  AddLink(baseSolid, Transform::getIdentity());
52 
53  baseRenderable = true;
54 }

72 void FeatherstoneEntity::getAABB(Vector3& min, Vector3& max)
73 {
74  //Initialize AABB
75  min = Vector3(BT_LARGE_FLOAT, BT_LARGE_FLOAT, BT_LARGE_FLOAT);
76  max = Vector3(-BT_LARGE_FLOAT, -BT_LARGE_FLOAT, -BT_LARGE_FLOAT);
77 
78  for(unsigned int i = 0; i < links.size(); i++)
79  {
80  //Get link AABB
81  Vector3 lmin;
82  Vector3 lmax;
83  links[i].solid->multibodyCollider->getCollisionShape()->getAabb(getLinkTransform(i), lmin, lmax);
84 
85  //Merge with other AABBs
86  min[0] = std::min(min[0], lmin[0]);
87  min[1] = std::min(min[1], lmin[1]);
88  min[2] = std::min(min[2], lmin[2]);
89 
90  max[0] = std::max(max[0], lmax[0]);
91  max[1] = std::max(max[1], lmax[1]);
92  max[2] = std::max(max[2], lmax[2]);
93  }
94 }

101 void FeatherstoneEntity::AddToSimulation(SimulationManager* sm, const Transform& origin)
102 {
103  //Creating joint limit constraints
104  for(size_t i=0; i<joints.size(); ++i)
105  {
106  if(joints[i].limit != nullptr)
107  sm->getDynamicsWorld()->addMultiBodyConstraint(joints[i].limit);
108  }
109 
110  //Creating motors (has to be after joint limits and not interleaved!)
111  for(size_t i=0; i<joints.size(); ++i)
112  {
113  if(joints[i].motor != nullptr)
114  sm->getDynamicsWorld()->addMultiBodyConstraint(joints[i].motor);
115  }
116 
117  //Resize matrices
118  multiBody->finalizeMultiDof();
119 
120  //Set origin position
121  setBaseTransform(origin);
122  multiBody->setBaseVel(Vector3(0,0,0));
123  multiBody->setBaseOmega(Vector3(0,0,0));
124 
125  //Move joints to the limits
126  for(size_t i=0; i<joints.size(); ++i)
127  {
128  if(joints[i].limit != nullptr)
129  {
130  if(joints[i].lowerLimit > Scalar(0))
131  multiBody->setJointPos((int)i, joints[i].lowerLimit);
132  else if(joints[i].upperLimit < Scalar(0))
133  multiBody->setJointPos((int)i, joints[i].upperLimit);
134  }
135  }
136 
137  //Calculate constrained link positions to avoid jump at the start of simulation
138  btAlignedObjectArray<Quaternion> scratchQ;
139  btAlignedObjectArray<Vector3> scratchM;
140  multiBody->forwardKinematics(scratchQ, scratchM);
141  multiBody->updateCollisionObjectWorldTransforms(scratchQ, scratchM);
142 
143  //Add multibody to the world
144  sm->getDynamicsWorld()->addMultiBody(multiBody);
145 }

432 int FeatherstoneEntity::AddRevoluteJoint(std::string name, unsigned int parent, unsigned int child, const Vector3& pivot, const Vector3& axis, bool collisionBetweenJointLinks)
433 {
434  //No self joint possible and base cannot be a child
435  if(parent == child || child == 0)
436  return -1;
437 
438  //Check if links exist
439  if(parent >= links.size() || child >= links.size())
440  return -1;
441 
442  //Instantiate joint structure
443  FeatherstoneJoint joint(name, btMultibodyLink::eRevolute, parent, child);
444 
445  //Setup joint
446  //q' = q2 * q1
447  Quaternion ornParentToChild = getLinkTransform(child).getRotation().inverse() * getLinkTransform(parent).getRotation();
448  Vector3 parentComToPivotOffset = getLinkTransform(parent).getBasis().inverse() * (pivot - getLinkTransform(parent).getOrigin());
449  Vector3 pivotToChildComOffset = getLinkTransform(child).getBasis().inverse() * (getLinkTransform(child).getOrigin() - pivot);
450 
451  //Get mass properties (including added mass)
452  Scalar M = links[child].solid->getAugmentedMass();
453  Vector3 I = links[child].solid->getAugmentedInertia();
454 
455  //Setup joint
456  joint.axisInChild = getLinkTransform(child).getBasis().inverse() * axis.normalized();
457  joint.pivotInChild = pivotToChildComOffset;
458  multiBody->setupRevolute(child - 1, M, I, parent - 1, ornParentToChild, joint.axisInChild, parentComToPivotOffset, pivotToChildComOffset, !collisionBetweenJointLinks);
459 
460  //Add feedback
461  joint.feedback = new btMultiBodyJointFeedback();
462  multiBody->getLink((int)child - 1).m_jointFeedback = joint.feedback;
463  joints.push_back(joint);
464 
465  return ((int)joints.size() - 1);
466 }

613 void FeatherstoneEntity::ApplyGravity(const Vector3& g)
614 {
615  bool isSleeping = false;
616 
617  if(multiBody->getBaseCollider() && multiBody->getBaseCollider()->getActivationState() == ISLAND_SLEEPING)
618  isSleeping = true;
619 
620  for(int i=0; i<multiBody->getNumLinks(); ++i)
621  {
622  if(multiBody->getLink(i).m_collider && multiBody->getLink(i).m_collider->getActivationState() == ISLAND_SLEEPING)
623  {
624  isSleeping = true;
625  break;
626  }
627  }
628 
629  if(!isSleeping)
630  {
631  multiBody->addBaseForce(g * links[0].solid->getMass());
632 
633  for(int i=0; i<multiBody->getNumLinks(); ++i)
634  {
635  multiBody->addLinkForce(i, g * links[i+1].solid->getMass());
636  }
637  }
638 }

640 void FeatherstoneEntity::ApplyDamping()
641 {
642  for(unsigned int i=0; i<joints.size(); ++i)
643  {
644  if(joints[i].sigDamping >= SIMD_EPSILON || joints[i].velDamping >= SIMD_EPSILON) //If damping factors not equal zero
645  {
646  Scalar velocity = multiBody->getJointVel(joints[i].child - 1);
647 
648  if(btFabs(velocity) >= SIMD_EPSILON) //If velocity higher than zero
649  {
650  Scalar damping = - velocity/btFabs(velocity) * joints[i].sigDamping - velocity * joints[i].velDamping;
651  multiBody->addJointTorque(joints[i].child - 1, damping);
652  }
653  }
654  }
