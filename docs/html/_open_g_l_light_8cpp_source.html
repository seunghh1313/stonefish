125 void OpenGLLight::Init(std::vector<OpenGLLight*>& lights)
126 {
127  if(lights.size() == 0)
128  return;
129 
130  //Count spotlights
131  unsigned int numOfSpotLights = 0;
132  for(unsigned int i=0; i < lights.size(); ++i)
133  if(lights[i]->getType() == LightType::SPOT) ++numOfSpotLights;
134 
135  //Generate shadowmap array
136  glGenTextures(1, &spotShadowArrayTex);
137  OpenGLState::BindTexture(TEX_BASE, GL_TEXTURE_2D_ARRAY, spotShadowArrayTex);
138  glTexImage3D(GL_TEXTURE_2D_ARRAY, 0, GL_DEPTH_COMPONENT32F, SPOT_LIGHT_SHADOWMAP_SIZE, SPOT_LIGHT_SHADOWMAP_SIZE, numOfSpotLights, 0, GL_DEPTH_COMPONENT, GL_FLOAT, NULL);
139  OpenGLState::UnbindTexture(TEX_BASE);
140 
141  //Generate samplers
142  glGenSamplers(1, &spotDepthSampler);
143  glSamplerParameteri(spotDepthSampler, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
144  glSamplerParameteri(spotDepthSampler, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
145  glSamplerParameteri(spotDepthSampler, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
146  glSamplerParameteri(spotDepthSampler, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
147 
148  glGenSamplers(1, &spotShadowSampler);
149  glSamplerParameteri(spotShadowSampler, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
150  glSamplerParameteri(spotShadowSampler, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
151  glSamplerParameteri(spotShadowSampler, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
152  glSamplerParameteri(spotShadowSampler, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
153  glSamplerParameteri(spotShadowSampler, GL_TEXTURE_COMPARE_MODE, GL_COMPARE_REF_TO_TEXTURE);
154  glSamplerParameteri(spotShadowSampler, GL_TEXTURE_COMPARE_FUNC, GL_LEQUAL);
155 
156  //Initialize lights shadow FBOs
157  numOfSpotLights = 0;
158  for(unsigned int i=0; i < lights.size(); ++i)
159  if(lights[i]->getType() == LightType::SPOT) lights[i]->InitShadowmap(numOfSpotLights++);
160 
161  //Bind textures and samplers
162  OpenGLState::BindTexture(TEX_SPOT_SHADOW, GL_TEXTURE_2D_ARRAY, spotShadowArrayTex);
163  glBindSampler(TEX_SPOT_SHADOW, spotShadowSampler);
164 
165  OpenGLState::BindTexture(TEX_SPOT_DEPTH, GL_TEXTURE_2D_ARRAY, spotShadowArrayTex);
166  glBindSampler(TEX_SPOT_DEPTH, spotDepthSampler);
167 }

169 void OpenGLLight::Destroy()
170 {
171  if(spotShadowArrayTex != 0) glDeleteTextures(1, &spotShadowArrayTex);
172  if(spotDepthSampler != 0) glDeleteSamplers(1, &spotDepthSampler);

174 }
