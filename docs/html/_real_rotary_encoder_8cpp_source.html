52 void RealRotaryEncoder::InternalUpdate(Scalar dt)
53 {
54  if(abs)
55  {
56  //to positive
57  angle = GetRawAngle();
58  if(angle < Scalar(0))
59  angle += FULL_ANGLE;
60 
61  //quantization
62  angle /= FULL_ANGLE;
63  angle = Scalar(trunc(angle * Scalar((1 << cpr_res) - 1))) / Scalar((1 << cpr_res) - 1) * FULL_ANGLE;
64  }
65  else
66  {
67  //new angle
68  Scalar actualAngle = GetRawAngle();
69 
70  //quantization
71  actualAngle /= FULL_ANGLE;
72  actualAngle = Scalar(trunc(actualAngle * cpr_res)) / Scalar(cpr_res) * FULL_ANGLE;
73 
74  //accumulate
75  if(lastAngle * actualAngle < Scalar(0))
76  {
77  if(lastAngle > M_PI_4)
78  angle += ((actualAngle + FULL_ANGLE) - lastAngle);
79  else if(lastAngle < -M_PI_4)
80  angle += (actualAngle - (lastAngle + FULL_ANGLE));
81  else
82  angle += (actualAngle-lastAngle);
83  }
84  else
85  angle += (actualAngle-lastAngle);
86 
87  lastAngle = actualAngle;
88  }
89 
90  //record sample
91  Scalar m[2];
92  m[0] = angle;
93  m[1] = Scalar(0.);
94 
95  Sample s(2, m);
96  AddSampleToHistory(s);
97 }
