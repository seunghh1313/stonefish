81 std::vector<Renderable> Camera::Render()
82 {

84  if(isRenderable())
85  {

90  //Create camera dummy

93  GLfloat aspect = (GLfloat)resX/(GLfloat)resY;
94  GLfloat y = x/aspect;
95 
96  item.points.push_back(glm::vec3(0,0,0));

103  item.points.push_back(glm::vec3(-x, y, iconSize));
104 
105  item.points.push_back(glm::vec3(x, -y, iconSize));
106  item.points.push_back(glm::vec3(x, y, iconSize));
107  item.points.push_back(glm::vec3(x, y, iconSize));
108  item.points.push_back(glm::vec3(-x, y, iconSize));
