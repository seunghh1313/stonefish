79 void Rudder::Update(Scalar dt)
80 {
81  //Update rudder angle
82  theta = setpoint;
83 
84  if(attach != NULL)
85  {
86  Quaternion rudderRot(theta, 0, 0);
87 
88  //Get transforms
89  Transform solidTrans = attach->getCGTransform();
90  // o2a is the transform of the actuator
91  Transform rudderTrans = attach->getOTransform() * o2a * Transform(rudderRot);
92  Vector3 relPos = rudderTrans.getOrigin() - solidTrans.getOrigin();
93 
94  Ocean* ocn = SimulationApp::getApp()->getSimulationManager()->getOcean();
95 
96  Vector3 absVel = attach->getLinearVelocityInLocalPoint(relPos);
97  Vector3 fluidVel = ocn->GetFluidVelocity(rudderTrans.getOrigin());
98  Vector3 velocity = rudderTrans.getBasis().transpose()*(absVel - fluidVel);
99 
100  Scalar angle = atan2(velocity.getY(), velocity.getX());
101 
102  // If rudder facing velocity, reverse equations
103  if (fabs(angle) > .5*M_PI)
104  {
105  angle = atan2(-velocity.getY(), -velocity.getX());
106  }
107 
108  if(ocn->IsInsideFluid(rudderTrans.getOrigin()) && !velocity.isZero())
109  {
110  // Calculate quadratic approximations for lift and drag
111 
112  // Equations from:
113  // Engelhardtsen, Oystein. 3D AUV Collision Avoidance.
114  // MS thesis. Institutt for teknisk kybernetikk, 2007.
115 
116  // Fluid density
117  Scalar rho = ocn->getLiquid().density;
118 
119  // Speed
120  Scalar u = velocity.length();
121 
122  // d = angle
123  Scalar du2 = angle * u*u;
124 
125  // Approximated rudder parameter
126  Scalar X = .5 * area * rho * liftCoeff;
127 
128  // Equation 3.2.19
129  Scalar lift = du2 * X;
130 
131  // Equation 3.2.21
132  Scalar drag = angle * du2 * X;
133 
134  // Drag is opposite to velocity
135  dragV = -velocity;
136  dragV.normalize() *= drag;
137 
138  // Lift is restricted to XY plane
139  // liftV = (-vy, vx, 0)
140  liftV = Vector3(0., 0., -1.).cross(velocity);
141  liftV.normalize() *= lift;
142 
143  // if angle greater than stall -> no lift
144  if (fabs(angle) > stallAngle)
145  {
146  liftV = Vector3(0, 0, 0);
147  }
148 
149  // Torque vectors
150  Vector3 dragT = relPos.cross(rudderTrans.getBasis() * dragV);
151  Vector3 liftT = relPos.cross(rudderTrans.getBasis() * liftV);
152 
153  // Apply forces and torques
154  attach->ApplyTorque(dragT);
155  attach->ApplyTorque(liftT);
156  attach->ApplyCentralForce(rudderTrans.getBasis() * dragV);
157  attach->ApplyCentralForce(rudderTrans.getBasis() * liftV);
158  }
159  }
160 }

162 std::vector<Renderable> Rudder::Render()
163 {
164  Transform rudderTrans = Transform::getIdentity();
165  if(attach != NULL)
166  rudderTrans = attach->getOTransform() * o2a;
167  else
168  LinkActuator::Render();
169 
170  //Rotate rudder
171  rudderTrans *= Transform(Quaternion(theta, 0, 0)) * rudder->getO2GTransform();
172 
173  //Add renderable
174  std::vector<Renderable> items(0);
175  Renderable item;
176  item.type = RenderableType::SOLID;
177  item.materialName = rudder->getMaterial().name;
178  item.objectId = rudder->getGraphicalObject();
179  item.lookId = dm == DisplayMode::GRAPHICAL ? rudder->getLook() : -1;
180  item.model = glMatrixFromTransform(rudderTrans);
181  items.push_back(item);
182 
183  item.type = RenderableType::ACTUATOR_LINES;
184  item.points.push_back(glm::vec3(0,0,0));
185  Vector3 VG = .1*(rudder->getO2GTransform().inverse().getBasis()*(liftV + dragV));
186  item.points.push_back(glm::vec3(VG.getX(),VG.getY(),VG.getZ()));
187  items.push_back(item);
188 
189  return items;
190 }
