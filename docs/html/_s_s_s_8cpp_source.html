36 SSS::SSS(std::string uniqueName, unsigned int numOfBins, unsigned int numOfLines, Scalar verticalBeamWidthDeg,
37  Scalar horizontalBeamWidthDeg, Scalar verticalTiltDeg, Scalar minRange, Scalar maxRange, ColorMap cm, Scalar frequency)

39 {

48  cMap = cm;
49  sonarData = NULL;
50  displayData = NULL;
51  newDataCallback = NULL;
52  glSSS = nullptr;

178 std::vector<Renderable> SSS::Render()
179 {
180  std::vector<Renderable> items = Sensor::Render();
181  if(isRenderable())
182  {
183  Renderable item;
184  item.type = RenderableType::SENSOR_LINES;
185 
186  //Create single transducer dummy
187  int div = 12;
188  GLfloat fovStep = glm::radians(fovH)/(GLfloat)div;
189  //Arcs min
190  GLfloat cosVAngle = cosf(glm::radians(fovV)/2.f) * range.x;
191  GLfloat sinVAngle = sinf(glm::radians(fovV)/2.f) * range.x;
192  GLfloat hAngle = -fovStep*(div/2);
193  for(int i=0; i<=div; ++i)
194  {
195  GLfloat z = cosf(hAngle) * cosVAngle;
196  GLfloat x = sinf(hAngle) * cosVAngle;
197  item.points.push_back(glm::vec3(x, sinVAngle, z));
198  if(i > 0 && i < div)
199  item.points.push_back(glm::vec3(x, sinVAngle, z));
200  hAngle += fovStep;
201  }
202  hAngle = -fovStep*(div/2);
203  for(int i=0; i<=div; ++i)
204  {
205  GLfloat z = cosf(hAngle) * cosVAngle;
206  GLfloat x = sinf(hAngle) * cosVAngle;
207  item.points.push_back(glm::vec3(x, -sinVAngle, z));
208  if(i > 0 && i < div)
209  item.points.push_back(glm::vec3(x, -sinVAngle, z));
210  hAngle += fovStep;
211  }
212  //Arcs max
213  cosVAngle = cosf(glm::radians(fovV)/2.f) * range.y;
214  sinVAngle = sinf(glm::radians(fovV)/2.f) * range.y;
215  hAngle = -fovStep*(div/2);
216  for(int i=0; i<=div; ++i)
217  {
218  GLfloat z = cosf(hAngle) * cosVAngle;
219  GLfloat x = sinf(hAngle) * cosVAngle;
220  item.points.push_back(glm::vec3(x, sinVAngle, z));
221  if(i > 0 && i < div)
222  item.points.push_back(glm::vec3(x, sinVAngle, z));
223  hAngle += fovStep;
224  }
225  hAngle = -fovStep*(div/2);
226  for(int i=0; i<=div; ++i)
227  {
228  GLfloat z = cosf(hAngle) * cosVAngle;
229  GLfloat x = sinf(hAngle) * cosVAngle;
230  item.points.push_back(glm::vec3(x, -sinVAngle, z));
231  if(i > 0 && i < div)
232  item.points.push_back(glm::vec3(x, -sinVAngle, z));
233  hAngle += fovStep;
234  }
235  //Ends
236  hAngle = -fovStep*(div/2);
237  GLfloat zs = cosf(hAngle) * cosVAngle;
238  GLfloat xs = sinf(hAngle) * cosVAngle;
239  item.points.push_back(glm::vec3(xs, sinVAngle, zs));
240  item.points.push_back(glm::vec3(xs, -sinVAngle, zs));
241  hAngle = fovStep*(div/2);
242  GLfloat ze = cosf(hAngle) * cosVAngle;
243  GLfloat xe = sinf(hAngle) * cosVAngle;
244  item.points.push_back(glm::vec3(xe, sinVAngle, ze));
245  item.points.push_back(glm::vec3(xe, -sinVAngle, ze));
246  //Pyramid
247  item.points.push_back(glm::vec3(0,0,0));
248  item.points.push_back(glm::vec3(xs, sinVAngle, zs));
249  item.points.push_back(glm::vec3(0,0,0));
250  item.points.push_back(glm::vec3(xs, -sinVAngle, zs));
251  item.points.push_back(glm::vec3(0,0,0));
252  item.points.push_back(glm::vec3(xe, sinVAngle, ze));
253  item.points.push_back(glm::vec3(0,0,0));
254  item.points.push_back(glm::vec3(xe, -sinVAngle, ze));
255 
256  //Add two transducer dummies
257  GLfloat offsetAngle = M_PI_2 - glm::radians(tilt);
258  glm::mat4 views[2];
259  views[0] = glm::rotate(-offsetAngle, glm::vec3(0.f,1.f,0.f));
260  views[1] = glm::rotate(offsetAngle, glm::vec3(0.f,1.f,0.f));
261  item.model = glMatrixFromTransform(getSensorFrame()) * views[0];
262  items.push_back(item);
263  item.model = glMatrixFromTransform(getSensorFrame()) * views[1];
264  items.push_back(item);
265  }
266  return items;
267 }
