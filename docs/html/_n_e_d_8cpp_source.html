43 void NED::Init(const Scalar lat, const Scalar lon, const Scalar height)
44 {
45  // Save NED origin
46  _init_lat = lat/Scalar(180) * M_PI;
47  _init_lon = lon/Scalar(180) * M_PI;
48  _init_h = height;
49 
50  // Compute ECEF of NED origin
51  Geodetic2Ecef(lat, lon, height, _init_ecef_x, _init_ecef_y, _init_ecef_z);
52 
53  // Compute ECEF to NED and NED to ECEF matrices
54  Scalar phiP = btAtan2(_init_ecef_z, btSqrt(_init_ecef_x*_init_ecef_x + _init_ecef_y*_init_ecef_y));
55 
56  _ecef_to_ned_matrix = __nRe__(phiP, _init_lon);
57  _ned_to_ecef_matrix = __nRe__(_init_lat, _init_lon).transpose();
58 }

60 void NED::Geodetic2Ecef(const Scalar lat, const Scalar lon, const Scalar height,
61  Scalar& x, Scalar& y, Scalar& z) const
62 {
63  // Convert geodetic coordinates to ECEF.
64  // http://code.google.com/p/pysatel/source/browse/trunk/coord.py?r=22
65  Scalar lat_rad = lat/180.0 * M_PI;
66  Scalar lon_rad = lon/180.0 * M_PI;
67  Scalar xi = btSqrt(1 - esq * btSin(lat_rad) * btSin(lat_rad));
68  x = (a / xi + height) * btCos(lat_rad) * btCos(lon_rad);
69  y = (a / xi + height) * btCos(lat_rad) * btSin(lon_rad);
70  z = (a / xi * (1 - esq) + height) * btSin(lat_rad);
71 }

73 void NED::Ecef2Geodetic(const Scalar x, const Scalar y, const Scalar z,
74  Scalar& lat, Scalar& lon, Scalar& height) const
75 {
76  // Convert ECEF coordinates to geodetic.
77  // J. Zhu, "Conversion of Earth-centered Earth-fixed coordinates
78  // to geodetic coordinates," IEEE Transactions on Aerospace and
79  // Electronic Systems, vol. 30, pp. 957-961, 1994.
80  Scalar r = btSqrt(x * x + y * y);
81  Scalar Esq = a * a - b * b;
82  Scalar F = 54 * b * b * z * z;
83  Scalar G = r * r + (1 - esq) * z * z - esq * Esq;
84  Scalar C = (esq * esq * F * r * r) / btPow(G, 3);
85  Scalar S = __cbrt__(1 + C + btSqrt(C * C + 2 * C));
86  Scalar P = F / (3 * btPow((S + 1 / S + 1), 2) * G * G);
87  Scalar Q = btSqrt(1 + 2 * esq * esq * P);
88  Scalar r_0 = -(P * esq * r) / (1 + Q) + btSqrt(0.5 * a * a * (1 + 1.0 / Q) - P * (1 - esq) * z * z / (Q * (1 + Q)) - 0.5 * P * r * r);
89  Scalar U = btSqrt(btPow((r - esq * r_0), 2) + z * z);
90  Scalar V = btSqrt(btPow((r - esq * r_0), 2) + (1 - esq) * z * z);
91  Scalar Z_0 = b * b * z / (a * V);
92  height = U * (1 - b * b / (a * V));
93  lat = (btAtan((z + e1sq * Z_0) / r))/M_PI * 180.0;
94  lon = (btAtan2(y, x))/M_PI * 180.0;
95 }

97 void NED::Ecef2Ned(const Scalar x, const Scalar y, const Scalar z,
98  Scalar& north, Scalar& east, Scalar& depth) const
99 {
100  // Converts ECEF coordinate pos into local-tangent-plane ENU
101  // coordinates relative to another ECEF coordinate ref. Returns a tuple
102  // (East, North, Up).
103  Vector3 vect, ret;
104  vect.setX(x - _init_ecef_x);
105  vect.setY(y - _init_ecef_y);
106  vect.setZ(z - _init_ecef_z);
107  ret = _ecef_to_ned_matrix * vect;
108  north = ret.getX();
109  east = ret.getY();
110  depth = -ret.getZ();
111 }

113 void NED::Ned2Ecef(const Scalar north, const Scalar east, const Scalar depth,
114  Scalar& x, Scalar& y, Scalar& z) const
115 {
116  // NED (north/east/down) to ECEF coordinate system conversion.
117  Vector3 ned, ret;
118  ned.setX(north);
119  ned.setY(east);
120  ned.setZ(-depth);
121  ret = _ned_to_ecef_matrix * ned;
122  x = ret.getX() + _init_ecef_x;
123  y = ret.getY() + _init_ecef_y;
124  z = ret.getZ() + _init_ecef_z;
125 }

146 {
147  if(x >= 0.0)

149  else
150  return -btPow(btFabs(x), 1.0/3.0);

153 Matrix3 NED::__nRe__(const Scalar lat_rad, const Scalar lon_rad) const
154 {
155  Scalar sLat = btSin(lat_rad);
156  Scalar sLon = btSin(lon_rad);
157  Scalar cLat = btCos(lat_rad);
158  Scalar cLon = btCos(lon_rad);
159 
160  Matrix3 ret(-sLat*cLon, -sLat*sLon, cLat,
161  -sLon, cLon, 0,
162  cLat*cLon, cLat*sLon, sLat);
163 
164  return ret;
165 }
