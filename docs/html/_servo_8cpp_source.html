56 void Servo::setDesiredPosition(Scalar pos)
57 {
58  if(btFuzzyZero(pSetpoint - pos)) //Check if setpoint changed
59  return;
60 
61  if(fe != nullptr)
62  {
63  FeatherstoneJoint jnt = fe->getJoint(jId);
64  if(jnt.lowerLimit < jnt.upperLimit) //Does it have joint limits?
65  pos = pos < jnt.lowerLimit ? jnt.lowerLimit : (pos > jnt.upperLimit ? jnt.upperLimit : pos);
66  }
67 
68  pSetpoint = pos;
69 }

105  case JointType::REVOLUTE:

107

188 void Servo::Update(Scalar dt)
189 {
190  if(j != nullptr)
191  {
192  Scalar vSetpoint2;
193  if(mode == ServoControlMode::POSITION || btFuzzyZero(vSetpoint))
194  {
195  Scalar err = pSetpoint - getPosition();
196  vSetpoint2 = Kp * err;
197  }
198  else
199  {
200  vSetpoint2 = vSetpoint; //Kv * (vSetpoint - getVelocity()) + vSetpoint; (not natural to use)
201  }
202 
203  switch(j->getType())
204  {
205  case JointType::REVOLUTE:
206  ((RevoluteJoint*)j)->setMotorVelocity(vSetpoint2);
207  break;
208 
209  default:
210  break;
211  }
212  }
213  else if(fe != nullptr)
214  {
215  //Use internal multibody motors
216  switch(mode)
217  {
218  case ServoControlMode::POSITION:
219  {
220  fe->MotorPositionSetpoint(jId, pSetpoint, Kp);
221  fe->MotorVelocitySetpoint(jId, Scalar(0), Kv);
222  }
223  break;
224 
225  case ServoControlMode::VELOCITY:
226  case ServoControlMode::TORQUE:
227  {
228  if(btFuzzyZero(vSetpoint))
229  {
230  fe->MotorPositionSetpoint(jId, pSetpoint, Kp);
231  fe->MotorVelocitySetpoint(jId, Scalar(0), Kv);
232  }
233  else
234  {
235  Scalar vSetpoint2 = vSetpoint;
236 
237  //Do not allow to cross limits by changing velocity setpoint
238  FeatherstoneJoint jnt = fe->getJoint(jId);
239  if(jnt.lowerLimit < jnt.upperLimit)
240  {
241  Scalar jpos = getPosition();
242  Scalar jpos2 = jpos + vSetpoint2 * dt;
243  vSetpoint2 = jpos2 < jnt.lowerLimit ? (jnt.lowerLimit - jpos)/dt : (jpos2 > jnt.upperLimit ? (jnt.upperLimit - jpos)/dt : vSetpoint2);
244  }
245 
246  fe->MotorPositionSetpoint(jId, Scalar(0), Scalar(0));
247  fe->MotorVelocitySetpoint(jId, vSetpoint2, Kv);
248  }
249  }
250  break;
251  }
252  }
253 }
