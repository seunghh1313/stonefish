36 RevoluteJoint::RevoluteJoint(std::string uniqueName, SolidEntity* solidA, SolidEntity* solidB, const Vector3& pivot, const Vector3& axis, bool collideLinked) : Joint(uniqueName, collideLinked)
37 {
38  Vector3 hingeAxis = axis.normalized();
39  btRigidBody* bodyA = solidA->rigidBody;
40  btRigidBody* bodyB = solidB->rigidBody;
41  axisInA = bodyA->getCenterOfMassTransform().getBasis().inverse() * hingeAxis;
42  Vector3 axisInB = bodyB->getCenterOfMassTransform().getBasis().inverse() * hingeAxis;
43  pivotInA = bodyA->getCenterOfMassTransform().inverse()(pivot);
44  Vector3 pivotInB = bodyB->getCenterOfMassTransform().inverse()(pivot);
45 
46  btHingeConstraint* hinge = new btHingeConstraint(*bodyA, *bodyB, pivotInA, pivotInB, axisInA, axisInB, true);
47  hinge->setLimit(Scalar(1), Scalar(-1)); //no limit (min > max)
48  setConstraint(hinge);
49 
50  sigDamping = Scalar(0);
51  velDamping = Scalar(0);
52  angleOffset = hinge->getHingeAngle();
53  cInfo("Created revolute joint '%s'. Offset: %lf rad.", uniqueName.c_str(), angleOffset);
54  setIC(Scalar(0));
55 }

81 void RevoluteJoint::setLimits(Scalar min, Scalar max)
82 {
83  if(min > max) // No limit
84  {
85  btHingeConstraint* hinge = (btHingeConstraint*)getConstraint();
86  hinge->setLimit(Scalar(1), Scalar(-1));
87  return;
88  }
89 
90  btHingeConstraint* hinge = (btHingeConstraint*)getConstraint();
91  min = btFmod(min + angleOffset, SIMD_2_PI);
92  max = btFmod(max + angleOffset, SIMD_2_PI);
93  hinge->setLimit(min, max);
94  cInfo("Setting limits of revolute joint '%s': %lf, %lf.", getName().c_str(), min, max);
95 }

114 {
115  btRigidBody& bodyA = getConstraint()->getRigidBodyA();
116  btRigidBody& bodyB = getConstraint()->getRigidBodyB();
117  Vector3 relativeAV = bodyA.getAngularVelocity() - bodyB.getAngularVelocity();

145 void RevoluteJoint::ApplyDamping()
146 {
147  if(sigDamping > Scalar(0.) || velDamping > Scalar(0.))
148  {
149  btRigidBody& bodyA = getConstraint()->getRigidBodyA();
150  btRigidBody& bodyB = getConstraint()->getRigidBodyB();
151  Vector3 axis = (bodyA.getCenterOfMassTransform().getBasis() * axisInA).normalized();
152  Vector3 relativeAV = bodyA.getAngularVelocity() - bodyB.getAngularVelocity();
153  Scalar av = relativeAV.dot(axis);
154 
155  if(av != Scalar(0.))
156  {
157  Scalar T = sigDamping * av/fabs(av) + velDamping * av;
158  Vector3 torque = axis * -T;
159 
160  bodyA.applyTorque(torque);
161  bodyB.applyTorque(-torque);
162  }
163  }
164 }

166 bool RevoluteJoint::SolvePositionIC(Scalar linearTolerance, Scalar angularTolerance)
167 {
168  Scalar lastAngleICError = angleICError;
169  angleICError = angleIC - getAngle();
170 
171  //Check if IC reached
172  if(btFabs(angleICError) < angularTolerance)
173  return true;
174 
175  //Move joint
176  Scalar torque = angleICError * Scalar(50) + (angleICError - lastAngleICError) * Scalar(300);
177  ApplyTorque(torque);
178 
179  return false;
180 }

182 std::vector<Renderable> RevoluteJoint::Render()
183 {
184  std::vector<Renderable> items(0);
185  Renderable item;
186  item.model = glm::mat4(1.f);
187  item.type = RenderableType::JOINT_LINES;
188 
189  btTypedConstraint* revo = getConstraint();
190  Vector3 A = revo->getRigidBodyA().getCenterOfMassPosition();
191  Vector3 B = revo->getRigidBodyB().getCenterOfMassPosition();
192  Vector3 pivot = revo->getRigidBodyA().getCenterOfMassTransform()(pivotInA);
193  Vector3 axis = (revo->getRigidBodyA().getCenterOfMassTransform().getBasis() * axisInA).normalized();
194 
195  //Calculate axis ends
196  Vector3 C1 = pivot;
197  Vector3 C2 = pivot + axis * btMax(0.05, btFabs((A-B).safeNorm())/Scalar(2));
198  item.points.push_back(glm::vec3(C1.getX(), C1.getY(), C1.getZ()));
199  item.points.push_back(glm::vec3(C2.getX(), C2.getY(), C2.getZ()));
200 
201  items.push_back(item);
202 
203  return items;
204 }
