37 Thruster::Thruster(std::string uniqueName, SolidEntity* propeller, Scalar diameter, std::pair<Scalar, Scalar> thrustCoeff, Scalar torqueCoeff, Scalar maxRPM, bool rightHand, bool inverted) : LinkActuator(uniqueName)
38 {
39  D = diameter;
40  kT = thrustCoeff;
41  kQ = torqueCoeff;
42  alpha = Scalar(-1) * kT.first;
43  beta = Scalar(-1) * kQ;
44  kp = Scalar(8.0);
45  ki = Scalar(3.0);
46  iLim = Scalar(2.0);
47  RH = rightHand;
48  inv = inverted;
49  omegaLim = maxRPM/Scalar(60) * Scalar(2) * M_PI; //[rad/s] (always positive)

51  theta = Scalar(0);
52  omega = Scalar(0);
53  thrust = Scalar(0);
54  torque = Scalar(0);
55  setpoint = Scalar(0);
56  iError = Scalar(0);
57 
58  prop = propeller;
59  prop->BuildGraphicalObject();
60 }

114 void Thruster::Update(Scalar dt)
115 {
116  if(attach != NULL)
117  {
118  //Update thruster velocity
119  Scalar error = setpoint * omegaLim - omega;
120  Scalar motorTorque = kp * error + ki * iError;
121  omega += (motorTorque - (-torque))*dt;
122  theta += omega * dt; //Just for animation
123 
124  //Integrate error
125  iError += error * dt;
126  iError = iError > iLim ? iLim : (iError < -iLim ? -iLim : iError);
127 
128  //Get transforms
129  Transform solidTrans = attach->getCGTransform();
130  Transform thrustTrans = attach->getOTransform() * o2a;
131  Vector3 relPos = thrustTrans.getOrigin() - solidTrans.getOrigin();
132  Vector3 velocity = attach->getLinearVelocityInLocalPoint(relPos);
133 
134  //Calculate thrust
135  Ocean* ocn = SimulationApp::getApp()->getSimulationManager()->getOcean();
136  if(ocn != nullptr && ocn->IsInsideFluid(thrustTrans.getOrigin()))
137  {
138  bool backward = (RH && omega < Scalar(0)) || (!RH && omega > Scalar(0));
139 
140  /*kT and kQ depend on the advance ratio J
141  J = u/(omega*D), where:
142  u - ambient velocity [m/s]
143  n - propeller rotational rate [1/s]
144  D - propeller diameter [m] */
145  Scalar n = (backward ? Scalar(-1) : Scalar(1)) * btFabs(omega)/(Scalar(2) * M_PI); // Accounts for propoller handedness
146  Scalar u = -thrustTrans.getBasis().getColumn(0).dot(ocn->GetFluidVelocity(thrustTrans.getOrigin()) - velocity); //Incoming water velocity
147 
148  //Thrust
149  Scalar kT0 = backward ? kT.second : kT.first; //In case of non-symmetrical thrusters the coefficient may be different
150  //kT(J) = kT0 + alpha * J --> approximated with linear function
151  thrust = ocn->getLiquid().density * D*D*D * btFabs(n) * (D*kT0*n + alpha*u);
152 
153  //Torque
154  Scalar kQ0 = kQ;
155  //kQ(J) = kQ0 + beta * J --> approximated with linear function
156  torque = (RH ? Scalar(-1) : Scalar(1)) * ocn->getLiquid().density * D*D*D*D * btFabs(n) * (D*kQ0*n + beta*u); //Torque is the loading of propeller due to water resistance (reaction force)
157 
158  //Apply forces and torques
159  Vector3 thrustV(thrust, 0, 0);
160  Vector3 torqueV(torque, 0, 0);
161  attach->ApplyCentralForce(thrustTrans.getBasis() * thrustV);
162  attach->ApplyTorque((thrustTrans.getOrigin() - solidTrans.getOrigin()).cross(thrustTrans.getBasis() * thrustV));
163  attach->ApplyTorque(thrustTrans.getBasis() * torqueV);
164  }
165  else
166  {
167  thrust = Scalar(0);
168  torque = Scalar(0);
169  }
170  }
171 }

173 std::vector<Renderable> Thruster::Render()
174 {
175  Transform thrustTrans = Transform::getIdentity();
176  if(attach != NULL)
177  thrustTrans = attach->getOTransform() * o2a;
178  else
179  LinkActuator::Render();
180 
181  //Rotate propeller
182  thrustTrans *= Transform(Quaternion(0, 0, theta), Vector3(0,0,0));
183 
184  //Add renderable
185  std::vector<Renderable> items(0);
186  Renderable item;
187  item.type = RenderableType::SOLID;
188  item.materialName = prop->getMaterial().name;
189  item.objectId = prop->getGraphicalObject();
190  item.lookId = dm == DisplayMode::GRAPHICAL ? prop->getLook() : -1;
191  item.model = glMatrixFromTransform(thrustTrans);
192  items.push_back(item);
193 
194  item.type = RenderableType::ACTUATOR_LINES;
195  item.points.push_back(glm::vec3(0,0,0));
196  item.points.push_back(glm::vec3(0.1f*thrust,0,0));
197  items.push_back(item);
198 
199  return items;
200 }
