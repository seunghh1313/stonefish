31 Pipe::Pipe(const Vector3& point1, const Vector3& point2, Scalar radius1, Scalar radius2, Scalar inletVelocity, Scalar exponent)
32 {

34  l = (point2-point1).length();
35  n = (point2-point1)/l;

37  r2 = radius2;

47 Vector3 Pipe::GetVelocityAtPoint(const Vector3& p) const
48 {
49  //Calculate distance to line
50  Vector3 p1p = p-p1;
51  Scalar d = p1p.cross(n).norm();
52 
53  //Calculate closest point on line section between P1 and P2
54  Scalar t = p1p.dot(n);
55  if(t < 0.0 || t > l) return Vector3(0,0,0);
56 
57  //Calculate radius at point
58  Scalar r = r1 + (r2-r1) * t/l;
59  if(d >= r) return Vector3(0,0,0);
60 
61  //Calculate central velocity
62  Vector3 v = r1/r * vin * n;
63 
64  //Calculate fraction of central velocity
65  Scalar f = btPow(Scalar(1)-d/r, gamma);
66 
67  return f*v;
68 }

70 std::vector<Renderable> Pipe::Render(VelocityFieldUBO& ubo)
71 {
72  std::vector<Renderable> items(0);
73  ubo.posR = glm::vec4((GLfloat)p1.getX(), (GLfloat)p1.getY(), (GLfloat)p1.getZ(), (GLfloat)r1);
74  ubo.dirV = glm::vec4((GLfloat)n.getX(), (GLfloat)n.getY(), (GLfloat)n.getZ(), (GLfloat)vin);
75  ubo.params = glm::vec3((GLfloat)l, (GLfloat)r2, (GLfloat)gamma);
76  ubo.type = 2;
77 
78  //Model matrix
79  glm::vec3 z_(n.x(), n.y(), n.z());
80  glm::vec3 x_(n.y(), n.x(), n.z());
81  glm::vec3 y_ = glm::cross(z_,x_);
82  glm::mat4 model(glm::vec4(x_, 0.f), glm::vec4(y_, 0.f), glm::vec4(z_, 0.f), glm::vec4(p1.x(), p1.y(), p1.z(), 1));
83 
84  //Inlet and outlet
85  Renderable inlet;
86  inlet.type = RenderableType::HYDRO_LINE_STRIP;
87  inlet.model = model;
88 
89  Renderable outlet;
90  outlet.type = RenderableType::HYDRO_LINE_STRIP;
91  outlet.model = model;
92 
93  //Pipe
94  Renderable pipe;
95  pipe.type = RenderableType::HYDRO_LINES;
96  pipe.model = model;
97  pipe.points.push_back(glm::vec3(0, 0, 0));
98  pipe.points.push_back(glm::vec3(0, 0, l));
99 
100  for(unsigned int i=0; i<12; ++i)
101  {
102  Scalar alpha = Scalar(i)/Scalar(12) * M_PI * Scalar(2);
103  Vector3 v1(btCos(alpha)*r1, btSin(alpha)*r1, 0);
104  Vector3 v2(v1.x()*r2/r1, v1.y()*r2/r1, l);
105  pipe.points.push_back(glm::vec3(v1.x(), v1.y(), v1.z()));
106  inlet.points.push_back(pipe.points.back());
107  pipe.points.push_back(glm::vec3(v2.x(), v2.y(), v2.z()));
108  outlet.points.push_back(pipe.points.back());
109  }
110 
111  inlet.points.push_back(inlet.points.front());
112  outlet.points.push_back(outlet.points.front());
113 
114  items.push_back(inlet);
115  items.push_back(outlet);
116  items.push_back(pipe);
117  return items;
118 }
