65 void DVL::InternalUpdate(Scalar dt)
66 {
67  /*
68  Status:
69  0 - bottom ping only
70  1 - water ping only
71  2 - bottom and water ping
72  3 - no ping at all
73  */
74  //Check hit with bottom
75  unsigned short status = 0;
76  Transform dvlTrans = getSensorFrame();
77 
78  //Simulate 4 beam DVL (typical design)
79  Vector3 dir[4];
80  Vector3 from[4];
81  Vector3 to[4];
82  for(unsigned int i=0; i<4; ++i)
83  {
84  Scalar alpha = M_PI_4 + i * M_PI_2;
85  dir[i] = dvlTrans.getBasis().getColumn(2) * btCos(beamAngle)
86  + (dvlTrans.getBasis().getColumn(0) * btCos(alpha) + dvlTrans.getBasis().getColumn(1) * btSin(alpha)) * btSin(beamAngle);
87  }
88 
89  Scalar dirFactor = beamPosZ ? Scalar(1) : Scalar(-1);
90 
91  unsigned int divs = ceil(channels[3].rangeMax - channels[3].rangeMin);
92  Scalar minRange(-1);
93 
94  for(unsigned int i=0; i<4; ++i)
95  {
96  from[i] = dvlTrans.getOrigin() + dirFactor * dir[i] * channels[3].rangeMin;
97  to[i] = dvlTrans.getOrigin() + dirFactor * dir[i] * channels[3].rangeMax;
98 
99  Vector3 step = (to[i]-from[i])/Scalar(divs);
100  range[i] = Scalar(-1);
101 
102  for(unsigned int h=0; h<divs; ++h)
103  {
104  Vector3 from_ = from[i] + step*Scalar(h);
105  Vector3 to_ = from[i] + step*Scalar(h+1);
106  btCollisionWorld::ClosestRayResultCallback closest(from_, to_);
107  closest.m_collisionFilterGroup = MASK_DYNAMIC;
108  closest.m_collisionFilterMask = MASK_STATIC | MASK_DYNAMIC | MASK_ANIMATED_COLLIDING;
109  SimulationApp::getApp()->getSimulationManager()->getDynamicsWorld()->rayTest(from_, to_, closest);
110 
111  if(closest.hasHit())
112  {
113  Vector3 p = from_.lerp(to_, closest.m_closestHitFraction);
114  range[i] = (p - dvlTrans.getOrigin()).length();
115  break;
116  }
117  }
118 
119  if(range[i] > Scalar(0) && (range[i] < minRange || minRange < Scalar(0)))
120  minRange = range[i];
121  }
122 
123  //Get altitude
124  Scalar altitude = channels[3].rangeMax;
125  Vector3 v = V0();
126  status = 3;
127 
128  //Check where the out of range happened
129  bool tooClose = false;
130  if(minRange < Scalar(0)) //No hit recorded in DVL operating range
131  {
132  for(unsigned int i=0; i<4; ++i)
133  {
134  range[i] = Scalar(-1);
135  from[i] = dvlTrans.getOrigin() + dirFactor * dir[i] * channels[3].rangeMin;
136  to[i] = dvlTrans.getOrigin();
137  btCollisionWorld::ClosestRayResultCallback closest(from[i], to[i]);
138  closest.m_collisionFilterGroup = MASK_DYNAMIC;
139  closest.m_collisionFilterMask = MASK_STATIC | MASK_DYNAMIC | MASK_ANIMATED_COLLIDING;
140  SimulationApp::getApp()->getSimulationManager()->getDynamicsWorld()->rayTest(from[i], to[i], closest);
141 
142  if(closest.hasHit() && btDot(closest.m_hitNormalWorld, dirFactor * dir[i]) > Scalar(0))
143  {
144  Vector3 p = from[i].lerp(to[i], closest.m_closestHitFraction);
145  range[i] = (p - dvlTrans.getOrigin()).length();
146  if(range[i] < minRange || minRange < Scalar(0)) minRange = range[i];
147  }
148  }
149  if(minRange > Scalar(0)) //Detection closer than minimum range of DVL - no water ping possible
150  tooClose = true;
151  }
152  else //Successful bottom ping
153  {
154  altitude = minRange * btCos(beamAngle);
155  v = dvlTrans.getBasis().inverse() * attach->getLinearVelocityInLocalPoint(dvlTrans.getOrigin() - attach->getCGTransform().getOrigin());
156  status = 0;
157  }
158 
159  //Get water velocity
160  Vector3 wv = V0();
161  //Water velocity is reported as 0 if:
162  //a) layer thickness = 0
163  //b) compressed layer thickness is less than minimum layer thickness
164  //ASSUME: Water layer far boundary has to be closer than 80% of altitude.
165  if(!tooClose && waterLayer.getX() > Scalar(0) && Scalar(0.8)*altitude > waterLayer.getX() + waterLayer.getY()) //Water layer ping possible
166  {
167  Ocean* ocn = SimulationApp::getApp()->getSimulationManager()->getOcean();
168  if(ocn != nullptr)
169  {
170  Vector3 zDir = dvlTrans.getBasis().getColumn(2);
171  //Layer compressed until minimum layer thinckness is reached.
172  Scalar layerSize = btClamped(Scalar(0.8) * altitude - waterLayer.getY(), waterLayer.getX(), waterLayer.getZ()-waterLayer.getY());
173  //Sample velocity
174  unsigned int n = ceil(layerSize/Scalar(0.1)); //ASSUME: Mesurement resolution = 0.1 m.
175  Scalar dist = waterLayer.getY();
176  Scalar weight(0);
177  for(unsigned int i=0; i<=n; ++i)
178  {
179  Scalar x = Scalar(i)/Scalar(n);
180  Scalar w = btMin(x, 1-x); //f(x) = mu min{x, 1-x} where mu changes the sharpness of the triangle
181  Vector3 p = dvlTrans.getOrigin() + zDir * dist;
182  wv += w * ocn->GetFluidVelocity(p);
183  weight += w;
184  dist += layerSize/Scalar(n);
185  }
186  wv /= weight;
187  }
188  //Transform to DVL frame and add DVL velocity
189  wv = dvlTrans.getBasis().inverse() * wv - v;
190  status = status == 3 ? 1 : 2;
191  }
192 
193  //Update noise characteristics
194  channels[0].setStdDev(mulNoiseFactor[0] * v.x() + addNoiseStdDev[0]);
195  channels[1].setStdDev(mulNoiseFactor[0] * v.y() + addNoiseStdDev[0]);
196  channels[2].setStdDev(mulNoiseFactor[0] * v.z() + addNoiseStdDev[0]);
197  channels[4].setStdDev(mulNoiseFactor[1] * wv.x() + addNoiseStdDev[1]);
198  channels[5].setStdDev(mulNoiseFactor[1] * wv.y() + addNoiseStdDev[1]);
199  channels[6].setStdDev(mulNoiseFactor[1] * wv.z() + addNoiseStdDev[1]);
200 
201  //Save data
202  Scalar data[8] = {v.x(), v.y(), v.z(), altitude, wv.x(), wv.y(), wv.z(), Scalar(status)};
203  Sample s(8, data);
204  AddSampleToHistory(s);
205 }

207 std::vector<Renderable> DVL::Render()
208 {
209  std::vector<Renderable> items = LinkSensor::Render();
210  if(isRenderable())
211  {
212  unsigned short status = (unsigned short)trunc(getLastValue(7));
213  Renderable item;
214  item.type = RenderableType::SENSOR_LINES;
215  item.model = glMatrixFromTransform(getSensorFrame());
216  //Bottom ping
217  if(status == 0 || status == 2) //Good bottom ping
218  {
219  //Beams
220  Vector3 dir[4];
221  for(unsigned int i=0; i<4; ++i)
222  {
223  Scalar alpha = M_PI_4 + i * M_PI_2;
224  dir[i] = Vector3(0,0,1) * btCos(beamAngle)
225  + (Vector3(1,0,0) * btCos(alpha) + Vector3(0,1,0) * btSin(alpha)) * btSin(beamAngle);
226  }
227 
228  if(!beamPosZ)
229  {
230  dir[0] = -dir[0];
231  dir[1] = -dir[1];
232  dir[2] = -dir[2];
233  dir[3] = -dir[3];
234  }
235 
236  if(range[0] > Scalar(0))
237  {
238  item.points.push_back(glm::vec3(0,0,0));
239  item.points.push_back(glm::vec3(dir[0].x()*range[0], dir[0].y()*range[0], dir[0].z()*range[0]));
240  }
241 
242  if(range[1] > Scalar(0))
243  {
244  item.points.push_back(glm::vec3(0,0,0));
245  item.points.push_back(glm::vec3(dir[1].x()*range[1], dir[1].y()*range[1], dir[1].z()*range[1]));
246  }
247 
248  if(range[2] > Scalar(0))
249  {
250  item.points.push_back(glm::vec3(0,0,0));
251  item.points.push_back(glm::vec3(dir[2].x()*range[2], dir[2].y()*range[2], dir[2].z()*range[2]));
252  }
253 
254  if(range[3] > Scalar(0))
255  {
256  item.points.push_back(glm::vec3(0,0,0));
257  item.points.push_back(glm::vec3(dir[3].x()*range[3], dir[3].y()*range[3], dir[3].z()*range[3]));
258  }
259  }
260  //Water ping
261  if(status == 1 || status == 2) //Good water ping
262  {
263  Scalar layerSize = btClamped(Scalar(0.8) * getLastValue(3) - waterLayer.getY(), waterLayer.getX(), waterLayer.getZ()-waterLayer.getY());
264  GLfloat a1 = (GLfloat)waterLayer.getY();
265  GLfloat a2 = (GLfloat)(waterLayer.getY() + layerSize);
266  GLfloat r1 = a1 * btSin(beamAngle);
267  GLfloat r2 = a2 * btSin(beamAngle);
268  for(unsigned int i=0; i<4; ++i)
269  {
270  GLfloat ang1 = (GLfloat)i/2.f * glm::pi<GLfloat>();
271  GLfloat ang2 = (GLfloat)(i+1)/2.f * glm::pi<GLfloat>();
272  glm::vec3 d1(glm::sin(ang1), glm::cos(ang1), 0.f);
273  glm::vec3 d2(glm::sin(ang2), glm::cos(ang2), 0.f);
274  item.points.push_back(r1 * d1 + glm::vec3(0.f, 0.f, -a1));
275  item.points.push_back(r1 * d2 + glm::vec3(0.f, 0.f, -a1));
276  item.points.push_back(r2 * d1 + glm::vec3(0.f, 0.f, -a2));
277  item.points.push_back(r2 * d2 + glm::vec3(0.f, 0.f, -a2));
278  }
279  }
280  items.push_back(item);
281  }
282  return items;
283 }

285 void DVL::setRange(const Vector3& velocityMax, Scalar altitudeMin, Scalar altitudeMax)
286 {
287  //Velocity
288  channels[0].rangeMin = -btClamped(velocityMax.getX(), Scalar(0), Scalar(BT_LARGE_FLOAT));
289  channels[1].rangeMin = -btClamped(velocityMax.getY(), Scalar(0), Scalar(BT_LARGE_FLOAT));
290  channels[2].rangeMin = -btClamped(velocityMax.getZ(), Scalar(0), Scalar(BT_LARGE_FLOAT));
291  channels[0].rangeMax = btClamped(velocityMax.getX(), Scalar(0), Scalar(BT_LARGE_FLOAT));
292  channels[1].rangeMax = btClamped(velocityMax.getY(), Scalar(0), Scalar(BT_LARGE_FLOAT));
293  channels[2].rangeMax = btClamped(velocityMax.getZ(), Scalar(0), Scalar(BT_LARGE_FLOAT));
294  //Altitude
295  channels[3].rangeMin = btClamped(altitudeMin, Scalar(0), Scalar(BT_LARGE_FLOAT));
296  channels[3].rangeMax = btClamped(altitudeMax, Scalar(0), Scalar(BT_LARGE_FLOAT));
297  //Water velocity
298  channels[4].rangeMin = -btClamped(velocityMax.getX(), Scalar(0), Scalar(BT_LARGE_FLOAT));
299  channels[5].rangeMin = -btClamped(velocityMax.getY(), Scalar(0), Scalar(BT_LARGE_FLOAT));
300  channels[6].rangeMin = -btClamped(velocityMax.getZ(), Scalar(0), Scalar(BT_LARGE_FLOAT));
301  channels[4].rangeMax = btClamped(velocityMax.getX(), Scalar(0), Scalar(BT_LARGE_FLOAT));
302  channels[5].rangeMax = btClamped(velocityMax.getY(), Scalar(0), Scalar(BT_LARGE_FLOAT));
303  channels[6].rangeMax = btClamped(velocityMax.getZ(), Scalar(0), Scalar(BT_LARGE_FLOAT));

306 void DVL::setNoise(Scalar velPercent, Scalar velStdDev, Scalar altitudeStdDev, Scalar waterVelPercent, Scalar waterVelStdDev)
307 {
308  channels[3].setStdDev(btClamped(altitudeStdDev, Scalar(0), Scalar(BT_LARGE_FLOAT)));
309  addNoiseStdDev[0] = btClamped(velStdDev, Scalar(0), Scalar(BT_LARGE_FLOAT));
310  addNoiseStdDev[1] = btClamped(waterVelStdDev, Scalar(0), Scalar(BT_LARGE_FLOAT));;
311  mulNoiseFactor[0] = btClamped(velPercent, Scalar(0), Scalar(100))/Scalar(100);
312  mulNoiseFactor[1] = btClamped(waterVelPercent, Scalar(0), Scalar(100))/Scalar(100);
313 }
