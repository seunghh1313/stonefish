35 Obstacle::Obstacle(std::string uniqueName,
36  std::string graphicsFilename, Scalar graphicsScale, const Transform& graphicsOrigin,
37  std::string physicsFilename, Scalar physicsScale, const Transform& physicsOrigin, bool convexHull,
38  std::string material, std::string look) : StaticEntity(uniqueName, material, look)
39 {
40  graMesh = OpenGLContent::LoadMesh(graphicsFilename, graphicsScale, false);
41  OpenGLContent::TransformMesh(graMesh, graphicsOrigin);
42 
43  if(physicsFilename != "")
44  {
45  phyMesh = OpenGLContent::LoadMesh(physicsFilename, physicsScale, false);
46  OpenGLContent::TransformMesh(phyMesh, physicsOrigin);
47  }
48  else
49  phyMesh = graMesh;
50 
51  graObjectId = -1;
52 
53  //Buidling collision shape
54  if(convexHull) // Convex approximation
55  {
56  btConvexHullShape* shape = new btConvexHullShape();
57  for(size_t i=0; i<phyMesh->getNumOfVertices(); ++i)
58  {
59  glm::vec3 pos = phyMesh->getVertexPos(i);
60  Vector3 v(pos.x, pos.y, pos.z);
61  shape->addPoint(v);
62  }
63  //shape->optimizeConvexHull();
64  shape->setMargin(0);
65  BuildRigidBody(shape);
66  }
67  else // Non-convex (arbitrary triangle mesh)
68  {
69  Scalar* vertices = new Scalar[phyMesh->getNumOfVertices() * 3];
70  int* indices = new int[phyMesh->faces.size()*3];
71 
72  for(size_t i=0; i<phyMesh->getNumOfVertices(); ++i)
73  {
74  glm::vec3 pos = phyMesh->getVertexPos(i);
75  vertices[i*3+0] = pos.x;
76  vertices[i*3+1] = pos.y;
77  vertices[i*3+2] = pos.z;
78  }
79 
80  for(size_t i=0; i<phyMesh->faces.size(); ++i)
81  {
82  indices[i*3+0] = phyMesh->faces[i].vertexID[0];
83  indices[i*3+1] = phyMesh->faces[i].vertexID[1];
84  indices[i*3+2] = phyMesh->faces[i].vertexID[2];
85  }
86 
87  btTriangleIndexVertexArray* triangleArray = new btTriangleIndexVertexArray((int)phyMesh->faces.size(), indices, 3*sizeof(int),
88  (int)phyMesh->getNumOfVertices(), vertices, 3*sizeof(Scalar));
89  btBvhTriangleMeshShape* shape = new btBvhTriangleMeshShape(triangleArray, true);
90  shape->setMargin(0);
91  BuildRigidBody(shape);
92  }
93 }

100 Obstacle::Obstacle(std::string uniqueName, Scalar sphereRadius, const Transform& origin, std::string material, std::string look) : StaticEntity(uniqueName, material, look)
101 {
102  phyMesh = OpenGLContent::BuildSphere(sphereRadius);
103  graMesh = phyMesh;
104  graObjectId = -1;
105 
106  btSphereShape* shape = new btSphereShape(sphereRadius);
107  if(origin == I4())
108  BuildRigidBody(shape);
109  else
110  {
111  OpenGLContent::TransformMesh(phyMesh, origin);
112  btCompoundShape* cShape = new btCompoundShape();
