56 int FeatherstoneRobot::getJoint(const std::string& name)
57 {

59  cCritical("Robot links not defined!");
60 
61  for(unsigned int i=0; i<dynamics->getNumOfJoints(); ++i)

95 void FeatherstoneRobot::DefineLinks(SolidEntity* baseLink, std::vector<SolidEntity*> otherLinks, bool selfCollision)
96 {
97  if(dynamics != nullptr)
98  cCritical("Robot cannot be redefined!");
99 
100  links.push_back(baseLink);
101  detachedLinks = otherLinks;
102  dynamics = new FeatherstoneEntity(name + "_Dynamics", (unsigned short)detachedLinks.size() + 1, baseLink, fixed);
103  dynamics->setSelfCollision(selfCollision);

106 void FeatherstoneRobot::BuildKinematicStructure()
107 {
108  cInfo("Building kinematic tree of robot '%s', consisting of %d links and %d joints.", getName().c_str(), detachedLinks.size()+1, jointsData.size());
109 
110  //Sort joints
111  std::vector<JointData> sortedJoints;
112 
113  //---Add joints connected to base
114  for(int i=(int)jointsData.size()-1; i>=0; --i)
115  {
116  if(jointsData[i].parent == links[0]->getName())
117  {
118  sortedJoints.push_back(jointsData[i]);
119  jointsData.erase(jointsData.begin() + i);
120  cInfo("Joint %ld: %s<-->%s", sortedJoints.size(),
121  sortedJoints.back().parent.c_str(),
122  sortedJoints.back().child.c_str());
123  }
124  }
125 
126  //---Traverse through tree
127  size_t i0=0;
128  size_t i1=sortedJoints.size()-1;
129 
130  while(jointsData.size() > 0)
131  {
132  for(size_t i=i0; i<=i1; ++i)
133  {
134  for(int h=(int)jointsData.size()-1; h>=0; --h)
135  {
136  if(jointsData[h].parent == sortedJoints[i].child)
137  {
138  sortedJoints.push_back(jointsData[h]);
139  jointsData.erase(jointsData.begin() + h);
140  cInfo("Joint %ld: %s<-->%s", sortedJoints.size(),
141  sortedJoints.back().parent.c_str(),
142  sortedJoints.back().child.c_str());
143  }
144  }
145  }
146  i0 = i1+1;
147  i1 = sortedJoints.size()-1;
148  }
149 
150  jointsData = sortedJoints;
151 
152  //Build kinematic tree
153  for(size_t i=0; i<jointsData.size(); ++i)
154  {
155  Transform linkTrans;
156 
157  // Check if connected links are already part of the model (parallel mechanisms)
158  unsigned int parentId = UINT32_MAX;
159  unsigned int childId = UINT32_MAX;
160  for(size_t h=0; h<dynamics->getNumOfLinks(); ++h)
161  {
162  if(dynamics->getLink(h).solid->getName() == jointsData[i].parent)
163  parentId = h;
164  else if(dynamics->getLink(h).solid->getName() == jointsData[i].child)
165  childId = h;
166  }
167 
168  if(parentId < dynamics->getNumOfLinks()
169  && childId < dynamics->getNumOfLinks()) // Kinematic loop
170  {
171  cCritical("Featherstone's algorithm does not support kinematic loops!");
172  }
173  else // Standard joint
174  {
175  if(parentId >= dynamics->getNumOfLinks())
176  cCritical("Parent link '%s' not yet joined with robot!", jointsData[i].parent.c_str());
177 
178  // Find child link
179  for(size_t h=0; h<detachedLinks.size(); ++h)
180  if(detachedLinks[h]->getName() == jointsData[i].child)
181  childId = h;
182 
183  if(childId >= detachedLinks.size())
184  {
185  cCritical("Child link '%s' doesn't exist!", jointsData[i].child.c_str());
186  }
187 
188  // Add link
189  linkTrans = dynamics->getLinkTransform(parentId) * dynamics->getLink(parentId).solid->getCG2OTransform() * jointsData[i].origin;
190  dynamics->AddLink(detachedLinks[childId], linkTrans);
191  links.push_back(detachedLinks[childId]);
192  detachedLinks.erase(detachedLinks.begin()+childId);
193  childId = dynamics->getNumOfLinks()-1;
194  }
195 
196  // Add joint
197  switch(jointsData[i].jtype)
198  {
199  case JointType::FIXED:
200  {
201  dynamics->AddFixedJoint(jointsData[i].name, parentId, childId, linkTrans.getOrigin());
202  }
203  break;
204 
205  case JointType::REVOLUTE:
206  {
207  dynamics->AddRevoluteJoint(jointsData[i].name, parentId, childId, linkTrans.getOrigin(), linkTrans.getBasis() * jointsData[i].axis);
208  dynamics->AddJointLimit(dynamics->getNumOfJoints()-1, jointsData[i].posLim.first, jointsData[i].posLim.second);
209 
210  if(jointsData[i].damping > Scalar(0))
211  {
212  dynamics->AddJointMotor(dynamics->getNumOfJoints()-1, jointsData[i].damping);
213  dynamics->MotorPositionSetpoint(dynamics->getNumOfJoints()-1, Scalar(0), Scalar(0));
214  dynamics->MotorVelocitySetpoint(dynamics->getNumOfJoints()-1, Scalar(0), Scalar(1));
215  }
216  }
217  break;
218 
219  case JointType::PRISMATIC:
220  {
221  dynamics->AddPrismaticJoint(jointsData[i].name, parentId, childId, linkTrans.getBasis() * jointsData[i].axis);
222  dynamics->AddJointLimit(dynamics->getNumOfJoints()-1, jointsData[i].posLim.first, jointsData[i].posLim.second);
223 
224  if(jointsData[i].damping > Scalar(0))
225  {
226  dynamics->AddJointMotor(dynamics->getNumOfJoints()-1, jointsData[i].damping);
227  dynamics->MotorPositionSetpoint(dynamics->getNumOfJoints()-1, Scalar(0), Scalar(0));
228  dynamics->MotorVelocitySetpoint(dynamics->getNumOfJoints()-1, Scalar(0), Scalar(1));
229  }
230  }
231  break;
232 
233  default:
234  break;
235  }
236  }

248 void FeatherstoneRobot::AddJointSensor(JointSensor* s, const std::string& monitoredJointName)
249 {
250  int jointId = getJoint(monitoredJointName);
251  if(jointId > -1)
252  {
253  s->AttachToJoint(dynamics, jointId);
254  sensors.push_back(s);
255  }
256  else
257  cCritical("Joint '%s' doesn't exist. Sensor '%s' cannot be attached!", monitoredJointName.c_str(), s->getName().c_str());
258 }

260 void FeatherstoneRobot::AddJointActuator(JointActuator* a, const std::string& actuatedJointName)
261 {
262  int jointId = getJoint(actuatedJointName);
263  if(jointId > -1)
264  {
265  a->AttachToJoint(dynamics, jointId);

267  }
268  else
269  cCritical("Joint '%s' doesn't exist. Actuator '%s' cannot be attached!", actuatedJointName.c_str(), a->getName().c_str());
270 }
