87 void Contact::AddContactPoint(const btPersistentManifold* manifold, bool swapped, Scalar dt)
88 {
89  size_t added = 0;
90  for(int i=0; i<manifold->getNumContacts(); ++i)
91  {
92  const btManifoldPoint& mp = manifold->getContactPoint(i);
93  Vector3 locationA = swapped ? mp.getPositionWorldOnB() : mp.getPositionWorldOnA();
94  Vector3 normalForceA = (swapped ? Scalar(1.) : Scalar(-1.)) * mp.m_normalWorldOnB * mp.getAppliedImpulse() / dt;
95 
96  //Filtering
97  if(points.size() > 0
98  && (locationA - points.back().locationA).length2() < (Scalar(0.001)*Scalar(0.001)) //Closer than 1 mm from the last point
99  && (normalForceA - points.back().normalForceA).fuzzyZero()) //No change in normal force
100  continue;
101 
102  ContactPoint p;
103  p.locationA = locationA;
104  p.locationB = swapped ? mp.getPositionWorldOnA() : mp.getPositionWorldOnB();
105  ContactInfo* cInfo = (ContactInfo*)mp.m_userPersistentData;
106  p.slippingVelocityA = (swapped ? Scalar(-1.) : Scalar(1.)) * cInfo->slip;
107  p.normalForceA = normalForceA;
108  AddContactPoint(p);
109  ++added;
110  }
111  pointsLastBeg = points.end()-added; // Save iterator to the beginning of new points
112 }

136 void Contact::SaveContactDataToOctaveFile(const std::string& path, bool includeTime)
137 {
138  if(points.size() == 0)
139  return;
140 
141  //build data structure
142  ScientificData data("");
143 
144  ScientificDataItem* it = new ScientificDataItem();
145  it->name = A->getName() + "_" + B->getName();
146  it->type = DATA_MATRIX;
147 
148  btMatrixXu* matrix = new btMatrixXu((unsigned int)points.size(), includeTime ? 10 : 9);
149  it->value = matrix;
150 
151  int offset = includeTime ? 1 : 0;
152 
153  for(unsigned int i = 0; i < points.size(); ++i)
154  {
155  if(includeTime)
156  matrix->setElem(i, 0, points[i].timeStamp);
157 
158  matrix->setElem(i, offset, points[i].locationA.x());
159  matrix->setElem(i, offset + 1, points[i].locationA.y());
160  matrix->setElem(i, offset + 2, points[i].locationA.z());
161  matrix->setElem(i, offset + 3, points[i].slippingVelocityA.x());
162  matrix->setElem(i, offset + 4, points[i].slippingVelocityA.y());
163  matrix->setElem(i, offset + 5, points[i].slippingVelocityA.z());
164  matrix->setElem(i, offset + 6, points[i].normalForceA.x());
165  matrix->setElem(i, offset + 7, points[i].normalForceA.y());
166  matrix->setElem(i, offset + 8, points[i].normalForceA.z());
167  }
168 
169  data.addItem(it);
170 
171  //save data structure to file
172  SaveOctaveData(path, data);
173 }

175 std::vector<Renderable> Contact::Render()
176 {

178 
179  if(points.size() == 0)
180  return items;
181 
182  //Drawing points
183  /*if(displayMask & CONTACT_DISPLAY_LAST_A)
184  vertices.push_back(glm::vec3((GLfloat)points.back().locationA.getX(), (GLfloat)points.back().locationA.getY(), (GLfloat)points.back().locationA.getZ()));

188 
189  OpenGLContent::getInstance()->DrawPrimitives(PrimitiveType::POINTS, vertices, CONTACT_COLOR);*/
190 
191  //Drawing lines
192  std::vector<glm::vec3> vertices;
193 
194  if(displayMask & CONTACT_DISPLAY_LAST_SLIP_VELOCITY_A)
195  {
196  Vector3 p1 = points.back().locationA;
197  Vector3 p2 = points.back().locationA + points.back().slippingVelocityA;
198  vertices.push_back(glm::vec3((GLfloat)p1.getX(), (GLfloat)p1.getY(), (GLfloat)p1.getZ()));
199  vertices.push_back(glm::vec3((GLfloat)p2.getX(), (GLfloat)p2.getY(), (GLfloat)p2.getZ()));
200  }
201 
202  if(displayMask & CONTACT_DISPLAY_LAST_SLIP_VELOCITY_B)
203  {
204  Vector3 p1 = points.back().locationB;
205  Vector3 p2 = points.back().locationB - points.back().slippingVelocityA;
206  vertices.push_back(glm::vec3((GLfloat)p1.getX(), (GLfloat)p1.getY(), (GLfloat)p1.getZ()));
207  vertices.push_back(glm::vec3((GLfloat)p2.getX(), (GLfloat)p2.getY(), (GLfloat)p2.getZ()));
208  }
209 
210  if(displayMask & CONTACT_DISPLAY_NORMAL_FORCE_A)
211  {
212  for(auto it=pointsLastBeg; it != points.end(); ++it)
213  {
214  Vector3 p1 = (*it).locationA;
215  Vector3 p2 = (*it).locationA + (*it).normalForceA;
216  vertices.push_back(glm::vec3((GLfloat)p1.getX(), (GLfloat)p1.getY(), (GLfloat)p1.getZ()));
217  vertices.push_back(glm::vec3((GLfloat)p2.getX(), (GLfloat)p2.getY(), (GLfloat)p2.getZ()));
218  }
219  }
220 
221  if(displayMask & CONTACT_DISPLAY_NORMAL_FORCE_B)
222  {
223  for(auto it=pointsLastBeg; it != points.end(); ++it)
224  {
225  Vector3 p1 = (*it).locationB;
226  Vector3 p2 = (*it).locationB - (*it).normalForceA;
227  vertices.push_back(glm::vec3((GLfloat)p1.getX(), (GLfloat)p1.getY(), (GLfloat)p1.getZ()));
228  vertices.push_back(glm::vec3((GLfloat)p2.getX(), (GLfloat)p2.getY(), (GLfloat)p2.getZ()));
229  }
230  }
231 
232  if(vertices.size() > 0)
233  {
234  Renderable item;
235  item.model = glm::mat4(1.f);
236  item.points = vertices;
237  item.type = RenderableType::SENSOR_LINES;
238  items.push_back(item);
239  }
240 
241  //Drawing line strips
242  if(displayMask & CONTACT_DISPLAY_PATH_A)
243  {
244  Renderable item;
245  item.model = glm::mat4(1.f);
246  item.type = RenderableType::SENSOR_POINTS;
247 
248  for(size_t i = 0; i < points.size(); ++i)
249  {
250  Vector3 p = points[i].locationA;
251  item.points.push_back(glm::vec3((GLfloat)p.getX(), (GLfloat)p.getY(), (GLfloat)p.getZ()));
252  }
253 
254  items.push_back(item);
255  }
256 
257  if(displayMask & CONTACT_DISPLAY_PATH_B)
258  {
259  Renderable item;
260  item.model = glm::mat4(1.f);
261  item.type = RenderableType::SENSOR_POINTS;
262 
263  for(size_t i = 0; i < points.size(); ++i)
264  {
265  Vector3 p = points[i].locationB;
266  item.points.push_back(glm::vec3((GLfloat)p.getX(), (GLfloat)p.getY(), (GLfloat)p.getZ()));
267  }
268 
269  items.push_back(item);
270  }
271 
272  return items;
273 }
