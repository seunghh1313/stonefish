94 void Sensor::setVisual(const std::string& meshFilename, Scalar scale, const std::string& look)
95 {
96  if(!SimulationApp::getApp()->hasGraphics())
97  return;
98 
99  Mesh* mesh = OpenGLContent::LoadMesh(meshFilename, scale, false);
100  if(mesh == nullptr)
101  return;
102 
103  graObjectId = ((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getContent()->BuildObject(mesh);
104  lookId = ((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getContent()->getLookId(look);
105  delete mesh;
106 }

114 void Sensor::Update(Scalar dt)
115 {
116  SDL_LockMutex(updateMutex);
117 
118  if(freq <= Scalar(0)) // Every simulation tick
119  {
120  InternalUpdate(dt);
121  newDataAvailable = true;
122  }
123  else //Fixed rate
124  {
125  eleapsedTime += dt;
126  Scalar invFreq = Scalar(1)/freq;
127 
128  if(eleapsedTime >= invFreq)
129  {
130  InternalUpdate(invFreq);
131  eleapsedTime -= invFreq;
132  newDataAvailable = true;
133  }
134  }
135 
136  SDL_UnlockMutex(updateMutex);
137 }
