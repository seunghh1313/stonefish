65 void Trajectory::Play(Scalar dt)
66 {
67  //Update time
68  switch(playMode)
69  {
70  case PlaybackMode::ONETIME:
71  {
72  playTime += dt;
73  if(playTime > endTime)
74  playTime = endTime;
75  }
76  break;
77 
78  case PlaybackMode::REPEAT:
79  {
80  playTime += dt;
81  if(playTime > endTime)
82  {
83  playTime -= endTime;
84  ++iteration;
85  }
86  }
87  break;
88 
89  case PlaybackMode::BOOMERANG:
90  {
91  playTime += forward ? dt : -dt;
92  if(playTime > endTime)
93  {
94  playTime = endTime - (playTime - endTime);
95  forward = false;
96  ++iteration;
97  }
98  else if(playTime < Scalar(0))
99  {
100  playTime = -playTime;
101  forward = true;
102  ++iteration;
103  }
104  }
105  break;
106  }
107 
108  //Compute new interpolants
109  Interpolate();
110 }
