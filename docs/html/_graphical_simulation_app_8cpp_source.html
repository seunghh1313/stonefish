180 void GraphicalSimulationApp::Init()
181 {
182  //General initialization
183  SimulationApp::Init();
184  //Window initialization + loading thread
185  loading = true;
186  InitializeSDL();
187 
188  //Continue initialization with console visible
189  cInfo("Initializing rendering pipeline:");
190  cInfo("Loading GUI...");
191  gui = new IMGUI(windowW, windowH);
192  InitializeGUI(); //Initialize non-standard graphical elements
193  glPipeline = new OpenGLPipeline(rSettings, hSettings);
194  ShowHUD();
195 
196  cInfo("Initializing simulation:");
197  InitializeSimulation();
198 
199  cInfo("Ready for running...");
200  SDL_Delay(1000);
201 
202  //Close loading console - exit loading thread
203  loading = false;
204  int status = 0;
205  SDL_WaitThread(loadingThread, &status);
206  SDL_GL_MakeCurrent(window, glMainContext);
207 
208  //Create performance counters
209  glGenQueries(2, timeQuery);
210 }

212 void GraphicalSimulationApp::InitializeSDL()
213 {
214  SDL_Init(SDL_INIT_VIDEO | SDL_INIT_EVENTS | SDL_INIT_JOYSTICK);
215 
216  //Create OpenGL contexts
217  SDL_GL_SetAttribute(SDL_GL_CONTEXT_PROFILE_MASK, SDL_GL_CONTEXT_PROFILE_CORE);
218  SDL_GL_SetAttribute(SDL_GL_CONTEXT_MAJOR_VERSION, 4);
219  SDL_GL_SetAttribute(SDL_GL_CONTEXT_MINOR_VERSION, 3);
220  SDL_GL_SetAttribute(SDL_GL_DOUBLEBUFFER, 1);
221  SDL_GL_SetAttribute(SDL_GL_DEPTH_SIZE, 24);
222  SDL_GL_SetAttribute(SDL_GL_STENCIL_SIZE, 8);
223  SDL_GL_SetAttribute(SDL_GL_RED_SIZE, 8);
224  SDL_GL_SetAttribute(SDL_GL_GREEN_SIZE, 8);
225  SDL_GL_SetAttribute(SDL_GL_BLUE_SIZE, 8);
226  SDL_GL_SetAttribute(SDL_GL_SHARE_WITH_CURRENT_CONTEXT, 1);
227 
228  //Create window
229  window = SDL_CreateWindow(getName().c_str(),
230  SDL_WINDOWPOS_CENTERED,
231  SDL_WINDOWPOS_CENTERED,
232  windowW,
233  windowH,
234  SDL_WINDOW_OPENGL | SDL_WINDOW_SHOWN// | SDL_WINDOW_ALLOW_HIGHDPI
235  );
236 
237  //Set window icon
238  uint32_t rmask, gmask, bmask, amask;
239 #if SDL_BYTEORDER == SDL_BIG_ENDIAN
240  int shift = (icon_image.bytes_per_pixel == 3) ? 8 : 0;
241  rmask = 0xff000000 >> shift;
242  gmask = 0x00ff0000 >> shift;
243  bmask = 0x0000ff00 >> shift;
244  amask = 0x000000ff >> shift;
245 #else
246  rmask = 0x000000ff;
247  gmask = 0x0000ff00;
248  bmask = 0x00ff0000;
249  amask = (icon_image.bytes_per_pixel == (3)) ? 0 : 0xff000000;
250 #endif
251 
252  SDL_Surface* icon = SDL_CreateRGBSurfaceFrom((void*)icon_image.pixel_data, icon_image.width, icon_image.height,
253  icon_image.bytes_per_pixel*8, icon_image.bytes_per_pixel*icon_image.width, rmask, gmask, bmask, amask);
254 
255  SDL_SetWindowIcon(window, icon);
256  SDL_FreeSurface(icon);
257 
258  //Create OpenGL contexts
259  glLoadingContext = SDL_GL_CreateContext(window);
260  if(glLoadingContext == NULL)
261  cCritical("SDL2: %s", SDL_GetError());
262 
263  glMainContext = SDL_GL_CreateContext(window);
264  if(glMainContext == NULL)
265  cCritical("SDL2: %s", SDL_GetError());
266 
267  //Disable vertical synchronization --> use framerate limitting instead (e.g. max 60 FPS)
268  if(SDL_GL_SetSwapInterval(0) == -1)
269  cError("SDL2: %s", SDL_GetError());
270 
271  //Initialize OpenGL function handlers
272  int version = gladLoadGL((GLADloadfunc) SDL_GL_GetProcAddress);
273  int vmajor = GLAD_VERSION_MAJOR(version);
274  int vminor = GLAD_VERSION_MINOR(version);
275  if(vmajor < 4 || (vmajor == 4 && vminor < 3))
276  cCritical("This program requires support for OpenGL 4.3, however OpenGL %d.%d was detected! Exiting...", vmajor, vminor);
277 
278  //Initialize OpenGL pipeline
279  cInfo("Window created. OpenGL %d.%d contexts created.", vmajor, vminor);
280  OpenGLState::Init();
281  GLSLShader::Init();
282 
283  //Initialize console output
284  std::vector<ConsoleMessage> textLines = console->getLines();
285  delete console;
286  console = new OpenGLConsole();
287  for(size_t i=0; i<textLines.size(); ++i)
288  console->AppendMessage(textLines[i]);
289  ((OpenGLConsole*)console)->Init(windowW, windowH);
290 
291  //Create loading thread
292  LoadingThreadData* data = new LoadingThreadData();
293  data->app = this;
294  data->mutex = console->getLinesMutex();
295  loadingThread = SDL_CreateThread(GraphicalSimulationApp::RenderLoadingScreen, "loadingThread", data);
296 
297  //Look for joysticks
298  int jcount = SDL_NumJoysticks();
299 
300  if(jcount > 0)
301  {
302  joystick = SDL_JoystickOpen(0);
303  joystickButtons = new bool[SDL_JoystickNumButtons(joystick)];
304  memset(joystickButtons, 0, SDL_JoystickNumButtons(joystick));
305  joystickAxes = new int16_t[SDL_JoystickNumAxes(joystick)];
306  memset(joystickAxes, 0, SDL_JoystickNumAxes(joystick) * sizeof(int16_t));
307  joystickHats = new uint8_t[SDL_JoystickNumHats(joystick)];
308  memset(joystickHats, 0, SDL_JoystickNumHats(joystick));
309  cInfo("Joystick %s connected (%d axes, %d hats, %d buttons)", SDL_JoystickName(joystick),
310  SDL_JoystickNumAxes(joystick),
311  SDL_JoystickNumHats(joystick),
312  SDL_JoystickNumButtons(joystick));
313  }

333 void GraphicalSimulationApp::KeyDown(SDL_Event *event)
334 {
335  GLfloat moveStep = 0.1f;
336  if(event->key.keysym.mod & KMOD_SHIFT)
337  moveStep = 1.f;
338 
339  switch (event->key.keysym.sym)
340  {
341  case SDLK_ESCAPE:
342  Quit();
343  break;
344 
345  // case SDLK_SPACE:
346  // selectedEntity = std::make_pair(nullptr, 0);
347  // if(!getSimulationManager()->isSimulationFresh())
348  // {
349  // StopSimulation();
350  // getSimulationManager()->RestartScenario();
351  // }
352  // StartSimulation();
353  // break;
354 
355  case SDLK_h:
356  displayHUD = !displayHUD;
357  break;
358 
359  case SDLK_k:
360  displayKeymap = !displayKeymap;
361  break;
362 
363  case SDLK_p:
364  displayPerformance = !displayPerformance;
365  break;
366 
367  case SDLK_c:
368  displayConsole = !displayConsole;
369  ((OpenGLConsole*)console)->ResetScroll();
370  break;
371 
372  case SDLK_w: //Forward
373  {
374  OpenGLTrackball* trackball = getSimulationManager()->getTrackball();
375  if(trackball->isEnabled())
376  trackball->MoveCenter(trackball->GetLookingDirection() * moveStep);
377  }
378  break;
379 
380  case SDLK_s: //Backward
381  {
382  OpenGLTrackball* trackball = getSimulationManager()->getTrackball();
383  if(trackball->isEnabled())
384  trackball->MoveCenter(-trackball->GetLookingDirection() * moveStep);
385  }
386  break;
387 
388  case SDLK_a: //Left
389  {
390  OpenGLTrackball* trackball = getSimulationManager()->getTrackball();
391  if(trackball->isEnabled())
392  {
393  glm::vec3 axis = glm::cross(trackball->GetLookingDirection(), trackball->GetUpDirection());
394  trackball->MoveCenter(-axis * moveStep);
395  }
396  }
397  break;
398 
399  case SDLK_d: //Right
400  {
401  OpenGLTrackball* trackball = getSimulationManager()->getTrackball();
402  if(trackball->isEnabled())
403  {
404  glm::vec3 axis = glm::cross(trackball->GetLookingDirection(), trackball->GetUpDirection());
405  trackball->MoveCenter(axis * moveStep);
406  }
407  }
408  break;
409 
410  case SDLK_q: //Up
411  {
412  OpenGLTrackball* trackball = getSimulationManager()->getTrackball();
413  if(trackball->isEnabled())
414  trackball->MoveCenter(glm::vec3(0.f, 0.f, -moveStep));
415  }
416  break;
417 
418  case SDLK_z: //Down
419  {
420  OpenGLTrackball* trackball = getSimulationManager()->getTrackball();
421  if(trackball->isEnabled())
422  trackball->MoveCenter(glm::vec3(0.f, 0.f, moveStep));
423  }
424  break;
425 
426  default:
427  break;
428  }
429 }

463 void GraphicalSimulationApp::Loop()
464 {
465  SDL_Event event;
466  bool mouseWasDown = false;
467  uint64_t startTime = GetTimeInMicroseconds();
468 
469  while(!hasFinished())
470  {
471  SDL_FlushEvents(SDL_FINGERDOWN, SDL_MULTIGESTURE);
472 
473  while(SDL_PollEvent(&event))
474  {
475  switch(event.type)
476  {
477  case SDL_WINDOWEVENT:
478  WindowEvent(&event);
479  break;
480 
481  case SDL_TEXTINPUT:
482  break;
483 
484  case SDL_KEYDOWN:
485  {
486  gui->KeyDown(event.key.keysym.sym);
487  KeyDown(&event);
488  break;
489  }
490 
491  case SDL_KEYUP:
492  {
493  gui->KeyUp(event.key.keysym.sym);
494  KeyUp(&event);
495  break;
496  }
497 
498  case SDL_MOUSEBUTTONDOWN:
499  {
500  gui->MouseDown(event.button.x, event.button.y, event.button.button == SDL_BUTTON_LEFT);
501  mouseWasDown = true;
502  }
503  break;
504 
505  case SDL_MOUSEBUTTONUP:
506  {
507  //GUI
508  gui->MouseUp(event.button.x, event.button.y, event.button.button == SDL_BUTTON_LEFT);
509 
510  //Trackball
511  if(event.button.button == SDL_BUTTON_RIGHT || event.button.button == SDL_BUTTON_MIDDLE)
512  {
513  OpenGLTrackball* trackball = getSimulationManager()->getTrackball();
514  if(trackball->isEnabled())
515  trackball->MouseUp();
516  }
517 
518  //Pass
519  MouseUp(&event);
520  }
521  break;
522 
523  case SDL_MOUSEMOTION:
524  {
525  //GUI
526  gui->MouseMove(event.motion.x, event.motion.y);
527 
528  OpenGLTrackball* trackball = getSimulationManager()->getTrackball();
529  if(trackball->isEnabled())
530  {
531  GLfloat xPos = (GLfloat)(event.motion.x-getWindowWidth()/2.f)/(GLfloat)(getWindowHeight()/2.f);
532  GLfloat yPos = -(GLfloat)(event.motion.y-getWindowHeight()/2.f)/(GLfloat)(getWindowHeight()/2.f);
533  trackball->MouseMove(xPos, yPos);
534  }
535 
536  //Pass
537  MouseMove(&event);
538  }
539  break;
540 
541  case SDL_MOUSEWHEEL:
542  {
543  if(displayConsole) //GUI
544  ((OpenGLConsole*)console)->Scroll((GLfloat)-event.wheel.y);
545  else
546  {
547  //Trackball
548  OpenGLTrackball* trackball = getSimulationManager()->getTrackball();
549  if(trackball->isEnabled())
550  trackball->MouseScroll(event.wheel.y * -1.f);
551 
552  //Pass
553  MouseScroll(&event);
554  }
555  }
556  break;
557 
558  case SDL_JOYBUTTONDOWN:
559  joystickButtons[event.jbutton.button] = true;
560  JoystickDown(&event);
561  break;
562 
563  case SDL_JOYBUTTONUP:
564  joystickButtons[event.jbutton.button] = false;
565  JoystickUp(&event);
566  break;
567 
568  case SDL_QUIT:
569  {
570  if(isRunning())
571  StopSimulation();
572 
573  Quit();
574  }
575  break;
576  }
577  }
578 
579  if(joystick != NULL)
580  {
581  for(int i=0; i<SDL_JoystickNumAxes(joystick); i++)
582  joystickAxes[i] = SDL_JoystickGetAxis(joystick, i);
583 
584  for(int i=0; i<SDL_JoystickNumHats(joystick); i++)
585  joystickHats[i] = SDL_JoystickGetHat(joystick, i);
586  }
587 
588  ProcessInputs();
589  RenderLoop();
590 
591  //workaround for checking if IMGUI is being manipulated
592  if(mouseWasDown && !gui->isAnyActive())
593  {
594  OpenGLTrackball* trackball = getSimulationManager()->getTrackball();
595 
596  if(trackball->isEnabled())
597  {
598  if(event.button.button == SDL_BUTTON_LEFT)
599  {
600  glm::vec3 eye = trackball->GetEyePosition();
601  glm::vec3 ray = trackball->Ray(event.button.x, event.button.y);
602  selectedEntity = getSimulationManager()->PickEntity(Vector3(eye.x, eye.y, eye.z), Vector3(ray.x, ray.y, ray.z));
603  }
604  else //RIGHT OR MIDDLE
605  {
606  GLfloat xPos = (GLfloat)(event.motion.x-getWindowWidth()/2.f)/(GLfloat)(getWindowHeight()/2.f);
607  GLfloat yPos = -(GLfloat)(event.motion.y-getWindowHeight()/2.f)/(GLfloat)(getWindowHeight()/2.f);
608  trackball->MouseDown(xPos, yPos, event.button.button == SDL_BUTTON_MIDDLE);
609  }
610  }
611 
612  //Pass
613  MouseDown(&event);
614  }
615  mouseWasDown = false;
616 
617  //Framerate limitting (60Hz)
618  uint64_t elapsedTime = GetTimeInMicroseconds() - startTime;
619  if(elapsedTime < 16000)
620  std::this_thread::sleep_for(std::chrono::microseconds(16000 - elapsedTime));
621  startTime = GetTimeInMicroseconds();
622  }
623 }

625 void GraphicalSimulationApp::RenderLoop()
626 {
627  //Do some updates
628  if(!isRunning())
629  {
630  getSimulationManager()->UpdateDrawingQueue();
631  }
632 
633  //Rendering
634  glBeginQuery(GL_TIME_ELAPSED, timeQuery[timeQueryPingpong]);
635  glPipeline->Render(getSimulationManager());
636  glPipeline->DrawDisplay();
637 
638  //GUI & Console
639  if(displayConsole)
640  {
641  gui->GenerateBackground();
642  ((OpenGLConsole*)console)->Render(true);
643  }
644  else
645  {
646  if(displayHUD) //Draw immediate mode GUI
647  {
648  gui->GenerateBackground();
649  gui->Begin();
650  DoHUD();
651  gui->End();
652  }
653  else //Just draw logo in the corner
654  {
655  gui->Begin();
656  gui->End();
657  }
658  }
659  glEndQuery(GL_TIME_ELAPSED);
660 
661  //Update drawing time
662  uint64_t drawTime;
663  glGetQueryObjectui64v(timeQuery[1-timeQueryPingpong], GL_QUERY_RESULT, &drawTime);
664  timeQueryPingpong = 1-timeQueryPingpong;
665  double dt = std::min(drawTime/1000000.0, 1000.0); //in ms
666  double f = 1.0/60.0;
667  drawingTime = f*dt + (1.0-f)*drawingTime;
668 
669  //Update maximum drawing time
670  if(maxCounter >= 60)
671  {
672  maxDrawingTime = drawingTime;
673  maxCounter = 0;
674  }
675  else
676  {
677  maxDrawingTime = std::max(maxDrawingTime, dt);
678  ++maxCounter;
679  }
680 
681  //glFinish(); //Ensure that the frame was fully rendered
682  SDL_GL_SwapWindow(window);
683 }
