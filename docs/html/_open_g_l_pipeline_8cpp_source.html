50 OpenGLPipeline::OpenGLPipeline(RenderSettings s, HelperSettings h) : rSettings(s), hSettings(h)
51 {
52  drawingQueueMutex = SDL_CreateMutex();
53 
54  //Set default OpenGL options
55  cInfo("Initialising OpenGL rendering pipeline...");
56 
57  //Load shaders and create rendering buffers
58  cInfo("Loading shaders...");
59  OpenGLAtmosphere::Init();
60  OpenGLCamera::Init(rSettings);
61  OpenGLDepthCamera::Init();
62  OpenGLSonar::Init();
63  OpenGLOceanParticles::Init();
64  content = new OpenGLContent();
65 
66  //Create display framebuffer
67  glGenFramebuffers(1, &screenFBO);
68  OpenGLState::BindFramebuffer(screenFBO);
69  glGenTextures(1, &screenTex);
70  OpenGLState::BindTexture(TEX_BASE, GL_TEXTURE_2D, screenTex);
71  glTexImage2D(GL_TEXTURE_2D, 0, GL_RGB8, rSettings.windowW, rSettings.windowH, 0, GL_RGB, GL_UNSIGNED_BYTE, NULL);
72  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR); //Cheaper/better gaussian blur for GUI background
73  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR); //Cheaper/better gaussian blur for GUI background
74  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
75  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
76  glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, screenTex, 0);
77  OpenGLState::UnbindTexture(TEX_BASE);
78  GLenum status = glCheckFramebufferStatus(GL_FRAMEBUFFER);
79  if(status != GL_FRAMEBUFFER_COMPLETE)
80  cError("Display FBO initialization failed!");
81  OpenGLState::BindFramebuffer(0);
82  lastSimTime = Scalar(0);
83 }

154 void OpenGLPipeline::PerformDrawingQueueCopy(SimulationManager* sim)
155 {
156  SDL_LockMutex(drawingQueueMutex);
157 
158  //Update vision sensor transforms and copy generated data to ensure consistency
159  glMemoryBarrier(GL_PIXEL_BUFFER_BARRIER_BIT);
160  for(unsigned int i=0; i < content->getViewsCount(); ++i)
161  content->getView(i)->UpdateTransform();
162  //Update light transforms to ensure consistency
163  for(unsigned int i=0; i < content->getLightsCount(); ++i)
164  content->getLight(i)->UpdateTransform();
165  //Update ocean currents for particle systems
166  Ocean* ocean = sim->getOcean();
167  if(ocean != NULL) ocean->UpdateCurrentsData();
168 
169  if(!drawingQueue.empty())
170  {
171  drawingQueueCopy.clear();
172  selectedDrawingQueueCopy.clear();
173  //Double buffering
174  drawingQueueCopy.insert(drawingQueueCopy.end(), drawingQueue.begin(), drawingQueue.end());
175  selectedDrawingQueueCopy.insert(selectedDrawingQueueCopy.end(), selectedDrawingQueue.begin(), selectedDrawingQueue.end());
176  //Enable update of drawing queue by clearing old queue
177  drawingQueue.clear();
178  selectedDrawingQueue.clear();
179  }
180 
181  SDL_UnlockMutex(drawingQueueMutex);
182 
183  //Sort objects by material to reduce uniform/texture switching
184  std::sort(drawingQueueCopy.begin(), drawingQueueCopy.end(), Renderable::SortByMaterial);
185 }

209 void OpenGLPipeline::DrawHelpers()
210 {
211  //Coordinate systems
212  if(hSettings.showCoordSys)
213  {
214  content->DrawCoordSystem(glm::mat4(1.f), 1.f);
215 
216  for(size_t h=0; h<drawingQueueCopy.size(); ++h)
217  {
218  if(drawingQueueCopy[h].type == RenderableType::SOLID_CS)
219  content->DrawCoordSystem(drawingQueueCopy[h].model, 0.25f);
220  }
221  }
222 
223  //Discrete and multibody joints
224  if(hSettings.showJoints)
225  {
226  for(size_t h=0; h<drawingQueueCopy.size(); ++h)
227  {
228  if(drawingQueueCopy[h].type == RenderableType::MULTIBODY_AXIS)
229  content->DrawPrimitives(PrimitiveType::LINES, drawingQueueCopy[h].points, glm::vec4(1.f,0.5f,1.f,1.f), drawingQueueCopy[h].model);
230  else if(drawingQueueCopy[h].type == RenderableType::JOINT_LINES)
231  content->DrawPrimitives(PrimitiveType::LINES, drawingQueueCopy[h].points, glm::vec4(1.f,0.5f,1.f,1.f), drawingQueueCopy[h].model);
232  else if(drawingQueueCopy[h].type == RenderableType::PATH_LINE_STRIP)
233  content->DrawPrimitives(PrimitiveType::LINE_STRIP, drawingQueueCopy[h].points, glm::vec4(1.f,0.5f,1.f,1.f), drawingQueueCopy[h].model);
234  }
235  }
236 
237  //Sensors
238  if(hSettings.showSensors)
239  {
240  for(size_t h=0; h<drawingQueueCopy.size(); ++h)
241  {
242  if(drawingQueueCopy[h].type == RenderableType::SENSOR_CS)
243  content->DrawCoordSystem(drawingQueueCopy[h].model, 0.25f);
244  else if(drawingQueueCopy[h].type == RenderableType::SENSOR_POINTS)
245  content->DrawPrimitives(PrimitiveType::POINTS, drawingQueueCopy[h].points, glm::vec4(1.f,1.f,0,1.f), drawingQueueCopy[h].model);
246  else if(drawingQueueCopy[h].type == RenderableType::SENSOR_LINES)
247  content->DrawPrimitives(PrimitiveType::LINES, drawingQueueCopy[h].points, glm::vec4(1.f,1.f,0,1.f), drawingQueueCopy[h].model);
248  else if(drawingQueueCopy[h].type == RenderableType::SENSOR_LINE_STRIP)
249  content->DrawPrimitives(PrimitiveType::LINE_STRIP, drawingQueueCopy[h].points, glm::vec4(1.f,1.f,0,1.f), drawingQueueCopy[h].model);
250  }
251  }
252 
253  //Actuators
254  if(hSettings.showActuators)
255  {
256  for(size_t h=0; h<drawingQueueCopy.size(); ++h)
257  {
258  if(drawingQueueCopy[h].type == RenderableType::ACTUATOR_LINES)
259  content->DrawPrimitives(PrimitiveType::LINES, drawingQueueCopy[h].points, glm::vec4(1.f,0.5f,0,1.f), drawingQueueCopy[h].model);
260  }
261  }
262 
263  //Fluid dynamics
264  if(hSettings.showFluidDynamics)
265  {
266  for(size_t h=0; h<drawingQueueCopy.size(); ++h)
267  {
268  switch(drawingQueueCopy[h].type)
269  {
270  case RenderableType::HYDRO_CS:
271  content->DrawEllipsoid(drawingQueueCopy[h].model, glm::vec3(0.005f), glm::vec4(0.3f, 0.7f, 1.f, 1.f));
272  break;
273 
274  case RenderableType::HYDRO_CYLINDER:
275  content->DrawCylinder(drawingQueueCopy[h].model, drawingQueueCopy[h].points[0], glm::vec4(0.2f, 0.5f, 1.f, 1.f));
276  break;
277 
278  case RenderableType::HYDRO_ELLIPSOID:
279  content->DrawEllipsoid(drawingQueueCopy[h].model, drawingQueueCopy[h].points[0], glm::vec4(0.2f, 0.5f, 1.f, 1.f));
280  break;
281 
282  case RenderableType::HYDRO_POINTS:
283  content->DrawPrimitives(PrimitiveType::POINTS, drawingQueueCopy[h].points, glm::vec4(0.3f, 0.7f, 1.f, 1.f), drawingQueueCopy[h].model);
284  break;
285 
286  case RenderableType::HYDRO_LINES:
287  content->DrawPrimitives(PrimitiveType::LINES, drawingQueueCopy[h].points, glm::vec4(0.2f, 0.5f, 1.f, 1.f), drawingQueueCopy[h].model);
288  break;
289 
290  case RenderableType::HYDRO_LINE_STRIP:
291  content->DrawPrimitives(PrimitiveType::LINE_STRIP, drawingQueueCopy[h].points, glm::vec4(0.2f, 0.5f, 1.f, 1.f), drawingQueueCopy[h].model);
292  break;
293 
294  case RenderableType::HYDRO_TRIANGLES:
295  content->DrawPrimitives(PrimitiveType::TRIANGLES, drawingQueueCopy[h].points, glm::vec4(0.2f, 0.5f, 1.f, 1.f), drawingQueueCopy[h].model);
296  break;
297 
298  default:
299  break;
300  }
301  }
302  }
303 
304  //Forces
305  if(hSettings.showForces)
306  {
307  for(size_t h=0; h<drawingQueueCopy.size(); ++h)
308  {
309  switch(drawingQueueCopy[h].type)
310  {
311  case RenderableType::FORCE_BUOYANCY:
312  content->DrawPrimitives(PrimitiveType::LINES, drawingQueueCopy[h].points, glm::vec4(0.f,0.f,1.f,1.f), drawingQueueCopy[h].model);
313  break;
314 
315  case RenderableType::FORCE_LINEAR_DRAG:
316  content->DrawPrimitives(PrimitiveType::LINES, drawingQueueCopy[h].points, glm::vec4(0.f,1.f,1.f,1.f), drawingQueueCopy[h].model);
317  break;
318 
319  case RenderableType::FORCE_QUADRATIC_DRAG:
320  content->DrawPrimitives(PrimitiveType::LINES, drawingQueueCopy[h].points, glm::vec4(1.f,0.f,1.f,1.f), drawingQueueCopy[h].model);
321  break;
322 
323  default:
324  break;
325  }
326  }
327  }

330 void OpenGLPipeline::Render(SimulationManager* sim)
331 {
332  //Update time step for animation purposes
333  Scalar now = sim->getSimulationTime();
334  Scalar dt = now-lastSimTime;
335  lastSimTime = now;
336 
337  //Double-buffering of drawing queue
338  PerformDrawingQueueCopy(sim);
339 
340  //Choose rendering mode
341  unsigned int renderMode = 0; //Defaults to rendering without ocean
342  Ocean* ocean = sim->getOcean();
343  if(ocean != NULL)
344  {
345  ocean->getOpenGLOcean()->Simulate(dt);
346  renderMode = rSettings.ocean > RenderQuality::DISABLED && ocean->isRenderable() ? 1 : 0;
347  }
348  Atmosphere* atm = sim->getAtmosphere();
349  OpenGLState::EnableDepthTest();
350  OpenGLState::EnableCullFace();
351 
352  //Bake shadow maps for lights (independent of view)
353  content->SetupLights();
354  if(rSettings.shadows > RenderQuality::DISABLED)
355  {
356  glCullFace(GL_FRONT);
357  glDisable(GL_DEPTH_CLAMP);
358  content->SetDrawingMode(DrawingMode::SHADOW);
359  for(unsigned int i=0; i<content->getLightsCount(); ++i)
360  content->getLight(i)->BakeShadowmap(this);
361  glEnable(GL_DEPTH_CLAMP);
362  glCullFace(GL_BACK);
363  }
364 
365  //Clear display framebuffer
366  OpenGLState::BindFramebuffer(screenFBO);
367  glClear(GL_COLOR_BUFFER_BIT);
368 
369  //Update the queue of views needing update
370  unsigned int updateCount = 0;
371  std::vector<unsigned int> viewsNoUpdate;
372  for(int i=content->getViewsCount()-1; i >= 0; --i)
373  {
374  OpenGLView* view = content->getView(i);
375  if(view->needsUpdate())
376  {
377  if(view->isContinuous())
378  {
379  viewsQueue.push_front(i);
380  ++updateCount;
381  }
382  else
383  {
384  viewsQueue.push_back(i);
385  viewsNoUpdate.push_back(i);
386  }
387  }
388  else
389  {
390  viewsNoUpdate.push_back(i);
391  }
392  }
393 
394  if(viewsQueue.size() > content->getViewsCount())
395  {
396  updateCount = viewsQueue.size();
397  viewsNoUpdate.clear();
398  }
399  else if(updateCount < (unsigned int)viewsQueue.size())
400  {
401  ++updateCount;
402  viewsNoUpdate.erase(std::find(viewsNoUpdate.begin(), viewsNoUpdate.end(), viewsQueue[updateCount-1]));
403  }
404 
405  //Loop through all views -> trackballs, cameras, depth cameras...
406  for(unsigned int i=0; i<updateCount; ++i)
407  {
408  OpenGLState::EnableDepthTest();
409  OpenGLState::EnableCullFace();
410  OpenGLState::DisableBlend();
411  OpenGLView* view = content->getView(viewsQueue[i]);
412 
413  if(view->getType() == ViewType::DEPTH_CAMERA)
414  {
415  OpenGLDepthCamera* camera = static_cast<OpenGLDepthCamera*>(view);
416  //Draw objects and compute depth data
417  camera->ComputeOutput(drawingQueueCopy);
418  //Draw camera output
419  camera->DrawLDR(screenFBO, true);
420  }
421  else if(view->getType() == ViewType::SONAR)
422  {
423  OpenGLSonar* sonar = static_cast<OpenGLSonar*>(view);
424  //Draw objects and compute sonar data
425  sonar->ComputeOutput(drawingQueueCopy);
426  //Draw sonar output
427  sonar->DrawLDR(screenFBO, true);
428  }
429  else if(view->getType() == ViewType::CAMERA || view->getType() == ViewType::TRACKBALL)
430  {
431  //Apply view properties
432  OpenGLCamera* camera = static_cast<OpenGLCamera*>(view);
433  OpenGLLight::SetCamera(camera);
434  GLint* viewport = camera->GetViewport();
435  content->SetViewportSize(viewport[2],viewport[3]);
436 
437  //Bake parallel-split shadowmaps for sun
438  if(rSettings.shadows > RenderQuality::DISABLED)
439  {
440  content->SetDrawingMode(DrawingMode::SHADOW);
441  atm->getOpenGLAtmosphere()->BakeShadowmaps(this, camera);
442  }
443 
444  atm->getOpenGLAtmosphere()->SetupMaterialShaders();
445 
446  //Clear main framebuffer and setup camera
447  OpenGLState::BindFramebuffer(camera->getRenderFBO());
448  camera->SetRenderBuffers(0, true, false);
449  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT | GL_STENCIL_BUFFER_BIT);
450 
451  camera->SetViewport();
452  content->SetCurrentView(camera);
453 
454  //Draw scene
455  if(renderMode == 0) //NO OCEAN
456  {
457  //Render all objects
458  content->SetDrawingMode(DrawingMode::FULL);
459  DrawObjects();
460  DrawLights();
461 
462  //Ambient occlusion
463  if(rSettings.ao > RenderQuality::DISABLED)
464  camera->DrawAO(1.0f);
465 
466  //Render sky (at the end to take profit of early bailing)
467  atm->getOpenGLAtmosphere()->DrawSkyAndSun(camera);
468  }
469  else if(renderMode == 1) //OCEAN
470  {
471  OpenGLOcean* glOcean = ocean->getOpenGLOcean();
472 
473  //Update ocean for this camera
474  if(ocean->hasWaves())
475  glOcean->UpdateSurface(camera);
476 
477  //Two separate rendering paths: above water and under water,
478  //possible because camera near plane is (virtually) removed with logarithmic depth buffer.
479  glm::vec3 eye = camera->GetEyePosition();
480  if(ocean->GetDepth(eye) > 0.0) //Underwater
481  {
482  content->SetDrawingMode(DrawingMode::UNDERWATER);
483  DrawObjects();
484  glOcean->DrawBackground(camera);
485  glOcean->DrawBacksurface(camera);
486  //camera->GenerateBloom();
487  DrawLights();
488 
489  if(rSettings.ssr > RenderQuality::DISABLED)
490  {
491  //Linear depth front faces
492  camera->GenerateLinearDepth(true);
493 
494  //Linear depth back faces
495  OpenGLState::BindFramebuffer(camera->getPostprocessFBO());
496  glClear(GL_DEPTH_BUFFER_BIT);
497  glColorMask(GL_FALSE, GL_FALSE, GL_FALSE, GL_FALSE);
498  glCullFace(GL_FRONT);
499  content->SetDrawingMode(DrawingMode::FLAT);
500  DrawObjects();
501  glCullFace(GL_BACK);
502  glColorMask(GL_TRUE, GL_TRUE, GL_TRUE, GL_TRUE);
503  camera->GenerateLinearDepth(false);
504 
505  //Draw screen-space reflections
506  camera->DrawSSR();
507  }
508 
509  //Draw bloom effect simulating scattering
510  //camera->DrawBloom((GLfloat)ocean->getWaterType());
511 
512  //Suspended particles only below surface
513  glDepthMask(GL_FALSE);
514  glOcean->DrawParticles(camera);
515  glDepthMask(GL_TRUE);
516 
517  }
518  else //Above water
519  {
520  content->SetDrawingMode(DrawingMode::UNDERWATER);
521  DrawObjects();
522  DrawLights();
523  glOcean->DrawBackground(camera);
524 
525  //Draw surface to back buffer
526  camera->SetRenderBuffers(1, false, true); //Clearing color buffer
527  camera->SetRenderBuffers(1, true, false); //Color + Normal
528  glOcean->DrawSurface(camera);
529  camera->SetRenderBuffers(0, false, false); //Color only
530 
531  //Blend surface on top of scene
532  OpenGLState::DisableDepthTest();
533  OpenGLState::EnableBlend();
534  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
535  content->DrawTexturedSAQ(camera->getColorTexture(1));
536  OpenGLState::DisableBlend();
537  OpenGLState::EnableDepthTest();
538 
539  //Draw all objects as above surface
540  //(depth testing will secure drawing only what is above water)
541  camera->SetRenderBuffers(0, true, false); //Color + Normal
542  content->SetDrawingMode(DrawingMode::FULL);
543  DrawObjects();
544  DrawLights();
545 
546  //Render sky (left for the end to only fill empty spaces)
547  atm->getOpenGLAtmosphere()->DrawSkyAndSun(camera);
548 
549  //Postprocess
550  if(rSettings.ssr > RenderQuality::DISABLED)
551  {
552  //Linear depth front faces
553  camera->GenerateLinearDepth(true);
554 
555  //Linear depth back faces
556  OpenGLState::BindFramebuffer(camera->getPostprocessFBO());
557  glClear(GL_DEPTH_BUFFER_BIT);
558  glColorMask(GL_FALSE, GL_FALSE, GL_FALSE, GL_FALSE);
559  glCullFace(GL_FRONT);
560  content->SetDrawingMode(DrawingMode::FLAT);
561  DrawObjects();
562  glCullFace(GL_BACK);
563  glColorMask(GL_TRUE, GL_TRUE, GL_TRUE, GL_TRUE);
564  camera->GenerateLinearDepth(false);
565 
566  //Draw screen-space reflections
567  camera->DrawSSR();
568  }
569  }
570  }
571 
572  //Tone mapping
573  camera->DrawLDR(screenFBO, true);
574 
575  //Helper objects
576  if(camera->getType() == ViewType::TRACKBALL)
577  {
578  //Overlay debugging info
579  OpenGLState::BindFramebuffer(screenFBO); //No depth buffer, just one color buffer
580  content->SetProjectionMatrix(camera->GetProjectionMatrix());
581  content->SetViewMatrix(camera->GetViewMatrix());
582  OpenGLState::DisableCullFace();
583 
584  //Simulation debugging
585  glPolygonMode(GL_FRONT_AND_BACK, GL_LINE);
586  //if(sim->getSolidDisplayMode() == DisplayMode::PHYSICAL) DrawObjects();
587  DrawHelpers();
588  if(hSettings.showOceanVelocityField && ocean != NULL) ocean->getOpenGLOcean()->DrawVelocityField(camera, 5.f);
589  if(hSettings.showBulletDebugInfo) sim->RenderBulletDebug();
590  glPolygonMode(GL_FRONT_AND_BACK, GL_FILL);
591 
592  //Overlay selection outline
593  ((OpenGLTrackball*)camera)->DrawSelection(selectedDrawingQueueCopy, screenFBO);
594 
595  //Graphics debugging
596  //if(ocean != NULL)
597  //{
598  //ocean->getOpenGLOcean()->ShowSpectrum(glm::vec2(512.f,512.f), glm::vec4(0.f,0.f,512.f,512.f));
599  //ocean->getOpenGLOcean()->ShowTexture(3, glm::vec4(0,0,512,512));
600  //}
601 
602  //atm->getOpenGLAtmosphere()->ShowSunShadowmaps(0, 0, 0.1f);
603 
604  //content->DrawTexturedQuad(0,0,800,600,camera->getPostprocessTexture(0));
605  //camera->ShowViewNormalTexture(glm::vec4(0,0,400,300));
606  //camera->ShowSceneTexture(glm::vec4(0,200,300,200));
607  //camera->ShowLinearDepthTexture(glm::vec4(0,0,600,500), true);
608  //camera->ShowLinearDepthTexture(glm::vec4(0,400,300,200), false);
609
