50 void USBLReal::ProcessMessages()
51 {
52  AcousticDataFrame* msg;
53  while((msg = (AcousticDataFrame*)ReadMessage()) != nullptr)
54  {
55  if(msg->data == "ACK")
56  {
57  //Get message data
58  AcousticModem* cNode = getNode(msg->source);
59  Vector3 cO = msg->txPosition;
60  Transform dT = getDeviceFrame();
61  Vector3 dO = dT.getOrigin();
62  Vector3 dir = getDeviceFrame().getBasis().inverse() * ((cO - dO).normalized()); //Direction in device frame
63  Scalar slantRange = msg->travelled/Scalar(2); //Distance to node is half of the full travelled distance
64  Scalar t = msg->timeStamp + slantRange/SOUND_VELOCITY_WATER;
65 
66  //Find plane coordinates
67  Scalar xLocal = CalcModel(slantRange, btAngle(dir, VX()));
68  Scalar yLocal = CalcModel(slantRange, btAngle(dir, VY()));
69 
70  //Find depth coordinate
71  Scalar zGlobal = cO.getZ();
72  if(noise)
73  {
74  zGlobal += noiseDepth(randomGenerator); //Transmitter
75  zGlobal -= noiseDepth(randomGenerator); //Receiver
76  }
77 
78  //Update position in the transponder and in the USBL
79  Vector3 worldPos = dT * Vector3(xLocal, yLocal, 0);
80  worldPos.setZ(zGlobal);
81  cNode->UpdatePosition(worldPos, true);
82 
83  //Populate beacon info structure
84  Scalar d = Vector3(dir.getX(), dir.getY(), Scalar(0)).length();
85  Scalar hAngle = atan2(dir.getY(), dir.getX());
86  Scalar vAngle = atan2(d, dir.getZ());
87 
88  BeaconInfo b;
89  b.t = t;
90  b.relPos = dT.inverse() * worldPos;
91  b.azimuth = hAngle;
92  b.elevation = M_PI_2 - vAngle;
93  b.range = slantRange;
94  b.localDepth = dO.getZ();
95  b.localOri = dT.getRotation();
96  beacons[msg->source] = b;
97 
98  newDataAvailable = true;
99  }
100 
101  delete msg;
102  }
103 }

105 Scalar USBLReal::CalcModel(Scalar R, Scalar theta)
106 {
107  Scalar result = R * btCos(theta);
108  if(noise)
109  {
110  result += R * btCos(theta) * (noiseTime(randomGenerator) + noiseSV(randomGenerator) - blError/bl);
111  result += R * SOUND_VELOCITY_WATER/freq * noisePhase(randomGenerator)/(Scalar(2.0*M_PI) * bl);
112  }
113  return result;
114 }
