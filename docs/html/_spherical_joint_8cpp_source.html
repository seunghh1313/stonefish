33 SphericalJoint::SphericalJoint(std::string uniqueName, SolidEntity* solidA, SolidEntity* solidB, const Vector3& pivot, bool collideLinked) : Joint(uniqueName, collideLinked)
34 {
35  btRigidBody* bodyA = solidA->rigidBody;
36  btRigidBody* bodyB = solidB->rigidBody;
37  Vector3 pivotInA = bodyA->getCenterOfMassTransform().inverse() * pivot;
38  Vector3 pivotInB = bodyB->getCenterOfMassTransform().inverse() * pivot;
39 
40  btPoint2PointConstraint* p2p = new btPoint2PointConstraint(*bodyA, *bodyB, pivotInA, pivotInB);
41  setConstraint(p2p);
42 
43  sigDamping = Vector3(0.,0.,0.);
44  velDamping = Vector3(0.,0.,0.);
45 
46  angleIC = Vector3(0.,0.,0.);
47 }

77 void SphericalJoint::ApplyDamping()
78 {
79  if(sigDamping.length2() > Scalar(0.) || velDamping.length2() > Scalar(0.))
80  {
81  btRigidBody& bodyA = getConstraint()->getRigidBodyA();
82  btRigidBody& bodyB = getConstraint()->getRigidBodyB();
83  Vector3 relativeAV = bodyA.getAngularVelocity() - bodyB.getAngularVelocity();
84  Vector3 torque(0.,0.,0.);
85 
86  if(relativeAV.x() != Scalar(0.))
87  torque[0] = -(sigDamping.x() * relativeAV.x()/fabs(relativeAV.x())) - velDamping.x() * relativeAV.x();
88  if(relativeAV.y() != Scalar(0.))
89  torque[1] = -(sigDamping.y() * relativeAV.y()/fabs(relativeAV.y())) - velDamping.y() * relativeAV.y();
90  if(relativeAV.z() != Scalar(0.))
91  torque[2] = -(sigDamping.z() * relativeAV.z()/fabs(relativeAV.z())) - velDamping.z() * relativeAV.z();
92 
93  bodyA.applyTorque(torque);
94  bodyB.applyTorque(-torque);
95  }
96 }

98 std::vector<Renderable> SphericalJoint::Render()
99 {
100  std::vector<Renderable> items(0);
101  Renderable item;
102  item.model = glm::mat4(1.f);
103  item.type = RenderableType::JOINT_LINES;
104 
105  btPoint2PointConstraint* p2p = (btPoint2PointConstraint*)getConstraint();
106  Vector3 pivot = p2p->getRigidBodyA().getCenterOfMassTransform()(p2p->getPivotInA());
107  Vector3 A = p2p->getRigidBodyA().getCenterOfMassPosition();
108  Vector3 B = p2p->getRigidBodyB().getCenterOfMassPosition();
109 
110  item.points.push_back(glm::vec3(A.getX(), A.getY(), A.getZ()));
111  item.points.push_back(glm::vec3(pivot.getX(), pivot.getY(), pivot.getZ()));
112  item.points.push_back(glm::vec3(B.getX(), B.getY(), B.getZ()));
113  item.points.push_back(glm::vec3(pivot.getX(), pivot.getY(), pivot.getZ()));
114 
115  items.push_back(item);
116  return items;
117 }
