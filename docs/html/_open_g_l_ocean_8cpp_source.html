348 void OpenGLOcean::setWaterType(GLfloat t)
349 {
350  t *= 63.f;
351  float iPart;
352  float fPart = modff(t, &iPart);
353  int id = (int)truncf(iPart);
354  lightAbsorption = absorption[id];
355  lightScattering = scattering[id];
356  if(fPart > 0.f)
357  {
358  lightAbsorption += fPart * (absorption[id+1] - absorption[id]);
359  lightScattering += fPart * (scattering[id+1] - scattering[id]);
360  }
361 }

425  oceanShaders["variance"]->SetUniform("fftSize", params.fftSize);

427  oceanShaders["variance"]->SetUniform("slopeVarianceDelta", slopeVarianceDelta);
428 
429  OpenGLState::BindTexture(TEX_POSTPROCESS1, GL_TEXTURE_2D, oceanTextures[0]);
430  OpenGLState::BindTexture(TEX_POSTPROCESS2, GL_TEXTURE_2D, oceanTextures[1]);

437  }
438 
439  OpenGLState::UnbindTexture(TEX_POSTPROCESS2);
440  OpenGLState::UnbindTexture(TEX_POSTPROCESS1);
441  OpenGLState::BindFramebuffer(0);

444 void OpenGLOcean::Simulate(GLfloat dt)
445 {
446  OpenGLState::DisableDepthTest();
447  OpenGLState::DisableCullFace();
448 
449  ((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getContent()->BindBaseVertexArray();
450 
451  //Init -> one triangle -> multiple outputs
452  OpenGLState::BindFramebuffer(oceanFBOs[0]);
453  OpenGLState::Viewport(0, 0, params.fftSize, params.fftSize);
454  oceanShaders["init"]->Use();
455  oceanShaders["init"]->SetUniform("texSpectrum12", TEX_POSTPROCESS1);
456  oceanShaders["init"]->SetUniform("texSpectrum34", TEX_POSTPROCESS2);
457  oceanShaders["init"]->SetUniform("fftSize", (GLfloat)params.fftSize);
458  oceanShaders["init"]->SetUniform("inverseGridSizes", glm::vec4(2.f*M_PI*(GLfloat)params.fftSize/params.gridSizes[0],
459  2.f*M_PI*(GLfloat)params.fftSize/params.gridSizes[1],
460  2.f*M_PI*(GLfloat)params.fftSize/params.gridSizes[2],
461  2.f*M_PI*(GLfloat)params.fftSize/params.gridSizes[3]));
462  oceanShaders["init"]->SetUniform("t", params.t);
463  params.t += dt;
464  OpenGLState::BindTexture(TEX_POSTPROCESS1, GL_TEXTURE_2D, oceanTextures[0]);
465  OpenGLState::BindTexture(TEX_POSTPROCESS2, GL_TEXTURE_2D, oceanTextures[1]);
466  glDrawArrays(GL_TRIANGLES, 0, 3); //1 Layer
467 
468  //FFT passes -> multiple triangles -> multiple layers
469  OpenGLState::BindFramebuffer(oceanFBOs[1]);
470  OpenGLState::Viewport(0, 0, params.fftSize, params.fftSize);
471 
472  oceanShaders["fftx"]->Use();
473  oceanShaders["fftx"]->SetUniform("texButterfly", TEX_POSTPROCESS1);
474  oceanShaders["fftx"]->SetUniform("texSource", TEX_POSTPROCESS2);
475 
476  OpenGLState::BindTexture(TEX_POSTPROCESS1, GL_TEXTURE_2D, oceanTextures[5]);
477  OpenGLState::UnbindTexture(TEX_POSTPROCESS2);
478 
479  for(unsigned int i = 0; i < params.passes; ++i)
480  {
481  oceanShaders["fftx"]->SetUniform("pass", ((float)i + 0.5f)/(float)params.passes);
482  if(i%2 == 0)
483  {
484  OpenGLState::BindTexture(TEX_POSTPROCESS2, GL_TEXTURE_2D_ARRAY, oceanTextures[3]);
485  glDrawBuffer(GL_COLOR_ATTACHMENT1);
486  }
487  else
488  {
489  OpenGLState::BindTexture(TEX_POSTPROCESS2, GL_TEXTURE_2D_ARRAY, oceanTextures[4]);
490  glDrawBuffer(GL_COLOR_ATTACHMENT0);
491  }
492  glDrawArrays(GL_TRIANGLES, 0, 3 * 3); //3 Layers
493  }
494 
495  oceanShaders["ffty"]->Use();
496  oceanShaders["ffty"]->SetUniform("texButterfly", TEX_POSTPROCESS1);
497  oceanShaders["ffty"]->SetUniform("texSource", TEX_POSTPROCESS2);
498 
499  for(unsigned int i = params.passes; i < 2 * params.passes; ++i)
500  {
501  oceanShaders["ffty"]->SetUniform("pass", ((float)i - params.passes + 0.5f)/(float)params.passes);
502  if (i%2 == 0)
503  {
504  OpenGLState::BindTexture(TEX_POSTPROCESS2, GL_TEXTURE_2D_ARRAY, oceanTextures[3]);
505  glDrawBuffer(GL_COLOR_ATTACHMENT1);
506  }
507  else
508  {
509  OpenGLState::BindTexture(TEX_POSTPROCESS2, GL_TEXTURE_2D_ARRAY, oceanTextures[4]);
510  glDrawBuffer(GL_COLOR_ATTACHMENT0);
511  }
512  glDrawArrays(GL_TRIANGLES, 0, 3 * 3); //3 Layers

814 {

836 GLfloat* OpenGLOcean::ComputeButterflyLookupTable(unsigned int size, unsigned int passes)
837 {
838  GLfloat *data = new GLfloat[size * passes * 4];
839 
840  for(unsigned int i = 0; i < passes; ++i)
841  {
842  int nBlocks = (int)powf(2.0, float(passes - 1 - (int)i));
843  int nHInputs = (int)powf(2.0, float(i));
844  for(int j = 0; j < nBlocks; ++j)
845  {
846  for(int k = 0; k < nHInputs; ++k)
847  {
848  int i1, i2, j1, j2;
849  if(i == 0)

851  i1 = j * nHInputs * 2 + k;
852  i2 = j * nHInputs * 2 + nHInputs + k;
853  j1 = bitReverse(i1, size);
854  j2 = bitReverse(i2, size);
855  }
856  else

859  i2 = j * nHInputs * 2 + nHInputs + k;
860  j1 = i1;
861  j2 = i2;
862  }
863 
864  float wr, wi;
865  computeWeight(size, k * nBlocks, wr, wi);
866 
867  int offset1 = 4 * (i1 + i * size);

896 // 1/kx and 1/ky in meters
897 float OpenGLOcean::spectrum(float kx, float ky, bool omnispectrum)
898 {
899  float U10 = params.wind;
900  float Omega = params.omega;
901 
902  // phase speed
903  float k = sqrt(kx * kx + ky * ky);
904  float c = omega(k) / k;
905 
906  // spectral peak
907  float kp = 9.81 * sqr(Omega / U10); // after Eq 3
908  float cp = omega(kp) / kp;
909 
910  // friction velocity
911  float z0 = 3.7e-5 * sqr(U10) / 9.81 * pow(U10 / cp, 0.9f); // Eq 66
912  float u_star = 0.41 * U10 / log(10.0 / z0); // Eq 60
913 
914  float Lpm = exp(- 5.0 / 4.0 * sqr(kp / k)); // after Eq 3
915  float gamma = Omega < 1.0 ? 1.7 : 1.7 + 6.0 * log(Omega); // after Eq 3 // log10 or log??
916  float sigma = 0.08 * (1.0 + 4.0 / pow(Omega, 3.0f)); // after Eq 3
917  float Gamma = exp(-1.0 / (2.0 * sqr(sigma)) * sqr(sqrt(k / kp) - 1.0));
918  float Jp = pow(gamma, Gamma); // Eq 3
919  float Fp = Lpm * Jp * exp(- Omega / sqrt(10.0) * (sqrt(k / kp) - 1.0)); // Eq 32
920  float alphap = 0.006 * sqrt(Omega); // Eq 34
921  float Bl = 0.5 * alphap * cp / c * Fp; // Eq 31
922 
923  float alpham = 0.01 * (u_star < params.cm ? 1.0 + log(u_star / params.cm) : 1.0 + 3.0 * log(u_star / params.cm)); // Eq 44
924  float Fm = exp(-0.25 * sqr(k / params.km - 1.0)); // Eq 41
925  float Bh = 0.5 * alpham * params.cm / c * Fm; // Eq 40
926 
927  Bh *= Lpm;
928 
929  if (omnispectrum)
930  {
931  return params.A * (Bl + Bh) / (k * sqr(k)); // Eq 30
932  }
933 
934  float a0 = log(2.0) / 4.0;
935  float ap = 4.0;
936  float am = 0.13 * u_star / params.cm; // Eq 59
937  float Delta = tanh(a0 + ap * pow(c / cp, 2.5f) + am * pow(params.cm / c, 2.5f)); // Eq 57
938 
939  float phi = atan2(ky, kx);
940 
941  if(params.propagate)
942  {
943  if (kx < 0.0)
944  {
945  return 0.0;
946  }
947  else
948  {
949  Bl *= 2.0;
950  Bh *= 2.0;
951  }
952  }
953 
954  // remove waves perpendicular to wind dir
955  float tweak = sqrtf( std::max( kx/sqrtf(kx*kx+ky*ky), 0.f) );
956  tweak = 1.0f;
957  return params.A * (Bl + Bh) * (1.0 + Delta * cos(2.0 * phi)) / (2.0 * M_PI * sqr(sqr(k))) * tweak; // Eq 67
958 }

960 void OpenGLOcean::GetSpectrumSample(int i, int j, float lengthScale, float kMin, float *result)
961 {
962  static long seed = 1234;
963  float dk = 2.0 * M_PI / lengthScale;
964  float kx = i * dk;
965  float ky = j * dk;
966  if(fabsf(kx) < kMin && fabsf(ky) < kMin)
967  {
968  result[0] = 0.0;
969  result[1] = 0.0;
970  }
971  else
972  {
973  float S = spectrum(kx, ky);
974  float h = sqrtf(S / 2.0) * dk;
975  float phi = frandom(&seed) * 2.0 * M_PI;
976  result[0] = h * cos(phi);
977  result[1] = h * sin(phi);
978  }

1017 float OpenGLOcean::ComputeSlopeVariance()
1018 {
1019  //slope variance due to all waves, by integrating over the full spectrum
1020  float theoreticSlopeVariance = 0.0;
1021  float k = 5e-3;
1022  while (k < 1e3)
1023  {
1024  float nextK = k * 1.001;
1025  theoreticSlopeVariance += k * k * spectrum(k, 0, true) * (nextK - k);
1026  k = nextK;
1027  }
1028 
1029  // slope variance due to waves, by integrating over the spectrum part
1030  // that is covered by the four nested grids. This can give a smaller result
1031  // than the theoretic total slope variance, because the higher frequencies
1032  // may not be covered by the four nested grid. Hence the difference between
1033  // the two is added as a "delta" slope variance in the "variances" shader,
1034  // to be sure not to lose the variance due to missing wave frequencies in
1035  // the four nested grids
1036  float totalSlopeVariance = 0.0;
1037  for (int y = 0; y < params.fftSize; ++y)
1038  {
1039  for (int x = 0; x < params.fftSize; ++x)
1040  {
1041  int offset = 4 * (x + y * params.fftSize);
1042  float i = 2.f * M_PI * (x >= params.fftSize / 2 ? x - params.fftSize : x);
1043  float j = 2.f * M_PI * (y >= params.fftSize / 2 ? y - params.fftSize : y);
1044  totalSlopeVariance += GetSlopeVariance(i / params.gridSizes[0], j / params.gridSizes[0], params.spectrum12 + offset);
1045  totalSlopeVariance += GetSlopeVariance(i / params.gridSizes[1], j / params.gridSizes[1], params.spectrum12 + offset + 2);
1046  totalSlopeVariance += GetSlopeVariance(i / params.gridSizes[2], j / params.gridSizes[2], params.spectrum34 + offset);
1047  totalSlopeVariance += GetSlopeVariance(i / params.gridSizes[3], j / params.gridSizes[3], params.spectrum34 + offset + 2);
1048  }
1049  }
1050 
1051  return theoreticSlopeVariance - totalSlopeVariance;
1052 }
