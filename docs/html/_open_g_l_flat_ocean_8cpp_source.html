38 OpenGLFlatOcean::OpenGLFlatOcean(GLfloat size) : OpenGLOcean(200000.f)
39 {
40  params.wind = 5.f;

43 
44  //Surface rendering
45  GLint compiled;
46  GLuint pcssFragment = GLSLShader::LoadShader(GL_FRAGMENT_SHADER, "lighting.frag", "", &compiled);
47  std::vector<GLuint> precompiled;
48  precompiled.push_back(OpenGLAtmosphere::getAtmosphereAPI());
49  precompiled.push_back(pcssFragment);
50 
51  std::vector<GLSLSource> sources;
52  sources.push_back(GLSLSource(GL_VERTEX_SHADER, "oceanSurface.vert"));
53  sources.push_back(GLSLSource(GL_FRAGMENT_SHADER, "oceanSurface.frag"));
54  oceanShaders["surface"] = new GLSLShader(sources, precompiled);
55  oceanShaders["surface"]->AddUniform("size", ParameterType::FLOAT);
56  oceanShaders["surface"]->AddUniform("texWaveFFT", ParameterType::INT);
57  oceanShaders["surface"]->AddUniform("texSlopeVariance", ParameterType::INT);
58  oceanShaders["surface"]->AddUniform("MVP", ParameterType::MAT4);
59  oceanShaders["surface"]->AddUniform("gridSizes", ParameterType::VEC4);
60  oceanShaders["surface"]->AddUniform("eyePos", ParameterType::VEC3);
61  oceanShaders["surface"]->AddUniform("viewDir", ParameterType::VEC3);
62  oceanShaders["surface"]->AddUniform("MV", ParameterType::MAT3);
63  oceanShaders["surface"]->AddUniform("FC", ParameterType::FLOAT);
64  oceanShaders["surface"]->AddUniform("viewport", ParameterType::VEC2);
65  oceanShaders["surface"]->AddUniform("transmittance_texture", ParameterType::INT);
66  oceanShaders["surface"]->AddUniform("scattering_texture", ParameterType::INT);
67  oceanShaders["surface"]->AddUniform("irradiance_texture", ParameterType::INT);
68  oceanShaders["surface"]->AddUniform("sunShadowMap", ParameterType::INT);
69  oceanShaders["surface"]->AddUniform("sunDepthMap", ParameterType::INT);
70  oceanShaders["surface"]->BindUniformBlock("SunSky", UBO_SUNSKY);
71 
72  oceanShaders["surface"]->Use();
73  oceanShaders["surface"]->SetUniform("transmittance_texture", TEX_ATM_TRANSMITTANCE);
74  oceanShaders["surface"]->SetUniform("scattering_texture", TEX_ATM_SCATTERING);
75  oceanShaders["surface"]->SetUniform("irradiance_texture", TEX_ATM_IRRADIANCE);
76  oceanShaders["surface"]->SetUniform("sunDepthMap", TEX_SUN_DEPTH);
77  oceanShaders["surface"]->SetUniform("sunShadowMap", TEX_SUN_SHADOW);
78 
79  //Backsurface rendering
80  GLuint oceanOpticsFragment = GLSLShader::LoadShader(GL_FRAGMENT_SHADER, "oceanOptics.frag", "", &compiled);
81  precompiled.pop_back();
82  precompiled.push_back(oceanOpticsFragment);
83 
84  sources.pop_back();
85  sources.push_back(GLSLSource(GL_FRAGMENT_SHADER, "oceanBacksurface.frag"));
86  oceanShaders["backsurface"] = new GLSLShader(sources, precompiled);
87  oceanShaders["backsurface"]->AddUniform("size", ParameterType::FLOAT);
88  oceanShaders["backsurface"]->AddUniform("texWaveFFT", ParameterType::INT);
89  oceanShaders["backsurface"]->AddUniform("texSlopeVariance", ParameterType::INT);
90  oceanShaders["backsurface"]->AddUniform("MVP", ParameterType::MAT4);
91  oceanShaders["backsurface"]->AddUniform("gridSizes", ParameterType::VEC4);
92  oceanShaders["backsurface"]->AddUniform("eyePos", ParameterType::VEC3);
93  oceanShaders["backsurface"]->AddUniform("MV", ParameterType::MAT3);
94  oceanShaders["backsurface"]->AddUniform("FC", ParameterType::FLOAT);
95  oceanShaders["backsurface"]->AddUniform("viewport", ParameterType::VEC2);
96  oceanShaders["backsurface"]->AddUniform("cWater", ParameterType::VEC3);
97  oceanShaders["backsurface"]->AddUniform("bWater", ParameterType::VEC3);
98  oceanShaders["backsurface"]->AddUniform("transmittance_texture", ParameterType::INT);
99  oceanShaders["backsurface"]->AddUniform("scattering_texture", ParameterType::INT);
100  oceanShaders["backsurface"]->AddUniform("irradiance_texture", ParameterType::INT);
101  oceanShaders["backsurface"]->BindUniformBlock("SunSky", UBO_SUNSKY);
102  oceanShaders["backsurface"]->BindUniformBlock("Lights", UBO_LIGHTS);
103  glDeleteShader(oceanOpticsFragment);
104 
105  oceanShaders["backsurface"]->Use();
106  oceanShaders["backsurface"]->SetUniform("transmittance_texture", TEX_ATM_TRANSMITTANCE);
107  oceanShaders["backsurface"]->SetUniform("scattering_texture", TEX_ATM_SCATTERING);
108  oceanShaders["backsurface"]->SetUniform("irradiance_texture", TEX_ATM_IRRADIANCE);
109 
110  //Mask rendering
111  sources.pop_back();
112  sources.push_back(GLSLSource(GL_FRAGMENT_SHADER, "flat.frag"));
113  oceanShaders["mask"] = new GLSLShader(sources);
114  oceanShaders["mask"]->AddUniform("size", ParameterType::FLOAT);
115  oceanShaders["mask"]->AddUniform("MVP", ParameterType::MAT4);
116  oceanShaders["mask"]->AddUniform("FC", ParameterType::FLOAT);
117

158 void OpenGLFlatOcean::DrawSurface(OpenGLCamera* cam)
159 {
160  GLint* viewport = cam->GetViewport();
161  OpenGLState::BindTexture(TEX_POSTPROCESS1, GL_TEXTURE_2D_ARRAY, oceanTextures[3]);
162  OpenGLState::BindTexture(TEX_POSTPROCESS2, GL_TEXTURE_3D, oceanTextures[2]);
163 
164  oceanShaders["surface"]->Use();
165  oceanShaders["surface"]->SetUniform("MVP", cam->GetProjectionMatrix() * cam->GetViewMatrix());
166  oceanShaders["surface"]->SetUniform("MV", glm::mat3(glm::transpose(glm::inverse(cam->GetViewMatrix()))));
167  oceanShaders["surface"]->SetUniform("FC", cam->GetLogDepthConstant());
168  oceanShaders["surface"]->SetUniform("size", oceanSize);
169  oceanShaders["surface"]->SetUniform("viewport", glm::vec2((GLfloat)viewport[2], (GLfloat)viewport[3]));
170  oceanShaders["surface"]->SetUniform("eyePos", cam->GetEyePosition());
171  oceanShaders["surface"]->SetUniform("viewDir", cam->GetLookingDirection());
172  oceanShaders["surface"]->SetUniform("gridSizes", params.gridSizes);
173  oceanShaders["surface"]->SetUniform("texWaveFFT", TEX_POSTPROCESS1);
174  oceanShaders["surface"]->SetUniform("texSlopeVariance", TEX_POSTPROCESS2);
175  OpenGLState::BindVertexArray(vao);
176  glDrawArrays(GL_TRIANGLES, 0, 12);
177  OpenGLState::BindVertexArray(0);
178 
179  OpenGLState::UseProgram(0);
180  OpenGLState::UnbindTexture(TEX_POSTPROCESS2);
181  OpenGLState::UnbindTexture(TEX_POSTPROCESS1);
182  delete [] viewport;
183 }

185 void OpenGLFlatOcean::DrawBacksurface(OpenGLCamera* cam)
186 {
187  GLint* viewport = cam->GetViewport();
188  OpenGLState::BindTexture(TEX_POSTPROCESS1, GL_TEXTURE_2D_ARRAY, oceanTextures[3]);
189  OpenGLState::BindTexture(TEX_POSTPROCESS2, GL_TEXTURE_3D, oceanTextures[2]);
190 
191  oceanShaders["backsurface"]->Use();
192  oceanShaders["backsurface"]->SetUniform("MVP", cam->GetProjectionMatrix() * cam->GetViewMatrix());
193  oceanShaders["backsurface"]->SetUniform("MV", glm::mat3(glm::transpose(glm::inverse(cam->GetViewMatrix()))));
194  oceanShaders["backsurface"]->SetUniform("FC", cam->GetLogDepthConstant());
195  oceanShaders["backsurface"]->SetUniform("size", oceanSize);
196  oceanShaders["backsurface"]->SetUniform("eyePos", cam->GetEyePosition());
197  oceanShaders["backsurface"]->SetUniform("gridSizes", params.gridSizes);
198  oceanShaders["backsurface"]->SetUniform("cWater", getLightAttenuation());
199  oceanShaders["backsurface"]->SetUniform("bWater", getLightScattering());
200  oceanShaders["backsurface"]->SetUniform("viewport", glm::vec2((GLfloat)viewport[2], (GLfloat)viewport[3]));
201  oceanShaders["backsurface"]->SetUniform("texWaveFFT", TEX_POSTPROCESS1);
202  oceanShaders["backsurface"]->SetUniform("texSlopeVariance", TEX_POSTPROCESS2);
203  OpenGLState::BindVertexArray(vao);
204  glCullFace(GL_FRONT);
205  glDrawArrays(GL_TRIANGLES, 0, 12);
206  glCullFace(GL_BACK);
207  OpenGLState::BindVertexArray(0);
208  OpenGLState::UnbindTexture(TEX_POSTPROCESS2);
209  OpenGLState::UnbindTexture(TEX_POSTPROCESS1);
210  OpenGLState::UseProgram(0);
211  delete [] viewport;
212 }

214 void OpenGLFlatOcean::DrawUnderwaterMask(OpenGLCamera* cam)
215 {
216  glColorMask(GL_FALSE, GL_FALSE, GL_FALSE, GL_FALSE);
217  oceanShaders["mask"]->Use();
218  oceanShaders["mask"]->SetUniform("MVP", cam->GetProjectionMatrix() * cam->GetViewMatrix());
219  oceanShaders["mask"]->SetUniform("FC", cam->GetLogDepthConstant());
220  oceanShaders["mask"]->SetUniform("size", oceanSize);
221 
222  //1. Draw surface to depth buffer
223  OpenGLState::BindVertexArray(vao);
224  glDrawArrays(GL_TRIANGLES, 0, 12);
225  OpenGLState::BindVertexArray(0);
226 
227  //2. Draw backsurface to depth and stencil buffer
228  OpenGLState::EnableStencilTest();
229  glStencilFunc(GL_ALWAYS, 1, 0xFF);
230  glStencilOp(GL_KEEP, GL_KEEP, GL_REPLACE);
231  glStencilMask(0xFF);
232  glClear(GL_STENCIL_BUFFER_BIT);
233  glCullFace(GL_FRONT);
234 
235  OpenGLState::BindVertexArray(vao);
236  glDrawArrays(GL_TRIANGLES, 0, 12);
237  OpenGLState::BindVertexArray(0);
238 
239  //3. Draw box around whole ocean
240  OpenGLOcean::DrawUnderwaterMask(cam);
241 
242  glCullFace(GL_BACK);
243  OpenGLState::DisableStencilTest();
244  glColorMask(GL_TRUE, GL_TRUE, GL_TRUE, GL_TRUE);
245  glClear(GL_DEPTH_BUFFER_BIT);
246 }
