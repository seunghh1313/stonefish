102 void Propeller::Update(Scalar dt)
103 {
104  //Update thruster velocity
105  Scalar error = setpoint * omegaLim - omega;
106  Scalar motorTorque = kp * error + ki * iError;
107  omega += (motorTorque - torque)*dt;
108  theta += omega * dt; //Just for animation
109 
110  //Integrate error
111  iError += error * dt;
112  iError = iError > iLim ? iLim : (iError < -iLim ? -iLim : iError);
113 
114  if(attach != NULL)
115  {
116  //Get transforms
117  Transform solidTrans = attach->getCGTransform();
118  Transform propTrans = attach->getOTransform() * o2a;
119  Vector3 relPos = propTrans.getOrigin() - solidTrans.getOrigin();
120  Vector3 velocity = attach->getLinearVelocityInLocalPoint(relPos);
121 
122  Atmosphere* atm = SimulationApp::getApp()->getSimulationManager()->getAtmosphere();
123 
124  if(atm->IsInsideFluid(propTrans.getOrigin()))
125  {
126  //Calculate thrust
127  //Thrust coefficient depends on advance ratio J = u/(n*D), kT0 -> J=0
128  //The lower the p/D the more linear the dependence of Kt on J
129  Scalar n = omega/(Scalar(2) * M_PI);
130  Scalar u = -propTrans.getBasis().getColumn(0).dot(atm->GetFluidVelocity(propTrans.getOrigin()) - velocity); //Incoming air velocity
131  Scalar alpha(-0.095/0.8);
132  //kT(J) = kT0 + alpha * J --> approximated with linear function
133  thrust = (RH ? Scalar(1) : Scalar(-1)) * atm->getGas().density * D*D*D * btFabs(n) * (D*kT0*n + alpha*u);
134  torque = -kQ0 * atm->getGas().density * btFabs(n)*n * D*D*D*D*D; //Torque is the loading of propeller due to drag
135 
136  //Apply forces and torques
137  Vector3 thrustV(thrust, 0, 0);
138  Vector3 torqueV(torque, 0, 0);
139  attach->ApplyCentralForce(propTrans.getBasis() * thrustV);
140  attach->ApplyTorque((propTrans.getOrigin() - solidTrans.getOrigin()).cross(propTrans.getBasis() * thrustV));
141  attach->ApplyTorque(propTrans.getBasis() * torqueV);
142  }
143  }
144 }

146 std::vector<Renderable> Propeller::Render()
147 {
148  Transform propTrans = Transform::getIdentity();
149  if(attach != NULL)
150  propTrans = attach->getOTransform() * o2a;
151  else
152  LinkActuator::Render();
153 
154  //Rotate propeller
155  propTrans *= Transform(Quaternion(0, 0, theta), Vector3(0,0,0));
156 
157  //Add renderable
158  std::vector<Renderable> items(0);
159  Renderable item;
160  item.type = RenderableType::SOLID;
161  item.materialName = prop->getMaterial().name;
162  item.objectId = prop->getGraphicalObject();
163  item.lookId = dm == DisplayMode::GRAPHICAL ? prop->getLook() : -1;
164  item.model = glMatrixFromTransform(propTrans);
165  items.push_back(item);
166 
167  item.type = RenderableType::ACTUATOR_LINES;
168  item.points.push_back(glm::vec3(0,0,0));
169  item.points.push_back(glm::vec3(0.1f*thrust,0,0));
170  items.push_back(item);
171 
172  return items;
173 }
