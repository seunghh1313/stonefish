102 void GeneralRobot::AddToSimulation(SimulationManager* sm, const Transform& origin)
103 {
104  Robot::AddToSimulation(sm, origin);
105 
106  // Base link
107  sm->AddSolidEntity(links[0], origin);
108  if(fixed)
109  {
110  FixedJoint* fix = new FixedJoint(name + "_fix", links[0]);
111  joints.push_back(fix);
112  }
113 
114  // Rest of the links and joints
115  for(size_t i = 1; i < links.size(); ++i)
116  {
117  sm->AddSolidEntity(links[i], origin);
118  }
119 
120  for(size_t i = 0; i < jointsData.size(); ++i)
121  {
122  size_t parentId;
123  size_t childId;
124 
125  for(size_t h = 0; h < links.size(); ++h)
126  {
127  if(links[h]->getName() == jointsData[i].parent)
128  parentId = h;
129  else if(links[h]->getName() == jointsData[i].child)
130  childId = h;
131  }
132  if(parentId >= links.size())
133  cCritical("Parent link '%s' doesn't exist!", jointsData[i].parent.c_str());
134  if(childId >= links.size())
135  cCritical("Child link '%s' doesn't exist!", jointsData[i].child.c_str());
136 
137  switch(jointsData[i].jtype)
138  {
139  case JointType::FIXED:
140  {
141  FixedJoint* fix = new FixedJoint(jointsData[i].name,
142  links[parentId],
143  links[childId]);
144  joints.push_back(fix);
145  }
146  break;
147 
148  case JointType::REVOLUTE:
149  {
150  RevoluteJoint* rev = new RevoluteJoint(jointsData[i].name,
151  links[parentId],
152  links[childId],
153  (origin * jointsData[i].origin).getOrigin(),
154  (origin * jointsData[i].origin).getBasis() * jointsData[i].axis, false);
155  rev->setLimits(jointsData[i].posLim.first, jointsData[i].posLim.second);
156  joints.push_back(rev);
157  }
158  break;
159 
160  case JointType::PRISMATIC:
161  {
162  PrismaticJoint* prism = new PrismaticJoint(jointsData[i].name,
163  links[parentId],
164  links[childId],
165  (origin * jointsData[i].origin).getBasis() * jointsData[i].axis, false);
166  prism->setLimits(jointsData[i].posLim.first, jointsData[i].posLim.second);
167  joints.push_back(prism);
168  }
169  break;
170 
171  default:
172  break;
173  }
174  }
175 
176  for(size_t i = 0; i < joints.size(); ++i)
177  sm->AddJoint(joints[i]);
178 
179  // Attach joint sensors
180  for(size_t i = 0; i < jsAttachments.size(); ++i)
181  {
182  Joint* j = getJoint(jsAttachments[i].second);
183  if(j != nullptr)
184  jsAttachments[i].first->AttachToJoint(j);
185  else
186  cCritical("Joint '%s' doesn't exist. Sensor '%s' cannot be attached!",
187  jsAttachments[i].second.c_str(),
188  jsAttachments[i].first->getName().c_str());
189  }
190 
191  // Attach joint actuators
192  for(size_t i = 0; i < jaAttachments.size(); ++i)
193  {
194  Joint* j = getJoint(jaAttachments[i].second);
195  if(j != nullptr)
196  jaAttachments[i].first->AttachToJoint(j);
197  else
198  cCritical("Joint '%s' doesn't exist. Actuator '%s' cannot be attached!",
199  jaAttachments[i].second.c_str(),
200  jaAttachments[i].first->getName().c_str());
201  }
202 }
