37 OpenGLTrackball::OpenGLTrackball(glm::vec3 centerPosition, GLfloat orbitRadius, glm::vec3 up,
38  GLint x, GLint y, GLint width, GLint height, GLfloat horizontalFovDeg,
39  glm::vec2 range) : OpenGLCamera(x, y, width, height, range)
40 {
41  this->up = glm::normalize(up);
42  rotation = glm::rotation(this->up, glm::vec3(0,0,1.f));
43  center = centerPosition;
44  radius = orbitRadius;
45  fovx = horizontalFovDeg/180.f*M_PI;
46  GLfloat fovy = 2.f * atanf( (GLfloat)viewportHeight/(GLfloat)viewportWidth * tanf(fovx/2.f) );
47  projection = glm::perspectiveFov(fovy, (GLfloat)viewportWidth, (GLfloat)viewportHeight, near, far);
48  dragging = false;
49  transMode = false;
50  continuous = true;
51  holdingEntity = nullptr;
52 
53  outlineShader[0] = new GLSLShader("outline.frag");
54  outlineShader[0]->AddUniform("color", ParameterType::VEC4);
55  outlineShader[0]->AddUniform("texStencil", ParameterType::INT);
56  outlineShader[0]->AddUniform("uvOffset", ParameterType::VEC2);
57 
58  outlineShader[1] = new GLSLShader("smallBlur.frag");
59  outlineShader[1]->AddUniform("tex", ParameterType::INT);
60  outlineShader[1]->AddUniform("invTexSize", ParameterType::VEC2);
61 
62  UpdateTransform();
63 }

108 void OpenGLTrackball::UpdateTransform()
109 {
110  if(holdingEntity != nullptr) center = tempCenter;
111  trackballTransform = glm::lookAt(GetEyePosition(), center, GetUpDirection());
112 
113  viewUBOData.VP = GetProjectionMatrix() * GetViewMatrix();
114  viewUBOData.eye = GetEyePosition();
115  ExtractFrustumFromVP(viewUBOData.frustum, viewUBOData.VP);

118 GLfloat OpenGLTrackball::calculateZ(GLfloat x, GLfloat y)
119 {
120  if(x*x+y*y <= 0.5f)
121  return sqrtf(1.f-(x*x+y*y));
122  else
123  return 0.5f/sqrtf(x*x+y*y);
124 }

151 void OpenGLTrackball::MouseMove(GLfloat x, GLfloat y)
152 {
153  if(dragging)
154  {
155  if(transMode)
156  {
157  glm::vec3 right = glm::normalize(glm::cross(GetLookingDirection(), GetUpDirection()));
158  center = translation_start + GetUpDirection() * (y-y_start) * -0.5f + right * (x-x_start) * -0.5f;
159  }
160  else //rotate
161  {
162  GLfloat z = calculateZ(x, y);
163  glm::quat rotation_new = glm::rotation(glm::normalize(glm::vec3(-x_start, z_start, y_start)), glm::normalize(glm::vec3(-x, z, y)));
164  rotation = rotation_new * rotation_start;
165  }
166  }
167 }

169 void OpenGLTrackball::MouseScroll(GLfloat s)
170 {
171  Scalar factor = pow(radius/5.0, 2.0);
172  factor = factor > 1.0 ? 1.0 : factor;

174  radius += s * factor;

198 void OpenGLTrackball::DrawSelection(const std::vector<Renderable>& r, GLuint destinationFBO)
199 {
200  if(r.size() == 0) //No selection
201  return;
202 
203  OpenGLContent* content = ((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getContent();
204 
205  //1. Draw flat shape to color and stencil buffer
206  OpenGLState::BindFramebuffer(getRenderFBO());
207  SetRenderBuffers(0, false, false);
208  OpenGLState::EnableStencilTest();
209  glStencilFunc(GL_ALWAYS, 1, 0xFF);
210  glStencilOp(GL_KEEP, GL_KEEP, GL_REPLACE);
211  glStencilMask(0xFF);
212  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT | GL_STENCIL_BUFFER_BIT);
213  content->SetDrawingMode(DrawingMode::FLAT);
214  for(size_t i=0; i<r.size(); ++i)
215  if(r[i].type == RenderableType::SOLID)
216  content->DrawObject(r[i].objectId, -1, r[i].model);
217  OpenGLState::DisableStencilTest();
218 
219  //2. Grow
220  outlineShader[0]->Use();
221  outlineShader[0]->SetUniform("color", glm::vec4(1.f,0.55f,0.1f,1.f)); //glm::vec4(0.4f,0.9f,1.f,1.f));
222  outlineShader[0]->SetUniform("texStencil", TEX_POSTPROCESS1);
223  OpenGLState::BindTexture(TEX_POSTPROCESS1, GL_TEXTURE_2D, getColorTexture(0));
224  SetRenderBuffers(1, false, false);
225  outlineShader[0]->SetUniform("uvOffset", glm::vec2(1.f/(GLfloat)viewportWidth, 1.f/(GLfloat)viewportHeight)*1.5f);
226  content->DrawSAQ();
227 
228  OpenGLState::BindTexture(TEX_POSTPROCESS1, GL_TEXTURE_2D, getColorTexture(1));
229  SetRenderBuffers(0, false, false);
230  content->DrawSAQ();
231 
232  //3. Cut outline
233  SetRenderBuffers(1, false, true);
234  OpenGLState::EnableStencilTest();
235  glStencilMask(0x00);
236  glStencilFunc(GL_EQUAL, 0, 0xFF);
237  content->DrawTexturedSAQ(getColorTexture(0));
238  OpenGLState::DisableStencilTest();
239 
240  //4. Gaussian blur 3x3
241  OpenGLState::BindTexture(TEX_POSTPROCESS1, GL_TEXTURE_2D, getColorTexture(1));
242  SetRenderBuffers(0, false, false);
243  outlineShader[1]->Use();
244  outlineShader[1]->SetUniform("tex", TEX_POSTPROCESS1);
245  outlineShader[1]->SetUniform("invTexSize", glm::vec2(1.f/(GLfloat)viewportWidth, 1.f/(GLfloat)viewportHeight));
246  content->DrawSAQ();
247 
248  OpenGLState::BindTexture(TEX_POSTPROCESS1, GL_TEXTURE_2D, getColorTexture(0));
249  SetRenderBuffers(1, false, false);
250  content->DrawSAQ();
251  OpenGLState::UnbindTexture(TEX_POSTPROCESS1);
252  OpenGLState::UseProgram(0);
253 
254  //5. Overlay
255  OpenGLState::BindFramebuffer(destinationFBO); //No depth buffer, just one color buffer
256  OpenGLState::EnableBlend();
257  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
258  content->DrawTexturedSAQ(getColorTexture(1));
259  OpenGLState::DisableBlend();
260 }
