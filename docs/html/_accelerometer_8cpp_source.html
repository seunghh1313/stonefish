41 void Accelerometer::InternalUpdate(Scalar dt)
42 {
43  //calculate transformation from global to imu frame
44  Transform accTrans = getSensorFrame();
45 
46  //get acceleration
47  Vector3 la = accTrans.getBasis().inverse() * (attach->getLinearAcceleration() + attach->getAngularAcceleration().cross(accTrans.getOrigin() - attach->getCGTransform().getOrigin()));
48 
49  //record sample
50  Scalar values[3] = {la.x(), la.y(), la.z()};
51  Sample s(3, values);
52  AddSampleToHistory(s);
53 }
