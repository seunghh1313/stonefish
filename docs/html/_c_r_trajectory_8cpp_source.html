36 void CRTrajectory::Interpolate()
37 {
38  if(points.size() < 3)
39  PWLTrajectory::Interpolate();
40  else
41  {
42  //Find current path segment
43  auto it = std::find_if(points.begin(), points.end(),
44  [&](const auto& key){ return key.t >= playTime; });
45 
46  Transform T0, T1, T2, T3;
47  Scalar t0, t1, t2, t3;
48 
49  if(it <= points.begin()+1) //Beginning
50  {
51  T1 = points[0].T;
52  T2 = points[1].T;
53  T3 = points[2].T;
54  T0.setOrigin(T1.getOrigin()-(T2.getOrigin()-T1.getOrigin()));
55  T0.setRotation(T1.getRotation());
56  t1 = points[0].t;
57  t2 = points[1].t;
58  t3 = points[2].t;
59  t0 = t1-(t2-t1);
60  }
61  else if(it >= points.end()-1) //End
62  {
63  T0 = points[points.size()-3].T;
64  T1 = points[points.size()-2].T;
65  T2 = points.back().T;
66  T3.setOrigin(T2.getOrigin()+(T2.getOrigin()-T1.getOrigin()));
67  T3.setRotation(T2.getRotation());
68  t0 = points[points.size()-3].t;
69  t1 = points[points.size()-2].t;
70  t2 = points.back().t;
71  t3 = t2 + (t2-t1);
72  }
73  else
74  {
75  T0 = (it-2)->T;
76  T1 = (it-1)->T;
77  T2 = it->T;
78  T3 = (it+1)->T;
79  t0 = (it-2)->t;
80  t1 = (it-1)->t;
81  t2 = it->t;
82  t3 = (it+1)->t;
83  }
84 
85  //Linear quantities
86  if(btFuzzyZero((T2.getOrigin()-T1.getOrigin()).safeNorm())) //Coinciding points
87  {
88  interpTrans.setOrigin(T2.getOrigin());
89  interpVel = V0();
90  }
91  else
92  {
93  interpTrans.setOrigin(catmullRom(T0.getOrigin(), T1.getOrigin(), T2.getOrigin(), T3.getOrigin(),
94  t0, t1, t2, t3, playTime));
95  interpVel = catmullRomDerivative(T0.getOrigin(), T1.getOrigin(), T2.getOrigin(), T3.getOrigin(),
96  t0, t1, t2, t3, playTime);
97  }
98  //Angular quantities
99  Vector3 dummy;
100  interpTrans.setRotation(slerp(T1.getRotation(), T2.getRotation(), (playTime-t1)/(t2-t1)));
101  btTransformUtil::calculateVelocity(T1, T2, t2-t1, dummy, interpAngVel);
102  }
103 }

105 void CRTrajectory::BuildGraphicalPath()
106 {
107  if(points.size() < 3)
108  PWLTrajectory::BuildGraphicalPath();
109  else
110  {
111  path.points.clear();
112  for(size_t i=0; i<points.size()-1; ++i)
113  {
114  Vector3 P1 = points[i].T.getOrigin();
115  Vector3 P2 = points[i+1].T.getOrigin();
116  Scalar t1 = points[i].t;
117  Scalar t2 = points[i+1].t;
118  Scalar dist = (P2-P1).safeNorm();
119  if(btFuzzyZero(dist))
120  continue;
121 
122  Vector3 P0, P3;
123  Scalar t0, t3;
124 
125  if(i==0) //Beginning
126  {
127  P3 = points[2].T.getOrigin();
128  P0 = P1-(P2-P1);
129  t3 = points[2].t;
130  t0 = t1-(t2-t1);
131  }
132  else if(i==points.size()-2) //End
133  {
134  P0 = points[i-1].T.getOrigin();
135  P3 = P2+(P2-P1);
136  t0 = points[i-1].t;
137  t3 = t2+(t2-t1);
138  }
139  else //Middle
140  {
141  P0 = points[i-1].T.getOrigin();
142  P3 = points[i+2].T.getOrigin();
143  t0 = points[i-1].t;
144  t3 = points[i+2].t;
145  }
146 
147  Scalar dt = (t2-t1)/Scalar(100.0);
148  for(Scalar t=t1; t<t2; t+=dt)
149  path.points.push_back(glVectorFromVector(catmullRom(P0, P1, P2, P3, t0, t1, t2, t3, t)));
150  }
151  path.points.push_back(glVectorFromVector(points.back().T.getOrigin()));
152  }
153 }

155 Vector3 CRTrajectory::catmullRom(Vector3 P0, Vector3 P1, Vector3 P2, Vector3 P3,
156  Scalar t0, Scalar t1, Scalar t2, Scalar t3, Scalar t)
157 {
158  Vector3 A1 = (t1-t)/(t1-t0)*P0 + (t-t0)/(t1-t0)*P1;
159  Vector3 A2 = (t2-t)/(t2-t1)*P1 + (t-t1)/(t2-t1)*P2;
160  Vector3 A3 = (t3-t)/(t3-t2)*P2 + (t-t2)/(t3-t2)*P3;
161  Vector3 B1 = (t2-t)/(t2-t0)*A1 + (t-t0)/(t2-t0)*A2;
162  Vector3 B2 = (t3-t)/(t3-t1)*A2 + (t-t1)/(t3-t1)*A3;
163  Vector3 C = (t2-t)/(t2-t1)*B1 + (t-t1)/(t2-t1)*B2;
164  return C;
165 }

167 Vector3 CRTrajectory::catmullRomDerivative(Vector3 P0, Vector3 P1, Vector3 P2, Vector3 P3,
168  Scalar t0, Scalar t1, Scalar t2, Scalar t3, Scalar t)
169 {
170  Vector3 A1 = (t1-t)/(t1-t0)*P0 + (t-t0)/(t1-t0)*P1;
171  Vector3 A2 = (t2-t)/(t2-t1)*P1 + (t-t1)/(t2-t1)*P2;
172  Vector3 A3 = (t3-t)/(t3-t2)*P2 + (t-t2)/(t3-t2)*P3;
173  Vector3 B1 = (t2-t)/(t2-t0)*A1 + (t-t0)/(t2-t0)*A2;
174  Vector3 B2 = (t3-t)/(t3-t1)*A2 + (t-t1)/(t3-t1)*A3;
175  Vector3 dA1 = Scalar(1)/(t1-t0)*(P1-P0);
176  Vector3 dA2 = Scalar(1)/(t2-t1)*(P2-P1);
177  Vector3 dA3 = Scalar(1)/(t3-t2)*(P3-P2);
178  Vector3 dB1 = Scalar(1)/(t2-t0)*(A2-A1) + (t2-t)/(t2-t0)*dA1 + (t-t0)/(t2-t0)*dA2;
179  Vector3 dB2 = Scalar(1)/(t3-t1)*(A3-A2) + (t3-t)/(t3-t1)*dA2 + (t-t1)/(t3-t1)*dA3;
180  Vector3 dC = Scalar(1)/(t2-t1)*(B2-B1) + (t2-t)/(t2-t1)*dB1 + (t-t1)/(t2-t1)*dB2;
181  return dC;
182 }
