33 CylindricalJoint::CylindricalJoint(std::string uniqueName, SolidEntity* solidA, SolidEntity* solidB, const Vector3& pivot, const Vector3& axis, bool collideLinked) : Joint(uniqueName, collideLinked)
34 {
35  btRigidBody* bodyA = solidA->rigidBody;
36  btRigidBody* bodyB = solidB->rigidBody;
37 
38  Vector3 sliderAxis = axis.normalized();
39  Vector3 v2;
40  if(fabs(sliderAxis.z()) > 0.8) v2 = Vector3(1,0,0); else v2 = Vector3(0,0,1);
41  Vector3 v3 = (sliderAxis.cross(v2)).normalized();
42  v2 = (v3.cross(sliderAxis)).normalized();
43  Matrix3 sliderBasis(sliderAxis.x(), v2.x(), v3.x(),
44  sliderAxis.y(), v2.y(), v3.y(),
45  sliderAxis.z(), v2.z(), v3.z());
46  Transform sliderFrame(sliderBasis, pivot);
47 
48  Transform frameInA = bodyA->getCenterOfMassTransform().inverse() * sliderFrame;
49  Transform frameInB = bodyB->getCenterOfMassTransform().inverse() * sliderFrame;
50  axisInA = frameInA.getBasis().getColumn(0).normalized();
51  pivotInA = frameInA.getOrigin();
52 
53  btSliderConstraint* slider = new btSliderConstraint(*bodyA, *bodyB, frameInA, frameInB, true);
54  slider->setLowerLinLimit(1.);
55  slider->setUpperLinLimit(-1.);
56  slider->setLowerAngLimit(1.);
57  slider->setUpperAngLimit(-1.);
58  setConstraint(slider);
59 
60  linSigDamping = Scalar(0.);
61  linVelDamping = Scalar(0.);
62  angSigDamping = Scalar(0.);
63  angVelDamping = Scalar(0.);
64 
65  displacementIC = Scalar(0.);
66  angleIC = Scalar(0.);
67 }

97 void CylindricalJoint::ApplyForce(Scalar F)
98 {
99  btRigidBody& bodyA = getConstraint()->getRigidBodyA();
100  btRigidBody& bodyB = getConstraint()->getRigidBodyB();
101  Vector3 axis = (bodyA.getCenterOfMassTransform().getBasis() * axisInA).normalized();
102  Vector3 force = axis * F;
103  bodyA.applyCentralForce(force);
104  bodyB.applyCentralForce(-force);

117 void CylindricalJoint::ApplyDamping()
118 {
119  if(linSigDamping > Scalar(0.) || linVelDamping > Scalar(0.) || angSigDamping > Scalar(0.) || angVelDamping > Scalar(0.))
120  {
121  btRigidBody& bodyA = getConstraint()->getRigidBodyA();
122  btRigidBody& bodyB = getConstraint()->getRigidBodyB();
123  Vector3 axis = (bodyA.getCenterOfMassTransform().getBasis() * axisInA).normalized();
124  Vector3 relativeV = bodyA.getLinearVelocity() - bodyB.getLinearVelocity();
125  Scalar v = relativeV.dot(axis);
126  Vector3 relativeAV = bodyA.getAngularVelocity() - bodyB.getAngularVelocity();
127  Scalar av = relativeAV.dot(axis);
128 
129  if(v != Scalar(0.))
130  {
131  Scalar F = linSigDamping * v/fabs(v) + linVelDamping * v;
132  Vector3 force = axis * -F;
133 
134  bodyA.applyCentralForce(force);
135  bodyB.applyCentralForce(-force);
136  }
137 
138  if(av != Scalar(0.))
139  {
140  Scalar T = angSigDamping * av/fabs(av) + angVelDamping * av;
141  Vector3 torque = axis * -T;
142 
143  bodyA.applyTorque(torque);
144  bodyB.applyTorque(-torque);
145  }
146  }

149 std::vector<Renderable> CylindricalJoint::Render()
150 {
151  std::vector<Renderable> items(0);
152  Renderable item;
153  item.model = glm::mat4(1.f);
154  item.type = RenderableType::JOINT_LINES;
155 
156  btTypedConstraint* cyli = getConstraint();
157  Vector3 A = cyli->getRigidBodyA().getCenterOfMassPosition();
158  Vector3 B = cyli->getRigidBodyB().getCenterOfMassPosition();
159  Vector3 pivot = cyli->getRigidBodyA().getCenterOfMassTransform()(pivotInA);
160  Vector3 axis = (cyli->getRigidBodyA().getCenterOfMassTransform().getBasis() * axisInA).normalized();
161 
162  //calculate axis ends
163  Scalar e1 = (A-pivot).dot(axis);
164  Scalar e2 = (B-pivot).dot(axis);
165  Vector3 C1 = pivot + e1 * axis;
166  Vector3 C2 = pivot + e2 * axis;
167 
168  item.points.push_back(glm::vec3(A.getX(), A.getY(), A.getZ()));
169  item.points.push_back(glm::vec3(C1.getX(), C1.getY(), C1.getZ()));
170  item.points.push_back(glm::vec3(B.getX(), B.getY(), B.getZ()));
171  item.points.push_back(glm::vec3(C2.getX(), C2.getY(), C2.getZ()));
172 
173  item.points.push_back(glm::vec3(C1.getX(), C1.getY(), C1.getZ()));
174  item.points.push_back(glm::vec3(C2.getX(), C2.getY(), C2.getZ()));
175 
176  items.push_back(item);
177  return items;
178 }
