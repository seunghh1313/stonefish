82 void Atmosphere::SetupSunPosition(Scalar longitudeDeg, Scalar latitudeDeg, std::tm& utc)
83 {
84  if(glAtmosphere == NULL) return;
85 
86  Scalar latitude = latitudeDeg/Scalar(180) * M_PI;
87  Scalar longitude = longitudeDeg/Scalar(180) * M_PI;
88  Scalar meridian(0); //GMT
89  Scalar decimalHours = (Scalar)utc.tm_hour + (Scalar)utc.tm_min/Scalar(60) + (Scalar)utc.tm_sec/Scalar(60);
90  std::tm utc0 = utc;
91  utc0.tm_mday = 1; //1
92  utc0.tm_mon = 0; //January
93  int J = JulianDay(utc) - JulianDay(utc0);
94  Scalar solarTime = decimalHours + Scalar(0.17)*btSin(Scalar(4)*M_PI*(J - 80)/Scalar(373)) - Scalar(0.129)*btSin(Scalar(2)*M_PI*(J - 8)/Scalar(355)) + Scalar(12)*(meridian - longitude)/M_PI;
95  Scalar declination = Scalar(0.4093)*btSin(Scalar(2)*M_PI*(J - 81)/Scalar(368));
96  Scalar elevation = btAsin(btSin(latitude)*btSin(declination) - btCos(latitude)*btCos(declination)*btCos(M_PI*solarTime/Scalar(12)));
97  Scalar azimuth = btAtan2(btSin(M_PI*solarTime/Scalar(12)), btCos(M_PI*solarTime/Scalar(12))*btSin(latitude) - btTan(declination)*btCos(latitude));
98 
99  glAtmosphere->SetSunPosition((float)(azimuth/M_PI*Scalar(180)), (float)(elevation/M_PI*Scalar(180)));
100 }

145 void Atmosphere::ApplyFluidForces(btDynamicsWorld* world, btCollisionObject* co, bool recompute)
146 {
147  Entity* ent;
148  btRigidBody* rb = btRigidBody::upcast(co);
149  btMultiBodyLinkCollider* mbl = btMultiBodyLinkCollider::upcast(co);
150 
151  if(rb != 0)
152  {
153  if(rb->isStaticOrKinematicObject())
154  return;
155  else
156  ent = (Entity*)rb->getUserPointer();
157  }
158  else if(mbl != 0)
159  {
160  if(mbl->isStaticOrKinematicObject())
161  return;
162  else
163  ent = (Entity*)mbl->getUserPointer();
164  }
165  else
166  return;
167 
168  if(ent->getType() == EntityType::SOLID)
169  {
170  if(recompute)
171  {
172  ((SolidEntity*)ent)->ComputeAerodynamicForces(this);
173  }
174 
175  ((SolidEntity*)ent)->ApplyAerodynamicForces();
176  }
177 }

179 int Atmosphere::JulianDay(std::tm& tm)
180 {
181  int m = tm.tm_mon + 1;
182  int y = tm.tm_year + 1900;
183  int d = tm.tm_mday;
184 
185  Scalar X = (m + 9) / 12.0;
186  int A = 4716 + y + (int)trunc(X);
187  Scalar Y = 275 * m / 9.0;
188  Scalar V = 7 * A / 4.0;
189  Scalar B = 1729279.5 + 367 * y + trunc(Y) - trunc(V) + d;
190  Scalar Q = (A + 83) / 100.0;
191  Scalar W = 3 * (trunc(Q) + 1) / 4.0;
192  return B + 38 - (int)trunc(W);
193 }
