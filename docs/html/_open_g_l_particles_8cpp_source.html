33 OpenGLParticles::OpenGLParticles(GLuint numOfParticles)
34 {
35  nParticles = numOfParticles;
36  glGenBuffers(1, &particlePosSSBO);
37  glBindBuffer(GL_SHADER_STORAGE_BUFFER, particlePosSSBO);
38  glBufferData(GL_SHADER_STORAGE_BUFFER, sizeof(glm::vec4) * nParticles, NULL, GL_DYNAMIC_DRAW);
39  glBindBuffer(GL_SHADER_STORAGE_BUFFER, 0);
40  glGenBuffers(1, &particleVelSSBO);
41  glBindBuffer(GL_SHADER_STORAGE_BUFFER, particleVelSSBO);
42  glBufferData(GL_SHADER_STORAGE_BUFFER, sizeof(glm::vec4) * nParticles, NULL, GL_DYNAMIC_DRAW);
43  glBindBuffer(GL_SHADER_STORAGE_BUFFER, 0);
44  glGenBuffers(1, &particleEAB);
45  glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, particleEAB);
46  glBufferData(GL_ELEMENT_ARRAY_BUFFER, sizeof(GLuint) * 6 * nParticles, NULL, GL_STATIC_DRAW);
47  GLuint* indices = (GLuint*)glMapBufferRange(GL_ELEMENT_ARRAY_BUFFER, 0, sizeof(GLuint) * 6 * nParticles, GL_MAP_WRITE_BIT | GL_MAP_INVALIDATE_BUFFER_BIT);
48  for(GLuint i=0; i<nParticles; ++i)
49  {
50  GLuint index = GLuint(i<<2);
51  *(indices++) = index;
52  *(indices++) = index+2;
53  *(indices++) = index+1;
54  *(indices++) = index;
55  *(indices++) = index+3;
56  *(indices++) = index+2;
57  }
58  glUnmapBuffer(GL_ELEMENT_ARRAY_BUFFER);
59  glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, 0);
60  glGenVertexArrays(1, &particleVAO);
61  OpenGLState::BindVertexArray(particleVAO);
62  glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, particleEAB);
63  OpenGLState::BindVertexArray(0);
64 }
