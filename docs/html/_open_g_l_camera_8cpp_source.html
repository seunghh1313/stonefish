52 OpenGLCamera::OpenGLCamera(GLint x, GLint y, GLint width, GLint height, glm::vec2 range) : OpenGLView(x, y, width, height)
53 {
54  fovx = 0.785f;
55  near = range.x;
56  far = range.y;
57  exposureComp = 0.f;
58  autoExposure = true;
59  toneMapping = true;
60  antiAliasing = false;
61  aoFactor = 0;
62 
63  if(((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getRenderSettings().ao != RenderQuality::DISABLED)
64  aoFactor = 1;
65  if(((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getRenderSettings().aa != RenderQuality::DISABLED)
66  antiAliasing = true;
67 
68  //----Geometry rendering----
69  renderColorTex[0] = OpenGLContent::GenerateTexture(GL_TEXTURE_2D, glm::uvec3(viewportWidth, viewportHeight, 0),
70  GL_RGBA32F, GL_RGBA, GL_FLOAT, NULL, FilteringMode::BILINEAR, false);
71  renderColorTex[1] = OpenGLContent::GenerateTexture(GL_TEXTURE_2D, glm::uvec3(viewportWidth, viewportHeight, 0),
72  GL_RGBA32F, GL_RGBA, GL_FLOAT, NULL, FilteringMode::BILINEAR, false);
73  renderViewNormalTex = OpenGLContent::GenerateTexture(GL_TEXTURE_2D, glm::uvec3(viewportWidth, viewportHeight, 0),
74  GL_RGBA8, GL_RGBA, GL_UNSIGNED_BYTE, NULL, FilteringMode::BILINEAR, false);
75  renderDepthStencilTex = OpenGLContent::GenerateTexture(GL_TEXTURE_2D, glm::uvec3(viewportWidth, viewportHeight, 0),
76  GL_DEPTH24_STENCIL8, GL_DEPTH_STENCIL, GL_UNSIGNED_INT_24_8, NULL, FilteringMode::NEAREST, false);
77  std::vector<FBOTexture> fboTextures;
78  fboTextures.push_back(FBOTexture(GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, renderColorTex[0]));
79  fboTextures.push_back(FBOTexture(GL_COLOR_ATTACHMENT1, GL_TEXTURE_2D, renderViewNormalTex));
80  fboTextures.push_back(FBOTexture(GL_COLOR_ATTACHMENT2, GL_TEXTURE_2D, renderColorTex[1]));
81  fboTextures.push_back(FBOTexture(GL_DEPTH_STENCIL_ATTACHMENT, GL_TEXTURE_2D, renderDepthStencilTex));
82  renderFBO = OpenGLContent::GenerateFramebuffer(fboTextures);
83  OpenGLState::BindFramebuffer(renderFBO);
84  SetRenderBuffers(0, true, false);
85  OpenGLState::BindFramebuffer(0);
86 
87  //----Postprocessing----
88  postprocessTex[0] = OpenGLContent::GenerateTexture(GL_TEXTURE_2D, glm::uvec3(viewportWidth, viewportHeight, 0),
89  GL_RGBA32F, GL_RGBA, GL_FLOAT, NULL, FilteringMode::NEAREST, false);
90  postprocessTex[1] = OpenGLContent::GenerateTexture(GL_TEXTURE_2D, glm::uvec3(viewportWidth, viewportHeight, 0),
91  GL_RGBA8, GL_RGBA, GL_UNSIGNED_BYTE, NULL, FilteringMode::BILINEAR, false);
92  postprocessStencilTex = OpenGLContent::GenerateTexture(GL_TEXTURE_2D, glm::uvec3(viewportWidth, viewportHeight, 0),
93  GL_DEPTH24_STENCIL8, GL_DEPTH_STENCIL, GL_UNSIGNED_INT_24_8, NULL, FilteringMode::NEAREST, false);
94  fboTextures.clear();
95  fboTextures.push_back(FBOTexture(GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, postprocessTex[0]));
96  fboTextures.push_back(FBOTexture(GL_COLOR_ATTACHMENT1, GL_TEXTURE_2D, postprocessTex[1]));
97  fboTextures.push_back(FBOTexture(GL_DEPTH_STENCIL_ATTACHMENT, GL_TEXTURE_2D, postprocessStencilTex));
98  postprocessFBO = OpenGLContent::GenerateFramebuffer(fboTextures);
99 
100  quaterPostprocessTex[0] = OpenGLContent::GenerateTexture(GL_TEXTURE_2D, glm::uvec3(viewportWidth/2, viewportHeight/2, 0),
101  GL_RGBA32F, GL_RGBA, GL_FLOAT, NULL, FilteringMode::BILINEAR, false);
102  quaterPostprocessTex[1] = OpenGLContent::GenerateTexture(GL_TEXTURE_2D, glm::uvec3(viewportWidth/2, viewportHeight/2, 0),
103  GL_RGBA32F, GL_RGBA, GL_FLOAT, NULL, FilteringMode::BILINEAR, false);
104  fboTextures.clear();
105  fboTextures.push_back(FBOTexture(GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, quaterPostprocessTex[0]));
106  fboTextures.push_back(FBOTexture(GL_COLOR_ATTACHMENT1, GL_TEXTURE_2D, quaterPostprocessTex[1]));
107  quaterPostprocessFBO = OpenGLContent::GenerateFramebuffer(fboTextures);
108 
109  //Linear depth
110  linearDepthTex[0] = OpenGLContent::GenerateTexture(GL_TEXTURE_2D, glm::uvec3(viewportWidth, viewportHeight, 0),
111  GL_R32F, GL_RED, GL_FLOAT, NULL, FilteringMode::NEAREST, false);
112  linearDepthTex[1] = OpenGLContent::GenerateTexture(GL_TEXTURE_2D, glm::uvec3(viewportWidth, viewportHeight, 0),
113  GL_R32F, GL_RED, GL_FLOAT, NULL, FilteringMode::NEAREST, false);
114  fboTextures.clear();
115  fboTextures.push_back(FBOTexture(GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, linearDepthTex[0]));
116  fboTextures.push_back(FBOTexture(GL_COLOR_ATTACHMENT1, GL_TEXTURE_2D, linearDepthTex[1]));
117  linearDepthFBO = OpenGLContent::GenerateFramebuffer(fboTextures);
118 
119  //---- Tonemapping ----
120  histogramBins = 256;
121  histogramRange = glm::vec2(-1.f,11.f);
122  GLuint histogram[histogramBins];
123  memset(histogram, 0, histogramBins * sizeof(GLuint));
124  glGenBuffers(1, &histogramSSBO);
125  glBindBuffer(GL_SHADER_STORAGE_BUFFER, histogramSSBO);
126  glBufferData(GL_SHADER_STORAGE_BUFFER, histogramBins * sizeof(GLuint), histogram, GL_STATIC_DRAW);
127  glBindBuffer(GL_SHADER_STORAGE_BUFFER, 0);
128 
129  GLfloat zero = 0.f;
130  exposureTex = OpenGLContent::GenerateTexture(GL_TEXTURE_2D, glm::uvec3(1,1,0),
131  GL_R32F, GL_RED, GL_FLOAT, &zero, FilteringMode::NEAREST, false);
132 
133  //----HBAO----
134  if(aoFactor > 0)
135  {
136  //Deinterleaved results
137  GLint swizzle[4] = {GL_RED,GL_GREEN,GL_ZERO,GL_ZERO};
138 
139  glGenTextures(1, &aoResultTex);
140  OpenGLState::BindTexture(TEX_BASE, GL_TEXTURE_2D, aoResultTex);
141  glTexStorage2D(GL_TEXTURE_2D, 1, GL_RG16F, viewportWidth, viewportHeight);
142  glTexParameteriv(GL_TEXTURE_2D, GL_TEXTURE_SWIZZLE_RGBA, swizzle);
143  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
144  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
145 
146  glGenTextures(1, &aoBlurTex);
147  OpenGLState::BindTexture(TEX_BASE, GL_TEXTURE_2D, aoBlurTex);
148  glTexStorage2D(GL_TEXTURE_2D, 1, GL_RG16F, viewportWidth, viewportHeight);
149  glTexParameteriv(GL_TEXTURE_2D, GL_TEXTURE_SWIZZLE_RGBA, swizzle);
150  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
151  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
152 
153  fboTextures.clear();
154  fboTextures.push_back(FBOTexture(GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, aoResultTex));
155  fboTextures.push_back(FBOTexture(GL_COLOR_ATTACHMENT1, GL_TEXTURE_2D, aoBlurTex));
156  aoFinalFBO = OpenGLContent::GenerateFramebuffer(fboTextures);
157 
158  //Interleaved rendering
159  int quarterWidth = ((viewportWidth+3)/4);
160  int quarterHeight = ((viewportHeight+3)/4);
161 
162  glGenTextures(1, &aoDepthArrayTex);
163  OpenGLState::BindTexture(TEX_BASE, GL_TEXTURE_2D_ARRAY, aoDepthArrayTex);
164  glTexStorage3D(GL_TEXTURE_2D_ARRAY, 1, GL_R32F, quarterWidth, quarterHeight, HBAO_RANDOM_ELEMENTS);
165  glTexParameteri(GL_TEXTURE_2D_ARRAY, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
166  glTexParameteri(GL_TEXTURE_2D_ARRAY, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
167  glTexParameteri(GL_TEXTURE_2D_ARRAY, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
168  glTexParameteri(GL_TEXTURE_2D_ARRAY, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
169 
170  glGenTextures(1, &aoResultArrayTex);
171  OpenGLState::BindTexture(TEX_BASE, GL_TEXTURE_2D_ARRAY, aoResultArrayTex);
172  glTexStorage3D(GL_TEXTURE_2D_ARRAY, 1, GL_RG16F, quarterWidth, quarterHeight, HBAO_RANDOM_ELEMENTS);
173  glTexParameteri(GL_TEXTURE_2D_ARRAY, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
174  glTexParameteri(GL_TEXTURE_2D_ARRAY, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
175  glTexParameteri(GL_TEXTURE_2D_ARRAY, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
176  glTexParameteri(GL_TEXTURE_2D_ARRAY, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
177 
178  OpenGLState::UnbindTexture(TEX_BASE);
179 
180  GLenum drawbuffers[NUM_MRT];
181  for(int layer = 0; layer < NUM_MRT; ++layer)
182  drawbuffers[layer] = GL_COLOR_ATTACHMENT0 + layer;
183 
184  glGenFramebuffers(1, &aoDeinterleaveFBO);
185  OpenGLState::BindFramebuffer(aoDeinterleaveFBO);
186  glDrawBuffers(NUM_MRT, drawbuffers);
187 
188  glGenFramebuffers(1, &aoCalcFBO);
189  OpenGLState::BindFramebuffer(aoCalcFBO);
190  glFramebufferTexture(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, aoResultArrayTex, 0);
191  OpenGLState::BindFramebuffer(0);
192 
193  glGenBuffers(1, &aoDataUBO);
194  glNamedBufferStorageEXT(aoDataUBO, sizeof(AOData), NULL, GL_DYNAMIC_STORAGE_BIT);
195 
196  //Generate random data
197  std::mt19937 rng((unsigned int)GetTimeInMicroseconds());
198  GLfloat numDir = 8; //Keep in sync with GLSL shader!!!
199  GLfloat rngMax = (GLfloat)rng.max() + 1.f;
200 
201  for(unsigned int i=0; i<HBAO_RANDOM_ELEMENTS; ++i)
202  {
203  GLfloat rand1 = (GLfloat)rng()/rngMax; //random in [0,1)
204  GLfloat rand2 = (GLfloat)rng()/rngMax; //random in [0,1)
205 
206  //Use random rotation angles in [0,2PI/NUM_DIRECTIONS)
207  GLfloat angle = 2.f * M_PI * rand1 / numDir;
208  aoData.jitters[i].x = cosf(angle);
209  aoData.jitters[i].y = sinf(angle);
210  aoData.jitters[i].z = rand2;
211  aoData.jitters[i].w = 0;
212 
213  aoData.float2Offsets[i] = glm::vec4((GLfloat)(i%4) + 0.5f, (GLfloat)(i/4) + 0.5f, 0.0, 0.0);
214  }
215  }
216 }

277 glm::vec3 OpenGLCamera::Ray(GLint x, GLint y)
278 {
279  //Translate point to view space
280  x -= originX;
281  y -= originY;
282 
283  //Check if point in view
284  if((x < 0) || (x >= viewportWidth) || (y < 0) || (y >= viewportHeight))
285  return glm::vec3(0);
286 
287  glm::vec2 pixPos = glm::vec2((GLfloat)x/(GLfloat)viewportWidth, (GLfloat)(viewportHeight-y)/(GLfloat)viewportHeight);
288  pixPos = (pixPos - glm::vec2(0.5f)) * 2.f;
289  glm::vec4 deviceRay = glm::vec4(pixPos, 0.f, 1.f);
290  glm::vec3 eyeRay = glm::vec3(glm::normalize((glm::inverse(GetProjectionMatrix()) * deviceRay)));
291  glm::vec3 worldRay = glm::normalize(glm::inverse(glm::mat3(GetViewMatrix())) * eyeRay);
292 
293  /*
294  //Calculate ray from point
295  glm::vec3 _eye = GetEyePosition();
296  glm::vec3 _lookingDir = GetLookingDirection();
297  glm::vec3 _up = GetUpDirection();
298 
299  glm::vec3 rayFrom = _eye;
300  glm::vec3 rayForward = _lookingDir * far;
301  glm::vec3 horizontal = glm::normalize(glm::cross(rayForward, _up));
302  glm::vec3 vertical = glm::normalize(glm::cross(horizontal, rayForward));
303 
304  GLfloat tanFov = tanf(0.5f*fovx);
305  horizontal *= 2.f * far * tanFov;
306  vertical *= 2.f * far * tanFov;
307  GLfloat aspect = (GLfloat)viewportWidth/(GLfloat)viewportHeight;
308  vertical /= aspect;
309 
310  glm::vec3 rayToCenter = rayFrom + rayForward;
311  glm::vec3 dH = horizontal * 1.f/(GLfloat)viewportWidth;
312  glm::vec3 dV = vertical * 1.f/(GLfloat)viewportHeight;
313 
314  glm::vec3 rayTo = rayToCenter - 0.5f * horizontal + 0.5f * vertical;
315  rayTo += Scalar(x) * dH;
316  rayTo -= Scalar(y) * dV;
317  */
318  return worldRay;
319 }

461 void OpenGLCamera::DrawAO(GLfloat intensity)
462 {
463  if(hasAO())
464  {
465  //Prepare data and set parameters
466  int quarterWidth = ((viewportWidth+3)/4);
467  int quarterHeight = ((viewportHeight+3)/4);
468  glm::mat4 proj = GetProjectionMatrix();
469  glm::vec4 projInfo(
470  2.0f/proj[0].x,
471  2.0f/proj[1].y,
472  -(1.f-proj[0].z)/proj[0].x,
473  -(1.f+proj[1].z)/proj[1].y
474  );
475 
476  glm::vec2 invFullRes(1.f/(GLfloat)viewportWidth, 1.f/(GLfloat)viewportHeight);
477  glm::vec2 invQuarterRes(1.f/(GLfloat)quarterWidth, 1.f/(GLfloat)quarterHeight);
478  GLfloat projScale = (GLfloat)viewportHeight/(tanf(GetFOVY() * 0.5f) * 2.0f);
479  GLfloat R = 0.5f;
480 
481  aoData.projInfo = projInfo;
482  aoData.R2 = R * R;
483  aoData.NegInvR2 = -1.f/aoData.R2;
484  aoData.RadiusToScreen = R * 0.5f * projScale;
485  aoData.PowExponent = intensity < 0.f ? 0.f : intensity; //intensity
486  aoData.NDotVBias = 0.01f; //<0,1>
487  aoData.AOMultiplier = 1.f/(1.f-aoData.NDotVBias);
488  aoData.InvQuarterResolution = invQuarterRes;
489  aoData.InvFullResolution = invFullRes;
490 
491  GLfloat blurSharpness = 40.0f;
492 
493  GenerateLinearDepth(true);
494  ((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getContent()->BindBaseVertexArray(); //Previous function unbinds vertex array
495 
496  //Deinterleave
497  OpenGLState::BindFramebuffer(aoDeinterleaveFBO);
498  OpenGLState::Viewport(0, 0, quarterWidth, quarterHeight);
499 
500  aoDeinterleaveShader->Use();
501  aoDeinterleaveShader->SetUniform("texLinearDepth", TEX_POSTPROCESS1);
502 
503  OpenGLState::BindTexture(TEX_POSTPROCESS1, GL_TEXTURE_2D, linearDepthTex[0]);
504  for(int i=0; i<HBAO_RANDOM_ELEMENTS; i+=NUM_MRT)
505  {
506  aoDeinterleaveShader->SetUniform("info", glm::vec4(float(i % 4) + 0.5f, float(i / 4) + 0.5f, invFullRes.x, invFullRes.y));
507 
508  for(int layer = 0; layer < NUM_MRT; ++layer)
509  glFramebufferTextureLayer(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0 + layer, aoDepthArrayTex, 0, i+layer);
510 
511  glDrawArrays(GL_TRIANGLES, 0, 3);
512  }
513 
514  OpenGLState::UnbindTexture(TEX_POSTPROCESS1);
515 
516  //Calculate HBAO
517  OpenGLState::BindFramebuffer(aoCalcFBO);
518  OpenGLState::Viewport(0, 0, quarterWidth, quarterHeight);
519  aoCalcShader->Use();
520  aoCalcShader->SetUniform("texLinearDepth", TEX_POSTPROCESS1);
521  aoCalcShader->SetUniform("texViewNormal", TEX_POSTPROCESS2);
522 
523  OpenGLState::BindTexture(TEX_POSTPROCESS2, GL_TEXTURE_2D, renderViewNormalTex);
524  OpenGLState::BindTexture(TEX_POSTPROCESS1, GL_TEXTURE_2D_ARRAY, aoDepthArrayTex);
525  glBindBufferBase(GL_UNIFORM_BUFFER, 0, aoDataUBO);
526  glNamedBufferSubDataEXT(aoDataUBO, 0, sizeof(AOData), &aoData);
527  glDrawArrays(GL_TRIANGLES, 0, 3 * HBAO_RANDOM_ELEMENTS);
528  OpenGLState::UnbindTexture(TEX_POSTPROCESS1);
529  OpenGLState::UnbindTexture(TEX_POSTPROCESS2);
530 
531  //Reinterleave
532  OpenGLState::BindFramebuffer(aoFinalFBO);
533  glDrawBuffer(GL_COLOR_ATTACHMENT0);
534  OpenGLState::Viewport(0, 0, viewportWidth, viewportHeight);
535  aoReinterleaveShader->Use();
536  aoReinterleaveShader->SetUniform("texResultArray", TEX_POSTPROCESS1);
537 
538  OpenGLState::BindTexture(TEX_POSTPROCESS1, GL_TEXTURE_2D_ARRAY, aoResultArrayTex);
539  glDrawArrays(GL_TRIANGLES, 0, 3);
540  OpenGLState::UnbindTexture(TEX_POSTPROCESS1);
541 
542  //Blur
543  glDrawBuffer(GL_COLOR_ATTACHMENT1);
544  aoBlurShader[0]->Use();
545  aoBlurShader[0]->SetUniform("sharpness", blurSharpness);
546  aoBlurShader[0]->SetUniform("invResolutionDirection", glm::vec2(1.f/(GLfloat)viewportWidth, 0));
547  aoBlurShader[0]->SetUniform("texSource", TEX_POSTPROCESS1);
548 
549  OpenGLState::BindTexture(TEX_POSTPROCESS1, GL_TEXTURE_2D, aoResultTex);
550  glDrawArrays(GL_TRIANGLES, 0, 3);
551  OpenGLState::UnbindTexture(TEX_POSTPROCESS1);
552 
553  //Final output to main fbo
554  OpenGLState::BindFramebuffer(renderFBO);
555  glDrawBuffer(GL_COLOR_ATTACHMENT0);
556  glDepthMask(GL_FALSE);
557  OpenGLState::DisableDepthTest();
558  OpenGLState::EnableBlend();
559  //glBlendFunc(GL_DST_COLOR, GL_ONE_MINUS_DST_ALPHA);
560  glBlendFunc(GL_ZERO, GL_SRC_COLOR);
561  aoBlurShader[1]->Use();
562  aoBlurShader[1]->SetUniform("sharpness", blurSharpness);
563  aoBlurShader[1]->SetUniform("invResolutionDirection", glm::vec2(0, 1.f/(GLfloat)viewportHeight));
564  aoBlurShader[1]->SetUniform("texSource", TEX_POSTPROCESS1);
565 
566  OpenGLState::BindTexture(TEX_POSTPROCESS1, GL_TEXTURE_2D, aoBlurTex);
567  glDrawArrays(GL_TRIANGLES, 0, 3);
568  OpenGLState::UnbindTexture(TEX_POSTPROCESS1);
569 
570  OpenGLState::EnableDepthTest();
571  OpenGLState::DisableBlend();
572  glDepthMask(GL_TRUE);
573 
574  OpenGLState::BindVertexArray(0);
575  OpenGLState::UseProgram(0);
576  OpenGLState::BindFramebuffer(renderFBO);

580 void OpenGLCamera::DrawSSR()
581 {
582  if(ssrShader == nullptr)
583  return;
584 
585  //Compute SSR
586  OpenGLState::BindTexture(TEX_POSTPROCESS1, GL_TEXTURE_2D, renderColorTex[0]);
587  OpenGLState::BindTexture(TEX_POSTPROCESS2, GL_TEXTURE_2D, renderViewNormalTex);
588  OpenGLState::BindTexture(TEX_POSTPROCESS3, GL_TEXTURE_2D, getLinearDepthTexture(true));
589  OpenGLState::BindTexture(TEX_POSTPROCESS4, GL_TEXTURE_2D, getLinearDepthTexture(false));
590 
591  GLfloat sx = viewportWidth/2.f;
592  GLfloat sy = viewportHeight/2.f;
593 
594  glm::mat4 proj = GetProjectionMatrix();
595 
596  glm::mat4 projPix = glm::transpose(glm::mat4(sx, 0, 0, sx,
597  0, sy, 0, sy,
598  0, 0, 1.f, 0,
599  0, 0, 0, 1.f)) * proj;
600 
601  glm::vec4 projInfo(
602  2.0f/proj[0].x,
603  2.0f/proj[1].y,
604  -(1.f-proj[0].z)/proj[0].x,
605  -(1.f+proj[1].z)/proj[1].y
606  );
607 
608  OpenGLState::BindFramebuffer(getRenderFBO());
609  OpenGLState::DisableDepthTest();
610  SetRenderBuffers(1, false, true);
611  OpenGLState::Viewport(0, 0, viewportWidth, viewportHeight);
612  ssrShader->Use();
613  ssrShader->SetUniform("texColor", TEX_POSTPROCESS1);
614  ssrShader->SetUniform("texViewNormal", TEX_POSTPROCESS2);
615  ssrShader->SetUniform("texLinearDepth", TEX_POSTPROCESS3);
616  ssrShader->SetUniform("texLinearBackfaceDepth", TEX_POSTPROCESS4);
617  ssrShader->SetUniform("P", projPix);
618  ssrShader->SetUniform("invP", glm::inverse(proj));
619  ssrShader->SetUniform("projInfo", projInfo);
620  ssrShader->SetUniform("viewportSize", glm::vec2(viewportWidth, viewportHeight));
621  ssrShader->SetUniform("invViewportSize", glm::vec2(1.f/(GLfloat)viewportWidth, 1.f/(GLfloat)viewportHeight));
622  ssrShader->SetUniform("near", near);
623  ssrShader->SetUniform("far", far);
624  ((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getContent()->DrawSAQ();
625  OpenGLState::UseProgram(0);
626  OpenGLState::UnbindTexture(TEX_POSTPROCESS4);
627  OpenGLState::UnbindTexture(TEX_POSTPROCESS3);
628  OpenGLState::UnbindTexture(TEX_POSTPROCESS2);
629  OpenGLState::UnbindTexture(TEX_POSTPROCESS1);
630 
631  OpenGLState::BindTexture(TEX_POSTPROCESS1, GL_TEXTURE_2D, renderColorTex[1]);
632  SetRenderBuffers(0, false, false);
633  OpenGLState::EnableBlend();
634  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
635  ssrBlur->Use();
636  ssrBlur->SetUniform("tex", TEX_POSTPROCESS1);
637  ssrBlur->SetUniform("invTexSize", glm::vec2(1.f/(GLfloat)viewportWidth, 1.f/(GLfloat)viewportHeight));
638  ((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getContent()->DrawSAQ();
639  OpenGLState::UseProgram(0);
640  OpenGLState::UnbindTexture(TEX_POSTPROCESS1);
641  OpenGLState::DisableBlend();
642  OpenGLState::EnableDepthTest();
