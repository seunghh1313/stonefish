57 Transform Comm::getDeviceFrame()
58 {
59  if(attach != nullptr)
60  {
61  if(attach->getType() == EntityType::STATIC)
62  return ((StaticEntity*)attach)->getTransform() * o2c;
63  else if(attach->getType() == EntityType::SOLID || attach->getType() == EntityType::ANIMATED)
64  return ((MovingEntity*)attach)->getOTransform() * o2c;
65  }
66  return o2c;
67 }

109 void Comm::SendMessage(std::string data)
110 {
111  CommDataFrame* msg = new CommDataFrame();
112  msg->seq = txSeq++;
113  msg->source = id;
114  msg->destination = cId;
115  msg->timeStamp = SimulationApp::getApp()->getSimulationManager()->getSimulationTime();
116  msg->data = data;
117  txBuffer.push_back(msg);
