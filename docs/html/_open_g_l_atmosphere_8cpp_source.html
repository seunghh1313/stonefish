47 OpenGLAtmosphere::OpenGLAtmosphere(const std::string& modelFilename, RenderQuality shadow)
48 {
49  //Prepare for rendering
50  for(unsigned short i=0; i<AtmosphereTextures::TEXTURE_COUNT; ++i) textures[i] = 0;
51  skySunShader = NULL;
52 
53  //Init shadow baking
54  sunShadowmapShader = NULL;
55  sunShadowmapArray = 0;
56  sunDepthSampler = 0;
57  sunShadowSampler = 0;
58  sunShadowmapSplits = 4;
59  sunShadowmapSize = 4096;
60  sunShadowFBO = 0;
61  sunSkyUBO = 0;
62  sunDirection = glm::vec3(0,0,1.f);
63  sunModelView = glm::mat4x4(0);
64  sunShadowFrustum = NULL;
65  sunShadowCPM = NULL;
66 
67  //Set standard sun position
68  SetSunPosition(0.0, 45.0);
69 
70  //Set shadow quality
71  switch(shadow)
72  {
73  case RenderQuality::DISABLED:
74  sunShadowmapSize = 0;
75  sunShadowmapSplits = 0;
76  break;
77 
78  case RenderQuality::LOW:
79  sunShadowmapSize = 1024;
80  sunShadowmapSplits = 4;
81  break;
82 
83  case RenderQuality::MEDIUM:
84  sunShadowmapSize = 2048;
85  sunShadowmapSplits = 4;
86  break;
87 
88  case RenderQuality::HIGH:
89  sunShadowmapSize = 4096;
90  sunShadowmapSplits = 4;
91  break;
92  }
93 
94  //Load API and Data
95  memset(&sunSkyUBOData, 0, sizeof(SunSkyUBO));
96  LoadAtmosphereData(modelFilename);
97 
98  //Permanently bind atmosphere textures
99  OpenGLState::BindTexture(TEX_ATM_TRANSMITTANCE, GL_TEXTURE_2D, textures[AtmosphereTextures::TRANSMITTANCE]);
100  OpenGLState::BindTexture(TEX_ATM_SCATTERING, GL_TEXTURE_3D, textures[AtmosphereTextures::SCATTERING]);
101  OpenGLState::BindTexture(TEX_ATM_IRRADIANCE, GL_TEXTURE_2D, textures[AtmosphereTextures::IRRADIANCE]);
102 
103  //Build rendering shaders
104  std::vector<GLuint> compiledShaders;
105  compiledShaders.push_back(atmosphereAPI);
106  std::vector<GLSLSource> sources;
107  sources.push_back(GLSLSource(GL_VERTEX_SHADER, "atmosphereSkySun.vert"));
108  sources.push_back(GLSLSource(GL_FRAGMENT_SHADER, "atmosphereSkySun.frag"));
109  skySunShader = new GLSLShader(sources, compiledShaders);
110  skySunShader->AddUniform("transmittance_texture", ParameterType::INT);
111  skySunShader->AddUniform("scattering_texture", ParameterType::INT);
112  skySunShader->AddUniform("single_mie_scattering_texture", ParameterType::INT);
113  skySunShader->AddUniform("eyePos", ParameterType::VEC3);
114  skySunShader->AddUniform("sunDir", ParameterType::VEC3);
115  skySunShader->AddUniform("invView", ParameterType::MAT4);
116  skySunShader->AddUniform("invProj", ParameterType::MAT4);
117  skySunShader->AddUniform("whitePoint", ParameterType::VEC3);
118  skySunShader->AddUniform("cosSunSize", ParameterType::FLOAT);
119  skySunShader->AddUniform("bottomRadius", ParameterType::FLOAT);
120  skySunShader->AddUniform("groundAlbedo", ParameterType::FLOAT);
121 
122  //Initialize shadows
123  sunShadowFrustum = new ViewFrustum[sunShadowmapSplits];
124  sunShadowCPM = new glm::mat4x4[sunShadowmapSplits];
125 
126  //Create sun & sky UBO
127  glGenBuffers(1, &sunSkyUBO);
128  glBindBuffer(GL_UNIFORM_BUFFER, sunSkyUBO);
129  glBufferData(GL_UNIFORM_BUFFER, sizeof(SunSkyUBO), NULL, GL_STATIC_DRAW);
130  glBindBuffer(GL_UNIFORM_BUFFER, 0);
131  glBindBufferRange(GL_UNIFORM_BUFFER, UBO_SUNSKY, sunSkyUBO, 0, sizeof(SunSkyUBO));
132 
133  if(shadow > RenderQuality::DISABLED)
134  {
135  sunShadowmapShader = new GLSLShader("sunCSM.frag");
136  sunShadowmapShader->AddUniform("shadowmapArray", ParameterType::INT);
137  sunShadowmapShader->AddUniform("shadowmapLayer", ParameterType::FLOAT);
138 
139  //Generate shadowmap array
140  glGenTextures(1, &sunShadowmapArray);
141  OpenGLState::BindTexture(TEX_BASE, GL_TEXTURE_2D_ARRAY, sunShadowmapArray);
142  glTexImage3D(GL_TEXTURE_2D_ARRAY, 0, GL_DEPTH_COMPONENT32F, sunShadowmapSize, sunShadowmapSize, sunShadowmapSplits, 0, GL_DEPTH_COMPONENT, GL_FLOAT, NULL);
143  OpenGLState::UnbindTexture(TEX_BASE);
144 
145  //Generate samplers
146  glGenSamplers(1, &sunDepthSampler);
147  glSamplerParameteri(sunDepthSampler, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
148  glSamplerParameteri(sunDepthSampler, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
149  glSamplerParameteri(sunDepthSampler, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
150  glSamplerParameteri(sunDepthSampler, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
151 
152  glGenSamplers(1, &sunShadowSampler);
153  glSamplerParameteri(sunShadowSampler, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
154  glSamplerParameteri(sunShadowSampler, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
155  glSamplerParameteri(sunShadowSampler, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
156  glSamplerParameteri(sunShadowSampler, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
157  glSamplerParameteri(sunShadowSampler, GL_TEXTURE_COMPARE_MODE, GL_COMPARE_REF_TO_TEXTURE);
158  glSamplerParameteri(sunShadowSampler, GL_TEXTURE_COMPARE_FUNC, GL_LEQUAL);
159 
160  //Create shadowmap framebuffer
161  glGenFramebuffers(1, &sunShadowFBO);
162  OpenGLState::BindFramebuffer(sunShadowFBO);
163  glFramebufferTextureLayer(GL_FRAMEBUFFER, GL_DEPTH_ATTACHMENT, sunShadowmapArray, 0, 0);
164  glDrawBuffer(GL_NONE);
165  glReadBuffer(GL_NONE);
166 
167  GLenum status = glCheckFramebufferStatus(GL_FRAMEBUFFER);
168  if(status != GL_FRAMEBUFFER_COMPLETE)
169  cError("Sun shadow FBO initialization failed!");
170 
171  OpenGLState::BindFramebuffer(0);
172 
173  //Permanently bind shadow textures and samplers
174  OpenGLState::BindTexture(TEX_SUN_SHADOW, GL_TEXTURE_2D_ARRAY, sunShadowmapArray);
175  glBindSampler(TEX_SUN_SHADOW, sunShadowSampler);
176 
177  OpenGLState::BindTexture(TEX_SUN_DEPTH, GL_TEXTURE_2D_ARRAY, sunShadowmapArray);
178  glBindSampler(TEX_SUN_DEPTH, sunDepthSampler);
179  }
180 }

205 void OpenGLAtmosphere::SetSunPosition(float azimuthDeg, float elevationDeg)
206 {
207  azimuthDeg = azimuthDeg < -180.f ? -180.f : (azimuthDeg > 180.f ? 180.f : azimuthDeg);
208  elevationDeg = elevationDeg < -90.f ? -90.f : (elevationDeg > 90.f ? 90.f : elevationDeg);
209 
210  sunAzimuth = azimuthDeg;
211  sunElevation = elevationDeg;
212 
213  //Calculate sun dir vector
214  float sunAzimuthAngle = -sunAzimuth/180.f*M_PI + M_PI;
215  float sunZenithAngle = (90.f - sunElevation)/180.f*M_PI;
216  sunDirection = glm::normalize(glm::rotate(glm::vec3(cos(sunAzimuthAngle) * sin(sunZenithAngle), sin(sunAzimuthAngle) * sin(sunZenithAngle), cos(sunZenithAngle)), (float)M_PI, glm::vec3(0,1.f,0)));
217 
218  //Build sun modelview matrix
219  glm::vec3 up(0,0,-1.f);
220  glm::vec3 right = glm::cross(sunDirection, up);
221  right = glm::normalize(right);
222  up = glm::normalize(glm::cross(right, sunDirection));
223  sunModelView = glm::lookAt(glm::vec3(0,0,0), -sunDirection, up) * glm::mat4(1.f);
224 }

242 void OpenGLAtmosphere::LoadAtmosphereData(const std::string& filename)
243 {
244  //Load parameters used to generate atmosphere data
245  glm::uvec3 transmittanceSize;
246  glm::uvec3 irradianceSize;
247  glm::uvec3 scatteringSize;
248  GLfloat lengthUnitInMeters;
249  GLfloat bottomRadius;
250 #ifdef EMBEDDED_RESOURCES
251  ResourceHandle rh(filename);
252  std::istringstream dataString(rh.string());
253  std::istream& data(dataString);
254 #else
255  std::ifstream dataFile(filename, std::ios::binary | std::ios::in);
256  std::istream& data(dataFile);
257 #endif
258  data.read((char*)&transmittanceSize, sizeof(transmittanceSize));
259  data.read((char*)&irradianceSize, sizeof(irradianceSize));
260  data.read((char*)&scatteringSize, sizeof(scatteringSize));
261  data.read((char*)&lengthUnitInMeters, sizeof(lengthUnitInMeters));
262  data.read((char*)&bottomRadius, sizeof(bottomRadius));
263  data.read((char*)&sunSkyUBOData.whitePoint, sizeof(sunSkyUBOData.whitePoint));
264  data.read((char*)&nPrecomputedWavelengths, sizeof(nPrecomputedWavelengths));
265  data.read((char*)&nScatteringOrders, sizeof(nScatteringOrders));
266 
267  sunSkyUBOData.atmLengthUnitInMeters = lengthUnitInMeters;
268  sunSkyUBOData.planetRadiusInUnits = bottomRadius/lengthUnitInMeters;
269 
270  //Load atmosphere textures
271  glm::vec4* pixels;
272  pixels = new glm::vec4[transmittanceSize.x * transmittanceSize.y];
273  data.read((char*)pixels, sizeof(glm::vec4) * transmittanceSize.x * transmittanceSize.y);
274  textures[AtmosphereTextures::TRANSMITTANCE] = OpenGLContent::GenerateTexture(GL_TEXTURE_2D, transmittanceSize, GL_RGBA32F, GL_RGBA, GL_FLOAT, pixels, FilteringMode::BILINEAR, false);
275  delete [] pixels;
276  pixels = new glm::vec4[irradianceSize.x * irradianceSize.y];
277  data.read((char*)pixels, sizeof(glm::vec4) * irradianceSize.x * irradianceSize.y);
278  textures[AtmosphereTextures::IRRADIANCE] = OpenGLContent::GenerateTexture(GL_TEXTURE_2D, irradianceSize, GL_RGBA32F, GL_RGBA, GL_FLOAT, pixels, FilteringMode::BILINEAR, false);
279  delete [] pixels;
280  pixels = new glm::vec4[scatteringSize.x * scatteringSize.y * scatteringSize.z];
281  data.read((char*)pixels, sizeof(glm::vec4) * scatteringSize.x * scatteringSize.y * scatteringSize.z);
282  textures[AtmosphereTextures::SCATTERING] = OpenGLContent::GenerateTexture(GL_TEXTURE_3D, scatteringSize, GL_RGBA16F, GL_RGBA, GL_FLOAT, pixels, FilteringMode::BILINEAR, false);
283  delete [] pixels;
284 #ifndef EMBEDDED_RESOURCES
285  dataFile.close();
286 #endif
287  cInfo("Loaded precomputed atmosphere model (%u wavelengths, %u scattering orders)",
288  nPrecomputedWavelengths, nScatteringOrders);
289 }

291 void OpenGLAtmosphere::DrawSkyAndSun(const OpenGLCamera* view)
292 {
293  glm::vec3 eyePos = view->GetEyePosition()/sunSkyUBOData.atmLengthUnitInMeters;
294  eyePos.z = eyePos.z > -0.5/sunSkyUBOData.atmLengthUnitInMeters ? -0.5/sunSkyUBOData.atmLengthUnitInMeters : eyePos.z;
295  glm::mat4 invViewMatrix = glm::inverse(view->GetViewMatrix());
296  invViewMatrix[3].x = eyePos.x;
297  invViewMatrix[3].y = eyePos.y;
298  invViewMatrix[3].z = eyePos.z;
299  glm::mat4 invProjectionMatrix = glm::inverse(view->GetProjectionMatrix());
300 
301  skySunShader->Use();
302  skySunShader->SetUniform("bottomRadius", sunSkyUBOData.planetRadiusInUnits);
303  skySunShader->SetUniform("groundAlbedo", 0.7f);
304  skySunShader->SetUniform("transmittance_texture", TEX_ATM_TRANSMITTANCE);
305  skySunShader->SetUniform("scattering_texture", TEX_ATM_SCATTERING);
306  skySunShader->SetUniform("single_mie_scattering_texture", TEX_ATM_SCATTERING);
307  skySunShader->SetUniform("eyePos", eyePos);
308  skySunShader->SetUniform("sunDir", GetSunDirection());
309  skySunShader->SetUniform("invProj", invProjectionMatrix);
310  skySunShader->SetUniform("invView", invViewMatrix);
311  skySunShader->SetUniform("whitePoint", sunSkyUBOData.whitePoint);
312  skySunShader->SetUniform("cosSunSize", (GLfloat)cosf(0.00935f/2.f));
313  ((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getContent()->DrawSAQ();
314  OpenGLState::UseProgram(0);
315 }

317 void OpenGLAtmosphere::BakeShadowmaps(OpenGLPipeline* pipe, OpenGLCamera* view)
318 {
319  if(sunShadowmapSize == 0)
320  return;
321 
322  //Pre-set splits
323  for(unsigned int i = 0; i < sunShadowmapSplits; ++i)
324  {
325  sunShadowFrustum[i].fov = view->GetFOVY() + 0.2f; //avoid artifacts in the borders
326  GLint* viewport = view->GetViewport();
327  sunShadowFrustum[i].ratio = (GLfloat)viewport[2]/(GLfloat)viewport[3];
328  delete [] viewport;
329  }
330 
331  //Compute the z-distances for each split as seen in camera space
332  UpdateSplitDist(view->GetNearClip(), view->GetFarClip());
333 
334  //Render maps
335  glCullFace(GL_FRONT); //GL_FRONT -> no shadow acne but problems with filtering
336  glDisable(GL_DEPTH_CLAMP);
337 
338  OpenGLState::BindFramebuffer(sunShadowFBO);
339  OpenGLState::Viewport(0, 0, sunShadowmapSize, sunShadowmapSize);
340 
341  glm::vec3 camPos = view->GetEyePosition();
342  glm::vec3 camDir = view->GetLookingDirection();
343  glm::vec3 camUp = view->GetUpDirection();
344 
345  // for all shadow splits
346  for(unsigned int i = 0; i < sunShadowmapSplits; ++i)
347  {
348  //Compute the camera frustum slice boundary points in world space
349  UpdateFrustumCorners(sunShadowFrustum[i], camPos, camDir, camUp);
350 
351  //Adjust the view frustum of the light, so that it encloses the camera frustum slice fully.
352  //note that this function sets the projection matrix as it sees best fit
353  glm::mat4 cp = BuildCropProjMatrix(sunShadowFrustum[i]);
354  sunShadowCPM[i] = cp * sunModelView;
355 
356  ((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getContent()->SetProjectionMatrix(cp);
357  ((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getContent()->SetViewMatrix(sunModelView);
358  //Draw current depth map
359  glFramebufferTextureLayer(GL_FRAMEBUFFER, GL_DEPTH_ATTACHMENT, sunShadowmapArray, 0, i);
360  glClear(GL_DEPTH_BUFFER_BIT);
361  pipe->DrawObjects();
362  }
363 
364  OpenGLState::BindFramebuffer(0);
365 
366  glEnable(GL_DEPTH_CLAMP);
367  glCullFace(GL_BACK);
368 }

370 //Computes the near and far distances for every frustum slice in camera eye space
371 void OpenGLAtmosphere::UpdateSplitDist(GLfloat nd, GLfloat fd)
372 {
373  fd = fd > 250.f ? 250.f : fd; //Limit camera frustum to max 250.0 m (to avoid shadow blocks)
374  nd = 0.1f;
375  GLfloat lambda = 0.95f;
376  GLfloat ratio = fd/nd;
377  sunShadowFrustum[0].near = nd;
378 
379  for(unsigned int i = 1; i < sunShadowmapSplits; ++i)
380  {
381  GLfloat si = i / (GLfloat)sunShadowmapSplits;
382 
383  sunShadowFrustum[i].near = lambda*(nd*powf(ratio, si)) + (1.f-lambda)*(nd + (fd - nd)*si);
384  sunShadowFrustum[i-1].far = sunShadowFrustum[i].near * 1.05f;
385  }
386 
387  sunShadowFrustum[sunShadowmapSplits-1].far = fd;
388  //for(int i=0; i<sunShadowmapSplits; i++) printf("Frustum%d Near: %f Far: %f\n", i, sunShadowFrustum[i].near, sunShadowFrustum[i].far);
389 }

391 //Computes the 8 corner points of the current view frustum
392 void OpenGLAtmosphere::UpdateFrustumCorners(ViewFrustum &f, glm::vec3 center, glm::vec3 dir, glm::vec3 up)
393 {
394  glm::vec3 right = glm::cross(dir, up);
395  right = glm::normalize(right);
396  //up = glm::normalize(glm::cross(right, dir));
397 
398  glm::vec3 fc = center + dir * f.far;
399  glm::vec3 nc = center + dir * f.near;
400 
401  // these heights and widths are half the heights and widths of
402  // the near and far plane rectangles
403  GLfloat nearHeight = tan(f.fov/2.0f) * f.near;
404  GLfloat nearWidth = nearHeight * f.ratio;
405  GLfloat farHeight = tan(f.fov/2.0f) * f.far;
406  GLfloat farWidth = farHeight * f.ratio;
407 
408  f.corners[0] = nc - up * nearHeight - right * nearWidth;
409  f.corners[1] = nc + up * nearHeight - right * nearWidth;
410  f.corners[2] = nc + up * nearHeight + right * nearWidth;
411  f.corners[3] = nc - up * nearHeight + right * nearWidth;
412  f.corners[4] = fc - up * farHeight - right * farWidth;
413  f.corners[5] = fc + up * farHeight - right * farWidth;
414  f.corners[6] = fc + up * farHeight + right * farWidth;
415  f.corners[7] = fc - up * farHeight + right * farWidth;
416 }

418 // this function builds a projection matrix for rendering from the shadow's POV.
419 // First, it computes the appropriate z-range and sets an orthogonal projection.
420 // Then, it translates and scales it, so that it exactly captures the bounding box
421 // of the current frustum slice
422 glm::mat4 OpenGLAtmosphere::BuildCropProjMatrix(ViewFrustum &f)
423 {
424  GLfloat maxX = -1000.0f;
425  GLfloat maxY = -1000.0f;
426  GLfloat maxZ;
427  GLfloat minX = 1000.0f;
428  GLfloat minY = 1000.0f;
429  GLfloat minZ;
430  glm::vec4 transf;
431 
432  //Find the z-range of the current frustum as seen from the light in order to increase precision
433  glm::mat4 shad_mv = sunModelView;
434 
435  //Note: only the z-component is needed and thus the multiplication can be simplified
436  //transf.z = shad_modelview[2] * f.point[0].x + shad_modelview[6] * f.point[0].y + shad_modelview[10] * f.point[0].z + shad_modelview[14]
437  transf = shad_mv * glm::vec4(f.corners[0], 1.0f);
438  minZ = transf.z;
439  maxZ = transf.z;
440 
441  for(int i = 1; i <8 ; i++) {
442  transf = shad_mv * glm::vec4(f.corners[i], 1.0f);
443  if(transf.z > maxZ) maxZ = transf.z;
444  if(transf.z < minZ) minZ = transf.z;
445  }
446 
447  //Make sure all relevant shadow casters are included - use object bounding boxes!
448  Vector3 aabbMin;
449  Vector3 aabbMax;
450  SimulationApp::getApp()->getSimulationManager()->getWorldAABB(aabbMin, aabbMax);
451 
452  transf = shad_mv * glm::vec4(aabbMin.x(), aabbMin.y(), aabbMin.z(), 1.f);
453  if(transf.z > maxZ) maxZ = transf.z;
454  if(transf.z < minZ) minZ = transf.z;
455 
456  transf = shad_mv * glm::vec4(aabbMax.x(), aabbMax.y(), aabbMax.z(), 1.f);
457  if(transf.z > maxZ) maxZ = transf.z;
458  if(transf.z < minZ) minZ = transf.z;
459 
460  //Set the projection matrix with the new z-bounds
461  //Note: there is inversion because the light looks at the neg z axis
462  glm::mat4 shad_proj = glm::ortho(-1.f, 1.f, -1.f, 1.f, -maxZ, -minZ);
463  glm::mat4 shad_mvp = shad_proj * shad_mv;
464 
465  //Find the extends of the frustum slice as projected in light's homogeneous coordinates
466  for(int i = 0; i < 8; i++) {
467  transf = shad_mvp * glm::vec4(f.corners[i], 1.0f);
468 
469  transf.x /= transf.w;
470  transf.y /= transf.w;
471 
472  if(transf.x > maxX) maxX = transf.x;
473  if(transf.x < minX) minX = transf.x;
474  if(transf.y > maxY) maxY = transf.y;
475  if(transf.y < minY) minY = transf.y;
476  }
477 
478  //Build crop matrix
479  glm::mat4 shad_crop = glm::mat4(1.f);
480  GLfloat scaleX = 2.0f/(maxX - minX);
481  GLfloat scaleY = 2.0f/(maxY - minY);
482  GLfloat offsetX = -0.5f*(maxX + minX)*scaleX;
483  GLfloat offsetY = -0.5f*(maxY + minY)*scaleY;
484  shad_crop[0][0] = scaleX;
485  shad_crop[1][1] = scaleY;
486  shad_crop[3][0] = offsetX;
487  shad_crop[3][1] = offsetY;
488 
489  //Build crop projection matrix
490  glm::mat4 shad_crop_proj = shad_crop * shad_proj;
491 
492  return shad_crop_proj;
493 }

495 void OpenGLAtmosphere::SetupMaterialShaders()
496 {
497  //Calculate shadow splits
498  glm::mat4 bias(0.5f, 0.f, 0.f, 0.f,
499  0.f, 0.5f, 0.f, 0.f,
500  0.f, 0.f, 0.5f, 0.f,
501  0.5f, 0.5f, 0.5f, 1.f);
502  GLfloat frustumFar[4];
503  GLfloat frustumNear[4];
504 
505  //For every inactive split
506  for(unsigned int i = sunShadowmapSplits; i<4; ++i)
507  {
508  frustumNear[i] = 0;
509  frustumFar[i] = 0;
