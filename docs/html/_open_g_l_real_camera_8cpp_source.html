38 OpenGLRealCamera::OpenGLRealCamera(glm::vec3 eyePosition, glm::vec3 direction, glm::vec3 cameraUp,
39  GLint x, GLint y, GLint width, GLint height,
40  GLfloat horizontalFovDeg, glm::vec2 range, bool continuousUpdate)
41  : OpenGLCamera(x, y, width, height, range)
42 {
43  _needsUpdate = false;
44  newData = false;
45  continuous = continuousUpdate;
46  camera = NULL;
47  cameraFBO = 0;
48 
49  //Setup view
50  SetupCamera(eyePosition, direction, cameraUp);
51  //Setup projection
52  fovx = horizontalFovDeg/180.f * M_PI;
53  GLfloat fovy = 2.f * atanf( (GLfloat)viewportHeight/(GLfloat)viewportWidth * tanf(fovx/2.f) );
54  projection = glm::perspectiveFov(fovy, (GLfloat)viewportWidth, (GLfloat)viewportHeight, near, far);
55 
56  UpdateTransform();
57 }

90 void OpenGLRealCamera::setCamera(ColorCamera* cam)
91 {
92  //Connect with camera sensor
93  camera = cam;
94 
95  //Generate buffers
96  cameraColorTex[0] = OpenGLContent::GenerateTexture(GL_TEXTURE_2D, glm::uvec3((GLuint)viewportWidth, (GLuint)viewportHeight, 0),
97  GL_RGB8, GL_RGB, GL_UNSIGNED_BYTE, NULL, sf::FilteringMode::NEAREST, false);
98  cameraColorTex[1] = OpenGLContent::GenerateTexture(GL_TEXTURE_2D, glm::uvec3((GLuint)viewportWidth, (GLuint)viewportHeight, 0),
99  GL_RGB8, GL_RGB, GL_UNSIGNED_BYTE, NULL, sf::FilteringMode::NEAREST, false);
100  std::vector<FBOTexture> textures;
101  textures.push_back(FBOTexture(GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, cameraColorTex[0]));
102  textures.push_back(FBOTexture(GL_COLOR_ATTACHMENT1, GL_TEXTURE_2D, cameraColorTex[1]));
103  cameraFBO = OpenGLContent::GenerateFramebuffer(textures);
104 
105  glGenBuffers(1, &cameraPBO);
106  glBindBuffer(GL_PIXEL_PACK_BUFFER, cameraPBO);
107  glBufferData(GL_PIXEL_PACK_BUFFER, viewportWidth * viewportHeight * 3, 0, GL_STREAM_READ);
108  glBindBuffer(GL_PIXEL_PACK_BUFFER, 0);
109 }

133 void OpenGLRealCamera::UpdateTransform()
134 {
135  eye = tempEye;
136  dir = tempDir;
137  up = tempUp;
138  SetupCamera();
139 
140  viewUBOData.VP = GetProjectionMatrix() * GetViewMatrix();
141  viewUBOData.eye = GetEyePosition();
142  ExtractFrustumFromVP(viewUBOData.frustum, viewUBOData.VP);
143 
144  //Inform camera to run callback
145  if(newData)
146  {
147  glBindBuffer(GL_PIXEL_PACK_BUFFER, cameraPBO);
148  GLubyte* src = (GLubyte*)glMapBuffer(GL_PIXEL_PACK_BUFFER, GL_READ_ONLY);
149  if(src)
150  {
151  camera->NewDataReady(src);
152  glUnmapBuffer(GL_PIXEL_PACK_BUFFER); //Release pointer to the mapped buffer
153  }
154  glBindBuffer(GL_PIXEL_PACK_BUFFER, 0);
155  newData = false;
156  }
157 }

169 void OpenGLRealCamera::DrawLDR(GLuint destinationFBO, bool updated)
170 {
171  //Check if there is a need to display image on screen
172  bool display = true;
173  unsigned int dispX, dispY;
174  GLfloat dispScale;
175  if(camera != NULL)
176  display = camera->getDisplayOnScreen(dispX, dispY, dispScale);
177 
178  //Draw on screen
179  if(display)
180  OpenGLCamera::DrawLDR(destinationFBO, updated);
181 
182  //Draw to camera buffer
183  if(camera != NULL && updated)
184  {
185  if(display) //No need to calculate exposure again
186  {
187  if(antiAliasing)
188  {
189  OpenGLState::BindTexture(TEX_POSTPROCESS1, GL_TEXTURE_2D, postprocessTex[1]);
190  OpenGLState::BindFramebuffer(cameraFBO);
191  OpenGLState::Viewport(0, 0, viewportWidth, viewportHeight);
192  glDrawBuffer(GL_COLOR_ATTACHMENT0);
193  fxaaShader->Use();
194  fxaaShader->SetUniform("texSource", TEX_POSTPROCESS1);
195  fxaaShader->SetUniform("RCPFrame", glm::vec2(1.f/(GLfloat)viewportWidth, 1.f/(GLfloat)viewportHeight));
196  ((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getContent()->DrawSAQ();
197 
198  glDrawBuffer(GL_COLOR_ATTACHMENT1);
199  OpenGLState::BindTexture(TEX_POSTPROCESS1, GL_TEXTURE_2D, cameraColorTex[0]);
200  flipShader->Use();
201  flipShader->SetUniform("texSource", TEX_POSTPROCESS1);
202  ((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getContent()->DrawSAQ();
203  OpenGLState::UseProgram(0);
204  OpenGLState::BindFramebuffer(0);
205 
206  OpenGLState::BindTexture(TEX_POSTPROCESS1, GL_TEXTURE_2D, cameraColorTex[1]);
207  glBindBuffer(GL_PIXEL_PACK_BUFFER, cameraPBO);
208  glGetTexImage(GL_TEXTURE_2D, 0, GL_RGB, GL_UNSIGNED_BYTE, NULL);
209  glBindBuffer(GL_PIXEL_PACK_BUFFER, 0);
210  OpenGLState::UnbindTexture(TEX_POSTPROCESS1);
211  }
212  else
213  {
214  OpenGLState::BindTexture(TEX_POSTPROCESS1, GL_TEXTURE_2D, renderColorTex[lastActiveRenderColorBuffer]);
215  OpenGLState::BindTexture(TEX_POSTPROCESS2, GL_TEXTURE_2D, exposureTex);
216  OpenGLState::BindFramebuffer(cameraFBO);
217  OpenGLState::Viewport(0, 0, viewportWidth, viewportHeight);
218  glDrawBuffer(GL_COLOR_ATTACHMENT0);
219  tonemappingShaders[2]->Use();
220  tonemappingShaders[2]->SetUniform("texSource", TEX_POSTPROCESS1);
221  tonemappingShaders[2]->SetUniform("texExposure", TEX_POSTPROCESS2);
222  tonemappingShaders[2]->SetUniform("exposureComp", (GLfloat)powf(2.f,exposureComp));
223  ((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getContent()->DrawSAQ();
224  OpenGLState::UnbindTexture(TEX_POSTPROCESS2);
225 
226  glDrawBuffer(GL_COLOR_ATTACHMENT1);
227  OpenGLState::BindTexture(TEX_POSTPROCESS1, GL_TEXTURE_2D, cameraColorTex[0]);
228  flipShader->Use();
229  flipShader->SetUniform("texSource", TEX_POSTPROCESS1);
230  ((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getContent()->DrawSAQ();
231  OpenGLState::UseProgram(0);
232  OpenGLState::BindFramebuffer(0);
233 
234  OpenGLState::BindTexture(TEX_POSTPROCESS1, GL_TEXTURE_2D, cameraColorTex[1]);
235  glBindBuffer(GL_PIXEL_PACK_BUFFER, cameraPBO);
236  glGetTexImage(GL_TEXTURE_2D, 0, GL_RGB, GL_UNSIGNED_BYTE, NULL);
237  glBindBuffer(GL_PIXEL_PACK_BUFFER, 0);
238  OpenGLState::UnbindTexture(TEX_POSTPROCESS1);
239  }
240  }
241  else
242  {
243  OpenGLCamera::DrawLDR(cameraFBO, updated);
244  OpenGLState::BindFramebuffer(cameraFBO);
245  OpenGLState::Viewport(0, 0, viewportWidth, viewportHeight);
246  glDrawBuffer(GL_COLOR_ATTACHMENT1);
247  OpenGLState::BindTexture(TEX_POSTPROCESS1, GL_TEXTURE_2D, cameraColorTex[0]);
248  flipShader->Use();
249  flipShader->SetUniform("texSource", TEX_POSTPROCESS1);
250  ((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getContent()->DrawSAQ();
251  OpenGLState::UseProgram(0);
252  OpenGLState::BindFramebuffer(0);
253 
254  OpenGLState::BindTexture(TEX_POSTPROCESS1, GL_TEXTURE_2D, cameraColorTex[1]);
255  glBindBuffer(GL_PIXEL_PACK_BUFFER, cameraPBO);
256  glGetTexImage(GL_TEXTURE_2D, 0, GL_RGB, GL_UNSIGNED_BYTE, NULL);
257  glBindBuffer(GL_PIXEL_PACK_BUFFER, 0);
258  OpenGLState::UnbindTexture(TEX_POSTPROCESS1);
259  }
260 
261  newData = true;
262  }

287  // constructor settings ( i . e . calibration constant K = 12.5)
288  // Reference: http://en.wikipedia.org/wiki/Film_speed
