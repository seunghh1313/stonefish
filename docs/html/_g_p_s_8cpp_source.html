46 void GPS::InternalUpdate(Scalar dt)
47 {
48  //get sensor frame in world
49  Transform gpsTrans = getSensorFrame();
50 
51  //GPS not updating underwater
52  Ocean* liq = SimulationApp::getApp()->getSimulationManager()->getOcean();
53  if(liq != nullptr && liq->IsInsideFluid(gpsTrans.getOrigin()))
54  {
55  Scalar data[4] = {BT_LARGE_FLOAT,
56  BT_LARGE_FLOAT,
57  Scalar(0), Scalar(0)};
58  Sample s(4, data);
59  AddSampleToHistory(s);
60  }
61  else
62  {
63  Vector3 gpsPos = gpsTrans.getOrigin();
64 
65  //add noise
66  if(!btFuzzyZero(nedStdDev))
67  {
68  gpsPos.setX(gpsPos.x() + noise(randomGenerator));
69  gpsPos.setY(gpsPos.y() + noise(randomGenerator));
70  }
71 
72  //convert NED to geodetic coordinates
73  double latitude;
74  double longitude;
75  double height;
76  SimulationApp::getApp()->getSimulationManager()->getNED()->Ned2Geodetic(gpsPos.x(), gpsPos.y(), 0.0, latitude, longitude, height);
77 
78  //record sample
79  Scalar data[4] = {latitude, longitude, gpsPos.x(), gpsPos.y()};
80  Sample s(4, data);
81  AddSampleToHistory(s);
82  }
83 }
