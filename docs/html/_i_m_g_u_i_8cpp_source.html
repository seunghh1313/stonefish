41 IMGUI::IMGUI(GLint windowWidth, GLint windowHeight, GLfloat hue)
42 {
43  shaders = false;
44  mouseX = 0;
45  mouseY = 0;
46  mouseLeftDown = false;
47  mouseRightDown = false;
48  clearActive();
49  guiVAO = 0;
50  translucentFBO = 0;
51  translucentTexture[0] = 0;
52  translucentTexture[1] = 0;
53  downsampleShader = NULL;
54  gaussianShader = NULL;
55  guiShader[0] = NULL;
56  guiShader[1] = NULL;
57  plainPrinter = NULL;
58  logoTexture = 0;
59  guiTexture = 0;
60 
61  //Set interface colors
62  theme[PANEL_COLOR] = glm::vec4(0.98f, 0.98f, 0.98f, 1.f);
63  theme[ACTIVE_TEXT_COLOR] = glm::vec4(0.1f, 0.1f, 0.1f, 1.f);
64  theme[INACTIVE_TEXT_COLOR] = glm::vec4(0.7f, 0.7f, 0.7f, 0.5f);
65  theme[ACTIVE_CONTROL_COLOR] = glm::vec4(1.0f, 1.0f, 1.0f, 0.8f);
66  theme[INACTIVE_CONTROL_COLOR] = glm::vec4(0.9f, 0.9f, 0.9f, 0.5f);
67  theme[HOT_CONTROL_COLOR] = glm::vec4(Color::HSV(hue, 0.1f, 1.0f).rgb, 1.f);
68  theme[PUSHED_CONTROL_COLOR] = glm::vec4(Color::HSV(hue, 0.2f, 0.8f).rgb, 1.f);
69  theme[EMPTY_COLOR] = glm::vec4(0.6f, 0.6f, 0.6f, 0.9f);
70  theme[FILLED_COLOR] = glm::vec4(Color::HSV(hue, 1.f, 0.8f).rgb, 0.9f);
71  theme[PLOT_COLOR] = glm::vec4(0.3f, 0.3f, 0.3f, 1.0f);
72  theme[PLOT_TEXT_COLOR] = glm::vec4(0.9f, 0.9f, 0.9f, 0.9f);
73 
74  //Set size
75  Resize(windowWidth, windowHeight);
76 
77  //Create printers
78  plainPrinter = new OpenGLPrinter(GetShaderPath() + std::string(STANDARD_FONT_NAME), STANDARD_FONT_SIZE);
79  backgroundMargin = 5.f;
80 
81  //Load logo texture
82  logoTexture = OpenGLContent::LoadInternalTexture("logo_gray_64.png", true);
83  //Load corner texture
84  guiTexture = OpenGLContent::LoadInternalTexture("gui.png", true);
85 
86  //Generate VAO
87  glGenVertexArrays(1, &guiVAO);
88  OpenGLState::BindVertexArray(guiVAO);
89  glEnableVertexAttribArray(0);
90  OpenGLState::BindVertexArray(0);
91 
92  //Load translucent shaders
93  downsampleShader = new GLSLShader("downsample2x.frag");
94  downsampleShader->AddUniform("source", INT);
95  downsampleShader->AddUniform("srcViewport", VEC2);
96  gaussianShader = new GLSLShader("gaussianBlur.frag", "gaussianBlur.vert");
97  gaussianShader->AddUniform("source", INT);
98  gaussianShader->AddUniform("texelOffset", VEC2);
99  guiShader[0] = new GLSLShader("guiFlat.frag","guiFlat.vert");
100  guiShader[0]->AddUniform("color", VEC4);
101  guiShader[1] = new GLSLShader("guiTex.frag","guiTex.vert");
102  guiShader[1]->AddUniform("tex", INT);
103  guiShader[1]->AddUniform("backTex", INT);
104  guiShader[1]->AddUniform("color", VEC4);
105 }

107 void IMGUI::Resize(GLint windowWidth, GLint windowHeight)
108 {
109  windowW = windowWidth;
110  windowH = windowHeight;
111 
112  //Resize printing area
113  OpenGLPrinter::SetWindowSize(windowW, windowH);
114 
115  //Destroy translucent background textures and framebuffers
116  if(translucentTexture[0] != 0) glDeleteTextures(2, translucentTexture);
117  if(translucentFBO != 0) glDeleteFramebuffers(1, &translucentFBO);
118 
119  //Create translucent background resources
120  glGenFramebuffers(1, &translucentFBO);
121  OpenGLState::BindFramebuffer(translucentFBO);
122 
123  glGenTextures(1, &translucentTexture[0]);
124  OpenGLState::BindTexture(TEX_BASE, GL_TEXTURE_2D, translucentTexture[0]);
125  glTexImage2D(GL_TEXTURE_2D, 0, GL_RGB8, windowW/4, windowH/4, 0, GL_RGB, GL_UNSIGNED_BYTE, NULL);
126  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
127  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
128  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
129  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
130  glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, translucentTexture[0], 0);
131 
132  glGenTextures(1, &translucentTexture[1]);
133  OpenGLState::BindTexture(TEX_BASE, GL_TEXTURE_2D, translucentTexture[1]);
134  glTexImage2D(GL_TEXTURE_2D, 0, GL_RGB8, windowW/4, windowH/4, 0, GL_RGB, GL_UNSIGNED_BYTE, NULL);
135  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
136  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
137  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
138  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
139  glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT1, GL_TEXTURE_2D, translucentTexture[1], 0);
140 
141  GLenum status = glCheckFramebufferStatus(GL_FRAMEBUFFER);
142  if(status != GL_FRAMEBUFFER_COMPLETE)
143  cError("Translucent background FBO initialization failed!");
144 
145  OpenGLState::BindFramebuffer(0);
146 }

242 void IMGUI::GenerateBackground()
243 {
244  OpenGLState::BindFramebuffer(translucentFBO);
245  OpenGLState::Viewport(0, 0, windowW/4, windowH/4);
246  OpenGLState::BindTexture(TEX_BASE, GL_TEXTURE_2D, ((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getScreenTexture());
247  downsampleShader->Use();
248  downsampleShader->SetUniform("source", TEX_BASE);
249  downsampleShader->SetUniform("srcViewport", glm::vec2((GLfloat)windowW, (GLfloat)windowH));
250  ((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getContent()->DrawSAQ();
251 
252  gaussianShader->Use();
253  gaussianShader->SetUniform("source", TEX_BASE);
254  for(int i=0; i<3; ++i)
255  {
256  glDrawBuffer(GL_COLOR_ATTACHMENT1);
257  OpenGLState::BindTexture(TEX_BASE, GL_TEXTURE_2D, translucentTexture[0]);
258  gaussianShader->SetUniform("texelOffset", glm::vec2(4.f/(GLfloat)windowW, 0.f));
259  ((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getContent()->DrawSAQ();
260 
261  glDrawBuffer(GL_COLOR_ATTACHMENT0);
262  OpenGLState::BindTexture(TEX_BASE, GL_TEXTURE_2D, translucentTexture[1]);
263  gaussianShader->SetUniform("texelOffset", glm::vec2(0.f, 4.f/(GLfloat)windowH));
264  ((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getContent()->DrawSAQ();
265  }
266  OpenGLState::UseProgram(0);
267  OpenGLState::UnbindTexture(TEX_BASE);
268  OpenGLState::BindFramebuffer(0);
269 }

271 void IMGUI::Begin()
272 {
273  clearHot();
274 
275  OpenGLState::DisableDepthTest();
276  OpenGLState::DisableCullFace();
277  OpenGLState::EnableBlend();
278  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
279 
280  glScissor(0, 0, windowW, windowH);
281  OpenGLState::Viewport(0, 0, windowW, windowH);
282  ((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getContent()->SetViewportSize(windowW, windowH);
283  OpenGLState::BindVertexArray(guiVAO);
284 }

364 void IMGUI::DrawArrow(GLfloat x, GLfloat y, GLfloat h, bool up, glm::vec4 color)
365 {
366  y = windowH - y;
367 
368  GLfloat hx = h/windowW * 2.f;
369  GLfloat hy = h/windowH * 2.f;
370  x = x/windowW * 2.f - 1.f;
371  y = y/windowH * 2.f - 1.f;
372 
373  GLfloat triData[3][2];
374  if(up)
375  {
376  triData[1][0] = x-hx/2.f;
377  triData[1][1] = y-hy/2.f;
378  triData[0][0] = x+hx/2.f;
379  triData[0][1] = y-hy/2.f;
380  triData[2][0] = x;
381  triData[2][1] = y+hy/2.f;
382  }
383  else
384  {
385  triData[1][0] = x-hx/2.f;
386  triData[1][1] = y+hy/2.f;
387  triData[0][0] = x+hx/2.f;
388  triData[0][1] = y+hy/2.f;
389  triData[2][0] = x;
390  triData[2][1] = y-hy/2.f;
391  }
392 
393  GLuint vbo;
394  glGenBuffers(1, &vbo);
395 
396  guiShader[0]->Use();
397  guiShader[0]->SetUniform("color", color);
398 
399  glBindBuffer(GL_ARRAY_BUFFER, vbo);
400  glBufferData(GL_ARRAY_BUFFER, sizeof(triData), triData, GL_STATIC_DRAW);
401  glVertexAttribPointer(0, 2, GL_FLOAT, GL_FALSE, 2*sizeof(GLfloat), (void*)0);
402  glBindBuffer(GL_ARRAY_BUFFER, 0);
403  glDrawArrays(GL_TRIANGLE_STRIP, 0, 3);
404 
405  OpenGLState::UseProgram(0);
406  glDeleteBuffers(1, &vbo);
407 }

439 void IMGUI::DrawRoundedRect(GLfloat x, GLfloat y, GLfloat w, GLfloat h, glm::vec4 color)
440 {
441  w = w < 4.f*CORNER_RADIUS ? 4.f*CORNER_RADIUS : w;
442  h = h < 4.f*CORNER_RADIUS ? 4.f*CORNER_RADIUS : h;
443  y = windowH - y;
444 
445  glm::vec2 cs(CORNER_RADIUS/windowW*2.f, CORNER_RADIUS/windowH*2.f);
446  w = w/windowW * 2.f;
447  h = h/windowH * 2.f;
448  x = x/windowW * 2.f - 1.f;
449  y = y/windowH * 2.f - 1.f;
450 
451  GLfloat rectData[24][4]= {{x, y, 0.f, 0.f},
452  {x, y-cs.y, 0.f, 0.333f},
453  {x+cs.x, y, 0.333f, 0.f},
454  {x+cs.x, y-cs.y, 0.333f, 0.333f},
455  {x+w-cs.x, y, 0.666f, 0.f},
456  {x+w-cs.x, y-cs.y, 0.666f, 0.333f},
457  {x+w, y, 1.f, 0.f},
458  {x+w, y-cs.y, 1.f, 0.333f},
459 
460  {x, y-cs.y, 0.f, 0.333f},
461  {x, y-h+cs.y, 0.f, 0.666f},
462  {x+cs.x, y-cs.y, 0.333f, 0.333f},
463  {x+cs.x, y-h+cs.y, 0.333f, 0.666f},
464  {x+w-cs.x, y-cs.y, 0.666f, 0.333f},
465  {x+w-cs.x, y-h+cs.y, 0.666f, 0.666f},
466  {x+w, y-cs.y, 1.f, 0.333f},
467  {x+w, y-h+cs.y, 1.f, 0.666f},
468 
469  {x, y-h+cs.y, 0.f, 0.666f},
470  {x, y-h, 0.f, 1.f},
471  {x+cs.x, y-h+cs.y, 0.333f, 0.666f},
472  {x+cs.x, y-h, 0.333f, 1.f},
473  {x+w-cs.x, y-h+cs.y, 0.666f, 0.666f},
474  {x+w-cs.x, y-h, 0.666f, 1.f},
475  {x+w, y-h+cs.y, 1.f, 0.666f},
476  {x+w, y-h, 1.f, 1.f}};
477 
478  //Get translucent texture
479  OpenGLState::BindTexture(TEX_BASE, GL_TEXTURE_2D, guiTexture);
480  OpenGLState::BindTexture(TEX_GUI1, GL_TEXTURE_2D, getTranslucentTexture());
481 
482  GLuint vbo;
483  glGenBuffers(1, &vbo);
484 
485  guiShader[1]->Use();
486  guiShader[1]->SetUniform("tex", 0);
487  guiShader[1]->SetUniform("backTex", 1);
488  guiShader[1]->SetUniform("color", color);
489 
490  glBindBuffer(GL_ARRAY_BUFFER, vbo);
491  glBufferData(GL_ARRAY_BUFFER, sizeof(rectData), rectData, GL_STATIC_DRAW);
492  glVertexAttribPointer(0, 4, GL_FLOAT, GL_FALSE, 4*sizeof(GLfloat), (void*)0);
493  glBindBuffer(GL_ARRAY_BUFFER, 0);
494  glDrawArrays(GL_TRIANGLE_STRIP, 0, 8);
495  glDrawArrays(GL_TRIANGLE_STRIP, 8, 8);
496  glDrawArrays(GL_TRIANGLE_STRIP, 16, 8);
497 
498  OpenGLState::UseProgram(0);
499  glDeleteBuffers(1, &vbo);
500 }

515 bool IMGUI::DoButton(Uid id, GLfloat x, GLfloat y, GLfloat w, GLfloat h, const std::string& title)
516 {
517  bool result = false;
518 
519  if(MouseInRect(x, y, w, h))
520  setHot(id);
521 
522  if(isActive(id))
523  {
524  if(!MouseIsDown(true)) //mouse went up
525  {
526  if(isHot(id))
527  result = true;
528  clearActive();
529  }
530  }
531  else if(isHot(id))
532  {
533  if(MouseIsDown(true)) //mouse went down
534  setActive(id);
535  }
536 
537  //drawing
538  if(isActive(id))
539  DrawRoundedRect(x, y, w, h, theme[PUSHED_CONTROL_COLOR]);
540  else if(isHot(id))
541  DrawRoundedRect(x, y, w, h, theme[HOT_CONTROL_COLOR]);
542  else
543  DrawRoundedRect(x, y, w, h, theme[PANEL_COLOR]);
544 
545  glm::vec2 textDim = PlainTextDimensions(title);
546  DrawPlainText(x + floorf((w - textDim.x)/2.f), y + floorf((h - textDim.y)/2.f), theme[ACTIVE_TEXT_COLOR], title);
547 
548  return result;
549 }

551 Scalar IMGUI::DoSlider(Uid id, GLfloat x, GLfloat y, GLfloat w, Scalar min, Scalar max, Scalar value, const std::string& title, unsigned int decimalPlaces)
552 {
553  //Check and correct dimensions
554  w = w < 8*backgroundMargin ? 8.f*backgroundMargin : w;
555 
556  GLfloat railW = w - 4.f*backgroundMargin;
557  GLfloat railH = 5.f;
558  GLfloat sliderW = 5.f;
559  GLfloat sliderH = 20.f;
560  GLfloat h = sliderH + 2.f * backgroundMargin + 5.f + STANDARD_FONT_SIZE;
561 
562  //Check mouse position
563  Scalar result = value;
564  GLfloat sliderPosition = (value-min)/(max-min);
565 
566  if(MouseInRect(x + backgroundMargin*2.f + sliderPosition * railW - sliderW/2.f, y + backgroundMargin + STANDARD_FONT_SIZE + 5.f, sliderW, sliderH))
567  setHot(id);
568 
569  if(isActive(id))
570  {
571  GLfloat mouseX = getMouseX();
572  if(mouseX <= x + backgroundMargin*2.f)
573  sliderPosition = 0;
574  else if(mouseX >= x + backgroundMargin*2.f + railW)
575  sliderPosition = 1.f;
576  else
577  sliderPosition = (mouseX - x - backgroundMargin*2.f)/railW;
578 
579  result = min + sliderPosition * (max - min);
580 
581  if(!MouseIsDown(true)) //mouse went up
582  clearActive();
583  }
584  else if(isHot(id))
585  {
586  if(MouseIsDown(true)) //mouse went down
587  setActive(id);
588  }
589 
590  //Drawing
591  //Background and text
