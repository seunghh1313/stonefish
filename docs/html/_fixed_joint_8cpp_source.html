45 FixedJoint::FixedJoint(std::string uniqueName, SolidEntity* solidA, SolidEntity* solidB) : Joint(uniqueName, false)
46 {
47  btRigidBody* bodyA = solidA->rigidBody;
48  btRigidBody* bodyB = solidB->rigidBody;
49  Transform frameInA = Transform::getIdentity();
50  Transform frameInB = solidB->getCGTransform().inverse() * solidA->getCGTransform();
51 
52  btFixedConstraint* fixed = new btFixedConstraint(*bodyA, *bodyB, frameInA, frameInB);
53  setConstraint(fixed);

56 FixedJoint::FixedJoint(std::string uniqueName, SolidEntity* solid, FeatherstoneEntity* fe, int linkId, const Vector3& pivot) : Joint(uniqueName, false)
57 {
58  Transform linkTransform = fe->getLinkTransform(linkId+1);
59  Transform solidTransform = solid->getCGTransform();
60  Vector3 pivotInA = linkTransform.inverse() * pivot;
61  Vector3 pivotInB = solidTransform.inverse() * pivot;
62  Matrix3 frameInB = solidTransform.getBasis().inverse() * linkTransform.getBasis();
63 
64  btMultiBodyFixedConstraint* fixed = new btMultiBodyFixedConstraint(fe->getMultiBody(), linkId, solid->rigidBody, pivotInA, pivotInB, Matrix3::getIdentity(), frameInB);
65  fixed->setMaxAppliedImpulse(BT_LARGE_FLOAT);
66  setConstraint(fixed);
67 
68  //Disable collision
69  SimulationApp::getApp()->getSimulationManager()->DisableCollision(fe->getLink(linkId+1).solid, solid);
70 }

93 std::vector<Renderable> FixedJoint::Render()
94 {
95  std::vector<Renderable> items(0);
96  btTypedConstraint* c = getConstraint();
97  if(c != nullptr)
98  {
99  Renderable item;
100  item.model = glm::mat4(1.f);
101  item.type = RenderableType::JOINT_LINES;
102  Vector3 A = c->getRigidBodyA().getCenterOfMassPosition();
103  Vector3 B = c->getRigidBodyB().getCenterOfMassPosition();
104  item.points.push_back(glm::vec3(A.getX(), A.getY(), A.getZ()));
105  item.points.push_back(glm::vec3(B.getX(), B.getY(), B.getZ()));
106  items.push_back(item);
107  }
108  return items;
109 }
