36 MSIS::MSIS(std::string uniqueName, Scalar stepAngleDeg, unsigned int numOfBins, Scalar horizontalBeamWidthDeg, Scalar verticalBeamWidthDeg,
37  Scalar minRotationDeg, Scalar maxRotationDeg, Scalar minRange, Scalar maxRange, ColorMap cm, Scalar frequency)
38  : Camera(uniqueName, (unsigned int)ceil(Scalar(360)/stepAngleDeg), numOfBins, horizontalBeamWidthDeg, frequency)
39 {
40  range.x = 0.f;
41  range.y = 0.f;
42  noise = glm::vec2(0.f);
43  fullRotation = false;
44  stepSize = btRadians(btScalar(360)/ceil(Scalar(360)/stepAngleDeg)); //Corrected step angle in radians
45  setRotationLimits(minRotationDeg, maxRotationDeg);
46  setRangeMax(maxRange);
47  setRangeMin(minRange);
48  setGain(1);
49  fovV = verticalBeamWidthDeg <= Scalar(0) ? Scalar(20) : (verticalBeamWidthDeg > Scalar(179) ? Scalar(179) : verticalBeamWidthDeg);
50  cMap = cm;
51  sonarData = NULL;
52  displayData = NULL;
53  newDataCallback = NULL;
54  glMSIS = nullptr;
55 }

63 void MSIS::setRotationLimits(Scalar l1Deg, Scalar l2Deg)
64 {
65  if(l1Deg == l2Deg) //Same
66  return;
67 
68  if(l1Deg > l2Deg) //Swap
69  {
70  Scalar tmp = l1Deg;
71  l1Deg = l2Deg;
72  l2Deg = tmp;
73  }
74 
75  //Clamp and convert to number of steps
76  btClamp(l1Deg, Scalar(-180), Scalar(180));
77  btClamp(l2Deg, Scalar(-180), Scalar(180));
78  roi.x = (GLint)round(btRadians(l1Deg)/stepSize);
79  roi.y = (GLint)round(btRadians(l2Deg)/stepSize);
80  fullRotation = (roi.x == -(int)resX/2) && (roi.y == (int)resX/2);
81  if(roi.y == (int)resX/2)
82  --roi.y;
83  currentStep = roi.x;
84  cw = true;
85 }

92 void MSIS::setRangeMax(Scalar r)
93 {
94  range.y = r > Scalar(range.x) ? (GLfloat)r : range.x;
95  Scalar pulseTime = (Scalar(2)*range.y/SOUND_VELOCITY_WATER) * Scalar(1.1);
96  setUpdateFrequency(Scalar(1)/pulseTime);

196 void MSIS::NewDataReady(void* data, unsigned int index)
197 {
198  if(newDataCallback != NULL)
199  {
200  if(index == 0)
201  {
202  unsigned int w, h;
203  getDisplayResolution(w, h);
204  memcpy(displayData, data, w*h*3);
205  }
206  else
207  {
208  sonarData = (GLubyte*)data;
209  newDataCallback(this);
210  sonarData = NULL;
211  }
212  }
213 
214  if(index == 1)
215  {
216  currentStep += cw ? 1 : -1;
217 
218  if(fullRotation)
219  {
220  if(cw && currentStep > roi.y)
221  currentStep = roi.x;
222  else if(!cw && currentStep < roi.x)
223  currentStep = roi.y;
224  }
225  else
226  {
227  if(cw && currentStep == roi.y)
228  cw = false;
229  else if(!cw && currentStep == roi.x)
230  cw = true;
231  }
232  }
233 }

241 std::vector<Renderable> MSIS::Render()
242 {
243  std::vector<Renderable> items = Sensor::Render();
244  if(isRenderable())
245  {
246  Renderable item;
247  item.model = glMatrixFromTransform(getSensorFrame());
248  item.type = RenderableType::SENSOR_LINES;
249 
250  //Create sonar dummy
251  int div = 24;
252  Scalar l1Deg, l2Deg;
253  getRotationLimits(l1Deg, l2Deg);
254  GLfloat fovStep = fullRotation ? 2.f*M_PI/(GLfloat)div : glm::radians(l2Deg-l1Deg)/(GLfloat)div;
255  //Arcs min
256  GLfloat cosVAngle = cosf(glm::radians(fovV)/2.f) * range.x;
257  GLfloat sinVAngle = sinf(glm::radians(fovV)/2.f) * range.x;
258  GLfloat hAngle = glm::radians(l1Deg);
259  for(int i=0; i<=div; ++i)
260  {
261  GLfloat z = cosf(hAngle) * cosVAngle;
262  GLfloat x = sinf(hAngle) * cosVAngle;
263  item.points.push_back(glm::vec3(x, sinVAngle, z));
264  if(i > 0 && i < div)
265  item.points.push_back(glm::vec3(x, sinVAngle, z));
266  hAngle += fovStep;
267  }
268  hAngle = glm::radians(l1Deg);
269  for(int i=0; i<=div; ++i)
270  {
271  GLfloat z = cosf(hAngle) * cosVAngle;
272  GLfloat x = sinf(hAngle) * cosVAngle;
273  item.points.push_back(glm::vec3(x, -sinVAngle, z));
274  if(i > 0 && i < div)
275  item.points.push_back(glm::vec3(x, -sinVAngle, z));
276  hAngle += fovStep;
277  }
278  //Arcs max
279  cosVAngle = cosf(glm::radians(fovV)/2.f) * range.y;
280  sinVAngle = sinf(glm::radians(fovV)/2.f) * range.y;
281  hAngle = glm::radians(l1Deg);
282  for(int i=0; i<=div; ++i)
283  {
284  GLfloat z = cosf(hAngle) * cosVAngle;
285  GLfloat x = sinf(hAngle) * cosVAngle;
286  item.points.push_back(glm::vec3(x, sinVAngle, z));
287  if(i > 0 && i < div)
288  item.points.push_back(glm::vec3(x, sinVAngle, z));
289  hAngle += fovStep;
290  }
291  hAngle = glm::radians(l1Deg);
292  for(int i=0; i<=div; ++i)
293  {
294  GLfloat z = cosf(hAngle) * cosVAngle;
295  GLfloat x = sinf(hAngle) * cosVAngle;
296  item.points.push_back(glm::vec3(x, -sinVAngle, z));
297  if(i > 0 && i < div)
298  item.points.push_back(glm::vec3(x, -sinVAngle, z));
299  hAngle += fovStep;
300  }
301  //Current beam position
302  hAngle = currentStep * stepSize;
303  GLfloat zc = cosf(hAngle) * cosVAngle;
304  GLfloat xc = sinf(hAngle) * cosVAngle;
305  item.points.push_back(glm::vec3(0,0,0));
306  item.points.push_back(glm::vec3(xc, sinVAngle, zc));
307  item.points.push_back(glm::vec3(xc, sinVAngle, zc));
308  item.points.push_back(glm::vec3(xc, -sinVAngle, zc));
309  item.points.push_back(glm::vec3(xc, -sinVAngle, zc));
310  item.points.push_back(glm::vec3(0,0,0));
311 
312  if(!fullRotation)
313  {
314  //Ends
315  hAngle = glm::radians(l1Deg);
316  GLfloat zs = cosf(hAngle) * cosVAngle;
317  GLfloat xs = sinf(hAngle) * cosVAngle;
318  item.points.push_back(glm::vec3(xs, sinVAngle, zs));
319  item.points.push_back(glm::vec3(xs, -sinVAngle, zs));
320  hAngle = glm::radians(l2Deg);
321  GLfloat ze = cosf(hAngle) * cosVAngle;
322  GLfloat xe = sinf(hAngle) * cosVAngle;
323  item.points.push_back(glm::vec3(xe, sinVAngle, ze));
324  item.points.push_back(glm::vec3(xe, -sinVAngle, ze));
325  //Pyramid
326  item.points.push_back(glm::vec3(0,0,0));
327  item.points.push_back(glm::vec3(xs, sinVAngle, zs));
328  item.points.push_back(glm::vec3(0,0,0));
329  item.points.push_back(glm::vec3(xs, -sinVAngle, zs));
330  item.points.push_back(glm::vec3(0,0,0));
331  item.points.push_back(glm::vec3(xe, sinVAngle, ze));
332  item.points.push_back(glm::vec3(0,0,0));
333  item.points.push_back(glm::vec3(xe, -sinVAngle, ze));
334  }
335 
336  items.push_back(item);
337  }
338  return items;
339 }
