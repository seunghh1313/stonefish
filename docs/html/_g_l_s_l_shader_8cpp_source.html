43 GLSLShader::GLSLShader(const std::vector<GLSLSource>& sources, const std::vector<GLuint>& precompiled)
44 {
45  valid = false;
46  program = 0;
47 
48  if(sources.size() > 0)
49  {
50  valid = true;
51 
52  std::vector<GLuint> shaders = precompiled;
53  GLint compiled = 0;
54 
55  for(size_t i=0; i<sources.size(); ++i)
56  {
57  GLuint shader = LoadShader(sources[i].type, sources[i].filename, sources[i].header, &compiled);
58  if(compiled == 0)
59  {
60  valid = false;
61  break;
62  }
63  shaders.push_back(shader);
64  }
65 
66  if(valid)
67  {
68  program = CreateProgram(shaders, precompiled.size());
69  if(program == 0)
70  valid = false;
71  }
72  }
73 }

84 GLSLShader::GLSLShader(std::string fragment, std::string vertex)
85 {
86  valid = false;
87  GLint compiled = 0;
88  GLuint vs;
89  GLuint fs;
90  std::string emptyHeader = "";
91 
92  if(vertex == "")
93  vs = saqVertexShader;
94  else
95  vs = LoadShader(GL_VERTEX_SHADER, vertex, emptyHeader, &compiled);
96 
97  fs = LoadShader(GL_FRAGMENT_SHADER, fragment, emptyHeader, &compiled);
98  program = CreateProgram({vs, fs}, vs == saqVertexShader ? 1 : 0);
99  valid = true;
100 }

442 GLuint GLSLShader::LoadShader(GLenum shaderType, const std::string& filename, const std::string& header, GLint *shaderCompiled)
443 {
444  GLuint shader = 0;
445  std::string sourcePath = GetShaderPath() + filename;
446 #ifdef EMBEDDED_RESOURCES
447  ResourceHandle rh(sourcePath);
448  if(!rh.isValid())
449  {
450  cCritical("Shader resource not found: %s", sourcePath.c_str());
451  return 0;
452  }
453  std::istringstream sourceString(rh.string());
454  std::istream& sourceBuf(sourceString);
455 #else
456  std::ifstream sourceFile(sourcePath);
457  if(!sourceFile.is_open())
458  {
459  cCritical("Shader file not found: %s", sourcePath.c_str());
460  return 0;
461  }
462  std::istream& sourceBuf(sourceFile);
463 #endif
464 #ifdef DEBUG
465  if(verbose)
466  cInfo("Loading shader from: %s", sourcePath.c_str());
467 #endif
468  std::string source = header + "\n";
469  std::string line;
470  while(!sourceBuf.eof())
471  {
472  std::getline(sourceBuf, line);
473 
474  if(line.find("#inject") == std::string::npos)
475  source.append(line + "\n");
476  else //inject code from another source file
477  {
478  size_t pos1 = line.find("\"");
479  size_t pos2 = line.find("\"", pos1+1);
480 
481  if(pos1 > 0 && pos2 > pos1)
482  {
483  std::string injectedPath = GetShaderPath() + line.substr(pos1+1, pos2-pos1-1);
484 #ifdef EMBEDDED_RESOURCES
485  ResourceHandle rh2(injectedPath);
486  if(!rh2.isValid())
487  {
488  cCritical("Shader include resource not found: %s", injectedPath.c_str());
489  return 0;
490  }
491  std::istringstream injectedString(rh2.string());
492  std::istream& injectedBuf(injectedString);
493 #else
494 
495  std::ifstream injectedFile(injectedPath);
496  if(!injectedFile.is_open())
497  {
498  sourceFile.close();
499  cCritical("Shader include file not found: %s", injectedPath.c_str());
500  return 0;
501  }
502  std::istream& injectedBuf(injectedFile);
503 #endif
504 #ifdef DEBUG
505  if(verbose)
506  cInfo("--> Injecting source from: %s", injectedPath.c_str());
507 #endif
508  while(!injectedBuf.eof())
509  {
510  std::getline(injectedBuf, line);
511  source.append(line + "\n");
512  }
513 #ifndef EMBEDDED_RESOURCES
514  injectedFile.close();
515 #endif
516  }
517  }
518  }
519 #ifndef EMBEDDED_RESOURCES
520  sourceFile.close();
521 #endif
522  const char* shaderSource = source.c_str();
523  if(shaderSource != NULL)
524  {
525  shader = glCreateShader(shaderType);
526  glShaderSource(shader, 1, (const GLchar**)&shaderSource, NULL);
527  glCompileShader(shader);
528  glGetShaderiv(shader, GL_COMPILE_STATUS, shaderCompiled);
529  if(*shaderCompiled == 0)
530  {
531  cError("Failed to compile shader: %s", shaderSource);
532  shader = 0;
533  }
534 #ifdef DEBUG
535  GLint infoLogLength = 0;
536  glGetShaderiv(shader, GL_INFO_LOG_LENGTH, &infoLogLength);
537  if(infoLogLength > 0)
538  {
539  std::vector<char> infoLog(infoLogLength+1);
540  glGetShaderInfoLog(shader, infoLogLength, NULL, &infoLog[0]);
541  cWarning("Shader compile log: %s", &infoLog[0]);
542  }
543 #endif
544  }
545  else
546  *shaderCompiled = 0;
547 
548  return shader;
549 }

551 GLuint GLSLShader::CreateProgram(const std::vector<GLuint>& compiledShaders, unsigned int doNotDeleteNFirstShaders)
552 {
553  GLint programLinked = 0;
554  GLuint program = glCreateProgram();
555 
556  for(unsigned int i=0; i<compiledShaders.size(); ++i)
557  if(compiledShaders[i] > 0)
558  glAttachShader(program, compiledShaders[i]);
559 
560  glLinkProgram(program);
561 #ifdef DEBUG
562  GLint infoLogLength = 0;
563  glGetProgramiv(program, GL_INFO_LOG_LENGTH, &infoLogLength);
564  if(infoLogLength > 0)
565  {
566  std::vector<char> infoLog(infoLogLength+1);
567  glGetProgramInfoLog(program, infoLogLength, NULL, &infoLog[0]);
568  cWarning("Program link log: %s", &infoLog[0]);
569  }
570 #endif
571  glGetProgramiv(program, GL_LINK_STATUS, &programLinked);
572 
573  for(unsigned int i=0; i<compiledShaders.size(); ++i)
574  {
575  if(compiledShaders[i] > 0)
576  {
577  glDetachShader(program, compiledShaders[i]);
578  if(i > doNotDeleteNFirstShaders-1)
579  glDeleteShader(compiledShaders[i]);
580  }
581  }
582 
583  if(programLinked == 0)
584  {
585  cError("Failed to link program!");
586  glDeleteProgram(program);
587  program = 0;
588  }
589 
590  return program;
591 }
