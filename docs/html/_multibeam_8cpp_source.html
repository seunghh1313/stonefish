38 Multibeam::Multibeam(std::string uniqueName, Scalar angleRangeDeg, unsigned int angleSteps, Scalar frequency, int historyLength) : LinkSensor(uniqueName, frequency, historyLength)
39 {
40  angRange = UnitSystem::Angle(true, angleRangeDeg);
41  angSteps = angleSteps;
42 
43  for(unsigned int i=0; i <= angSteps; ++i)
44  {
45  angles.push_back(i/(Scalar)angSteps * angRange - Scalar(0.5) * angRange);
46 
47  channels.push_back(SensorChannel("Distance", QuantityType::LENGTH));
48  channels.back().rangeMin = Scalar(0);
49  channels.back().rangeMax = BT_LARGE_FLOAT;
50  }
51 
52  distances = std::vector<Scalar>(angSteps+1, Scalar(0));
53 }

55 void Multibeam::InternalUpdate(Scalar dt)
56 {
57  //get sensor frame in world
58  Transform mbTrans = getSensorFrame();
59 
60  //shoot rays
61  for(unsigned int i=0; i<=angSteps; ++i)
62  {
63  Vector3 dir = mbTrans.getBasis().getColumn(0) * btCos(angles[i]) + mbTrans.getBasis().getColumn(1) * btSin(angles[i]);
64  Vector3 from = mbTrans.getOrigin() + dir * channels[1].rangeMin;
65  Vector3 to = mbTrans.getOrigin() + dir * channels[1].rangeMax;
66 
67  btCollisionWorld::ClosestRayResultCallback closest(from, to);
68  closest.m_collisionFilterGroup = MASK_DYNAMIC;
69  closest.m_collisionFilterMask = MASK_STATIC | MASK_DYNAMIC | MASK_ANIMATED_COLLIDING;
70  SimulationApp::getApp()->getSimulationManager()->getDynamicsWorld()->rayTest(from, to, closest);
71 
72  if(closest.hasHit())
73  {
74  Vector3 p = from.lerp(to, closest.m_closestHitFraction);
75  distances[i] = (p - mbTrans.getOrigin()).length();
76  }
77  else
78  distances[i] = channels[i].rangeMax;
79  }
80 
81  //record sample
82  Sample s(angSteps+1, distances.data());
83  AddSampleToHistory(s);
84 }

86 std::vector<Renderable> Multibeam::Render()
87 {
88  std::vector<Renderable> items = Sensor::Render();
89  if(isRenderable())
90  {
91  Renderable item;
92  item.type = RenderableType::SENSOR_LINES;
93  item.model = glMatrixFromTransform(getSensorFrame());
94  for(unsigned int i=0; i <= angSteps; ++i)
95  {
96  Vector3 dir = Vector3(1, 0, 0) * btCos(angles[i]) + Vector3(0, 1, 0) * btSin(angles[i]);
97  item.points.push_back(glm::vec3(0,0,0));
98  item.points.push_back(glm::vec3(dir.x() * distances[i], dir.y() * distances[i], dir.z() * distances[i]));
99  }
100  items.push_back(item);
101  }
102  return items;
103 }
