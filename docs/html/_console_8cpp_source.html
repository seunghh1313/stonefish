56 void Console::Print(MessageType t, std::string format, ...)
57 {
58  va_list args;
59  char buffer[4096];
60  va_start(args, format);
61  vsnprintf(buffer, sizeof(buffer), format.c_str(), args);
62  va_end(args);
63 
64  if(stdoutEnabled)
65  {
66 #ifdef COLOR_CONSOLE
67  switch(t)
68  {
69  default:
70  case MessageType::INFO:
71  printf("[INFO] %s\n", buffer);
72  break;
73 
74  case MessageType::WARNING:
75  printf("\033[33m[WARN] %s\033[0m\n", buffer);
76  break;
77 
78  case MessageType::ERROR:
79  printf("\033[31m[ERROR] %s\033[0m\n", buffer);
80  break;
81 
82  case MessageType::CRITICAL:
83  printf("\033[1;31m[CRITICAL] %s\033[0m\n", buffer);
84  break;
85  }
86 #else
87  switch(t)
88  {
89  default:
90  case MessageType::INFO:
91  printf("[INFO] %s\n", buffer);
92  break;
93 
94  case MessageType::WARNING:
95  printf("[WARN] %s\n", buffer);
96  break;
97 
98  case MessageType::ERROR:
99  printf("[ERROR] %s\n", buffer);
100  break;
101 
102  case MessageType::CRITICAL:
103  printf("[CRITICAL] %s\n", buffer);
104  break;
105  }
106 #endif
107  }
108 
109  ConsoleMessage msg;
110  msg.type = t;
111  msg.text = std::string(buffer);
112  SDL_LockMutex(linesMutex);
113  lines.push_back(msg);
114  SDL_UnlockMutex(linesMutex);
115 }

131 bool Console::SaveToFile(std::string filename)
132 {
133  std::ofstream outFile(filename);
134  if(outFile.is_open())
135  {
136  SDL_LockMutex(linesMutex);
137  for(size_t i=0; i<lines.size(); ++i)
138  {
139  switch(lines[i].type)
140  {
141  case MessageType::INFO:
142  outFile << "[INFO] " << lines[i].text << std::endl;
143  break;
144  case MessageType::WARNING:
145  outFile << "[WARN] " << lines[i].text << std::endl;
146  break;
147  case MessageType::ERROR:
148  outFile << "[ERROR] " << lines[i].text << std::endl;
149  break;
150  case MessageType::CRITICAL:
151  outFile << "[CRITICAL] " << lines[i].text << std::endl;
152  break;
153  }
154  }
155  SDL_UnlockMutex(linesMutex);
156  outFile.close();
157  return true;
158  }
159  else
160  return false;
161 }
