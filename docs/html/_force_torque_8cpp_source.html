60 void ForceTorque::InternalUpdate(Scalar dt)
61 {
62  if(j != NULL && attach != NULL)
63  {
64  Vector3 force(j->getFeedback(0), j->getFeedback(1), j->getFeedback(2));
65  Vector3 torque(j->getFeedback(3), j->getFeedback(4), j->getFeedback(5));
66 
67  if(j->isMultibodyJoint())
68  {
69  force /= dt;
70  torque /= dt;
71  }
72 
73  lastFrame = attach->getOTransform() * o2s;
74  Matrix3 toSensor = lastFrame.getBasis().inverse();
75  force = toSensor * force;
76  torque = toSensor * torque;
77 
78  Scalar values[6] = {force.getX(), force.getY(), force.getZ(), torque.getX(), torque.getY(), torque.getZ()};
79  Sample s(6, values);
80  AddSampleToHistory(s);
81  }
82  else
83  {
84  Vector3 force, torque;
85  unsigned int childId = fe->getJointFeedback(jId, force, torque);
86  lastFrame = fe->getLink(childId).solid->getCG2OTransform() * o2s;
87  Matrix3 toSensor = lastFrame.getBasis().inverse();
88  force = toSensor * force;
89  torque = toSensor * torque;
90  lastFrame = fe->getLink(childId).solid->getCGTransform() * lastFrame; //From local to global
91 
92  Scalar values[6] = {force.getX(), force.getY(), force.getZ(), torque.getX(), torque.getY(), torque.getZ()};
93  Sample s(6, values);
94  AddSampleToHistory(s);
95  }
96 }
