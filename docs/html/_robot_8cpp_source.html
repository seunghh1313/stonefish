172 void Robot::AddLinkSensor(LinkSensor* s, const std::string& monitoredLinkName, const Transform& origin)
173 {
174  SolidEntity* link = getLink(monitoredLinkName);
175  if(link != nullptr)
176  {
177  s->AttachToSolid(link, origin);
178  sensors.push_back(s);
179  }
180  else
181  cCritical("Link '%s' doesn't exist. Sensor '%s' cannot be attached!", monitoredLinkName.c_str(), s->getName().c_str());
182 }

220 void Robot::AddToSimulation(SimulationManager* sm, const Transform& origin)
221 {
222  for(size_t i=0; i<sensors.size(); ++i)
223  sm->AddSensor(sensors[i]);
224  for(size_t i=0; i<actuators.size(); ++i)
225  sm->AddActuator(actuators[i]);
226  for(size_t i=0; i<comms.size(); ++i)
227  sm->AddComm(comms[i]);
228 }
