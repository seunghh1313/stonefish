36 Polyhedron::Polyhedron(std::string uniqueName, BodyPhysicsSettings phy,
37  std::string graphicsFilename, Scalar graphicsScale, const Transform& graphicsOrigin,
38  std::string physicsFilename, Scalar physicsScale, const Transform& physicsOrigin,
39  std::string material, std::string look, Scalar thickness, GeometryApproxType approx)
40  : SolidEntity(uniqueName, phy, material, look, thickness)
41 {
42  //1.Load geometry from file
43  graMesh = OpenGLContent::LoadMesh(graphicsFilename, graphicsScale, false);
44  T_O2G = graphicsOrigin;
45 
46  if(physicsFilename != "")
47  {
48  phyMesh = OpenGLContent::LoadMesh(physicsFilename, physicsScale, false);
49  T_O2C = physicsOrigin;
50  }
51  else
52  {
53  phyMesh = graMesh;
54  T_O2C = T_O2G;
55  }
56 
57  OpenGLContent::Refine(phyMesh, 3.f);
58 
59  //2. Compute physical properties
60  Vector3 CG;
61  Matrix3 Irot;
62  ComputePhysicalProperties(phyMesh, thickness, mat.density, mass, CG, volume, Ipri, Irot);
63  T_CG2C.setOrigin(-CG); //Set CG position
64  T_CG2C = Transform(Irot, Vector3(0,0,0)).inverse() * T_CG2C; //Align CG frame to principal axes of inertia
65 
66  //3.Calculate equivalent ellipsoid for hydrodynamic force computation
67  ComputeFluidDynamicsApprox(approx);
68 
69  //4. Compute missing transformations
70  T_CG2O = T_CG2C * T_O2C.inverse();
71  T_CG2G = T_CG2O * T_O2G;
72  T_O2H = T_CG2O.inverse() * T_CG2H;
73  P_CB = Vector3(0,0,0);
74 }

97 btCollisionShape* Polyhedron::BuildCollisionShape()
98 {

100  for(size_t i=0; i<phyMesh->getNumOfVertices(); ++i)
101  {
102  glm::vec3 pos = phyMesh->getVertexPos(i);
103  Vector3 v(pos.x, pos.y, pos.z);
104  convex->addPoint(v);
105  }
106  //convex->optimizeConvexHull();
