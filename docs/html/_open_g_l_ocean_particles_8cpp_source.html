59 void OpenGLOceanParticles::Create(glm::vec3 eyePos)
60 {
61  lastEyePos = eyePos;
62  //Create particles randomly (uniformly) distributed inside a sphere
63  glBindBuffer(GL_SHADER_STORAGE_BUFFER, particlePosSSBO);
64  glm::vec4* positions = (glm::vec4*)glMapBufferRange(GL_SHADER_STORAGE_BUFFER, 0, sizeof(glm::vec4) * nParticles, GL_MAP_WRITE_BIT | GL_MAP_INVALIDATE_BUFFER_BIT);
65  for(GLuint i=0; i<nParticles; ++i)
66  {
67  GLfloat r = cbrtf(uniformd(generator)) * range; //cbrtf for uniform distribution in sphere volume
68  *(positions++) = glm::vec4(r * glm::normalize(glm::vec3(normald(generator), normald(generator), normald(generator))) + eyePos, uniformd(generator)*0.005f + 0.002f);
69  }
70  glUnmapBuffer(GL_SHADER_STORAGE_BUFFER);
71 
72  glBindBuffer(GL_SHADER_STORAGE_BUFFER, particleVelSSBO);
73  glm::vec4* velocities = (glm::vec4*)glMapBufferRange(GL_SHADER_STORAGE_BUFFER, 0, sizeof(glm::vec4) * nParticles, GL_MAP_WRITE_BIT | GL_MAP_INVALIDATE_BUFFER_BIT);
74  memset(velocities, 0, sizeof(glm::vec4) * nParticles);
75  glUnmapBuffer(GL_SHADER_STORAGE_BUFFER);
76  glBindBuffer(GL_SHADER_STORAGE_BUFFER, 0);
77  initialised = true;
78 }

80 void OpenGLOceanParticles::Update(OpenGLCamera* cam, GLfloat dt)
81 {
82  glm::vec3 eyePos = cam->GetEyePosition();
83  lastEyePos = eyePos;
84 
85  //Check if ever updated
86  if(!initialised)
87  {
88  Create(eyePos);
89  return;
90  }
91 
92  //Move particles
93  updateShader->Use();
94  updateShader->SetUniform("dt", dt);
95  updateShader->SetUniform("numParticles", nParticles);
96  updateShader->SetUniform("eyePos", eyePos);
97  updateShader->SetUniform("R", range);
98  OpenGLState::BindTexture(TEX_MAT_ALBEDO, GL_TEXTURE_3D, noiseTexture);
99  glBindBufferBase(GL_SHADER_STORAGE_BUFFER, SSBO_PARTICLE_POS, particlePosSSBO);
100  glBindBufferBase(GL_SHADER_STORAGE_BUFFER, SSBO_PARTICLE_VEL, particleVelSSBO);
101  glDispatchCompute((GLuint)ceil(nParticles/256.0), 1, 1);
102  OpenGLState::UnbindTexture(TEX_MAT_ALBEDO);
103  OpenGLState::UseProgram(0);
104  glMemoryBarrier(GL_SHADER_STORAGE_BARRIER_BIT);
105 }

107 void OpenGLOceanParticles::Draw(OpenGLCamera* cam, OpenGLOcean* glOcn)
108 {
109  glm::mat4 MV = cam->GetViewMatrix();
110  renderShader->Use();
111  renderShader->SetUniform("MV", MV);
112  renderShader->SetUniform("iMV", glm::inverse(MV));
113  renderShader->SetUniform("P", cam->GetProjectionMatrix());
114  renderShader->SetUniform("FC", cam->GetLogDepthConstant());
115  renderShader->SetUniform("eyePos", cam->GetEyePosition());
116  renderShader->SetUniform("viewDir", cam->GetLookingDirection());
117  renderShader->SetUniform("cWater", glOcn->getLightAttenuation());
118  renderShader->SetUniform("bWater", glOcn->getLightScattering());
119  glBindBufferBase(GL_SHADER_STORAGE_BUFFER, SSBO_PARTICLE_POS, particlePosSSBO);
120  OpenGLState::BindTexture(TEX_MAT_ALBEDO, GL_TEXTURE_2D, flakeTexture);
121  OpenGLState::EnableBlend();
122  glBlendFunc(GL_ONE, GL_ONE);
123  OpenGLState::BindVertexArray(particleVAO);
124  glDrawElements(GL_TRIANGLES, nParticles * 6, GL_UNSIGNED_INT, 0);
125  OpenGLState::BindVertexArray(0);
126  OpenGLState::DisableBlend();
127  OpenGLState::UnbindTexture(TEX_MAT_ALBEDO);
128  OpenGLState::UseProgram(0);
129 }

131 void OpenGLOceanParticles::Init()
132 {
133  //Load shaders
134  std::vector<GLuint> precompiled;
135  precompiled.push_back(OpenGLAtmosphere::getAtmosphereAPI());
136 
137  std::vector<GLSLSource> sources;
138  sources.push_back(GLSLSource(GL_COMPUTE_SHADER, "oceanParticle.comp"));
139  updateShader = new GLSLShader(sources);
140  updateShader->AddUniform("dt", ParameterType::FLOAT);
141  updateShader->AddUniform("numParticles", ParameterType::UINT);
142  updateShader->AddUniform("eyePos", ParameterType::VEC3);
143  updateShader->AddUniform("R", ParameterType::FLOAT);
144  updateShader->AddUniform("texNoise", ParameterType::INT);
145  updateShader->AddUniform("invNoiseSize", ParameterType::FLOAT);
146  updateShader->BindUniformBlock("OceanCurrents", UBO_OCEAN_CURRENTS);
147  updateShader->BindShaderStorageBlock("Positions", SSBO_PARTICLE_POS);
148  updateShader->BindShaderStorageBlock("Velocities", SSBO_PARTICLE_VEL);
149 
150  GLuint noiseSize = 16;
151  updateShader->Use();
152  updateShader->SetUniform("texNoise", TEX_MAT_ALBEDO);
153  updateShader->SetUniform("invNoiseSize", 1.f/(GLfloat)noiseSize);
154  OpenGLState::UseProgram(0);
155 
156  sources.clear();
157  sources.push_back(GLSLSource(GL_VERTEX_SHADER, "particle.vert"));
158  sources.push_back(GLSLSource(GL_FRAGMENT_SHADER, "lightingNoShadow.frag"));
159  sources.push_back(GLSLSource(GL_FRAGMENT_SHADER, "oceanOptics.frag"));
160  sources.push_back(GLSLSource(GL_FRAGMENT_SHADER, "oceanSurfaceFlat.glsl"));
161  sources.push_back(GLSLSource(GL_FRAGMENT_SHADER, "oceanParticle.frag"));
162  sources.push_back(GLSLSource(GL_FRAGMENT_SHADER, "materialUUv.frag"));
163 
164  renderShader = new GLSLShader(sources, precompiled);
165  renderShader->AddUniform("MV", ParameterType::MAT4);
166  renderShader->AddUniform("iMV", ParameterType::MAT4);
167  renderShader->AddUniform("P", ParameterType::MAT4);
168  renderShader->AddUniform("FC", ParameterType::FLOAT);
169  renderShader->AddUniform("eyePos", ParameterType::VEC3);
170  renderShader->AddUniform("viewDir", ParameterType::VEC3);
171  renderShader->AddUniform("color", ParameterType::VEC4);
172  renderShader->AddUniform("texAlbedo", ParameterType::INT);
173  renderShader->AddUniform("enableAlbedoTex", ParameterType::BOOLEAN);
174  renderShader->AddUniform("enableNormalTex", ParameterType::BOOLEAN);
175  renderShader->AddUniform("reflectivity", ParameterType::FLOAT);
176  renderShader->AddUniform("cWater", ParameterType::VEC3);
177  renderShader->AddUniform("bWater", ParameterType::VEC3);
178  renderShader->AddUniform("transmittance_texture", ParameterType::INT);
179  renderShader->AddUniform("scattering_texture", ParameterType::INT);
180  renderShader->AddUniform("irradiance_texture", ParameterType::INT);
181  renderShader->BindUniformBlock("SunSky", UBO_SUNSKY);
182  renderShader->BindUniformBlock("Lights", UBO_LIGHTS);
183  renderShader->BindShaderStorageBlock("Positions", SSBO_PARTICLE_POS);
184 
185  renderShader->Use();
186  renderShader->SetUniform("texAlbedo", TEX_MAT_ALBEDO);
187  renderShader->SetUniform("enableAlbedoTex", true);
188  renderShader->SetUniform("enableNormalTex", false);
189  renderShader->SetUniform("transmittance_texture", TEX_ATM_TRANSMITTANCE);
190  renderShader->SetUniform("scattering_texture", TEX_ATM_SCATTERING);
191  renderShader->SetUniform("irradiance_texture", TEX_ATM_IRRADIANCE);
192  renderShader->SetUniform("reflectivity", 0.f);
193  renderShader->SetUniform("color", glm::vec4(0.f,0.f,0.f,0.3f));
194  OpenGLState::UseProgram(0);
195 
196  //Load textures
197  flakeTexture = OpenGLContent::LoadInternalTexture("flake.png", true, 0.f);
198 
199  unsigned int seed = (unsigned int)GetTimeInMicroseconds();
200  std::mt19937 generator(seed);
201  std::uniform_int_distribution<int8_t> dist(-127,127);
202  glm::uvec3 noiseSize3(noiseSize, noiseSize, noiseSize);
203  int8_t* noiseData = new int8_t[noiseSize3.x * noiseSize3.y * noiseSize3.z * 4];
204  int8_t *ptr = noiseData;
205  for(unsigned int z=0; z<noiseSize3.z; ++z)
206  for(unsigned int y=0; y<noiseSize3.y; ++y)
207  for(unsigned int x=0; x<noiseSize3.x; ++x)
208  {
209  *ptr++ = dist(generator);
210  *ptr++ = dist(generator);
211  *ptr++ = dist(generator);
212  *ptr++ = dist(generator);
213  }
214  noiseTexture = OpenGLContent::GenerateTexture(GL_TEXTURE_3D, noiseSize3,
215  GL_RGBA8_SNORM, GL_RGBA, GL_BYTE, noiseData, FilteringMode::BILINEAR, true);
216  delete [] noiseData;
217 }
