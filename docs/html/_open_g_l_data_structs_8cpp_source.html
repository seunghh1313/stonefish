34 GLfloat Color::bbSpectrum(GLfloat wavelength, GLfloat temperature)
35 {
36  double wlm = wavelength * 1e-9; /* Wavelength in meters */
37  return (3.74183e-16 * pow(wlm, -5.0)) / (exp(1.4388e-2 / (wlm * temperature)) - 1.0);
38 }

89 void Color::xyzToRGB(GLfloat x, GLfloat y, GLfloat z, GLfloat& r, GLfloat& g, GLfloat& b, ColorSystem cs)
90 {
91  GLfloat xr, yr, zr, xg, yg, zg, xb, yb, zb;
92  GLfloat xw, yw, zw;
93  GLfloat rx, ry, rz, gx, gy, gz, bx, by, bz;
94  GLfloat rw, gw, bw;
95 
96  xr = cs.xRed; yr = cs.yRed; zr = 1 - (xr + yr);
97  xg = cs.xGreen; yg = cs.yGreen; zg = 1 - (xg + yg);
98  xb = cs.xBlue; yb = cs.yBlue; zb = 1 - (xb + yb);
99 
100  xw = cs.xWhite; yw = cs.yWhite; zw = 1 - (xw + yw);
101 
102  /* xyz -> rgb matrix, before scaling to white. */
103  rx = (yg * zb) - (yb * zg); ry = (xb * zg) - (xg * zb); rz = (xg * yb) - (xb * yg);
104  gx = (yb * zr) - (yr * zb); gy = (xr * zb) - (xb * zr); gz = (xb * yr) - (xr * yb);
105  bx = (yr * zg) - (yg * zr); by = (xg * zr) - (xr * zg); bz = (xr * yg) - (xg * yr);
106 
107  /* White scaling factors.
108  Dividing by yw scales the white luminance to unity, as conventional. */
109  rw = ((rx * xw) + (ry * yw) + (rz * zw)) / yw;
110  gw = ((gx * xw) + (gy * yw) + (gz * zw)) / yw;
111  bw = ((bx * xw) + (by * yw) + (bz * zw)) / yw;
112 
113  /* xyz -> rgb matrix, correctly scaled to white. */
114  rx = rx / rw; ry = ry / rw; rz = rz / rw;
115  gx = gx / gw; gy = gy / gw; gz = gz / gw;
116  bx = bx / bw; by = by / bw; bz = bz / bw;
117 
118  /* rgb of the desired point */
119  r = (rx * x) + (ry * y) + (rz * z);
120  g = (gx * x) + (gy * y) + (gz * z);
121  b = (bx * x) + (by * y) + (bz * z);
122 
123  /* constrain rgb */
124  GLfloat w;
125  w = (0 < r) ? 0 : r;
126  w = (w < g) ? w : g;
127  w = (w < b) ? w : b;
128  w = -w;
129 
130  /* add just enough white to make r, g, b all positive. */
131  if(w > 0)
132  {
133  r += w;
134  g += w;
135  b += w;
136  }
137 
138  /* normalize rgb */
139  GLfloat maximum = Max(r, Max(g,b));
140  if(maximum > 1.f)
141  {
142  r /= maximum;
143  g /= maximum;
144  b /= maximum;
145  }
146 }
