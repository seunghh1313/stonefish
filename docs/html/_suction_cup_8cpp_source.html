71 void SuctionCup::Engage(SimulationManager* sm)
72 {
73  if(attach != nullptr && spring == nullptr && pump)
74  {
75  btDispatcher* dispatcher = sm->getDynamicsWorld()->getDispatcher();
76  int numManifolds = dispatcher->getNumManifolds();
77  for(int i=0; i<numManifolds; ++i)
78  {
79  btPersistentManifold* contactManifold = dispatcher->getManifoldByIndexInternal(i);
80  if(contactManifold->getNumContacts() == 0)
81  continue;
82 
83  btCollisionObject* coA = (btCollisionObject*)contactManifold->getBody0();
84  btCollisionObject* coB = (btCollisionObject*)contactManifold->getBody1();
85  Entity* entA = (Entity*)coA->getUserPointer();
86  Entity* entB = (Entity*)coB->getUserPointer();
87 
88  if(entA == attach && entB->getType() == EntityType::SOLID)
89  {
90  Transform springFrame = ((SolidEntity*)entA)->getCG2CTransform();
91  springFrame.setOrigin(contactManifold->getContactPoint(0).getPositionWorldOnA());
92  spring = new SpringJoint(getName() + "/Spring", (SolidEntity*)entA, (SolidEntity*)entB, springFrame,
93  Vector3(10,10,10), Vector3(10,10,10), Vector3(0.5,0.5,0.5), Vector3(0.5,0.5,0.5));
94  }
95  else if(entB == attach && entA->getType() == EntityType::SOLID)
96  {
97  Transform springFrame = ((SolidEntity*)entB)->getCG2CTransform();
98  springFrame.setOrigin(contactManifold->getContactPoint(0).getPositionWorldOnB());
99  spring = new SpringJoint(getName() + "/Spring", (SolidEntity*)entB, (SolidEntity*)entA, springFrame,
100  Vector3(10,10,10), Vector3(10,10,10), Vector3(0.5,0.5,0.5), Vector3(0.5,0.5,0.5));
101  }
102 
103  if(spring != nullptr)
104  {
105  contactManifold->clearManifold();
106  sm->AddJoint(spring);
107  break;
108  }
109  }
110  }
111  else if(spring != nullptr && !pump)
112  {
113  sm->RemoveJoint(spring);
114  spring = nullptr;
115  }
116 }
117 
118 }
