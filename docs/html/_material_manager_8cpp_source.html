51 std::string MaterialManager::CreateMaterial(const std::string& uniqueName, Scalar density, Scalar restitution, Scalar magnetic)
52 {
53  //Create and add new material
54  Material mat;
55  mat.name = materialNameManager.AddName(uniqueName);
56  mat.density = density;
57  mat.restitution = restitution;
58  mat.magnetic = magnetic;
59  materials.push_back(mat);
60 
61  cInfo("Material %s (%d) created.", mat.name.c_str(), materials.size()-1);
62 
63  //Set initial friction coefficients
64  Friction f;
65  f.fStatic = Scalar(1);
66  f.fDynamic = Scalar(1);
67 
68  MaterialPair p;
69  p.mat1Id = (int)materials.size()-1;
70 
71  for(unsigned int i=0; i < materials.size(); ++i)
72  {
73  p.mat2Id = i;
74  interactions.insert({p,f});
75  }
76 
77  //Display interactions
78  //for(std::pair<MaterialPair, Friction> element : interactions)
79  // std::cout << "(" << element.first.mat1Id << "," << element.first.mat2Id << ")" << std::endl;
80 
81  return mat.name;
82 }

96 bool MaterialManager::SetMaterialsInteraction(const std::string& firstMaterialName, const std::string& secondMaterialName, Scalar staticFricCoeff, Scalar dynamicFricCoeff)
97 {
98  MaterialPair p;
99  p.mat1Id = getMaterialIndex(firstMaterialName);
100  p.mat2Id = getMaterialIndex(secondMaterialName);
101 
102  Friction f;
103  f.fStatic = staticFricCoeff;
104  f.fDynamic = dynamicFricCoeff;
105 
106  try
107  {
108  interactions.at(p) = f;
109  return true;
110  }
111  catch(const std::out_of_range& e)
112  {
113  cError("Material pair (%s,%s) not found!", firstMaterialName.c_str(), secondMaterialName.c_str());
114  return false;
115  }
116 }
