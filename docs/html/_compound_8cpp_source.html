158 void Compound::RecalculatePhysicalProperties()
159 {
160  //Calculate rigid body properties
161  /*
162  1. Calculate compound mass and compound CG (sum of location - local * m / M)
163  2. Calculate inertia of part in global frame and compound CG (rotate and translate inertia tensor) 3x3
164  and calculate compound inertia 3x3 (sum of parts inertia)
165  3. Calculate primary moments of inertia
166  4. Find primary axes of inertia
167  5. Rotate frame to match primary axes and move to CG
168  */
169 
170  //1. Calculate compound mass, CG and CB
171  Vector3 compoundCG(0,0,0); //In compound body origin frame
172  Vector3 compoundCB(0,0,0); //In compound body origin frame
173  Scalar compoundMass = 0;
174  Scalar compoundAugmentedMass = 0;
175  Scalar compoundVolume = 0;
176 
177  T_CG2O = T_CG2C = T_CG2G = Transform::getIdentity();
178  P_CB = Vector3(0,0,0);
179 
180  for(size_t i=0; i<parts.size(); ++i)
181  {
182  //Mechanical parameters
183  compoundMass += parts[i].solid->getMass();
184  compoundAugmentedMass += parts[i].isExternal ? parts[i].solid->getAugmentedMass() : parts[i].solid->getMass();
185  compoundCG += (parts[i].origin * parts[i].solid->getCG2OTransform().inverse()).getOrigin() * parts[i].solid->getMass();
186 
187  if(parts[i].solid->isBuoyant())
188  {
189  compoundVolume += parts[i].solid->getVolume();
190  compoundCB += parts[i].origin * parts[i].solid->getCG2OTransform().inverse() * parts[i].solid->getCB() * parts[i].solid->getVolume();
191  }
192  }
193 
194  //Set transform origin
195  compoundCG /= compoundMass;
196  if(compoundVolume > Scalar(0)) compoundCB /= compoundVolume;
197  T_CG2O.setOrigin(-compoundCG);
198 
199  //2. Calculate compound inertia matrix
200  Matrix3 I = Matrix3(0,0,0,0,0,0,0,0,0);
201 
202  for(unsigned int i=0; i<parts.size(); ++i)
203  {
204  //Calculate inertia matrix 3x3 of solid in the compound body origin frame and transform to CB
205  Vector3 solidPriInertia = parts[i].isExternal ? parts[i].solid->getAugmentedInertia() : parts[i].solid->getInertia();
206  Matrix3 solidInertia = Matrix3(solidPriInertia.x(), 0, 0, 0, solidPriInertia.y(), 0, 0, 0, solidPriInertia.z());
207 
208  //Rotate inertia tensor from part CG to compound CG
209  Transform compToPart = T_CG2O * parts[i].origin * parts[i].solid->getCG2OTransform().inverse();
210  solidInertia = compToPart.getBasis() * solidInertia * compToPart.getBasis().transpose();
211 
212  //Translate inertia tensor from part CG to compound CG
213  Vector3 t = compToPart.getOrigin();
214  Scalar m = parts[i].isExternal ? parts[i].solid->getAugmentedMass() : parts[i].solid->getMass();
215  solidInertia += Matrix3(t.y()*t.y()+t.z()*t.z(), -t.x()*t.y(), -t.x()*t.z(),
216  -t.y()*t.x(), t.x()*t.x()+t.z()*t.z(), -t.y()*t.z(),
217  -t.z()*t.x(), -t.z()*t.y(), t.x()*t.x()+t.y()*t.y()).scaled(Vector3(m, m, m));
218 
219  //Accumulate inertia tensor
220  I += solidInertia;
221  }
222 
223  //3. Find compound moments of inertia
224  Vector3 compoundPriInertia(I.getRow(0).getX(), I.getRow(1).getY(), I.getRow(2).getZ());
225 
226  //Check if inertia matrix is not diagonal
227  if(!(btFuzzyZero(I.getRow(0).getY()) && btFuzzyZero(I.getRow(0).getZ())
228  && btFuzzyZero(I.getRow(1).getX()) && btFuzzyZero(I.getRow(1).getZ())
229  && btFuzzyZero(I.getRow(2).getX()) && btFuzzyZero(I.getRow(2).getY())))
230  {
231  //3.1. Calculate principal moments of inertia
232  Scalar T = I[0][0] + I[1][1] + I[2][2]; //Ixx + Iyy + Izz
233  Scalar II = I[0][0]*I[1][1] + I[0][0]*I[2][2] + I[1][1]*I[2][2] - I[0][1]*I[0][1] - I[0][2]*I[0][2] - I[1][2]*I[1][2]; //Ixx Iyy + Ixx Izz + Iyy Izz - Ixy^2 - Ixz^2 - Iyz^2
234  Scalar U = btSqrt(T*T-Scalar(3.)*II)/Scalar(3.);
235  Scalar theta = btAcos((-Scalar(2.)*T*T*T + Scalar(9.)*T*II - Scalar(27.)*I.determinant())/(Scalar(54.)*U*U*U));
236  Scalar A = T/Scalar(3.) - Scalar(2.)*U*btCos(theta/Scalar(3.));
237  Scalar B = T/Scalar(3.) - Scalar(2.)*U*btCos(theta/Scalar(3.) - Scalar(2.)*M_PI/Scalar(3.));
238  Scalar C = T/Scalar(3.) - Scalar(2.)*U*btCos(theta/Scalar(3.) + Scalar(2.)*M_PI/Scalar(3.));
239  compoundPriInertia = Vector3(A, B, C);
240 
241  //3.2. Calculate principal axes of inertia
242  Matrix3 L;
243  Vector3 axis1,axis2,axis3;
244  axis1 = FindInertialAxis(I, A);
245  axis2 = FindInertialAxis(I, B);
246  axis3 = axis1.cross(axis2);
247  axis2 = axis3.cross(axis1);
248 
249  //3.3. Rotate body so that principal axes are parallel to (x,y,z) system
250  Matrix3 rotMat(axis1[0],axis2[0],axis3[0], axis1[1],axis2[1],axis3[1], axis1[2],axis2[2],axis3[2]);
251  T_CG2O = Transform(rotMat, Vector3(0,0,0)).inverse() * T_CG2O;
252  }
253 
254  T_CG2C = T_CG2G = T_CG2O;
255 
256  //Move CB to compound CG frame
257  P_CB = T_CG2O * compoundCB;
258 
259  mass = compoundMass;
260  aMass.setX(compoundAugmentedMass - compoundMass);
261  aMass.setY(compoundAugmentedMass - compoundMass);
262  aMass.setZ(compoundAugmentedMass - compoundMass);
263  volume = compoundVolume;
264  Ipri = compoundPriInertia;
265 }

284 void Compound::ComputeHydrodynamicForces(HydrodynamicsSettings settings, Ocean* ocn)
285 {
286  if(phy.mode != BodyPhysicsMode::FLOATING && phy.mode != BodyPhysicsMode::SUBMERGED) return;
287 
288  submerged.points.clear();
289 
290  BodyFluidPosition bf = CheckBodyFluidPosition(ocn);
291 
292  //If completely outside fluid just set all torques and forces to 0
293  if(bf == BodyFluidPosition::OUTSIDE)
294  {
295  Fb.setZero();
296  Tb.setZero();
297  Fdl.setZero();
298  Tdl.setZero();
299  Fdq.setZero();
300  Tdq.setZero();
301  Fds.setZero();
302  Tds.setZero();
303  return;
304  }
305 
306  if(bf == BodyFluidPosition::INSIDE)
307  {
308  //Compute buoyancy based on CB position
309  if(isBuoyant())
310  {
311  Fb = -volume*ocn->getLiquid().density * SimulationApp::getApp()->getSimulationManager()->getGravity();
312  Tb = (getCGTransform() * P_CB - getCGTransform().getOrigin()).cross(Fb);
313  }
314 
315  if(settings.dampingForces)
316  {
317  //Set zero
318  Fdl.setZero();
319  Tdl.setZero();
320  Fdq.setZero();
321  Tdq.setZero();
322  Fds.setZero();
323  Tds.setZero();
324 
325  //Get velocity data
326  Vector3 v = getLinearVelocity();
327  Vector3 omega = getAngularVelocity();
328 
329  //Create temporary vectors for summing
330  Vector3 Fdlp(0,0,0);
331  Vector3 Tdlp(0,0,0);
332  Vector3 Fdqp(0,0,0);
333  Vector3 Tdqp(0,0,0);
334  Vector3 Fdsp(0,0,0);
335  Vector3 Tdsp(0,0,0);
336 
337  for(size_t i=0; i<parts.size(); ++i) //Go through all parts
338  if(parts[i].isExternal) //Compute drag only for external parts
339  {
340  Transform T_C_part = getOTransform() * parts[i].origin * parts[i].solid->getO2CTransform();
341  ComputeHydrodynamicForcesSubmerged(parts[i].solid->getPhysicsMesh(), ocn, getCGTransform(), T_C_part, v, omega, Fdlp, Tdlp, Fdqp, Tdqp, Fdsp, Tdsp);
342  parts[i].solid->CorrectHydrodynamicForces(ocn, Fdlp, Tdlp, Fdqp, Tdqp, Fdsp, Tdsp);
343  Fdl += Fdlp;
344  Tdl += Tdlp;
345  Fdq += Fdqp;
346  Tdq += Tdqp;
347  Fds += Fdsp;
348  Tds += Tdsp;
349  }
350  }
351  }
352  else //CROSSING FLUID SURFACE (compound body but not necessarily all parts!)
353  {
354  if(settings.reallisticBuoyancy || settings.dampingForces)
355  {
356  //Clear forces that will be recomputed
357  if(settings.reallisticBuoyancy)
358  {
359  Fb.setZero();
360  Tb.setZero();
361  }
362 
363  if(settings.dampingForces)
364  {
365  Fdl.setZero();
366  Tdl.setZero();
367  Fdq.setZero();
368  Tdq.setZero();
369  Fds.setZero();
370  Tds.setZero();
371  }
372 
373  //Get velocity data
374  Vector3 v = getLinearVelocity();
375  Vector3 omega = getAngularVelocity();
376 
377  //Create temporary vectors for summing
378  Vector3 Fbp(0,0,0);
379  Vector3 Tbp(0,0,0);
380 
381  Vector3 Fdlp(0,0,0);
382  Vector3 Tdlp(0,0,0);
383  Vector3 Fdqp(0,0,0);
384  Vector3 Tdqp(0,0,0);
385  Vector3 Fdsp(0,0,0);
386  Vector3 Tdsp(0,0,0);
387 
388  for(size_t i=0; i<parts.size(); ++i) //Loop through all parts
389  {
390  Transform T_C_part = getOTransform() * parts[i].origin * parts[i].solid->getO2CTransform();
391  HydrodynamicsSettings pSettings = settings;
392  pSettings.reallisticBuoyancy &= parts[i].solid->isBuoyant();
393 
394  if(parts[i].isExternal) //Compute buoyancy and drag
395  {
396  ComputeHydrodynamicForcesSurface(pSettings, parts[i].solid->getPhysicsMesh(), ocn, getCGTransform(), T_C_part, v, omega, Fbp, Tbp, Fdlp, Tdlp, Fdqp, Tdqp, Fdsp, Tdsp, submerged);
397  parts[i].solid->CorrectHydrodynamicForces(ocn, Fdlp, Tdlp, Fdqp, Tdqp, Fdsp, Tdsp);
398  Fb += Fbp;
399  Tb += Tbp;
400  Fdl += Fdlp;
401  Tdl += Tdlp;
402  Fdq += Fdqp;
403  Tdq += Tdqp;
404  Fds += Fdsp;
405  Tds += Tdsp;
406  }
407  else if(pSettings.reallisticBuoyancy) //Compute only buoyancy
408  {
409  pSettings.dampingForces = false;
410  ComputeHydrodynamicForcesSurface(pSettings, parts[i].solid->getPhysicsMesh(), ocn, getCGTransform(), T_C_part, v, omega, Fbp, Tbp, Fdlp, Tdlp, Fdqp, Tdqp, Fdsp, Tdsp, submerged);
411  Fb += Fbp;
412  Tb += Tbp;
413  }
414  }
415  }
416  }
417 }

452 std::vector<Renderable> Compound::Render(size_t partId)
453 {
454  std::vector<Renderable> items(0);
455  Transform oCompoundTrans = getOTransform();
456 
457  try
458  {
459  Renderable item;
460 
461  if((parts.at(partId).isExternal && !displayInternals)
462  || (!parts.at(partId).isExternal && displayInternals))
463  {
464  item.type = RenderableType::SOLID;
465  item.materialName = parts.at(partId).solid->getMaterial().name;
466 
467  if(dm == DisplayMode::GRAPHICAL)
468  {
469  Transform oTrans = oCompoundTrans * parts.at(partId).origin * parts.at(partId).solid->getO2GTransform();
470  item.objectId = parts.at(partId).solid->getGraphicalObject();
471  item.lookId = parts.at(partId).solid->getLook();
472  item.model = glMatrixFromTransform(oTrans);
473  items.push_back(item);
474  }
475  else if(dm == DisplayMode::PHYSICAL)
476  {
477  Transform oTrans = oCompoundTrans * parts.at(partId).origin * parts.at(partId).solid->getO2CTransform();
478  item.objectId = parts.at(partId).solid->getPhysicalObject();
479  item.lookId = -1;
480  item.model = glMatrixFromTransform(oTrans);
481  items.push_back(item);
482  }
483  }
484 
485 #ifndef DEBUG_HYDRO
486  GeometryApproxType atype;
487  std::vector<Scalar> aparams;
488  parts.at(partId).solid->getGeometryApprox(atype, aparams);
489  item.model = glMatrixFromTransform(oCompoundTrans * parts.at(partId).origin * parts.at(partId).solid->getO2HTransform());
490 
491  switch(atype)
492  {
493  case GeometryApproxType::AUTO:
494  break;
495 
496  case GeometryApproxType::SPHERE:
497  item.type = RenderableType::HYDRO_ELLIPSOID;
498  item.points.push_back(glm::vec3((GLfloat)aparams[0], (GLfloat)aparams[0], (GLfloat)aparams[0]));
499  items.push_back(item);
500  break;
501 
502  case GeometryApproxType::CYLINDER:
503  item.type = RenderableType::HYDRO_CYLINDER;
504  item.points.push_back(glm::vec3((GLfloat)aparams[0], (GLfloat)aparams[0], (GLfloat)aparams[1]));
505  items.push_back(item);
506  break;
507 
508  case GeometryApproxType::ELLIPSOID:
509  item.type = RenderableType::HYDRO_ELLIPSOID;
510  item.points.push_back(glm::vec3((GLfloat)aparams[0], (GLfloat)aparams[1], (GLfloat)aparams[2]));
511  items.push_back(item);
512  break;
513  }
514 #endif
515  }
516  catch(const std::exception& e)
517  {
518  //Error finding part id
519  }
520 
521  return items;
522 }
