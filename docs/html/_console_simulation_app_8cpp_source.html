95 int ConsoleSimulationApp::RunSimulation(void* data)
96 {
97  ConsoleSimulationThreadData* stdata = (ConsoleSimulationThreadData*)data;
98  SimulationManager* sim = stdata->app->getSimulationManager();
99 
100  int maxThreads = std::max(omp_get_max_threads()/2, 1);
101  omp_set_num_threads(maxThreads);
102 
103  while(stdata->app->isRunning())
104  sim->AdvanceSimulation();
105 
106  return 0;
107 }
