148 void OpenGLSonar::Init()
149 {
150  sonarInputShader[0] = new GLSLShader("sonarInput.frag", "sonarInput.vert");
151  sonarInputShader[0]->AddUniform("MVP", ParameterType::MAT4);
152  sonarInputShader[0]->AddUniform("M", ParameterType::MAT4);
153  sonarInputShader[0]->AddUniform("N", ParameterType::MAT3);
154  sonarInputShader[0]->AddUniform("eyePos", ParameterType::VEC3);
155  sonarInputShader[0]->AddUniform("restitution", ParameterType::FLOAT);
156 
157  sonarInputShader[1] = new GLSLShader("sonarInputUv.frag", "sonarInputUv.vert");
158  sonarInputShader[1]->AddUniform("MVP", ParameterType::MAT4);
159  sonarInputShader[1]->AddUniform("M", ParameterType::MAT4);
160  sonarInputShader[1]->AddUniform("N", ParameterType::MAT3);
161  sonarInputShader[1]->AddUniform("eyePos", ParameterType::VEC3);
162  sonarInputShader[1]->AddUniform("restitution", ParameterType::FLOAT);
163  sonarInputShader[1]->AddUniform("texNormal", ParameterType::INT);
164  sonarInputShader[1]->Use();
165  sonarInputShader[1]->SetUniform("texNormal", TEX_MAT_NORMAL);
166  OpenGLState::UseProgram(0);
167 
168  sonarVisualizeShader = new GLSLShader("sonarVisualize.frag", "printer.vert");
169  sonarVisualizeShader->AddUniform("texSonarData", ParameterType::INT);
170  sonarVisualizeShader->AddUniform("colormap", ParameterType::INT);
171 }
