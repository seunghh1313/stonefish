280 int SimulationManager::CheckCollision(const Entity *entA, const Entity *entB)
281 {
282  for(size_t i = 0; i < collisions.size(); ++i)
283  {
284  if((collisions[i].A == entA && collisions[i].B == entB)
285  || (collisions[i].B == entA && collisions[i].A == entB))
286  return (int)i;
287  }
288 
289  return -1;
290 }

292 void SimulationManager::EnableCollision(const Entity* entA, const Entity* entB)
293 {
294  int colId = CheckCollision(entA, entB);
295 
296  if(collisionFilter == CollisionFilteringType::COLLISION_INCLUSIVE && colId == -1)
297  {
298  Collision c;
299  c.A = const_cast<Entity*>(entA);
300  c.B = const_cast<Entity*>(entB);
301  collisions.push_back(c);
302  }
303  else if(collisionFilter == CollisionFilteringType::COLLISION_EXCLUSIVE && colId > -1)
304  {
305  collisions.erase(collisions.begin() + colId);
306  }
307 }

309 void SimulationManager::DisableCollision(const Entity* entA, const Entity* entB)
310 {
311  int colId = CheckCollision(entA, entB);
312  if(collisionFilter == CollisionFilteringType::COLLISION_EXCLUSIVE && colId == -1)
313  {
314  Collision c;
315  c.A = const_cast<Entity*>(entA);
316  c.B = const_cast<Entity*>(entB);
317  collisions.push_back(c);
318  cInfo("Disabling collisions between '%s' and '%s'.", entA->getName().c_str(), entB->getName().c_str());
319  }
320  else if(collisionFilter == CollisionFilteringType::COLLISION_INCLUSIVE && colId > -1)
321  {
322  collisions.erase(collisions.begin() + colId);
323  cInfo("Disabling collisions between '%s' and '%s'.", entA->getName().c_str(), entB->getName().c_str());
324  }
325 }

683 void SimulationManager::InitializeSolver()
684 {
685  dwBroadphase = new btDbvtBroadphase(); //btAxisSweep3(Vector3(-50000.0, -50000.0, -10000.0), Vector3(50000.0, 50000.0, 10000.0));
686  dwCollisionConfig = new btSoftBodyRigidBodyCollisionConfiguration();
687 
688  //Choose collision dispatcher
689  switch(collisionFilter)
690  {
691  case CollisionFilteringType::COLLISION_INCLUSIVE:
692  dwDispatcher = new FilteredCollisionDispatcher(dwCollisionConfig, true);
693  break;
694 
695  case CollisionFilteringType::COLLISION_EXCLUSIVE:
696  dwDispatcher = new FilteredCollisionDispatcher(dwCollisionConfig, false);
697  break;
698  }
699  //dwDispatcher = new btCollisionDispatcher(dwCollisionConfig);
700 
701  //Choose constraint solver
702  if(solver == SolverType::SOLVER_SI)
703  {
704  mbSolver = new btMultiBodyConstraintSolver();
705  }
706  else
707  {
708  btMLCPSolverInterface* mlcp;
709 
710  switch(solver)
711  {
712  default:
713  case SolverType::SOLVER_DANTZIG:
714  mlcp = new btDantzigSolver();
715  break;
716 
717  case SolverType::SOLVER_PGS:
718  mlcp = new btSolveProjectedGaussSeidel();
719  break;
720 
721  case SolverType::SOLVER_LEMKE:
722  mlcp = new btLemkeSolver();
723  //((btLemkeSolver*)mlcp)->m_maxLoops = 10000;
724  break;
725  }
726 
727  mbSolver = new btMultiBodyMLCPConstraintSolver(mlcp); //ResearchConstraintSolver(mlcp);
728  }
729 
730  sbSolver = new btDefaultSoftBodySolver();
731 
732  //Create dynamics world
733  dynamicsWorld = new btSoftMultiBodyDynamicsWorld(dwDispatcher, dwBroadphase, mbSolver, dwCollisionConfig, sbSolver);
734 
735  //Basic configuration

972  return false;

1005 {

1007  if(!icProblemSolved)
1008  return;

1011  uint64_t deltaTime;
1012 
1013  if(currentTime == 0) //Start of simulation
1014  {

1017  return;
1018  }
1019 
1020  uint64_t timeInMicroseconds = getSimulationClock(); //Realtime factor included in clock
1021  deltaTime = timeInMicroseconds - currentTime;
1022  currentTime = timeInMicroseconds;

1025  {
1026  SimulationClockSleep(ssus - deltaTime);
1027  timeInMicroseconds = getSimulationClock();
1028  deltaTime += timeInMicroseconds - currentTime;
1029  currentTime = timeInMicroseconds;
1030  }
1031

1034  perfMon.PhysicsStarted();

1036  perfMon.PhysicsFinished();
1037  SDL_UnlockMutex(simSettingsMutex);
1038 
1039  SDL_LockMutex(simInfoMutex);
1040  Scalar cpuUsageNow = (Scalar)perfMon.getPhysicsTime()/(Scalar)deltaTime * Scalar(100);
1041  Scalar filter(0.001);
1042  cpuUsage = filter * cpuUsageNow + (Scalar(1)-filter) * cpuUsage;

1171  Entity* ent1 = (Entity*)colObj1Wrap->getCollisionObject()->getUserPointer();

1177  cp.m_combinedRollingFriction = Scalar(0.);
1178  cp.m_combinedRestitution = Scalar(0.);

1181

1183  MaterialManager* mm = SimulationApp::getApp()->getSimulationManager()->getMaterialManager();
1184

1186  Vector3 contactVelocity0;
1187  Scalar contactAngularVelocity0;
1188

1190  {
1191  StaticEntity* sent0 = (StaticEntity*)ent0;
1192  mat0 = sent0->getMaterial();

1206  contactAngularVelocity0 = sent0->getAngularVelocity().dot(-cp.m_normalWorldOnB);

1208  else
1209  {
1210  cp.m_combinedFriction = Scalar(0);
1211  cp.m_combinedRollingFriction = Scalar(0);
1212  cp.m_combinedRestitution = Scalar(0);
1213  return true;
1214  }
1215 
1216  Material mat1;
1217  Vector3 contactVelocity1;
1218  Scalar contactAngularVelocity1;
1219 
1220  if(ent1->getType() == EntityType::STATIC)
1221  {
1222  StaticEntity* sent1 = (StaticEntity*)ent1;
1223  mat1 = sent1->getMaterial();
1224  contactVelocity1.setZero();
1225  contactAngularVelocity1 = Scalar(0);
1226  }
1227  else if(ent1->getType() == EntityType::SOLID)
1228  {
1229  SolidEntity* sent1 = (SolidEntity*)ent1;
1230  if(sent1->getSolidType() == SolidType::COMPOUND)
1231  mat1 = ((Compound*)sent1)->getMaterial(((Compound*)sent1)->getPartId(index1));
1232  else
1233  mat1 = sent1->getMaterial();
1234  //Vector3 localPoint1 = sent1->getTransform().getBasis() * cp.m_localPointB;
1235  Vector3 localPoint1 = sent1->getCGTransform().inverse() * cp.getPositionWorldOnB();
1236  contactVelocity1 = sent1->getLinearVelocityInLocalPoint(localPoint1);
1237  contactAngularVelocity1 = sent1->getAngularVelocity().dot(cp.m_normalWorldOnB);
1238  }
1239  else
1240  {
1241  cp.m_combinedFriction = Scalar(0);
1242  cp.m_combinedRollingFriction = Scalar(0);
1243  cp.m_combinedRestitution = Scalar(0);
1244  return true;
1245  }
1246 
1247  //Calculate contact forces
1248  //A. Stribeck friction model
1249  Vector3 relLocalVel = contactVelocity1 - contactVelocity0;
1250  Vector3 normalVel = cp.m_normalWorldOnB * cp.m_normalWorldOnB.dot(relLocalVel);
1251  Vector3 slipVel = relLocalVel - normalVel;
1252  Scalar sigma = 1000;
1253  // f = (static - dynamic)/(sigma * v^2 + 1) + dynamic
1254  Friction f = mm->GetMaterialsInteraction(mat0.name, mat1.name);
1255  cp.m_combinedFriction = (f.fStatic - f.fDynamic)/(sigma * slipVel.length2() + Scalar(1)) + f.fDynamic;
1256 
1257  //Rolling friction not possible to generalize - needs special treatment
1258  cp.m_combinedRollingFriction = Scalar(0);
1259  cp.m_combinedSpinningFriction = Scalar(0);
1260

1262  ContactInfo* cInfo = new ContactInfo();
1263  cInfo->totalAppliedImpulse = Scalar(0);
1264  cInfo->slip = slipVel;
1265  cp.m_userPersistentData = cInfo;
1266

1268  //calculate relative angular velocity
1269  Scalar relAngularVelocity01 = contactAngularVelocity0 - contactAngularVelocity1;
1270  Scalar relAngularVelocity10 = contactAngularVelocity1 - contactAngularVelocity0;
1271

1273  Scalar normalForce = cp.m_appliedImpulse * SimulationApp::getApp()->getSimulationManager()->getStepsPerSecond();
1274  Scalar T = cp.m_combinedFriction * normalForce * 0.002;
1275

1277  if(ent0->getType() == EntityType::SOLID && !btFuzzyZero(relAngularVelocity01))
1278  ((SolidEntity*)ent0)->ApplyTorque(cp.m_normalWorldOnB * relAngularVelocity01/btFabs(relAngularVelocity01) * T);
1279 
1280  if(ent1->getType() == EntityType::SOLID && !btFuzzyZero(relAngularVelocity10))
1281  ((SolidEntity*)ent1)->ApplyTorque(cp.m_normalWorldOnB * relAngularVelocity10/btFabs(relAngularVelocity10) * T);
1282 
1283  //Restitution
1284  cp.m_combinedRestitution = mat0.restitution * mat1.restitution;
1285 
1286  //B. Magnetic attraction (only between magnet and ferromagnetic body, no magnet-magnet support)
1287  if((mat0.magnetic < Scalar(0) && mat1.magnetic > Scalar(0))
1288  || (mat0.magnetic > Scalar(0) && mat1.magnetic < Scalar(0)))
1289  {
1290  Scalar d = btClamped(cp.getDistance(), Scalar(0.0001), BT_LARGE_FLOAT);
1291  Scalar mag = (btFabs(mat0.magnetic) * btFabs(mat1.magnetic))/(d*d)/Scalar(1e4);
1292  btClamp(mag, Scalar(0), Scalar(10000)); //Arbitrary limit of 10kN
1293  Vector3 mForce = cp.m_normalWorldOnB * mag;
1294 
1295  if(ent0->getType() == EntityType::SOLID)
1296  {
1297  SolidEntity* sent0 = (SolidEntity*)ent0;
1298  sent0->ApplyCentralForce(-mForce);
1299  sent0->ApplyTorque((cp.m_positionWorldOnA - sent0->getCGTransform().getOrigin()).cross(-mForce));
1300  }
1301  if(ent1->getType() == EntityType::SOLID)
1302  {
1303  SolidEntity* sent1 = (SolidEntity*)ent1;
1304  sent1->ApplyCentralForce(mForce);
1305  sent1->ApplyTorque((cp.m_positionWorldOnB - sent1->getCGTransform().getOrigin()).cross(mForce));
1306  }
1307

1309  }
1310
