109 void Trigger::Activate(btCollisionObject* co)
110 {
111  if(solids.size() == 0)
112  return;
113 
114  Entity* ent;
115  btRigidBody* rb = btRigidBody::upcast(co);
116  btMultiBodyLinkCollider* mbl = btMultiBodyLinkCollider::upcast(co);
117 
118  if(rb != 0)
119  {
120  if(rb->isStaticOrKinematicObject())
121  return;
122  else
123  ent = (Entity*)rb->getUserPointer();
124  }
125  else if(mbl != 0)
126  {
127  if(mbl->isStaticOrKinematicObject())
128  return;
129  else
130  ent = (Entity*)mbl->getUserPointer();
131  }
132  else
133  return;
134 
135  if(ent->getType() == EntityType::SOLID)
136  {
137  SolidEntity* solid = (SolidEntity*)ent;
138  for(unsigned int i=0; i<solids.size(); ++i)
139  if(solids[i] == solid)
140  {
141  active = true;
142  return;
143  }
144  }
145 }
