35 VariableBuoyancy::VariableBuoyancy(std::string uniqueName, const std::vector<std::string>& volumeMeshPaths, Scalar initialVolume) : LinkActuator(uniqueName)
36 {
37  if(volumeMeshPaths.size() < 2)
38  cCritical("VBS volume definition requires loading at least two meshes - for the full/empty states!");
39 
40  density = Scalar(1000.0);
41  Ocean* ocn;
42  if((ocn = SimulationApp::getApp()->getSimulationManager()->getOcean()) != nullptr)
43  density = ocn->getLiquid().density;
44  gravity = SimulationApp::getApp()->getSimulationManager()->getGravity();
45 
46  for(size_t i=0; i<volumeMeshPaths.size(); ++i)
47  {
48  Mesh* mesh = LoadGeometryFromFile(volumeMeshPaths[i], 1.f);
49  if(mesh == NULL)
50  abort();
51  Vprops.push_back(ComputePhysicalProperties(mesh, Scalar(0), density));
52  delete mesh;
53  }
54  auto volumeCompare = [](MeshProperties& mp1, MeshProperties& mp2) { return mp1.volume < mp2.volume; };
55  std::sort(Vprops.begin(), Vprops.end(), volumeCompare);
56 
57  flowRate = Scalar(0);
58  force = V0();
59  Vmin = Vprops.front().volume;
60  Vmax = Vprops.back().volume;
61 
62  cInfo("VBS created with Vmin: %1.3lf Vmax: %1.3lf", Vmin, Vmax);
63 
64  V = initialVolume > Vmax ? Vmax : (initialVolume < Vmin ? Vmin : initialVolume);
65 
66  Scalar mass;
67  InterpolateVProps(V, mass, CG);
68 }

95 void VariableBuoyancy::InterpolateVProps(Scalar volume, Scalar& m, Vector3& cg)
96 {
97  if(volume <= Vmin)
98  {
99  m = Vprops.front().mass;
100  cg = Vprops.front().CG;
101  }
102  else if(volume >= Vmax)
103  {
104  m = Vprops.back().mass;
105  cg = Vprops.back().CG;
106  }
107  else
108  {
109  m = volume*density;
110 
111  //Find interpolation pair
112  size_t i;
113  for(i=0; i<Vprops.size(); ++i)
114  if(Vprops[i].volume > volume)
115  break;
116 
117  //Linear interpolation
118  cg = Vprops[i-1].CG + (Vprops[i].CG - Vprops[i-1].CG)/(Vprops[i].volume - Vprops[i-1].volume) * (volume - Vprops[i-1].volume);
119  }
120 }

122 void VariableBuoyancy::Update(Scalar dt)
123 {
124  Ocean* ocn = SimulationApp::getApp()->getSimulationManager()->getOcean();
125  if(ocn != nullptr && attach != NULL)
126  {
127  //Update volume
128  V += flowRate*dt;
129  V = V < Vmin ? Vmin : (V > Vmax ? Vmax : V);
130  Vector3 cg;
131  Scalar m;
132  InterpolateVProps(V, m, CG);
133 
134  //Compute forces
135  force = m*gravity;
136 
137  //Apply forces and torques
138  Vector3 solidCG = attach->getCGTransform().getOrigin();
139  Vector3 vbsCG = attach->getOTransform() * o2a * CG;
140  attach->ApplyCentralForce(force);
141  attach->ApplyTorque((vbsCG - solidCG).cross(force));
142  }
143 }

145 std::vector<Renderable> VariableBuoyancy::Render()
146 {
147  Transform vbsTrans = Transform::getIdentity();
148  if(attach != NULL)
149  vbsTrans.setOrigin(attach->getOTransform() * o2a * CG);
150  else
151  LinkActuator::Render();
152 
153  //Add renderable
154  std::vector<Renderable> items(0);
155  Renderable item;
156  item.type = RenderableType::ACTUATOR_LINES;
157  item.model = glMatrixFromTransform(vbsTrans);
158  item.points.push_back(glm::vec3(0,0,0));
159  item.points.push_back(0.1f * glm::vec3((GLfloat)force.x(), (GLfloat)force.y(), (GLfloat)force.z()));
160  items.push_back(item);
161 
162  return items;
163 }
