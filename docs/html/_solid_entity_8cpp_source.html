143 void SolidEntity::SetContactProperties(bool soft, Scalar stiffness, Scalar damping)
144 {
145  if(soft)
146  {
147  contactK = stiffness > Scalar(0) ? stiffness : Scalar(1000);
148  contactD = damping >= Scalar(0) ? damping : Scalar(0);

150  else

153  contactD = Scalar(0);
154  }
155 
156  if(rigidBody != nullptr)

161  {
162  int cflags = rigidBody->getCollisionFlags();
163  cflags &= ~(btCollisionObject::CollisionFlags::CF_HAS_CONTACT_STIFFNESS_DAMPING);
164  rigidBody->setCollisionFlags(cflags);
165  }
166  }
167  else if(multibodyCollider != nullptr)
168  {
169  if(soft)
170  multibodyCollider->setContactStiffnessAndDamping(contactK, contactD);
171  else
172  {
173  int cflags = multibodyCollider->getCollisionFlags();
174  cflags &= ~(btCollisionObject::CollisionFlags::CF_HAS_CONTACT_STIFFNESS_DAMPING);

428 Vector3 SolidEntity::getLinearVelocity() const
429 {
430  if(rigidBody != nullptr)
431  {
432  return rigidBody->getLinearVelocity();
433  }
434  else if(multibodyCollider != nullptr)
435  {
436  //Get multibody and link id
437  btMultiBody* multiBody = multibodyCollider->m_multiBody;
438  int index = multibodyCollider->m_link;
439 
440  //Start with base velocity
441  Vector3 linVelocity = multiBody->getBaseVel(); //Global
442  Vector3 angVelocity = multiBody->getBaseOmega(); //Global
443 
444  if(index >= 0) //If collider is not base
445  {
446  for(int i = 0; i <= index; ++i) //Accumulate velocity resulting from joints
447  {
448  //Add velocity resulting from rotation of previous links
449  linVelocity += angVelocity.cross(multiBody->localDirToWorld(i, multiBody->getRVector(i)));
450 
451  if(multiBody->getLink(i).m_jointType == btMultibodyLink::ePrismatic) //Just add linear velocity
452  {
453  Vector3 axis = multiBody->getLink(i).getAxisBottom(0); //Local axis
454  Vector3 vel = multiBody->getJointVel(i) * axis; //Local velocity
455  Vector3 gvel = multiBody->localDirToWorld(i, vel); //Global velocity
456  linVelocity += gvel;
457  }
458  else if(multiBody->getLink(i).m_jointType == btMultibodyLink::eRevolute) //Add linear velocity due to rotation
459  {
460  //Vector3 axis = multiBody->getLink(i).getAxisBottom(0); //Local linear motion
461  //Vector3 vel = multiBody->getJointVel(i) * axis; //Local velocity
462 
463  Vector3 axis = multiBody->getLink(i).getAxisTop(0); //Axis of joint
464  Vector3 aVel = multiBody->getJointVel(i) * axis;
465  Vector3 vel = aVel.cross(multiBody->getLink(i).m_dVector); //Local velocity
466  Vector3 gvel = multiBody->localDirToWorld(i, vel); //Global linear velocity
467  linVelocity += gvel;
468  angVelocity += multiBody->localDirToWorld(i, aVel); //Global angular velocity
469  }
470  }
471  }
472 
473  return linVelocity;
474  }
475  else
476  return Vector3(0,0,0);
477 }

654 void SolidEntity::ComputeSphericalApprox()
655 {
656  std::vector<Vector3>* x = getMeshVertices();
657  if(x->size() < 2)
658  return;
659  for(size_t i=0; i<x->size(); ++i)
660  x->at(i) = T_CG2C * x->at(i) - P_CB;
661 
662  Scalar r(0);
663  for(size_t i=0; i<x->size(); ++i)
664  {
665  Scalar rc = x->at(i).length2();
666  if(rc > r)
667  r = rc;
668  }
669
