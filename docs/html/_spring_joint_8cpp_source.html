100 std::vector<Renderable> SpringJoint::Render()
101 {
102  std::vector<Renderable> items(0);
103  btGeneric6DofSpring2Constraint* c = (btGeneric6DofSpring2Constraint*)getConstraint();
104  if(c != nullptr)
105  {
106  Renderable item;
107  item.model = glm::mat4(1.f);
108  item.type = RenderableType::JOINT_LINES;
109  Vector3 A = (c->getRigidBodyA().getCenterOfMassTransform() * c->getFrameOffsetA()).getOrigin();
110  Vector3 B = (c->getRigidBodyB().getCenterOfMassTransform() * c->getFrameOffsetB()).getOrigin();
111  item.points.push_back(glm::vec3(A.getX(), A.getY(), A.getZ()));
112  item.points.push_back(glm::vec3(B.getX(), B.getY(), B.getZ()));
113  items.push_back(item);
114  }
115  return items;
116 }
