59 Scalar RotaryEncoder::GetRawAngle()
60 {
61  if(j != nullptr && j->getType() == JointType::REVOLUTE)
62  {
63  return ((RevoluteJoint*)j)->getAngle();
64  }
65  else if(fe != nullptr)
66  {
67  Scalar mbAngle;
68  btMultibodyLink::eFeatherstoneJointType jt = btMultibodyLink::eInvalid;
69  fe->getJointPosition(jId, mbAngle, jt);
70 
71  if(jt == btMultibodyLink::eRevolute)
72  return mbAngle;
73  else
74  return Scalar(0);
75  }
76  else if(motor != nullptr)
77  {
78  return motor->getAngle();
79  }
80  else if(thrust != nullptr)
81  {
82  return thrust->getAngle();
83  }
84  else
85  return Scalar(0);
86 }

117 void RotaryEncoder::InternalUpdate(Scalar dt)
118 {
119  //new angle
120  Scalar actualAngle = GetRawAngle();
121 
122  //accumulate
123  Scalar angle0 = angle;
124 
125  if(lastAngle * actualAngle < Scalar(0))
126  {
127  if(lastAngle > M_PI_4)
128  angle += ((actualAngle + 2 * M_PI) - lastAngle);
129  else if(lastAngle < -M_PI_4)
130  angle += (actualAngle - (lastAngle + 2 * M_PI));
131  else
132  angle += (actualAngle - lastAngle);
133  }
134  else
135  angle += (actualAngle - lastAngle);
136 
137  lastAngle = actualAngle;
138 
139  //angular velocity
140  Scalar angularVelocity = (angle - angle0)/dt; // Less noisy than reading raw velocity
141 
142  //record sample
143  Scalar m[2];
144  m[0] = angle;
145  m[1] = angularVelocity;
146  Sample s(2, m);
147  AddSampleToHistory(s);
148 }
