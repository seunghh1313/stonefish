75 void INS::InternalUpdate(Scalar dt)
76 {
77  Scalar now = SimulationApp::getApp()->getSimulationManager()->getSimulationTime();
78 
79  //--- internal sensors
80  //get sensor frame in world
81  Transform imuTrans = getSensorFrame();
82  Vector3 R = imuTrans.getOrigin() - attach->getCGTransform().getOrigin();
83  //get angular velocity
84  Vector3 av = imuTrans.getBasis().inverse() * attach->getAngularVelocity();
85  //get acceleration
86  Vector3 acc = imuTrans.getBasis().inverse() * (
87  attach->getLinearAcceleration()
88  + attach->getAngularAcceleration().cross(R))
89  + attach->getAngularVelocity().cross(attach->getAngularVelocity().cross(R));
90  //get angular acceleration
91  Vector3 aacc = imuTrans.getBasis().inverse() * attach->getAngularAcceleration();
92 
93  //noise
94  if(imuNoise)
95  {
96  av.setX(av.x() + avNoiseX(randomGenerator));
97  av.setY(av.y() + avNoiseY(randomGenerator));
98  av.setZ(av.z() + avNoiseZ(randomGenerator));
99  acc.setX(acc.x() + accNoiseX(randomGenerator));
100  acc.setY(acc.y() + accNoiseY(randomGenerator));
101  acc.setZ(acc.z() + accNoiseZ(randomGenerator));
102  }
103 
104  //Predict (implicit Euler)
105  velocity += acc * dt; //In body frame (accumulated velocity)
106 
107  //--- external sensors
108  DVL* dvl;
109  if(dvlName != "" && (dvl = (DVL*)SimulationApp::getApp()->getSimulationManager()->getSensor(dvlName)) != nullptr) //Correct velocities
110  {
111  Sample s = dvl->getLastSample();
112  Scalar ts = s.getTimestamp();
113  if(ts >= 0.0 && now-ts <= dt) //Is DVL valid?
114  {
115  Transform dvlTrans = dvl->getSensorFrame();
116  Vector3 avv = imuTrans.getBasis() * av; // Includes noise
117  velocity = imuTrans.getBasis().inverse()* (
118  dvlTrans.getBasis() * Vector3(s.getValue(0), s.getValue(1), s.getValue(2)) // Pure linear velocity component
119  + avv.cross(imuTrans.getOrigin() - dvlTrans.getOrigin()) // Angular velocity component
120  );
121  altitude = s.getValue(3);
122  }
123  }
124 
125  Vector3 dp = velocity * dt; //In body frame
126  ned += imuTrans.getBasis() * dp; //In NED frame
127 
128  GPS* gps;
129  if(gpsName != "" && (gps = (GPS*)SimulationApp::getApp()->getSimulationManager()->getSensor(gpsName)) != nullptr) //Correct global position
130  {
131  Sample s = gps->getLastSample();
132  Scalar ts = s.getTimestamp();
133  if(ts >= 0.0 && now-ts <= dt && s.getValue(0) <= Scalar(90) && s.getValue(1) <= Scalar(180)) //Is GPS valid?
134  {
135  Vector3 trans = imuTrans.getOrigin() - gps->getSensorFrame().getOrigin();
136  ned.setX(trans.x() + s.getValue(2));
137  ned.setY(trans.y() + s.getValue(3));
138  }
139  }
140 
141  Pressure* press;
142  if(pressName != "" && (press = (Pressure*)SimulationApp::getApp()->getSimulationManager()->getSensor(pressName)) != nullptr) //Correct depth
143  {
144  Sample s = press->getLastSample();
145  Scalar ts = s.getTimestamp();
146  if(ts >= 0.0 && now-ts <= dt) //Is pressure valid?
147  {
148  Ocean* liq = SimulationApp::getApp()->getSimulationManager()->getOcean();
149  if(liq != NULL)
150  {
151  Scalar rho = liq->getLiquid().density;
152  Scalar g = SimulationApp::getApp()->getSimulationManager()->getGravity().z();
153  Scalar depth = s.getValue(0)/(rho*g);
154  ned.setZ(depth + imuTrans.getOrigin().z() - press->getSensorFrame().getOrigin().z() ); //Depth at INS frame
155  }
156  }
157  }
158 
159  //transformed output
160  Vector3 nedo = ned + (imuTrans.getBasis() * out.getOrigin());
161  Vector3 velo = out.getBasis().inverse() * (velocity + av.cross(out.getOrigin()));
162  Vector3 avo = out.getBasis().inverse() * av;
163  Vector3 acco = out.getBasis().inverse() * (acc + aacc.cross(out.getOrigin()) + av.cross(av.cross(out.getOrigin())));
164 
165  //compute geodetic position
166  Scalar height;
167  SimulationApp::getApp()->getSimulationManager()->getNED()->Ned2Geodetic(nedo.x(), nedo.y(), nedo.z(), latitude, longitude, height);
168 
169  //get angles
170  Scalar yaw, pitch, roll;
171  (imuTrans * out).getBasis().getEulerYPR(yaw, pitch, roll);
172 
173  //record sample
174  Scalar values[18] = {nedo.x(), nedo.y(), nedo.z(), altitude, latitude, longitude,
175  velo.x(), velo.y(), velo.z(), roll, pitch, yaw,
176  avo.x(), avo.y(), avo.z(), acco.x(), acco.y(), acco.z()};
177  Sample s(18, values);
178  AddSampleToHistory(s); //Adds noise.....:(
179 }

207  channels[13].rangeMax = btClamped(angularVelocityMax.y(), Scalar(0), Scalar(BT_LARGE_FLOAT));
208  channels[14].rangeMax = btClamped(angularVelocityMax.z(), Scalar(0), Scalar(BT_LARGE_FLOAT));
209  channels[15].rangeMin = -btClamped(linearAccelerationMax.x(), Scalar(0), Scalar(BT_LARGE_FLOAT));
210  channels[16].rangeMin = -btClamped(linearAccelerationMax.y(), Scalar(0), Scalar(BT_LARGE_FLOAT));
211  channels[17].rangeMin = -btClamped(linearAccelerationMax.z(), Scalar(0), Scalar(BT_LARGE_FLOAT));
212  channels[15].rangeMax = btClamped(linearAccelerationMax.x(), Scalar(0), Scalar(BT_LARGE_FLOAT));

217 void INS::setNoise(Vector3 angularVelocityStdDev, Vector3 linearAccelerationStdDev)
218 {
219  avNoiseX = std::normal_distribution<Scalar>(Scalar(0), btClamped(angularVelocityStdDev.x(), Scalar(0), BT_LARGE_FLOAT));
220  avNoiseY = std::normal_distribution<Scalar>(Scalar(0), btClamped(angularVelocityStdDev.y(), Scalar(0), BT_LARGE_FLOAT));
221  avNoiseZ = std::normal_distribution<Scalar>(Scalar(0), btClamped(angularVelocityStdDev.z(), Scalar(0), BT_LARGE_FLOAT));
222 
223  accNoiseX = std::normal_distribution<Scalar>(Scalar(0), btClamped(linearAccelerationStdDev.x(), Scalar(0), BT_LARGE_FLOAT));
224  accNoiseY = std::normal_distribution<Scalar>(Scalar(0), btClamped(linearAccelerationStdDev.y(), Scalar(0), BT_LARGE_FLOAT));
225  accNoiseZ = std::normal_distribution<Scalar>(Scalar(0), btClamped(linearAccelerationStdDev.z(), Scalar(0), BT_LARGE_FLOAT));
226 
227  imuNoise = true;
228 }
