78 bool AcousticModem::mutualContact(uint64_t device1Id, uint64_t device2Id)
79 {
80  AcousticModem* node1 = getNode(device1Id);
81  AcousticModem* node2 = getNode(device2Id);
82 
83  if(node1 == NULL || node2 == NULL)
84  return false;
85 
86  Vector3 pos1 = node1->getDeviceFrame().getOrigin();
87  Vector3 pos2 = node2->getDeviceFrame().getOrigin();
88  Vector3 dir = pos2-pos1;
89  Scalar distance = dir.length();
90 
91  if(!node1->isReceptionPossible(dir, distance) || !node2->isReceptionPossible(-dir, distance))
92  return false;
93 
94  if(node1->getOcclusionTest() || node2->getOcclusionTest())
95  {
96  btCollisionWorld::ClosestRayResultCallback closest(pos1, pos2);
97  closest.m_collisionFilterGroup = MASK_DYNAMIC;
98  closest.m_collisionFilterMask = MASK_STATIC | MASK_DYNAMIC | MASK_ANIMATED_COLLIDING;
99  SimulationApp::getApp()->getSimulationManager()->getDynamicsWorld()->rayTest(pos1, pos2, closest);
100  return !closest.hasHit();
101  }
102  else
103  return true;
104 }

126 bool AcousticModem::isReceptionPossible(Vector3 worldDir, Scalar distance)
127 {
128  //Check if modems are close enough
129  if(distance > range) return false;
130 
131  //Check if direction is in the FOV of the device
132  Vector3 dir = (getDeviceFrame().inverse() * worldDir).normalized();
133  Scalar d = Vector3(dir.getX(), dir.getY(), Scalar(0)).length();
134  Scalar vAngle = atan2(d, dir.getZ());
135  return btFabs(vAngle) >= minFov2 && btFabs(vAngle) <= maxFov2;
136 }

159 void AcousticModem::SendMessage(std::string data)
160 {
161  if(!mutualContact(getDeviceId(), getConnectedId()))

163 
164  AcousticDataFrame* msg = new AcousticDataFrame();

166  msg->seq = txSeq++;
167  msg->source = getDeviceId();
168  msg->destination = getConnectedId();
169  msg->data = data;

171  msg->travelled = Scalar(0);

176 {
177  AcousticDataFrame* msg;
178  while((msg = (AcousticDataFrame*)ReadMessage()) != nullptr)
179  {
180  //Different responses to messages should be implemented here

183  //timestamp and sequence don't change
184  msg->destination = msg->source;
185  msg->source = getDeviceId();
186  msg->data = "ACK";

188  txBuffer.push_back(msg);
189  }
190  else

197 void AcousticModem::InternalUpdate(Scalar dt)
198 {
199  //Propagate messages already sent
200  std::map<AcousticDataFrame*, Vector3>::iterator mIt;
201  for(mIt = propagating.begin(); mIt != propagating.end(); )
202  {
203  AcousticModem* dest = getNode(mIt->first->destination);
204  Vector3 dO = dest->getDeviceFrame().getOrigin();
205  Vector3 sO = mIt->second;
206  Vector3 dir = dO - sO;
207  Scalar d = dir.length();
208 
209  if(d <= SOUND_VELOCITY_WATER*dt) //Message reached?
210  {
211  mIt->first->travelled += d;
212  dest->MessageReceived(mIt->first);
213  mIt = propagating.erase(mIt);
214  }
215  else //Advance pulse
216  {
217  dir /= d; //Normalize direction
218  d = SOUND_VELOCITY_WATER * dt;
219  mIt->second += dir * d;
220  mIt->first->travelled += d;
221  ++mIt;
222  }
223  }
224 
225  //Send first message from the tx buffer
226  if(txBuffer.size() > 0)
227  {
228  AcousticDataFrame* msg = (AcousticDataFrame*)txBuffer[0];
229  if(mutualContact(msg->source, msg->destination))
230  propagating[msg] = msg->txPosition;
231  else
232  delete msg;
233 
234  txBuffer.pop_front();
235  }
236 }

248 std::vector<Renderable> AcousticModem::Render()
249 {
250  std::vector<Renderable> items(0);
251 
252  //Fov indicator
253  Renderable item;
254  item.model = glMatrixFromTransform(getDeviceFrame());
255  item.type = RenderableType::SENSOR_LINES;
256  GLfloat iconSize = 0.25f;
257  int div = 24;
258  //Upper circle
259  if(minFov2 > Scalar(0))
260  {
261  GLfloat r = iconSize * glm::sin((GLfloat)minFov2);
262  GLfloat h = iconSize * glm::cos((GLfloat)minFov2);
263  for(int i=0; i<=div; ++i)
264  {
265  GLfloat angle = (GLfloat)i/(GLfloat)div * 2.f * M_PI;
266  item.points.push_back(glm::vec3(glm::cos(angle)*r, glm::sin(angle)*r, -h));
267  if(i > 0 && i < div)
268  item.points.push_back(item.points.back());
269  }
270  }
271  //Lower circle
272  if(maxFov2 < Scalar(M_PI))
273  {
274  GLfloat r = iconSize * glm::sin((GLfloat)maxFov2);
275  GLfloat h = iconSize * glm::cos((GLfloat)maxFov2);
276  for(int i=0; i<=div; ++i)
277  {
278  GLfloat angle = (GLfloat)i/(GLfloat)div * 2.f * M_PI;
279  item.points.push_back(glm::vec3(glm::cos(angle)*r, glm::sin(angle)*r, -h));
280  if(i > 0 && i < div)
281  item.points.push_back(item.points.back());
282  }
283  }
284  //4 bars
285  div = 16;
286  if(maxFov2 - minFov2 > Scalar(0))
287  {
288  for(int i=0; i<4; ++i)
289  {
290  GLfloat hangle = (GLfloat)(i * M_PI_2);
291  GLfloat x = iconSize * glm::cos(hangle);
292  GLfloat y = iconSize * glm::sin(hangle);
293  for(int h=0; h<=div; ++h)
294  {
295  GLfloat angle = (GLfloat)h/(GLfloat)div * (maxFov2-minFov2) + minFov2;
296  item.points.push_back(glm::vec3(glm::sin(angle)*x, glm::sin(angle)*y, -glm::cos(angle)*iconSize));
297  if(h == 0 && minFov2 > Scalar(0))
298  {
299  glm::vec3 v = item.points.back();
300  item.points.push_back(glm::vec3(0.f,0.f,0.f));
301  item.points.push_back(v);
302  }
303  else if(h == div && maxFov2 < Scalar(M_PI))
304  {
305  item.points.push_back(item.points.back());
306  item.points.push_back(glm::vec3(0.f,0.f,0.f));
307  }
308  else if(h > 0 && h < div)
309  item.points.push_back(item.points.back());
310  }
311  }
312  }
313  items.push_back(item);
314 
315  //Axes
316  item.type = RenderableType::SENSOR_CS;
317  items.push_back(item);
318 
319 #ifdef DEBUG
320  item.type = RenderableType::SENSOR_POINTS;
321  item.model = glm::mat4(1.f);
322  std::map<AcousticDataFrame*, Vector3>::iterator mIt;
323  for(mIt = propagating.begin(); mIt != propagating.end(); ++mIt)
324  {
325  Vector3 mPos = mIt->second;
326  item.points.push_back(glm::vec3((GLfloat)mPos.getX(), (GLfloat)mPos.getY(), (GLfloat)mPos.getZ()));
327  }
328  items.push_back(item);
329 #endif
330 
331  return items;
332 }
