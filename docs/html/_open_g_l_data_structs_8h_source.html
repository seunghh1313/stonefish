280  void ComputeFaceTangent(size_t faceID, glm::vec3& tangent) const
281  {
282  //Shortcuts
283  const TexturableVertex& v0 = vertices[faces[faceID].vertexID[0]];
284  const TexturableVertex& v1 = vertices[faces[faceID].vertexID[1]];
285  const TexturableVertex& v2 = vertices[faces[faceID].vertexID[2]];
286  //Position delta
287  glm::vec3 deltaPos1 = v1.pos - v0.pos;
288  glm::vec3 deltaPos2 = v2.pos - v0.pos;
289  //UV delta
290  glm::vec2 deltaUV1 = v1.uv - v0.uv;
291  glm::vec2 deltaUV2 = v2.uv - v0.uv;
292  //Denominator
293  GLfloat r = 1.0f/(deltaUV1.x * deltaUV2.y - deltaUV1.y * deltaUV2.x);
294  tangent = (deltaPos1 * deltaUV2.y - deltaPos2 * deltaUV1.y)*r;
295  //bitangent = (deltaPos2 * deltaUV1.x - deltaPos1 * deltaUV2.x)*r;
296  }

436  static Color HSV(GLfloat H, GLfloat S, GLfloat V)
437  {
438  const glm::vec4 K(1.f, 2.f/3.f, 1.f/3.f, 3.f);
439  glm::vec3 p = glm::abs(glm::fract(glm::vec3(H) + glm::vec3(K.x, K.y, K.z)) * 6.f - glm::vec3(K.w));
440  glm::vec3 c = V * glm::mix(glm::vec3(K.x), glm::clamp(p - glm::vec3(K.x), 0.f, 1.f), S);
441  return Color(c.r, c.g, c.b);
442  }
