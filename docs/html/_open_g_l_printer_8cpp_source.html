48 OpenGLPrinter::OpenGLPrinter(const std::string& fontPath, GLuint size)
49 {
50  initialized = false;
51  fontVBO = 0;
52  nativeFontSize = size;
53 
54  FT_Error error = 0;
55  FT_Library ft;
56  FT_Face face;
57 
58  //Load Freetype library and font
59  if((error = FT_Init_FreeType(&ft)))
60  printf("Freetype: Could not init library!\n");
61  else
62  {
63 #ifdef EMBEDDED_RESOURCES
64  ResourceHandle rh(fontPath);
65  error = FT_New_Memory_Face(ft, rh.data(), rh.size(), 0, &face);
66 #else
67  error = FT_New_Face(ft, fontPath.c_str(), 0, &face);
68 #endif
69  if(error)
70  {
71  printf("Freetype: Could not load font from: %s!\n", fontPath.c_str());
72  FT_Done_FreeType(ft);
73  }
74  }
75 
76  if(!error)
77  {
78  //Load shader if not already loaded
79  if(printShader == NULL)
80  {
81  printShader = new GLSLShader("printer.frag","printer.vert");
82  printShader->AddUniform("tex", ParameterType::INT);
83  printShader->AddUniform("color", ParameterType::VEC4);
84  }
85 
86  if(printShader->isValid())
87  {
88  //Calculate texture atlas dimensions
89  FT_Set_Pixel_Sizes(face, 0, nativeFontSize);
90 
91  unsigned int w = 0;
92  unsigned int h = 0;
93 
94  for(int i = 32; i < 128; ++i)
95  {
96  if(FT_Load_Char(face, i, FT_LOAD_RENDER))
97  {
98  fprintf(stderr, "Loading character %c failed!\n", i);
99  continue;
100  }
101 
102  w += face->glyph->bitmap.width;
103  h = face->glyph->bitmap.rows > h ? face->glyph->bitmap.rows : h;
104  }
105 
106  texWidth = (GLfloat)w;
107  texHeight = (GLfloat)h;
108 
109  //Create texture
110  glGenTextures(1, &fontTexture);
111  OpenGLState::BindTexture(TEX_GUI1, GL_TEXTURE_2D, fontTexture);
112  glPixelStorei(GL_UNPACK_ALIGNMENT, 1);
113  glTexImage2D(GL_TEXTURE_2D, 0, GL_R8, (GLint)w, (GLint)h, 0, GL_RED, GL_UNSIGNED_BYTE, 0);
114  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
115  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
116  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
117  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
118 
119  //Load all characters
120  GLuint x = 0;
121 
122  for(int i = 32; i < 128; ++i)
123  {
124  if(FT_Load_Char(face, i, FT_LOAD_RENDER))
125  continue;
126 
127  glTexSubImage2D(GL_TEXTURE_2D, 0, x, 0, face->glyph->bitmap.width, face->glyph->bitmap.rows, GL_RED, GL_UNSIGNED_BYTE, face->glyph->bitmap.buffer);
128 
129  chars[i-32] = {glm::vec2((GLfloat)(face->glyph->advance.x >> 6), (GLfloat)(face->glyph->advance.y >> 6)),
130  glm::vec2((GLfloat)face->glyph->bitmap.width, (GLfloat)face->glyph->bitmap.rows),
131  glm::vec2((GLfloat)face->glyph->bitmap_left, (GLfloat)face->glyph->bitmap_top),
132  (GLfloat)x/texWidth};
133 
134  x += face->glyph->bitmap.width;
135  }
136 
137  OpenGLState::UnbindTexture(TEX_GUI1);
138 
139  //Freetype not needed any more
140  FT_Done_Face(face);
141  FT_Done_FreeType(ft);
142 
143  glGenBuffers(1, &fontVBO); //Generate VBO for rendering textured quads
144 
145  //Successfully initialized!
146  initialized = true;
147  }
148  }
149 }

161 void OpenGLPrinter::Print(const std::string& text, glm::vec4 color, GLuint x, GLuint y, GLfloat size, bool raw)
162 {
163  if(!initialized)
164  return;
165 
166  struct Point
167  {
168  GLfloat x;
169  GLfloat y;
170  GLfloat s;
171  GLfloat t;
172  } coords[6 * text.length()];
173 
174  memset(coords, 0, sizeof coords);
175 
176  unsigned int n = 0;
177 
178  GLfloat scale = (GLfloat)size/(GLfloat)nativeFontSize;
179  GLfloat xf = (GLfloat)x/(GLfloat)windowW * 2.f - 1.f;
180  GLfloat yf = (GLfloat)y/(GLfloat)windowH * 2.f - 1.f;
181  GLfloat sx = 2.f/(GLfloat)windowW;
182  GLfloat sy = 2.f/(GLfloat)windowH;
183 
184  if(raw)
185  {
186  glActiveTexture(GL_TEXTURE0 + TEX_GUI1);
187  glBindTexture(GL_TEXTURE_2D, fontTexture);
188  glUseProgram(printShader->getProgramHandle());
189  }
190  else
191  {
192  OpenGLState::BindTexture(TEX_GUI1, GL_TEXTURE_2D, fontTexture);
193  printShader->Use();
194  }
195 
196  printShader->SetUniform("color", color);
197  printShader->SetUniform("tex", TEX_GUI1);
198 
199  const char* ctext = text.c_str();
200  for(const char *c = ctext; *c; ++c)
201  {
202  Character ch = chars[*c-32];
203  GLfloat x2 = xf + ch.bearing.x * scale * sx;
204  GLfloat y2 = -yf - ch.bearing.y * scale * sy;
205  GLfloat w = ch.size.x * scale * sx;
206  GLfloat h = ch.size.y * scale * sy;
207  xf += ch.advance.x * scale * sx;
208  yf += ch.advance.y * scale * sy;
209 
210  if(!w || !h)
211  continue;
212 
213  coords[n++] = (Point){x2, -y2, ch.offset, 0};
214  coords[n++] = (Point){x2+w, -y2, ch.offset + ch.size.x/texWidth, 0};
215  coords[n++] = (Point){x2, -y2-h, ch.offset, ch.size.y/texHeight};
216  coords[n++] = (Point){x2+w, -y2, ch.offset + ch.size.x/texWidth, 0};
217  coords[n++] = (Point){x2, -y2-h, ch.offset, ch.size.y/texHeight};
218  coords[n++] = (Point){x2+w, -y2-h, ch.offset + ch.size.x/texWidth, ch.size.y/texHeight};
219  }
220 
221  glBindBuffer(GL_ARRAY_BUFFER, fontVBO);
222  glVertexAttribPointer(0, 4, GL_FLOAT, GL_FALSE, 0, 0);
223  glBufferData(GL_ARRAY_BUFFER, sizeof coords, coords, GL_DYNAMIC_DRAW);
224  glBindBuffer(GL_ARRAY_BUFFER, 0);
225 
226  glDrawArrays(GL_TRIANGLES, 0, n);
227 
228  if(raw)
229  {
230  glBindTexture(GL_TEXTURE_2D, 0);
231  glUseProgram(0);
232  }
233  else
234  {
235  OpenGLState::UnbindTexture(TEX_GUI1);
236  OpenGLState::UseProgram(0);
237  }
238 }

240 GLuint OpenGLPrinter::TextLength(const std::string& text)
241 {
242  GLuint length = 0;
243  const char* ctext = text.c_str();
244  for(const char *c = ctext; *c; ++c)
245  length += chars[*c-32].advance.x + chars[*c-32].bearing.x;
246  return length;
247 }
