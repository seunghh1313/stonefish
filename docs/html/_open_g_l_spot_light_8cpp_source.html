39 OpenGLSpotLight::OpenGLSpotLight(glm::vec3 position, glm::vec3 direction, GLfloat radius, GLfloat coneAngleDeg, glm::vec3 color, GLfloat lum)
40  : OpenGLLight(position, radius, color, lum)
41 {
42  coneAngle = glm::max(0.1f, coneAngleDeg)/180.f*M_PI;
43  GLfloat S = 2.f*M_PI*(1.f-cosf(coneAngle/2.f));
44  colorLi = glm::vec4(color, lum/S);
45  clipSpace = glm::mat4();
46  GLfloat near = getSourceRadius()/tanf(coneAngle/2.f);
47  zNear = glm::max(0.05f, near);
48  zFar = sqrtf(colorLi.a/MIN_INTENSITY_THRESHOLD);
49 
50  PlainMesh* m = new PlainMesh;
51  Vertex vt;
52  Face f;
53  unsigned int slices = 24;
54  vt.normal = glm::vec3(0,0,-1.f);
55  vt.pos = glm::vec3(0,0,0);
56  m->vertices.push_back(vt);
57  //vertices
58  for(unsigned int i=0; i<=slices; ++i)
59  {
60  vt.pos = glm::vec3(sinf(i/(GLfloat)slices*M_PI*2.f)*getSourceRadius(), -cosf(i/(GLfloat)slices*M_PI*2.f)*getSourceRadius(), 0.0);
61  m->vertices.push_back(vt);
62  }
63  //faces
64  for(unsigned int i=0; i<slices; ++i)
65  {
66  f.vertexID[0] = 0;
67  f.vertexID[1] = i + 2;
68  f.vertexID[2] = i + 1;
69  m->faces.push_back(f);
70  }
71  sourceObject = ((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getContent()->BuildObject(m);
72  delete m;
73 
74  UpdatePosition(position);
75  UpdateDirection(direction);
76  UpdateTransform();
77 }

120 glm::mat4 OpenGLSpotLight::getTransform()
121 {
122  glm::vec3 pos = getPosition() + getDirection() * (zNear - 1e-3f);
123  return glm::inverse(glm::lookAt(pos, pos + getDirection(), glm::vec3(0,0,1.f)));

137 void OpenGLSpotLight::SetupShader(LightUBO* ubo)
138 {
139  SpotLightUBO* spotUbo = (SpotLightUBO*)ubo;
140  spotUbo->position = getPosition();
141  spotUbo->color = glm::vec3(colorLi) * colorLi.a;
142  spotUbo->direction = getDirection();
143  spotUbo->cone = (GLfloat)cosf(getAngle()/2.f);
144  spotUbo->clipSpace = getClipSpace();
145  spotUbo->frustumNear = zNear;
146  spotUbo->frustumFar = zFar;
147  spotUbo->radius = glm::vec3(0.1f,0.1f,getSourceRadius());
148 }

150 void OpenGLSpotLight::BakeShadowmap(OpenGLPipeline* pipe)
151 {
152  glm::mat4 proj = glm::perspective((GLfloat)(2.f * coneAngle), 1.f, zNear, zFar);
153  glm::mat4 view = glm::lookAt(getPosition(),
154  getPosition() + getDirection(),
155  glm::vec3(0,0,1.f));
156 
157  glm::mat4 bias(0.5f, 0.f, 0.f, 0.f,
158  0.f, 0.5f, 0.f, 0.f,
159  0.f, 0.f, 0.5f, 0.f,
160  0.5f, 0.5f, 0.5f, 1.f);
161  clipSpace = bias * (proj * view);
162 
163  ((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getContent()->SetProjectionMatrix(proj);
164  ((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getContent()->SetViewMatrix(view);
165 
166  OpenGLState::BindFramebuffer(shadowFBO);
167  OpenGLState::Viewport(0, 0, SPOT_LIGHT_SHADOWMAP_SIZE, SPOT_LIGHT_SHADOWMAP_SIZE);
168  glClear(GL_DEPTH_BUFFER_BIT);
169  //glEnable(GL_POLYGON_OFFSET_FILL);
170  //glPolygonOffset(4.0f, 32.0f);
171  pipe->DrawObjects();
172  //glDisable(GL_POLYGON_OFFSET_FILL);
173  OpenGLState::BindFramebuffer(0);
174 }
