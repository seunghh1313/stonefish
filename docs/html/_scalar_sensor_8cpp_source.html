111 void ScalarSensor::AddSampleToHistory(const Sample& s)
112 {
113  if(historyLen < 0 && history.size() > 0) //No history
114  {
115  delete history[0];
116  history.pop_front();
117  }
118  else if(historyLen > 0 && (int)history.size() == historyLen) //Specified history length
119  {
120  delete history[0];
121  history.pop_front();
122  }
123  //else == 0 --> unlimited history
124 
125  Sample* sample = new Sample(s, sampleCount);
126  ++sampleCount;
127 
128  for(unsigned int i=0; i<sample->getNumOfDimensions(); ++i)
129  {
130  Scalar* data = sample->getDataPointer();
131 
132  //Add noise
133  if(channels[i].stdDev > Scalar(0) && data[i] < channels[i].rangeMax && data[i] > channels[i].rangeMin)
134  data[i] += channels[i].noise(randomGenerator);
135 
136  //Limit readings
137  if(data[i] > channels[i].rangeMax)
138  data[i] = channels[i].rangeMax;
139  else if(data[i] < channels[i].rangeMin)
140  data[i] = channels[i].rangeMin;
141  }
142 
143  //Add to history
144  history.push_back(sample);
145 }

155 void ScalarSensor::SaveMeasurementsToTextFile(const std::string& path, bool includeTime, unsigned int fixedPrecision)
156 {
157  if(history.size() == 0)
158  return;
159 
160  cInfo("Saving %s measurements to: %s", getName().c_str(), path.c_str());
161 
162  FILE* fp = fopen(path.c_str(), "wt");
163  if(fp == NULL)
164  {
165  cError("File could not be opened!");
166  return;
167  }
168 
169  //Write header
170  fprintf(fp, "#Measurements from %s\n", getName().c_str());
171  fprintf(fp, "#Number of channels: %ld\n", channels.size());
172  fprintf(fp, "#Number of samples: %ld\n", history.size());
173  if(freq <= Scalar(0.))
174  fprintf(fp, "#Frequency: %1.3lf Hz\n", SimulationApp::getApp()->getSimulationManager()->getStepsPerSecond());
175  else
176  fprintf(fp, "#Frequency: %1.3lf Hz\n", freq);
177  fprintf(fp, "#Unit system: SI\n\n");
178 
179  //Write data header
180  if(includeTime)
181  fprintf(fp, "#Time\t");
182  else
183  fprintf(fp, "#");
184 
185  for(unsigned int i = 0; i < channels.size(); i++)
186  {
187  fprintf(fp, "%s", channels[i].name.c_str());
188 
189  if(i < channels.size() - 1)
190  fprintf(fp, "\t");
191  else
192  fprintf(fp, "\n");
193  }
194 
195  //Write data
196  std::string format = "%1." + std::to_string(fixedPrecision) + "lf";
197 
198  for(unsigned int i = 0; i < history.size(); i++)
199  {
200  Sample* s = history[i];
201 
202  if(includeTime)
203  {
204  fprintf(fp, format.c_str(), s->getTimestamp());
205  fprintf(fp, "\t");
206  }
207 
208  for(unsigned int h = 0; h < channels.size(); h++)
209  {
210  Scalar v = s->getValue(h);
211 
212  fprintf(fp, format.c_str(), v);
213 
214  if(h < channels.size() - 1)
215  fprintf(fp, "\t");
216  else
217  fprintf(fp, "\n");
218  }
219  }
220 
221  fclose(fp);
222 }

224 void ScalarSensor::SaveMeasurementsToOctaveFile(const std::string& path, bool includeTime, bool separateChannels)
225 {
226  if(history.size() == 0)
227  return;
228 
229  //build data structure
230  ScientificData data("");
231 
232  if(separateChannels) //channels separated and saved in vecotors
233  {
234  if(includeTime) //write time vector
235  {
236  ScientificDataItem* it = new ScientificDataItem();
237  it->name = "Time";
238  it->type = DATA_VECTOR;
239 
240  btVectorXu* vector = new btVectorXu((unsigned int)history.size());
241  it->value = vector;
242 
243  for(unsigned int i = 0; i < history.size(); ++i)
244  {
245  Sample* s = history[i];
246  (*vector)[i] = s->getTimestamp();
247  }
248 
249  data.addItem(it);
250  }
251 
252  //write channel vectors
253  for(unsigned int i = 0; i < channels.size(); ++i)
254  {
255  ScientificDataItem* it = new ScientificDataItem();
256  it->name = channels[i].name;
257  it->type = DATA_VECTOR;
258 
259  btVectorXu* vector = new btVectorXu((unsigned int)history.size());
260  it->value = vector;
261 
262  for(unsigned int h = 0; h < history.size(); ++h)
263  {
264  Sample* s = history[h];
265  Scalar v = s->getValue(i);
266  (*vector)[h] = v;
267  }
268 
269  data.addItem(it);
270  }
271  }
272  else //channels combined in a matrix
273  {
274  ScientificDataItem* it = new ScientificDataItem();
275  it->name = getName();
276  it->type = DATA_MATRIX;
277 
278  btMatrixXu* matrix = new btMatrixXu((unsigned int)history.size(), (unsigned int)channels.size() + (includeTime ? 1 : 0));
279  it->value = matrix;
280 
281  for(unsigned int i = 0; i < history.size(); ++i)
282  {
283  Sample* s = history[i];
284 
285  if(includeTime)
286  matrix->setElem(i, 0, s->getTimestamp());
287 
288  for(unsigned int h = 0; h < channels.size(); ++h)
289  {
290  Scalar v = s->getValue(h);
291  matrix->setElem(i, h + (includeTime ? 1 : 0), v);
292  }
293  }
294 
295  data.addItem(it);
296  }
297 
298  //save data structure to file
299  SaveOctaveData(path, data);
300 }
