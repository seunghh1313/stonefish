41 Ocean::Ocean(std::string uniqueName, Scalar waves, Fluid l) : ForcefieldEntity(uniqueName)
42 {
43  ghost->setCollisionFlags(ghost->getCollisionFlags() | btCollisionObject::CF_STATIC_OBJECT);
44  oceanState = waves > Scalar(2.0) ? Scalar(2.0) : waves;
45 
46  Scalar size(100000);
47  depth = size;
48  Vector3 halfExtents = Vector3(size/Scalar(2), size/Scalar(2), size/Scalar(2));
49  ghost->setWorldTransform(Transform(Quaternion::getIdentity(), Vector3(0, 0, size/Scalar(2) - oceanState*Scalar(3)))); //Move ocean influence zone a bit up to account for waves
50  ghost->setCollisionShape(new btBoxShape(halfExtents));
51 
52  currents = std::vector<VelocityField*>(0);
53  currentsEnabled = false;
54 
55  liquid = l;
56  wavesDebug.type = RenderableType::HYDRO_POINTS;
57  wavesDebug.model = glm::mat4(1.f);
58  waterType = Scalar(0.0);
59  glOcean = nullptr;
60 }

141 float Ocean::GetDepth(const glm::vec3& point)
142 {
143  if(hasWaves()) //Geometric waves
144  {
145  GLfloat waveHeight = glOcean->ComputeWaveHeight(point.x, point.y);
146  glm::vec3 wavePoint(point.x, point.y, waveHeight);
147 #ifdef DEBUG_WAVES
148  wavesDebug.points.push_back(wavePoint);
149 #endif
150  return point.z - waveHeight;
151  }
152  else //Flat surface
153  {
154  glm::vec3 wavePoint(point.x, point.y, 0.f);
155 #ifdef DEBUG_WAVES
156  wavesDebug.points.push_back(wavePoint);
157 #endif
158  return point.z;
159  }
160 }

167 Scalar Ocean::GetPressure(const Vector3& point)
168 {
169  Scalar g = SimulationApp::getApp()->getSimulationManager()->getGravity().getZ();
170  Scalar d = GetDepth(point);
171  Scalar pressure = d > Scalar(0) ? d*liquid.density*g : Scalar(0);
172  return pressure;

211 void Ocean::ApplyFluidForces(btDynamicsWorld* world, btCollisionObject* co, bool recompute)
212 {
213  Entity* ent;
214  btRigidBody* rb = btRigidBody::upcast(co);
215  btMultiBodyLinkCollider* mbl = btMultiBodyLinkCollider::upcast(co);
216 
217  if(rb != 0)
218  {
219  if(rb->isStaticOrKinematicObject())
220  return;
221  else
222  ent = (Entity*)rb->getUserPointer();
223  }
224  else if(mbl != 0)
225  {
226  if(mbl->isStaticOrKinematicObject())
227  return;
228  else
229  ent = (Entity*)mbl->getUserPointer();
230  }
231  else
232  return;
233 
234  HydrodynamicsSettings settings;
235 
236  if(ent->getType() == EntityType::SOLID)
237  {
238  if(recompute)
239  {
240  settings.dampingForces = true;
241  settings.reallisticBuoyancy = true;
242  ((SolidEntity*)ent)->ComputeHydrodynamicForces(settings, this);
243  }
244 
245  ((SolidEntity*)ent)->ApplyHydrodynamicForces();
246  }

264 std::vector<Renderable> Ocean::Render(const std::vector<Actuator*>& act)
265 {
266  std::vector<Renderable> items(0);
267 
268  //Update currents data
269  glOceanCurrentsUBOData.gravity = glm::vec3(0.f,0.f,9.81f);
270  glOceanCurrentsUBOData.numCurrents = 0;
271 
272  if(currentsEnabled)
273  {
274  for(size_t i=0; i<currents.size(); ++i)
275  if(currents[i]->isEnabled())
276  {
277  std::vector<Renderable> citems = currents[i]->Render(glOceanCurrentsUBOData.currents[glOceanCurrentsUBOData.numCurrents]);
278  items.insert(items.end(), citems.begin(), citems.end());
279  ++glOceanCurrentsUBOData.numCurrents;
280  }
281  }
282 
283  for(size_t i=0; i<act.size(); ++i)
284  if(act[i]->getType() == ActuatorType::THRUSTER)
285  {
286  Thruster* th = (Thruster*)act[i];
287  Transform thFrame = th->getActuatorFrame();
288  Vector3 thPos = thFrame.getOrigin();
289  Vector3 thDir = -thFrame.getBasis().getColumn(0);
290  Scalar R = th->getDiameter()/Scalar(2);
291  Scalar vel = (th->isPropellerRight() ? Scalar(0.1) : Scalar(-0.1)) * th->getThrust();
292  glOceanCurrentsUBOData.currents[glOceanCurrentsUBOData.numCurrents].posR = glm::vec4((GLfloat)thPos.getX(),
293  (GLfloat)thPos.getY(),
294  (GLfloat)thPos.getZ(), (GLfloat)R);
295  glOceanCurrentsUBOData.currents[glOceanCurrentsUBOData.numCurrents].dirV = glm::vec4((GLfloat)thDir.getX(),
296  (GLfloat)thDir.getY(),
297  (GLfloat)thDir.getZ(),
298  (GLfloat)vel);
299  glOceanCurrentsUBOData.currents[glOceanCurrentsUBOData.numCurrents].params = glm::vec3(0.f);
300  glOceanCurrentsUBOData.currents[glOceanCurrentsUBOData.numCurrents].type = 10;
301  ++glOceanCurrentsUBOData.numCurrents;
302  }
303 
304  if(wavesDebug.points.size() > 0)
305  {
306  items.push_back(wavesDebug);
307  wavesDebug.points.clear();
308  }
309 
310  return items;
311 }
