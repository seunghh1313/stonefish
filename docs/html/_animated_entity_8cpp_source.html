121 AnimatedEntity::AnimatedEntity(std::string uniqueName, Trajectory* traj, std::string graphicsFilename, Scalar graphicsScale, const Transform& graphicsOrigin,
122  std::string physicsFilename, Scalar physicsScale, const Transform& physicsOrigin, std::string material, std::string look, bool collides)
123  : MovingEntity(uniqueName, material, look), tr(traj)
124 {
125  if(traj == nullptr)
126  return;
127 
128  //Load geometry from files
129  Mesh* graMesh = OpenGLContent::LoadMesh(graphicsFilename, graphicsScale, false);
130  Mesh* phyMesh;
131  T_O2G = graphicsOrigin;
132  T_CG2O = I4();
133 
134  if(physicsFilename != "")
135  {
136  phyMesh = OpenGLContent::LoadMesh(physicsFilename, physicsScale, false);
137  T_O2C = physicsOrigin;
138  }
139  else
140  {
141  phyMesh = graMesh;
142  T_O2C = T_O2G;
143  }
144 
145  //Build rigid body
146  btConvexHullShape* shape = new btConvexHullShape();
147  for(size_t i=0; i<phyMesh->getNumOfVertices(); ++i)
148  {
149  glm::vec3 pos = phyMesh->getVertexPos(i);
150  Vector3 v(pos.x, pos.y, pos.z);
151  shape->addPoint(v);
152  }
153  shape->optimizeConvexHull();
154  BuildRigidBody(shape, collides);
155 
156  //Build graphical objects
157  if(SimulationApp::getApp()->hasGraphics())
158  {
159  phyObjectId = ((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getContent()->BuildObject(phyMesh);
160  if(graMesh != phyMesh)
161  graObjectId = ((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getContent()->BuildObject(graMesh);
162  else
163  graObjectId = phyObjectId;
164  }
165 
166  //Delete mesh data
167  if(graMesh != phyMesh)
168  {
169  delete phyMesh;
170  delete graMesh;
171  }
172  else
173  delete phyMesh;
174 }

251 void AnimatedEntity::BuildRigidBody(btCollisionShape* shape, bool collides)
252 {
253  btDefaultMotionState* motionState = new btDefaultMotionState(tr->getInterpolatedTransform());
254  shape->setMargin(0.0);
255 
256  btRigidBody::btRigidBodyConstructionInfo rigidBodyCI(Scalar(0), motionState, shape, V0());
257  rigidBodyCI.m_friction = rigidBodyCI.m_rollingFriction = rigidBodyCI.m_restitution = Scalar(0); //not used
258  rigidBodyCI.m_linearDamping = rigidBodyCI.m_angularDamping = Scalar(0); //not used
259  rigidBodyCI.m_linearSleepingThreshold = rigidBodyCI.m_angularSleepingThreshold = Scalar(0); //not used
260  rigidBodyCI.m_additionalDamping = false;
261 
262  rigidBody = new btRigidBody(rigidBodyCI);
263  rigidBody->setUserPointer(this);
264  rigidBody->setCollisionFlags(rigidBody->getCollisionFlags()
265  | btCollisionObject::CF_KINEMATIC_OBJECT
266  | btCollisionObject::CF_CUSTOM_MATERIAL_CALLBACK);
267  if(!collides) rigidBody->setCollisionFlags(rigidBody->getCollisionFlags() | btCollisionObject::CF_NO_CONTACT_RESPONSE);
268  rigidBody->setActivationState(DISABLE_DEACTIVATION);
269 }

287 void AnimatedEntity::Update(Scalar dt)
288 {
