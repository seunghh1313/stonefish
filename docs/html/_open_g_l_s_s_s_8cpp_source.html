45 OpenGLSSS::OpenGLSSS(glm::vec3 centerPosition, glm::vec3 direction, glm::vec3 forward,
46  GLfloat verticalBeamWidthDeg, GLfloat horizontalBeamWidthDeg,
47  GLuint numOfBins, GLuint numOfLines, GLfloat verticalTiltDeg, glm::vec2 range_)
48 : OpenGLSonar(centerPosition, direction, forward, glm::uvec2(numOfBins, numOfLines), range_)
49 {
50  //SSS specs
51  sonar = nullptr;
52  tilt = glm::radians(verticalTiltDeg);
53  fov.x = glm::radians(verticalBeamWidthDeg);
54  fov.y = glm::radians(horizontalBeamWidthDeg);
55  nBeamSamples.x = glm::min((GLuint)ceilf(verticalBeamWidthDeg * (GLfloat)numOfBins/2.f * SSS_VRES_FACTOR), (GLuint)2048);
56  nBeamSamples.y = glm::min((GLuint)ceilf(horizontalBeamWidthDeg * SSS_HRES_FACTOR), (GLuint)2048);
57  noise = glm::vec2(0.f);
58  UpdateTransform();
59 
60  //Setup matrices
61  GLfloat near = range.x * glm::cos(glm::max(fov.x/2.f, fov.y/2.f));
62  GLfloat far = range.y;
63  projection = glm::perspective(fov.y, tanf(fov.x/2.f)/tanf(fov.y/2.f), near, far);
64  GLfloat offsetAngle = M_PI_2 - tilt;
65  views[0] = glm::rotate(-offsetAngle, glm::vec3(0.f,1.f,0.f));
66  views[1] = glm::rotate(offsetAngle, glm::vec3(0.f,1.f,0.f));
67 
68  //Allocate resources
69  inputRangeIntensityTex = OpenGLContent::GenerateTexture(GL_TEXTURE_2D_ARRAY, glm::uvec3(nBeamSamples.x, nBeamSamples.y, 2),
70  GL_RG32F, GL_RG, GL_FLOAT, NULL, FilteringMode::NEAREST, false);
71  glGenRenderbuffers(1, &inputDepthRBO);
72  glBindRenderbuffer(GL_RENDERBUFFER, inputDepthRBO);
73  glRenderbufferStorage(GL_RENDERBUFFER, GL_DEPTH_COMPONENT32F, nBeamSamples.x, nBeamSamples.y);
74  glBindRenderbuffer(GL_RENDERBUFFER, 0);
75 
76  glGenFramebuffers(1, &renderFBO);
77  OpenGLState::BindFramebuffer(renderFBO);
78  glFramebufferRenderbuffer(GL_FRAMEBUFFER, GL_DEPTH_ATTACHMENT, GL_RENDERBUFFER, inputDepthRBO);
79  glFramebufferTextureLayer(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, inputRangeIntensityTex, 0, 0);
80  glFramebufferTextureLayer(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT1, inputRangeIntensityTex, 0, 1);
81  GLenum status = glCheckFramebufferStatus(GL_FRAMEBUFFER);
82  if(status != GL_FRAMEBUFFER_COMPLETE)
83  cError("Sonar input FBO initialization failed!");
84 
85  //Output shader: sonar range data
86  //a. Histograms for vertical beam FOV
87  outputTex[0] = OpenGLContent::GenerateTexture(GL_TEXTURE_2D_ARRAY, glm::uvec3(nBeamSamples.x, viewportWidth/2, 2),
88  GL_RG32F, GL_RG, GL_FLOAT, NULL, FilteringMode::NEAREST, false);
89  //b. Waterfall output (ping-pong)
90  outputTex[1] = OpenGLContent::GenerateTexture(GL_TEXTURE_2D, glm::uvec3(viewportWidth, viewportHeight, 1),
91  GL_R8, GL_RED, GL_UNSIGNED_BYTE, NULL, FilteringMode::NEAREST, false);
92  outputTex[2] = OpenGLContent::GenerateTexture(GL_TEXTURE_2D, glm::uvec3(viewportWidth, viewportHeight, 1),
93  GL_R8, GL_RED, GL_UNSIGNED_BYTE, NULL, FilteringMode::NEAREST, false);
94  pingpong = 0;
95  //Visualisation shader: sonar data color mapped
96  glGenTextures(1, &displayTex);
97  OpenGLState::BindTexture(TEX_BASE, GL_TEXTURE_2D, displayTex);
98  glPixelStorei(GL_PACK_ALIGNMENT, 1);
99  glTexImage2D(GL_TEXTURE_2D, 0, GL_RGB8, viewportWidth, viewportHeight, 0, GL_RGB, GL_UNSIGNED_BYTE, NULL); //RGB image
100  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
101  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
102  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
103  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
104  OpenGLState::UnbindTexture(TEX_BASE);
105  std::vector<FBOTexture> textures;
106  textures.push_back(FBOTexture(GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, displayTex));
107  displayFBO = OpenGLContent::GenerateFramebuffer(textures);
108 
109  //Display quad
110  GLfloat quadData[4][4];
111  quadData[0][0] = -1.f;
112  quadData[0][1] = 1.f;
113  quadData[0][2] = 0.f;
114  quadData[0][3] = 1.f;
115  quadData[1][0] = -1.f;
116  quadData[1][1] = -1.f;
117  quadData[1][2] = 0.f;
118  quadData[1][3] = 0.f;
119  quadData[2][0] = 1.f;
120  quadData[2][1] = 1.f;
121  quadData[2][2] = 1.f;
122  quadData[2][3] = 1.f;
123  quadData[3][0] = 1.f;
124  quadData[3][1] = -1.f;
125  quadData[3][2] = 1.f;
126  quadData[3][3] = 0.f;
127 
128  glGenVertexArrays(1, &displayVAO);
129  OpenGLState::BindVertexArray(displayVAO);
130  glEnableVertexAttribArray(0);
131  glGenBuffers(1, &displayVBO);
132  glBindBuffer(GL_ARRAY_BUFFER, displayVBO);
133  glBufferData(GL_ARRAY_BUFFER, sizeof(quadData), quadData, GL_STATIC_DRAW);
134  glVertexAttribPointer(0, 4, GL_FLOAT, GL_FALSE, 4*sizeof(GLfloat), 0);
135  glBindBuffer(GL_ARRAY_BUFFER, 0);
136  OpenGLState::BindVertexArray(0);
137 
138  //Load output shader
139  std::string header = "#version 430\n#define N_HALF_BINS " + std::to_string(viewportWidth/2)
140  + "\n#define N_HORI_BEAM_SAMPLES " + std::to_string(nBeamSamples.y)
141  + "\n#define N_VERT_BEAM_SAMPLES " + std::to_string(nBeamSamples.x) + "\n"; ;
142  std::vector<GLSLSource> sources;
143  sources.push_back(GLSLSource(GL_COMPUTE_SHADER, "sonarOutput2.comp", header));
144  sonarOutputShader[0] = new GLSLShader(sources);
145  sonarOutputShader[0]->AddUniform("sonarInput", ParameterType::INT);
146  sonarOutputShader[0]->AddUniform("sonarHist", ParameterType::INT);
147  sonarOutputShader[0]->AddUniform("range", ParameterType::VEC3);
148 
149  sonarOutputShader[0]->Use();
150  sonarOutputShader[0]->SetUniform("sonarInput", TEX_POSTPROCESS1);
151  sonarOutputShader[0]->SetUniform("sonarHist", TEX_POSTPROCESS2);
152  sonarOutputShader[0]->SetUniform("range", glm::vec3(range.x, range.y, 2*(range.y-range.x)/(GLfloat)viewportWidth));
153  OpenGLState::UseProgram(0);
154 
155  sources.clear();
156  sources.push_back(GLSLSource(GL_COMPUTE_SHADER, "sonarLine.comp", header));
157  sonarOutputShader[1] = new GLSLShader(sources);
158  sonarOutputShader[1]->AddUniform("sonarHist", ParameterType::INT);
159  sonarOutputShader[1]->AddUniform("sonarOutput", ParameterType::INT);
160  sonarOutputShader[1]->AddUniform("noiseSeed", ParameterType::VEC3);
161  sonarOutputShader[1]->AddUniform("noiseStddev", ParameterType::VEC2);
162  sonarOutputShader[1]->AddUniform("gain", ParameterType::FLOAT);
163  sonarOutputShader[1]->AddUniform("vfov", ParameterType::FLOAT);
164  sonarOutputShader[1]->AddUniform("tilt", ParameterType::FLOAT);
165  sonarOutputShader[1]->Use();
166  sonarOutputShader[1]->SetUniform("sonarHist", TEX_POSTPROCESS1);
167  sonarOutputShader[1]->SetUniform("sonarOutput", TEX_POSTPROCESS2);
168  sonarOutputShader[1]->SetUniform("gain", gain);
169  sonarOutputShader[1]->SetUniform("vfov", fov.x);
170  sonarOutputShader[1]->SetUniform("tilt", tilt);
171  OpenGLState::UseProgram(0);
172 
173  sources.clear();
174  sources.push_back(GLSLSource(GL_COMPUTE_SHADER, "sonarShift.comp"));
175  sonarShiftShader = new GLSLShader(sources);
176  sonarShiftShader->AddUniform("sonarOutputIn", ParameterType::INT);
177  sonarShiftShader->AddUniform("sonarOutputOut", ParameterType::INT);
178  sonarShiftShader->Use();
179  sonarShiftShader->SetUniform("sonarOutputIn", TEX_POSTPROCESS1);
180  sonarShiftShader->SetUniform("sonarOutputOut", TEX_POSTPROCESS2);
181  OpenGLState::UseProgram(0);
182 }

192 void OpenGLSSS::UpdateTransform()
193 {
194  OpenGLSonar::UpdateTransform();
195 
196  if(sonar == nullptr)
197  return;
198 
199  //Update settings if necessary
200  bool updateProjection = false;
201  glm::vec3 rangeGain((GLfloat)sonar->getRangeMin(), (GLfloat)sonar->getRangeMax(), (GLfloat)sonar->getGain());
202  if(rangeGain.x != range.x)
203  {
204  range.x = rangeGain.x;
205  updateProjection = true;
206  settingsUpdated = true;
207  }
208  if(rangeGain.y != range.y)
209  {
210  range.y = rangeGain.y;
211  updateProjection = true;
212  settingsUpdated = true;
213  }
214  if(rangeGain.z != gain)
215  {
216  gain = rangeGain.z;
217  //settingsUpdated = true;
218  }
219  if(updateProjection)
220  {
221  GLfloat near = range.x/2.f;
222  GLfloat far = range.y;
223  projection[0] = glm::vec4(near/(near*tanf(fov.x/2.f)), 0.f, 0.f, 0.f);
224  projection[1] = glm::vec4(0.f, near/(near*tanf(fov.y/2.f)), 0.f, 0.f);
225  projection[2] = glm::vec4(0.f, 0.f, -(far + near)/(far-near), -1.f);
226  projection[3] = glm::vec4(0.f, 0.f, -2.f*far*near/(far-near), 0.f);
227  }
228 
229  //Inform sonar to run callback
230  if(newData)
231  {
232  glBindBuffer(GL_PIXEL_PACK_BUFFER, displayPBO);
233  GLubyte* src = (GLubyte*)glMapBuffer(GL_PIXEL_PACK_BUFFER, GL_READ_ONLY);
234  if(src)
235  {
236  sonar->NewDataReady(src, 0);
237  glUnmapBuffer(GL_PIXEL_PACK_BUFFER); //Release pointer to the mapped buffer
238  }
239 
240  glBindBuffer(GL_PIXEL_PACK_BUFFER, outputPBO);
241  src = (GLubyte*)glMapBuffer(GL_PIXEL_PACK_BUFFER, GL_READ_ONLY);
242  if(src)
243  {
244  sonar->NewDataReady(src, 1);
245  glUnmapBuffer(GL_PIXEL_PACK_BUFFER); //Release pointer to the mapped buffer
246  }
247  glBindBuffer(GL_PIXEL_PACK_BUFFER, 0);
248  newData = false;
249  }

272 void OpenGLSSS::ComputeOutput(std::vector<Renderable>& objects)
273 {
274  OpenGLContent* content = ((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getContent();
275  content->SetDrawingMode(DrawingMode::RAW);
276  //Generate sonar input
277  OpenGLState::BindFramebuffer(renderFBO);
278  OpenGLState::Viewport(0, 0, nBeamSamples.x, nBeamSamples.y);
279  glDisable(GL_DEPTH_CLAMP);
280  sonarInputShader[1]->Use();
281  sonarInputShader[1]->SetUniform("eyePos", GetEyePosition());
282  sonarInputShader[0]->Use();
283  sonarInputShader[0]->SetUniform("eyePos", GetEyePosition());
284  GLSLShader* shader;
285  for(size_t i=0; i<2; ++i) //For each of the sonar views
286  {
287  //Compute matrices
288  glm::mat4 VP = GetProjectionMatrix() * views[i] * GetViewMatrix();
289  //Clear color and depth for particular framebuffer layer
290  glDrawBuffer(GL_COLOR_ATTACHMENT0 + (GLuint)i);
291  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
292  //Draw objects
293  for(size_t h=0; h<objects.size(); ++h)
294  {
295  if(objects[h].type != RenderableType::SOLID)
296  continue;
297  const Object& obj = content->getObject(objects[h].objectId);
298  const Look& look = content->getLook(objects[h].lookId);
299  glm::mat4 M = objects[h].model;
300  Material mat = SimulationApp::getApp()->getSimulationManager()->getMaterialManager()->getMaterial(objects[h].materialName);
301  bool normalMapping = obj.texturable && (look.normalTexture > 0);
302  shader = normalMapping ? sonarInputShader[1] : sonarInputShader[0];
303  shader->Use();
304  shader->SetUniform("MVP", VP * M);
305  shader->SetUniform("M", M);
306  shader->SetUniform("N", glm::mat3(glm::transpose(glm::inverse(M))));
307  shader->SetUniform("restitution", (GLfloat)mat.restitution);
308  if(normalMapping)
309  OpenGLState::BindTexture(TEX_MAT_NORMAL, GL_TEXTURE_2D, look.normalTexture);
310  content->DrawObject(objects[h].objectId, objects[h].lookId, objects[h].model);
311  }
312  }
313  glEnable(GL_DEPTH_CLAMP);
314  OpenGLState::UnbindTexture(TEX_MAT_NORMAL);
315  OpenGLState::BindFramebuffer(0);
316 
317  //Compute sonar output histogram
318  glBindImageTexture(TEX_POSTPROCESS1, inputRangeIntensityTex, 0, GL_TRUE, 0, GL_READ_ONLY, GL_RG32F);
319  glBindImageTexture(TEX_POSTPROCESS2, outputTex[0], 0, GL_TRUE, 0, GL_WRITE_ONLY, GL_RG32F);
320  sonarOutputShader[0]->Use();
321  if(settingsUpdated)
322  {
323  sonarOutputShader[0]->SetUniform("range", glm::vec3(range.x, range.y, 2*(range.y-range.x)/(GLfloat)viewportWidth));
324  settingsUpdated = false;
325  }
326  glMemoryBarrier(GL_FRAMEBUFFER_BARRIER_BIT);
327  glDispatchCompute((GLuint)ceilf(nBeamSamples.x/64.f), 2, 1);
328 
329  //Shift old sonar output
330  glBindImageTexture(TEX_POSTPROCESS1, outputTex[pingpong + 1], 0, GL_FALSE, 0, GL_READ_ONLY, GL_R8);
331  glBindImageTexture(TEX_POSTPROCESS2, outputTex[1-pingpong + 1], 0, GL_FALSE, 0, GL_WRITE_ONLY, GL_R8);
332  sonarShiftShader->Use();
333  glDispatchCompute((GLuint)ceilf(viewportWidth/16.f), (GLuint)ceilf(viewportHeight/16.f), 1);
334 
335  //Postprocess sonar output
336  glBindImageTexture(TEX_POSTPROCESS1, outputTex[0], 0, GL_TRUE, 0, GL_READ_ONLY, GL_RG32F);
337  sonarOutputShader[1]->Use();
338  sonarOutputShader[1]->SetUniform("noiseSeed", glm::vec3(randDist(randGen), randDist(randGen), randDist(randGen)));
339  sonarOutputShader[1]->SetUniform("noiseStddev", noise); // Multiplicative, additive (0.01f, 0.02f)
340  sonarOutputShader[1]->SetUniform("gain", gain);
341  glMemoryBarrier(GL_SHADER_IMAGE_ACCESS_BARRIER_BIT);
342  glDispatchCompute((GLuint)ceilf(viewportWidth/2.f/64.f), 2, 1);
343 
344  OpenGLState::BindFramebuffer(displayFBO);
345  OpenGLState::Viewport(0, 0, viewportWidth, viewportHeight);
346  glClear(GL_COLOR_BUFFER_BIT);
347  OpenGLState::BindTexture(TEX_POSTPROCESS2, GL_TEXTURE_2D, outputTex[1-pingpong + 1]);
348  sonarVisualizeShader->Use();
349  sonarVisualizeShader->SetUniform("texSonarData", TEX_POSTPROCESS2);
350  sonarVisualizeShader->SetUniform("colormap", static_cast<GLint>(cMap));
351  glMemoryBarrier(GL_SHADER_IMAGE_ACCESS_BARRIER_BIT);
352  OpenGLState::BindVertexArray(displayVAO);
353  glDrawArrays(GL_TRIANGLE_STRIP, 0, 4);
354  OpenGLState::BindVertexArray(0);
355  OpenGLState::BindFramebuffer(0);
356  OpenGLState::UseProgram(0);
357  OpenGLState::UnbindTexture(TEX_POSTPROCESS2);
358 
359  ++pingpong;
360  if(pingpong > 1)
361  pingpong = 0;
362 }

364 void OpenGLSSS::DrawLDR(GLuint destinationFBO, bool updated)
365 {
366  //Check if there is a need to display image on screen
367  bool display = true;
368  unsigned int dispX, dispY;
369  GLfloat dispScale;
370  if(sonar != nullptr)
371  display = sonar->getDisplayOnScreen(dispX, dispY, dispScale);
372 
373  //Draw on screen
374  if(display)
375  {
376  OpenGLContent* content = ((GraphicalSimulationApp*)SimulationApp::getApp())->getGLPipeline()->getContent();
377  if(0)
378  {
379  content->SetViewportSize(nBeamSamples.x/2, nBeamSamples.y/4);
380  OpenGLState::BindFramebuffer(destinationFBO);
381  OpenGLState::Viewport(0, 0, nBeamSamples.x/2, nBeamSamples.y/4);
382  content->DrawTexturedQuad(0.f, 0.f, (GLfloat)nBeamSamples.x/4, (GLfloat)nBeamSamples.y/4, inputRangeIntensityTex, 0, true);
383  content->DrawTexturedQuad((GLfloat)nBeamSamples.x/4, 0.f, (GLfloat)nBeamSamples.x/4, (GLfloat)nBeamSamples.y/4, inputRangeIntensityTex, 1, true);
384  OpenGLState::BindFramebuffer(0);
385  }
386  else
387  {
388  int windowHeight = ((GraphicalSimulationApp*)SimulationApp::getApp())->getWindowHeight();
389  int windowWidth = ((GraphicalSimulationApp*)SimulationApp::getApp())->getWindowWidth();
390  OpenGLState::BindFramebuffer(destinationFBO);
391  OpenGLState::Viewport(0, 0, windowWidth, windowHeight);
392  OpenGLState::DisableCullFace();
393  content->DrawTexturedQuad(dispX, dispY+viewportHeight*dispScale, viewportWidth*dispScale, -viewportHeight*dispScale, displayTex);
394  OpenGLState::EnableCullFace();
395  OpenGLState::BindFramebuffer(0);
396  }
397  }
398 
399  //Copy texture to sonar buffer
400  if(sonar != nullptr && updated)
401  {
402  OpenGLState::BindTexture(TEX_POSTPROCESS1, GL_TEXTURE_2D, outputTex[pingpong+1]);
403  glBindBuffer(GL_PIXEL_PACK_BUFFER, outputPBO);
404  glGetTexImage(GL_TEXTURE_2D, 0, GL_RED, GL_UNSIGNED_BYTE, NULL);
405  glBindBuffer(GL_PIXEL_PACK_BUFFER, 0);
406  OpenGLState::BindTexture(TEX_POSTPROCESS1, GL_TEXTURE_2D, displayTex);
407  glBindBuffer(GL_PIXEL_PACK_BUFFER, displayPBO);
408  glGetTexImage(GL_TEXTURE_2D, 0, GL_RGB, GL_UNSIGNED_BYTE, NULL);
409  glBindBuffer(GL_PIXEL_PACK_BUFFER, 0);
410  OpenGLState::UnbindTexture(TEX_POSTPROCESS1);
411  newData = true;
412  }
413 }
