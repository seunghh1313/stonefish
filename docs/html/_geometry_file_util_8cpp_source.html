35 Mesh* LoadGeometryFromFile(const std::string& path, GLfloat scale)
36 {
37  std::string extension = path.substr(path.length()-3,3);
38  Mesh* mesh = nullptr;
39 
40  if(extension == "stl" || extension == "STL")
41  mesh = LoadSTL(path, scale);
42  else if(extension == "obj" || extension == "OBJ")
43  mesh = LoadOBJ(path, scale);
44  else
45  cError("Unsupported geometry file type: %s!", extension.c_str());
46 
47  return mesh;
48 }

50 Mesh* LoadOBJ(const std::string& path, GLfloat scale)
51 {
52  //Read OBJ data
53  FILE* file = fopen(path.c_str(), "rb");
54 
55  if(file == NULL)
56  {
57  cCritical("Failed to open geometry file: %s", path.c_str());
58  return nullptr;
59  }
60 
61  cInfo("Loading geometry from: %s", path.c_str());
62 
63  char line[1024];
64  std::vector<glm::vec3> positions;
65  std::vector<glm::vec3> normals;
66  std::vector<glm::vec2> uvs;
67  Mesh* mesh_ = nullptr;
68  int64_t start = GetTimeInMicroseconds();
69 
70  //Read vertices
71  while(fgets(line, 1024, file))
72  {
73  if(line[0] == 'v')
74  {
75  if(line[1] == ' ')
76  {
77  glm::vec3 v;
78  sscanf(line, "v %f %f %f\n", &v.x, &v.y, &v.z);
79  v *= scale; //Scaling
80  positions.push_back(v);
81  }
82  else if(line[1] == 'n')
83  {
84  glm::vec3 n;
85  sscanf(line, "vn %f %f %f\n", &n.x, &n.y, &n.z);
86  normals.push_back(n);
87  }
88  else if(line[1] == 't')
89  {
90  glm::vec2 uv;
91  sscanf(line, "vt %f %f\n", &uv.x, &uv.y);
92  uvs.push_back(uv);
93  }
94  }
95  }
96  fseek(file, 0, SEEK_SET); //Go back to beginning of file
97 
98  size_t genVStart = positions.size();
99  bool hasNormals = normals.size() > 0;
100  bool hasUVs = uvs.size() > 0;
101 
102 #ifdef DEBUG
103  printf("Vertices: %ld Normals: %ld\n", genVStart, normals.size());
104 #endif
105 
106  if(hasUVs)
107  {
108  TexturableMesh* mesh = new TexturableMesh;
109 
110  //Initialize positions
111  for(size_t i=0; i<positions.size(); ++i)
112  {
113  TexturableVertex v;
114  v.pos = positions[i];
115  mesh->vertices.push_back(v);
116  }
117 
118  //Read faces
119  while(fgets(line, 1024, file))
120  {
121  if(line[0] != 'f')
122  continue;
123 
124  Face face;
125  unsigned int vID[3];
126  unsigned int uvID[3];
127  unsigned int nID[3];
128  sscanf(line, "f %u/%u/%u %u/%u/%u %u/%u/%u\n", &vID[0], &uvID[0], &nID[0], &vID[1], &uvID[1], &nID[1], &vID[2], &uvID[2], &nID[2]);
129 
130  for(short unsigned int i=0; i<3; ++i)
131  {
132  TexturableVertex v = mesh->vertices[vID[i]-1]; //Vertex from previously read pool
133 
134  if(glm::length2(v.normal) == 0.f) //Is it a fresh vertex?
135  {
136  mesh->vertices[vID[i]-1].normal = normals[nID[i]-1];
137  mesh->vertices[vID[i]-1].uv = uvs[uvID[i]-1];
138  face.vertexID[i] = vID[i]-1;
139  }
140  else if((v.normal == normals[nID[i]-1]) && (v.uv == uvs[uvID[i]-1])) //Does it have the same normal and UV?
141  {
142  face.vertexID[i] = vID[i]-1;
143  }
144  else //Otherwise search the generated pool
145  {
146  v.normal = normals[nID[i]-1];
147  v.uv = uvs[uvID[i]-1];
148 
149  std::vector<TexturableVertex>::iterator it;
150  if((it = std::find(mesh->vertices.begin()+genVStart, mesh->vertices.end(), v)) != mesh->vertices.end()) //If vertex exists
151  {
152  face.vertexID[i] = (GLuint)(it - mesh->vertices.begin());
153  }
154  else
155  {
156  mesh->vertices.push_back(v);
157  face.vertexID[i] = (GLuint)mesh->vertices.size()-1;
158  }
159  }
160  }
161 
162  mesh->faces.push_back(face);
163  }
164  mesh_ = mesh;
165  }
166  else
167  {
168  PlainMesh* mesh = new PlainMesh;
169 
170  //Initialize positions
171  for(size_t i=0; i<positions.size(); ++i)
172  {
173  Vertex v;
174  v.pos = positions[i];
175  mesh->vertices.push_back(v);
176  }
177 
178  //Read faces
179  while(fgets(line, 1024, file))
180  {
181  if(line[0] != 'f')
182  continue;
183 
184  Face face;
185 
186  if(hasNormals)
187  {
188  unsigned int vID[3];
189  unsigned int nID[3];
190  sscanf(line, "f %u//%u %u//%u %u//%u\n", &vID[0], &nID[0], &vID[1], &nID[1], &vID[2], &nID[2]);
191 
192  for(short unsigned int i=0; i<3; ++i)
193  {
194  Vertex v = mesh->vertices[vID[i]-1]; //Vertex from previously read pool
195 
196  if(glm::length2(v.normal) == 0.f) //Is it a fresh vertex?
197  {
198  mesh->vertices[vID[i]-1].normal = normals[nID[i]-1];
199  face.vertexID[i] = vID[i]-1;
200  }
201  else if(v.normal == normals[nID[i]-1]) //Does it have the same normal?
202  {
203  face.vertexID[i] = vID[i]-1;
204  }
205  else //Otherwise search the generated pool
206  {
207  v.normal = normals[nID[i]-1];
208 
209  std::vector<Vertex>::iterator it;
210  if((it = std::find(mesh->vertices.begin()+genVStart, mesh->vertices.end(), v)) != mesh->vertices.end()) //If vertex exists
211  {
212  face.vertexID[i] = (GLuint)(it - mesh->vertices.begin());
213  }
214  else
215  {
216  mesh->vertices.push_back(v);
217  face.vertexID[i] = (GLuint)mesh->vertices.size()-1;
218  }
219  }
220  }
221  }
222  else
223  {
224  unsigned int vID[3];
225  sscanf(line, "f %u %u %u\n", &vID[0], &vID[1], &vID[2]);
226 
227  face.vertexID[0] = vID[0]-1;
228  face.vertexID[1] = vID[1]-1;
229  face.vertexID[2] = vID[2]-1;
230  }
231 
232  mesh->faces.push_back(face);
233  }
234  mesh_ = mesh;
235  }
236  fclose(file);
237 
238  int64_t end = GetTimeInMicroseconds();
239 
240 #ifdef DEBUG
241  printf("Loaded: %ld Generated: %ld\n", genVStart, mesh_->getNumOfVertices()-genVStart);
242  printf("Total time: %ld\n", (long int)(end-start));
243 #endif
244  cInfo("Loaded mesh with %ld faces in %ld ms.", mesh_->faces.size(), (end-start)/1000);
245  return mesh_;
246 }

248 Mesh* LoadSTL(const std::string& path, GLfloat scale)
249 {
250  //Read STL data
251  FILE* file = fopen(path.c_str(), "rb");
252 
253  if(file == NULL)
254  {
255  cCritical("Failed to open geometry file: %s", path.c_str());
256  return nullptr;
257  }
258 
259  cInfo("Loading geometry from: %s", path.c_str());
260 
261  char line[128];
262  char keyword[10];
263  PlainMesh* mesh = new PlainMesh;
264  Vertex v;
265 
266  while(fgets(line, 128, file))
267  {
268  sscanf(line, "%s", keyword);
269 
270  if(strcmp(keyword, "facet")==0)
271  {
272  sscanf(line, " facet normal %f %f %f\n", &v.normal.x, &v.normal.y, &v.normal.z);
273  }
274  else if(strcmp(keyword, "vertex")==0)
275  {
276  sscanf(line, " vertex %f %f %f\n", &v.pos.x, &v.pos.y, &v.pos.z);
277  v.pos *= scale;
278  mesh->vertices.push_back(v);
279  }
280  else if(strcmp(keyword, "endfacet")==0)
281  {
282  unsigned int lastVertexID = (GLuint)mesh->vertices.size()-1;
283 
284  Face f;
285  f.vertexID[0] = lastVertexID-2;
286  f.vertexID[1] = lastVertexID-1;
287  f.vertexID[2] = lastVertexID;
288  mesh->faces.push_back(f);
289  }
290  }
291 
292  fclose(file);
293 
294  //Remove duplicates (so that it becomes equivalent to OBJ file representation)
295 
296  return mesh;
297 }

299 void ComputePhysicalProperties(const Mesh* mesh, Scalar thickness, Scalar density, Scalar& mass, Vector3& CG, Scalar& volume, Vector3& Ipri, Matrix3& Irot)
300 {
301  //1.Calculate mesh volume, CG and mass
302  CG = Vector3(0,0,0);
303  volume = 0;
304 
305  if(thickness > Scalar(0)) //Shell
306  {
307  for(size_t i=0; i<mesh->faces.size(); ++i)
308  {
309  //Get triangle, convert from OpenGL to physics
310  glm::vec3 v1gl = mesh->getVertexPos(i, 0);
311  glm::vec3 v2gl = mesh->getVertexPos(i, 1);
312  glm::vec3 v3gl = mesh->getVertexPos(i, 2);
313  Vector3 v1(v1gl.x,v1gl.y,v1gl.z);
314  Vector3 v2(v2gl.x,v2gl.y,v2gl.z);
315  Vector3 v3(v3gl.x,v3gl.y,v3gl.z);
316 
317  //Calculate volume of shell triangle
318  Scalar A = (v2-v1).cross(v3-v1).length()/Scalar(2);
319  Vector3 triCG = (v1+v2+v3)/Scalar(3);
320  Scalar triVolume = A * thickness;
321  CG += triCG * triVolume;
322  volume += triVolume;
323  }
324 
325  //Compute mesh CG
326  if(volume > Scalar(0))
327  CG /= volume;
328  else
329  CG = Vector3(0,0,0);
330  }
331  else //Solid body
332  {
333  for(size_t i=0; i<mesh->faces.size(); ++i)
334  {
335  //Get triangle, convert from OpenGL to physics
336  glm::vec3 v1gl = mesh->getVertexPos(i, 0);
337  glm::vec3 v2gl = mesh->getVertexPos(i, 1);
338  glm::vec3 v3gl = mesh->getVertexPos(i, 2);
339  Vector3 v1(v1gl.x,v1gl.y,v1gl.z);
340  Vector3 v2(v2gl.x,v2gl.y,v2gl.z);
341  Vector3 v3(v3gl.x,v3gl.y,v3gl.z);
342 
343  //Calculate signed volume of a tetrahedra
344  Vector3 tetraCG = (v1+v2+v3)/Scalar(4);
345  Scalar tetraVolume6 = v1.dot(v2.cross(v3));
346  CG += tetraCG * tetraVolume6;
347  volume += tetraVolume6;
348  }
349 
350  //Compute mesh CG
351  if(volume > Scalar(0))
352  CG /= volume;
353  else
354  CG = Vector3(0,0,0);
355 
356  //Compute mesh volume
357  volume /= Scalar(6);
358  }
359 
360  mass = volume * density;
361 
362  //2.Calculate moments of inertia for local coordinate system located in CG (not necessarily principal)
363  Matrix3 I;
364 
365  if(thickness > Scalar(0)) //Shell - I have doubts if it is correct!
366  {
367  //Compute properties a shell by subtracting the inner solid from the outer solid
368  //Outer solid -> eternal surface
369  Scalar Pxx = Scalar(0);
370  Scalar Pyy = Scalar(0);
371  Scalar Pzz = Scalar(0);
372  Scalar Pxy = Scalar(0);
373  Scalar Pxz = Scalar(0);
374  Scalar Pyz = Scalar(0);
375 
376  for(size_t i=0; i<mesh->faces.size(); ++i)
377  {
378  //Triangle verticies with respect to CG
379  glm::vec3 v1gl = mesh->getVertexPos(i, 0);
380  glm::vec3 v2gl = mesh->getVertexPos(i, 1);
381  glm::vec3 v3gl = mesh->getVertexPos(i, 2);
382 
383  Vector3 v1(v1gl.x,v1gl.y,v1gl.z);
384  Vector3 v2(v2gl.x,v2gl.y,v2gl.z);
385  Vector3 v3(v3gl.x,v3gl.y,v3gl.z);
386  Vector3 n = (v2-v1).cross(v3-v1).normalize();
387  v1 = v1 + n*thickness/Scalar(2) - CG;
388  v2 = v2 + n*thickness/Scalar(2) - CG;
389  v3 = v3 + n*thickness/Scalar(2) - CG;
390 
391  //Pjk = const * dV * (2*Aj*Ak + 2*Bj*Bk + 2*Cj*Ck + Aj*Bk + Ak*Bj + Aj*Ck + Ak*Cj + Bj*Ck + Bk*Cj)
392  Scalar V6 = v1.dot(v2.cross(v3));
393  Pxx += V6 * 2 *(v1.x()*v1.x() + v2.x()*v2.x() + v3.x()*v3.x() + v1.x()*v2.x() + v1.x()*v3.x() + v2.x()*v3.x());
394  Pyy += V6 * 2 *(v1.y()*v1.y() + v2.y()*v2.y() + v3.y()*v3.y() + v1.y()*v2.y() + v1.y()*v3.y() + v2.y()*v3.y());
395  Pzz += V6 * 2 *(v1.z()*v1.z() + v2.z()*v2.z() + v3.z()*v3.z() + v1.z()*v2.z() + v1.z()*v3.z() + v2.z()*v3.z());
396  Pxy += V6 * (2*(v1.x()*v1.y() + v2.x()*v2.y() + v3.x()*v3.y()) + v1.x()*v2.y() + v1.y()*v2.x() + v1.x()*v3.y() + v1.y()*v3.x() + v2.x()*v3.y() + v2.y()*v3.x());

495  //3. Find primary moments of inertia
496  Ipri = Vector3(I.getRow(0).getX(), I.getRow(1).getY(), I.getRow(2).getZ());
497  Irot = I3();
498 
499  if(!IsDiagonal(I)) // If inertia matrix is not diagonal
500  {
501  //3.1. Calculate principal moments of inertia
502  Scalar T = I[0][0] + I[1][1] + I[2][2]; //Ixx + Iyy + Izz
503  Scalar II = I[0][0]*I[1][1] + I[0][0]*I[2][2] + I[1][1]*I[2][2] - I[0][1]*I[0][1] - I[0][2]*I[0][2] - I[1][2]*I[1][2]; //Ixx Iyy + Ixx Izz + Iyy Izz - Ixy^2 - Ixz^2 - Iyz^2
504  Scalar U = btSqrt(T*T-Scalar(3)*II)/Scalar(3);
505  Scalar theta = btAcos((-Scalar(2)*T*T*T + Scalar(9)*T*II - Scalar(27)*I.determinant())/(Scalar(54)*U*U*U));
506  Scalar A = T/Scalar(3) - Scalar(2)*U*btCos(theta/Scalar(3));
507  Scalar B = T/Scalar(3) - Scalar(2)*U*btCos(theta/Scalar(3) - Scalar(2)*M_PI/Scalar(3));
508  Scalar C = T/Scalar(3) - Scalar(2)*U*btCos(theta/Scalar(3) + Scalar(2)*M_PI/Scalar(3));
509  Ipri = Vector3(A, B, C);
510 
511  //3.2. Calculate principal axes of inertia
512  Matrix3 L;
513  Vector3 axis1,axis2,axis3;
514  axis1 = FindInertialAxis(I, A);
515  axis2 = FindInertialAxis(I, B);
516  axis3 = axis1.cross(axis2);
517  axis2 = axis3.cross(axis1);
518 
519  //3.3. Rotate body so that principal axes are parallel to (x,y,z) system
520  Irot = Matrix3(axis1[0],axis2[0],axis3[0], axis1[1],axis2[1],axis3[1], axis1[2],axis2[2],axis3[2]);
521  }
522 }
