106 void DCMotor::Update(Scalar dt)
107 {
108  //Get joint angular velocity in radians
109  Scalar aVelocity = getAngularVelocity();
110 
111  //Calculate internal state and output
112  torque = (I * Kt - aVelocity * B) * gearRatio * gearEff;
113  Scalar VoverL = (V - aVelocity * Ke * 9.5493 - I * R)/L;
114  I += VoverL * dt;
115 
116  //Hack to avoid system blowup when the motor starts (shortcut)
117  if((btFabs(I) > btFabs(V/R)) && (I*V > Scalar(0)))
118  I = V/R;
119 
120  //I += Scalar(0.5) * (VoverL + lastVoverL) * dt; //Integration (mid-point)
121  //lastVoverL = VoverL;
122 
123  //Drive the joint
124  Motor::Update(dt);
125 }
