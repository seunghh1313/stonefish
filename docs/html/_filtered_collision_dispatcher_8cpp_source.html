43 bool FilteredCollisionDispatcher::needsCollision(const btCollisionObject* body0, const btCollisionObject* body1)
44 {
45  bool needs = btCollisionDispatcher::needsCollision(body0, body1);
46  if(!needs)
47  return false;
48 
49  Entity* ent0 = (Entity*)body0->getUserPointer();
50  if(ent0 == nullptr)
51  return false;
52 
53  Entity* ent1 = (Entity*)body1->getUserPointer();
54  if(ent1 == nullptr)
55  return false;
56 
57  if(inclusive)
58  needs = SimulationApp::getApp()->getSimulationManager()->CheckCollision(ent0, ent1) > -1;
59  else //exclusive
60  needs = SimulationApp::getApp()->getSimulationManager()->CheckCollision(ent0, ent1) == -1;
61 
62  return needs;
63 }

65 void FilteredCollisionDispatcher::myNearCallback(btBroadphasePair& collisionPair, btCollisionDispatcher& dispatcher, const btDispatcherInfo& dispatchInfo)
66 {
67  btCollisionObject* colObj0 = (btCollisionObject*)collisionPair.m_pProxy0->m_clientObject;
68  btCollisionObject* colObj1 = (btCollisionObject*)collisionPair.m_pProxy1->m_clientObject;
69 
70  if(dispatcher.needsCollision(colObj0,colObj1))
71  {
72  btCollisionObjectWrapper obj0Wrap(0, colObj0->getCollisionShape(), colObj0, colObj0->getWorldTransform(), -1, -1);
73  btCollisionObjectWrapper obj1Wrap(0, colObj1->getCollisionShape(), colObj1, colObj1->getWorldTransform(), -1, -1);
74 
75  //Dispatcher will keep algorithms persistent in the collision pair
76  if(!collisionPair.m_algorithm)
77  collisionPair.m_algorithm = dispatcher.findAlgorithm(&obj0Wrap, &obj1Wrap, 0, BT_CONTACT_POINT_ALGORITHMS);
78 
79  if(collisionPair.m_algorithm)
80  {
81  btManifoldResult contactPointResult(&obj0Wrap,&obj1Wrap);
82 
83  if (dispatchInfo.m_dispatchFunc == btDispatcherInfo::DISPATCH_DISCRETE)
84  {
85  //discrete collision detection query
86  collisionPair.m_algorithm->processCollision(&obj0Wrap, &obj1Wrap, dispatchInfo, &contactPointResult);
87  }
88  else
89  {
90  //continuous collision detection query, time of impact (toi)
91  Scalar toi = collisionPair.m_algorithm->calculateTimeOfImpact(colObj0, colObj1, dispatchInfo, &contactPointResult);
92  if (dispatchInfo.m_timeOfImpact > toi)
93  dispatchInfo.m_timeOfImpact = toi;
94  }
95  }
96  }
97 }
