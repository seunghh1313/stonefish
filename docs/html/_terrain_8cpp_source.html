36 Terrain::Terrain(std::string uniqueName, std::string pathToHeightmap, Scalar scaleX, Scalar scaleY, Scalar height, std::string material, std::string look, float uvScale)
37  : StaticEntity(uniqueName, material, look)
38 {
39  //Load heightmap data
40  int w, h, ch;
41  GLfloat* heightmap;
42 
43  if(stbi_is_16_bit(pathToHeightmap.c_str())) //16 bit image
44  {
45  stbi_us* data = stbi_load_16(pathToHeightmap.c_str(), &w, &h, &ch, 1);
46  if(data == NULL) cCritical("Failed to load heightmap from file '%s'!", pathToHeightmap.c_str());
47  heightmap = new GLfloat[w*h];
48  for(int i=0; i<h; ++i)
49  for(int j=0; j<w; ++j)
50  heightmap[i*w+j] = (1.f - data[i*w+j]/(GLfloat)(__UINT16_MAX__)) * height;
51  stbi_image_free(data);
52  }
53  else //8 bit image
54  {
55  stbi_uc* data = stbi_load(pathToHeightmap.c_str(), &w, &h, &ch, 1);
56  if(data == NULL) cCritical("Failed to load heightmap from file '%s'!", pathToHeightmap.c_str());
57  heightmap = new GLfloat[w*h];
58  for(int i=0; i<h; ++i)
59  for(int j=0; j<w; ++j)
60  heightmap[i*w+j] = (1.f - data[i*w+j]/(GLfloat)(__UINT8_MAX__)) * height;
61  stbi_image_free(data);
62  }
63 
64  //Calculate max height
65  maxHeight = Scalar(0);
66  heightfield = new Scalar[w*h];
67  for(int i=0; i<h; ++i)
68  for(int j=0; j<w; ++j)
69  {
70  heightfield[i*w+j] = Scalar(heightmap[i*w+j]);
71  maxHeight = heightfield[i*w+j] > maxHeight ? heightfield[i*w+j] : maxHeight;
72  }
73 
74  //Generate graphical mesh
75  phyMesh = OpenGLContent::BuildTerrain(heightmap, w, h, scaleX, scaleY, (GLfloat)maxHeight, uvScale);
76  delete [] heightmap;
77 
78  //Generate collision mesh
79  btHeightfieldTerrainShape* shape = new btHeightfieldTerrainShape(w, h, heightfield, Scalar(1), Scalar(0), maxHeight, 2, PHY_FLOAT, false);
80  shape->setLocalScaling(Vector3(scaleX, scaleY, 1.0));
81  shape->setUseDiamondSubdivision(true);
82  shape->setMargin(0);
83  BuildRigidBody(shape);
84 }
