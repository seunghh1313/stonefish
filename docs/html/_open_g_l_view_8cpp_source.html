95 void OpenGLView::ExtractFrustumFromVP(glm::vec4 frustum[6], const glm::mat4& VP)
96 {
97  // left
98  frustum[0].x = VP[0][3] + VP[0][0];
99  frustum[0].y = VP[1][3] + VP[1][0];
100  frustum[0].z = VP[2][3] + VP[2][0];
101  frustum[0].w = VP[3][3] + VP[3][0];
102  frustum[0] = glm::normalize(frustum[0]);
103 
104  // right
105  frustum[1].x = VP[0][3] - VP[0][0];
106  frustum[1].y = VP[1][3] - VP[1][0];
107  frustum[1].z = VP[2][3] - VP[2][0];
108  frustum[1].w = VP[3][3] - VP[3][0];
109  frustum[1] = glm::normalize(frustum[1]);
110 
111  // top
112  frustum[2].x = VP[0][3] + VP[0][1];
113  frustum[2].y = VP[1][3] + VP[1][1];
114  frustum[2].z = VP[2][3] + VP[2][1];
115  frustum[2].w = VP[3][3] + VP[3][1];
116  frustum[2] = glm::normalize(frustum[2]);
117 
118  // bottom
119  frustum[3].x = VP[0][3] - VP[0][1];
120  frustum[3].y = VP[1][3] - VP[1][1];
121  frustum[3].z = VP[2][3] - VP[2][1];
122  frustum[3].w = VP[3][3] - VP[3][1];
123  frustum[3] = glm::normalize(frustum[3]);
124 
125  // near
126  frustum[4].x = VP[0][3] + VP[0][2];
127  frustum[4].y = VP[1][3] + VP[1][2];
128  frustum[4].z = VP[2][3] + VP[2][2];
129  frustum[4].w = VP[3][3] + VP[3][2];
130  frustum[4] = glm::normalize(frustum[4]);
131 
132  // far
133  frustum[5].x = VP[0][3] - VP[0][2];
134  frustum[5].y = VP[1][3] - VP[1][2];
135  frustum[5].z = VP[2][3] - VP[2][2];
136  frustum[5].w = VP[3][3] - VP[3][2];
137  frustum[5] = glm::normalize(frustum[5]);
138 }
